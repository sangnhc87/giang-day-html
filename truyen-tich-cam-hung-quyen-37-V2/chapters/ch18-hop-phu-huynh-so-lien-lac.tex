\chapter{Họp Phụ Huynh Và Sổ Liên Lạc}
\chapterintro{Có những ngày học sinh ngoan bất thường vì lịch họp đã đến gần}{Nỗi sợ họp phụ huynh là cảm xúc chung đáng được thơ ghi lại.}

\begin{lucbatbox}{Lục bát: Trước ngày họp phụ huynh}
Mấy hôm nay em chăm ngoan
Lễ phép hơn cả bình thường gấp ba
Mẹ hỏi sao lại thế mà
Em cười không nói thật ra sắp họp
Sổ kia ghi đủ bao dòng
Nên em cố sống như không lỗi gì
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Sổ liên lạc mở ra}
Sổ liên lạc mở giữa đêm nay
Mẹ con cùng thấy lạnh trong tay
Con chữ nằm im không nói tiếng
Mà nghe vang động cả căn nhà
\end{tutuyetbox}

\begin{ghichu}
Họp phụ huynh là lúc nhiều học sinh bỗng nhớ lại toàn bộ lời nhắc của giáo viên trong học kỳ.
\end{ghichu}

\begin{lucbatbox}{Lục bát: Xin cô đừng nói kỹ}
Em mong cô nói nhẹ nhàng
Nếu cần góp ý xin sang ý hiền
Đừng đào quá khứ triền miên
Kẻo về nhà lại thêm phiên điều trần
Mong sao buổi họp êm đềm
Để tim em đỡ thức thêm mấy ngày
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Tan họp đi về}
Tan họp em đi cạnh mẹ thôi
Không khí gia đình nhẹ chẳng thôi
Mỗi bước lại nghe lòng tự hỏi
Từ nay có chịu học hơn không
\end{tutuyetbox}
