\chapter{Ban be, phe nhom va ap luc hoc duong}

\begin{leadbox}
Hoc sinh it khi sai mot minh. Xung quanh cac em la nhom ban, su thua nhan, noi so bi tach khoi tap the. Chuong nay nhin vao cai gia cua viec muon thuoc ve va nghe thuat giup hoc sinh dung truoc ap luc dong dang.
\end{leadbox}

\storycard{31. Ban ngoi chung va phe nho}{The Small Clique}{Nhung nhom than co the la noi nang do, nhung cung co the thanh vung loai tru trong lop.}
\storycard{32. Bi dat biet danh}{The Nickname}{Mot biet danh vui voi nguoi nay co the la vet dau với nguoi khac.}
\storycard{33. Chuyen choi xau tren nhom chat}{The Group Chat Trap}{Bat nat hoc duong hien dai thuong bat dau bang icon cuoi va man hinh sang.}
\storycard{34. Doi ban gian lan}{Cheating as a Team}{Khi quay cop tro thanh tinh dong doi, can day hoc sinh rang su lien ket cung co dao duc cua no.}
\storycard{35. Ban moi chuyen truong}{The New Student}{Mot lop hoc duoc do bang cach no tiep nhan nguoi moi chu khong phai bang khau hieu treo tren tuong.}
\storycard{36. Nguoi dung ra benh ban}{The One Who Takes the Side}{Mot hoc sinh dung len noi cong bang co the bi co lap, nhung do la khoanh khac nhan cach sang nhat.}
\storycard{37. Thich duoc cong nhan}{Needing Approval}{Rat nhieu hanh vi xau chi la cach vuong ve de noi rang em muon duoc thay.}
\storycard{38. Cuoc thi ngam trong lop}{The Silent Competition}{Gan dua lien tuc khien ban be thanh doi thu, va lop hoc mat di niem vui hoc.}
\storycard{39. Phe cua nguoi gioi}{The Top-Student Circle}{Ngay ca hoc sinh gioi cung co the vo tinh tao ra buc tuong vo hinh ngan cach voi phan con lai.}
\storycard{40. Mot cho ngoi trong}{The Empty Seat}{Khi ca lop tranh ngoi canh mot nguoi, do la tin hieu can can thiep som hon moi bai ky luat.}
