\chapter{Nhắn Tin Xuyên Đêm}
\chapterintro{Ban ngày gật gù, ban đêm lại tỉnh như chưa từng buồn ngủ}{Nhiều cuộc trò chuyện đẹp bắt đầu sau mười một giờ và kết thúc bằng tiết đầu gục mặt xuống bàn.}

\begin{lucbatbox}{Lục bát: Chỉ nhắn chút thôi}
Đêm nay em nhắn một câu\
Ngờ đâu câu ấy kéo nhau tới gần\
Mười hai rồi một, rồi gần\
Hai giờ mắt vẫn tần ngần chưa thôi\
Sáng ra đi học bơ phờ\
Tình thì sáng tỏ, còn giờ thì đau
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Báo thức rất oan}
Đêm qua trò chuyện quá vui rồi
Sáng nay báo thức nghe buồn ghê
Không phải chuông này làm em ghét
Chỉ vì em ngủ ít quá thôi
\end{tutuyetbox}

\begin{ghichu}
Nhắn tin xuyên đêm luôn cho cảm giác mình đang sống rất sâu. Tiết một hôm sau thường có cảm nhận khác.
\end{ghichu}

\begin{lucbatbox}{Lục bát: Seen rồi lại chờ}
Tin đi em đợi từng hồi\
Thấy dòng đã xem mà vẫn bồi hồi hơn\
Ngủ đâu chẳng thấy, chỉ còn\
Màn hình sáng nhẹ như dồn tim nghe\
Nếu mà chờ bài như thế\
Đời em chắc đã khác ghê một đời
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Tỉnh vì một chữ}
Chỉ một tin nhắn rất ngắn thôi
Mà em thức trắng đến gần trời
Nếu việc học cần em bền vậy
Tên đâu đứng mãi phía sau rồi
\end{tutuyetbox}
