\chapter{Bai Hoc: Khiem Ton Tao Nen Vi Dai (Humility Creates Greatness)}
\begin{center}
  \textit{\color{truyenblue}Cay cao nhat cung coi nhieu nhat trong bao; vi dai biet khi nao can khiem ton.}\\
  \textit{(The tallest trees bow the most in storms; greatness knows when to be humble.)}\\[4pt]
  \small\color{truyengold}--- Thien nhien day ta ---
\end{center}
\ngancach

\section{Tuyet Tung Cung Ra Qua Nho}
\begin{truyen}{Tuyet Tung Cung Ra Qua Nho}{Cay Tuyet Tung}
\chuhoa{D}{}D ai bang cao lon nhat rung, cay tuyet tung thu khong he phu bong ngan non cac cay nho.\\
\textit{(Though tallest in the forest, the cedar does not shade and block young trees below.)}

Re no toa song giup cac cay nho hut nuoc va khoang chat tu dat am bao quanh.\\
\textit{(Its roots spread sharing with young trees to absorb water and minerals from the surrounding moist soil.)}

Moi nam no cho qua nho xiu chua chac, khong sang phat rang rang nhu nhieu loai khac.\\
\textit{(Each year it bears tiny sour fruits, nothing flashy and dazzling like many other species.)}

Nhung khu rung no tao ra la noi dieu duong cua ca tram loai sinh vat khac nhau.\\
\textit{(But the forest it creates shelters hundreds of different species of creatures.)}

Vi dai that su khong can phai duoc chung kien hay ta thuong; no tu the hien qua nhung gi no tao nen.\\
\textit{(True greatness needs no witness or praise; it expresses itself through what it creates.)}

\end{truyen}
\begin{baihoc}
  Nguoi vi dai nhat trong phong khong phai nguoi noi nhieu nhat ma nguoi de nguoi khac toa sang.\\
  \textit{(The greatest person in the room is not who speaks most but who lets others shine.)}
\end{baihoc}

\section{Vooc Chua Phong La Gui}
\begin{truyen}{Vooc Chua Phong La Gui}{Vooc Gia}
\chuhoa{K}{}K hi vooc tre thang loi trong cuoc tu do chon lanh dao moi, vooc gia dung lai nhin.\\
\textit{(When a young vervet wins the contest to choose a new leader, the old one stands aside and watches.)}

No khong chong cu, khong tim cach quan ly bong gio; no trao quyen mot cach yen lang.\\
\textit{(It does not resist or seek to manage from behind the scenes; it surrenders power quietly.)}

Suot nhung tuan tiep theo no hoa minh vao bays, di sau gan cuoi hang de khong gay ra canh tranh.\\
\textit{(In the following weeks it blends into the troop, walking near the back to avoid creating competition.)}

Nhung con tre van tim den no khi can loi khuyen; no tra loi roi bo di khong de lai bong.\\
\textit{(Young ones still seek it for advice; it answers then walks away leaving no shadow.)}

Nghiet ngha khiem ton la cho qua va binh an khi ta con u quyen luc, khong phai khi ta bat buoc phai tu bo.\\
\textit{(The hardest humility is surrendering power while still possessing it, not when forced to relinquish it.)}

\end{truyen}
\begin{baihoc}
  Biet lui ve de nguoi khac tien len la hanh dong vi dai hon la cu bam giu quyen luc den hoi thu cuoi cung.\\
  \textit{(Knowing when to step back so others can step forward is a greater act than clinging to power until the last breath.)}
\end{baihoc}

\section{Chim Non Hoc Bay}
\begin{truyen}{Chim Non Hoc Bay}{Chim Non}
\chuhoa{C}{}C on chim non truoc khi bay lan dau dung o vach to, nhin xuong vuc sau manh dat xa.\\
\textit{(The young bird before its first flight stands on the nest's edge, looking down at the distant far ground.)}

No run ram, but lui lai, nhay len mac du co canh nhung van so hai khon ta.\\
\textit{(It trembles, steps back, pushes forward though having wings but still frightened beyond words.)}

Me chim dung phia sau, khong day manh; chi tao khong gian de no tu quyet dinh.\\
\textit{(The mother bird stands behind, not pushing hard; only creating space for it to decide on its own.)}

Khi no nhay xo tu vach to, no cam thay gio nang do cap duoi canh lan dau tien.\\
\textit{(When it leaps from the nest's edge, it feels the wind lifting beneath its wings for the first time.)}

Khiem ton la nho rang du ta co canh, lan dau bay ta van can gio cua hoan canh nang do.\\
\textit{(Humility is remembering that even though we have wings, the first flight still needs the wind of circumstance to lift us.)}

\end{truyen}
\begin{baihoc}
  Kho khan va canh truong thanh luon di doi: khiem ton truoc canh truong la khon ngoan that su.\\
  \textit{(Difficulty and growth always go together: humility before growth is true wisdom.)}
\end{baihoc}

\section{Cay Lua Cong Than}
\begin{truyen}{Cay Lua Cong Than}{Cay Lua}
\chuhoa{C}{}C ay lua khac cay co dan biet rang hat lu nang hon tra ne cang to cang cong than.\\
\textit{(The rice plant unlike arrogant grass: the heavier the grain the more the stem bends.)}

Nong dan nhin dan lua nhin nua bong lua la biet cay nao chin tot cay nao con xanh non.\\
\textit{(Farmers by looking at how the rice bends know which plants are ripe and which are still young.)}

Cay nao than thang dung toc doc la cay con re, con xanh hoac de trong.\\
\textit{(Any plant with a stalk standing stiff and straight is young, immature, or has few grains.)}

Nhung cay gi nhe nhat la cay khinh thuong nhat; nhung cay gi nang nhat cung cong tan nhat.\\
\textit{(The lightest stalks are the most arrogant; the heaviest stalks bow the deepest.)}

Cuoc doi giong dong lua: Cang tich luy nhieu bai hoc va trai nghiem, ta cang bien ra khiem nhung hon.\\
\textit{(Life is like a rice field: The more lessons and experiences we accumulate, the more humble we become.)}

\end{truyen}
\begin{baihoc}
  Kien thuc that su lam con nguoi khiem ton; chi co su thieu hieu biet moi khien nguoi ta kieu ngao.\\
  \textit{(True knowledge makes a person humble; only lack of understanding makes people arrogant.)}
\end{baihoc}

\section{Nhen Co Tu Chi Viet Truyen}
\begin{truyen}{Nhen Co Tu Chi Viet Truyen}{Nhen Nho}
\chuhoa{N}{}N hen nho day ra khoi bao ca vao vach nua bong toi, minh mot khong ai quan tam.\\
\textit{(A tiny spider climbs from mid-web onto the half-dark wall, alone with no one paying attention.)}

No bat dau dan to moi tu nhung cuoi chi mong manh nhat ma no co tu thanh phan than the.\\
\textit{(It begins weaving a new web from the finest silk threads it has from components in its own body.)}

Tam luoi tuyen my hinh thanh qua dem, bot suong bam doi vo dem thanh chuoi hat ngoc.\\
\textit{(The exquisite web forms overnight, dew drops clinging in the dark becoming a string of pearls.)}

Nhen khong noi rang no da tao nen tac pham sang tao; no chi tiep tuc tao va song.\\
\textit{(The spider does not announce it has created a masterpiece; it simply continues creating and living.)}

Khiem ton trong viec tao ra dieu dep tuc la lam viec vi chinh cong viec, khong vi tieng vang.\\
\textit{(Humility in creating beauty means working for the work itself, not for the echo of applause.)}

\end{truyen}
\begin{baihoc}
  Giam tri gia cua cong viec minh lam bang viec koe hang nhu la gi to lon chang la dau; hay lam tot roi de doi noi.\\
  \textit{(Do not diminish your work's value by loud boasting; do it well and let the world speak.)}
\end{baihoc}

\section{Ca Voi Nho Vang Tieng}
\begin{truyen}{Ca Voi Nho Vang Tieng}{Ca Voi}
\chuhoa{K}{}K hi ca voi lua den vung sinh san, no khong het len goi dan khong gap mat lan nhau.\\
\textit{(When a young whale arrives at the breeding ground, it does not shout loudly calling others before meeting anyone.)}

No phat ra am thanh nho, chu dong lang nghe phan hong tu ca voi khac truoc.\\
\textit{(It emits soft sounds, actively listening for responses from other whales first.)}

Chi khi am thanh no duoc tra loi ung thuan no moi phat am thanh lon hon va tro nen ro mat.\\
\textit{(Only when its sounds receive affirming responses does it emit louder sounds and make itself known.)}

He thong giao tiep nay ngan chan xung dot va dam bao noi thu nhat khong bi de doa.\\
\textit{(This communication system prevents conflict and ensures the newcomer is not threatened.)}

Den vao mit cua mot cong dong la biet lang nghe truoc roi moi noi, bat ke ban vi dai den dau.\\
\textit{(Entering the silence of a community means listening before speaking, no matter how great you are.)}

\end{truyen}
\begin{baihoc}
  Khiem ton la lang nghe du ta biet; vi khi ta thuc su lang nghe ta se hoc duoc dieu minh CHUA biet.\\
  \textit{(Humility is listening even when you already know; because when you truly listen you learn what you do NOT yet know.)}
\end{baihoc}

\section{Bao Chua Biet Viec Hay Khong}
\begin{truyen}{Bao Chua Biet Viec Hay Khong}{Con Bao}
\chuhoa{T}{}T ro ve trong chau Phi mot con bao khong lo, cac con thiet xau gap bao lan chau bao.\\
\textit{(In African leopard territory, a leopard unmatched in size compared to leopards many times smaller.)}

No san moi mot minh, giu xac moi tren cay cao de khong chia se voi ai khac.\\
\textit{(It hunts alone, keeps prey up high in trees to not share with anyone else.)}

Nhung mot cua khi gia hoang loai gay tro nan khong buoc duoc tren dat bang phang nua.\\
\textit{(But the day of old age came and it could no longer leap or walk on flat ground properly.)}

Khong mot con vat nao trong luc du no tung goi mai luong cuu giup; no chet don doc.\\
\textit{(Not one animal came when it once only would have called for food sharing; it died alone.)}

Su tu cach tram hoc khong chia se su giup do cuoi cung khi can cung se khong den.\\
\textit{(The arrogance of never sharing nor helping ultimately ensures no help will come when needed.)}

\end{truyen}
\begin{baihoc}
  Khiem ton va chia se khi manh me la dau tu cho ngay ban yeu duoi va can nguoi khac.\\
  \textit{(Humility and sharing when strong is an investment for the day you are weak and need others.)}
\end{baihoc}

\section{Trau Co Biet Co Non}
\begin{truyen}{Trau Co Biet Co Non}{Trau Co}
\chuhoa{T}{}T rau do rung lay mot ngon co non va non toi truoc mat no trong ngay nang nong.\\
\textit{(A wild buffalo spotted a tender sprout right in front of it on a hot dry day.)}

Thay vi an ngay, no de lai va di tim co kho va gia hon xa hoi cho cai mieng no.\\
\textit{(Instead of eating right away, it left it and walked to find older drier grass farther away to satisfy its hunger.)}

No thay rang ngon co non con nhieu co hoi lon len nhieu hon neu no de cho no song.\\
\textit{(It understood the young sprout had more chances to grow larger if allowed to live.)}

Mot con sat de lai mot ngon co non vai ngay, cho no xanh um roi moi an no.\\
\textit{(A few days later it returned to find the sprout fully green and lush, then it ate it.)}

Khiem ton cua loai an co la biet rang khong can phai an het ngay hom nay de dam bao ngay mai.\\
\textit{(The humility of a grazer: understanding there is no need to devour everything today to ensure tomorrow.)}

\end{truyen}
\begin{baihoc}
  Khiem ton vat chat va kinh te: dung an het tat ca khi co the; hay de lai de ngay mai co cai ma an.\\
  \textit{(Material and economic humility: don't consume everything when you can; leave some for tomorrow to have something to eat.)}
\end{baihoc}

\section{Cay Ho Phong Tran Bao Tu}
\begin{truyen}{Cay Ho Phong Tran Bao Tu}{Ca Ho}
\chuhoa{T}{}T rong mua sinh san, ca ho duc doc dao va long lay de thu hut ca ho cai bang mau sac ruc ro.\\
\textit{(During mating season, male peacock wrasse dives deep and glows to attract females with brilliant colors.)}

Nhung khi nguoi san bat den, mau sac ruc ro nay tro thanh su dich chuyen nguy hiem.\\
\textit{(But when hunters come, these brilliant colors become a dangerous billboard.)}

Ca ho duc hoc duoc cach tat mau den tuc thi khi cam nhan moi nguy hiem xung quanh.\\
\textit{(The male wrasse learned to instantly darken its colors when sensing any nearby danger.)}

Biet tat di su noi bat khi hoan canh doi hoi la dang khiem ton linh hoat bac nhat.\\
\textit{(Knowing how to dim your prominence when circumstances demand it is the highest form of flexible humility.)}

Khong phai luc nao ta cung can phu truong; biet khi nao im lang cung la tri tue.\\
\textit{(Not every moment requires display; knowing when to be quiet is also wisdom.)}

\end{truyen}
\begin{baihoc}
  Lanh dao thong minh biet luc nao nen sang tao noi bat va luc nao nen hoa minh vao nen de bao toan.\\
  \textit{(The smart leader knows when to stand out brilliantly and when to blend into the background for safety.)}
\end{baihoc}

\section{Khi Co Bau Nhuong Thuc An}
\begin{truyen}{Khi Co Bau Nhuong Thuc An}{Khi}
\chuhoa{K}{}K hi co khi cai mang bau trong bays khi, cac con duc tim kiem va mang lai nhieu thuc an nhat.\\
\textit{(When a female monkey in the troop is pregnant, the males seek and bring the most food.)}

Khi co bau an truoc, cac con khac doi cho du chung cung dang doi va thieu nguon an.\\
\textit{(The pregnant female eats first, others wait even though they too are hungry and food is scarce.)}

Day khong phai cam thay vui ve boi cam giac doi; day la lua chon khiem ton truoc su suu tam thuyet.\\
\textit{(This is not a pleasant feeling from hunger; it is the humble choice before the survival of the species.)}

Su to truong cua loai duoc dat len tren su tiet hanh riet cua ca nhan; do la tri tue tap the.\\
\textit{(The growth of the species is placed above individual gratification; that is collective wisdom.)}

Nhung xa hoi manh nhat la nhung xa hoi biet nhuong truoc cho nguoi de bi ton thuong nhat.\\
\textit{(The strongest societies are those that know to yield first for their most vulnerable members.)}

\end{truyen}
\begin{baihoc}
  Khiem ton don gian la biet rang ban khong phai trung tam vu tru; co nguoi xung dang duoc uu tien hon.\\
  \textit{(Humility is simply knowing you are not the center of the universe; someone else deserves priority more.)}
\end{baihoc}
