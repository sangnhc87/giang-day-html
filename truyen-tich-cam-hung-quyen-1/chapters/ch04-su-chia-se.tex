\chapter{Sự Chia Sẻ (Sharing)}

\begin{center}
  \textit{\color{truyenblue}``Happiness is quite unshared can scarcely be called happiness; it has no taste.''}\\
  \textit{(Hạnh phúc mà không được chia sẻ thì hiếm khi được gọi là hạnh phúc; nó không có hương vị gì cả.)}\\[4pt]
  \small\color{truyengold}--- Charlotte Brontë ---

\ngayDay{}
\section{Cho Đi Vô Điều Kiện}
\begin{truyen}{Cho Đi Vô Điều Kiện}{Chuyện ngụ ngôn}
\chuhoa{A}{ } \textbf{large apple tree loved} a \textbf{little boy}.\\
\textit{(Một cây táo khổng lồ vô cùng yêu thương một cậu bé nhỏ.)}

The \textbf{boy played} around the \textbf{tree every day}.\\
\textit{(Cậu bé cứ vô tư vui đùa lẩn quẩn quanh gốc cây mỗi ngày.)}

When \textbf{he was hungry}, it \textbf{gave him sweet apples}.\\
\textit{(Khi cậu cảm thấy đói bụng, nó tận tụy thả cho cậu những quả táo ngọt ngào.)}

When \textbf{he was tired}, it \textbf{provided cool shade}.\\
\textit{(Khi cậu thấm mệt, nó bung xòe bộ tán cung cấp cho cậu một bóng râm mát mẻ.)}

The \textbf{tree was happy} just to \textbf{give}.\\
\textit{(Chỉ riêng một việc được cho đi thôi cũng đủ làm cái cây đó thấy ngập tràn hạnh phúc.)}
\end{truyen}

\begin{baihoc}
  \textbf{True love is giving without expecting anything in return.} \textit{(Tình yêu thương đích thực luôn là một sự cho đi mà không màng mong ngóng nhận lại bất cứ một điều xa hoa gì.)}
\end{baihoc}

\ngayDay{}
\section{Chia Vớt}
\begin{truyen}{Chia Vớt}{Cuộc sống}
\chuhoa{T}{wo} \textbf{poor brothers had} only \textbf{one piece of bread}.\\
\textit{(Hai anh em trai nghèo xơ xác chỉ sở hữu lỏng thỏng đúng một mảnh vụn bánh mì dài duy nhất.)}

Both \textbf{were extremely hungry}.\\
\textit{(Và dĩ nhiên cả hai đứa đểu đang chịu trận vô cùng cồn cào đói khát.)}

The \textbf{older brother split} it into \textbf{two parts}.\\
\textit{(Người anh trai vội vã bẻ đôi mẩu bánh làm hai phần rưỡi tách biệt.)}

He \textbf{gave the bigger piece} to his \textbf{younger brother}.\\
\textit{(Rồi cậu nhường phần bánh nhỉnh miếng to hơn ấy cho đứa em non nớt của mình.)}

\textbf{Hunger faded} when \textbf{they smiled} at each other.\\
\textit{(Cảm giác thèm khát đói ăn bỗng mờ tịt nhạt nhòa khi chúng tươi rói mỉm cười nhìn vào mặt nhau.)}
\end{truyen}

\begin{baihoc}
  \textbf{When you share, love multiplies.} \textit{(Một khi bạn chịu sẻ nhường, dung lượng của tình yêu thương sẽ được tự do nhân rộng lên vô biên.)}
\end{baihoc}

\section{Món Quà Nhỏ}
\begin{truyen}{Món Quà Nhỏ}{Thiên nhiên}
\chuhoa{A}{ } \textbf{blade of grass was thirsty} in the \textbf{hot sun}.\\
\textit{(Một cọng cỏ gai dại đang trong tình trạng cực kỳ khát khô cổ dưới một bầu mặt trời bỏng oi ả hực hực.)}

A \textbf{drop of dew fell} onto its \textbf{leaf}.\\
\textit{(Một hạt sương sớm giọt trong veo đã mỏng manh thả rơi đậu đậu đậu lên chiếc mặt lá xanh của nó.)}

The \textbf{dew said}: \textit{"Drink me to live."}\\
\textit{(Giọt sương thủ thỉ nỉ non nhỏ nhẹ: "Hãy hút uống cạn ta đi để cậu được tiếp sức sống sót.")}

The \textbf{grass lived}, but the \textbf{dew disappeared}.\\
\textit{(Cuổi cùng cọng cỏ đã sống sót qua cơn thiêu, nhưng hạt sương kiều diễm kia lại vĩnh viễn bốc hơi tan bay đi mất tích.)}
\end{truyen}

\begin{baihoc}
  \textbf{Some gifts require brave sacrifice.} \textit{(Rất nhiều món quà trị giá đòi hỏi cần bắt buộc phải đánh đổi bằng cả một sự hy sinh dũng cảm.)}
\end{baihoc}

\section{Soi Rọi}
\begin{truyen}{Soi Rọi}{Đời thường}
\chuhoa{A}n \textbf{old man walked} in the \textbf{dark holding a bright lamp}.\\
\textit{(Một bóng ông lão già cả đang lẫm chẫm bước bộ lầm lũi trong một màn đêm khuya khoắt tay cầm xách theo một chiếc đèn lồng dầu bừng vụt sáng tỏa tỏ.)}

A \textbf{young man asked}: \textit{"Why \textbf{do you need a lamp}? You \textbf{are blind}."}\\
\textit{(Một gã thanh niên trẻ tuổi nhạo hiếu kì cắc cớ hỏi vặn lại: "Lão cần quái gì phải rọi vác theo cái thứ đèn củi kia? Lão vốn là một gã lão mù mà.")}

He \textbf{agreed politely}: \textit{"Yes, I \textbf{can not see}. The \textbf{light is for others}."}\\
\textit{(Cụ già lịch duyệt khẽ gật lẫy mỉm cười điềm đạm thuận theo: "Đúng vậy, ta quả thực là chả thể dòm ngó thấy gì. Ánh sáng này đơn thuần là để soi rọi dành cho những lữ kẻ khác.")}

\textit{"So \textbf{they don't bump into me}."}\\
\textit{("Để họ không vô ý vô tứ sơ suất lậu đụng vấp xô vấp xát vào tấm thân già này.")}
\end{truyen}

\begin{baihoc}
  \textbf{Thinking of others protects everyone.} \textit{(Biết nghĩ cặn kẽ cho những người khác cũng là một phương thức vĩ đại bảo vệ vẹn bảo đảm an toàn chung cho tất cả mọi chùm sinh linh.)}
\end{baihoc}

\section{Mảnh Ghép}
\begin{truyen}{Mảnh Ghép}{Đoàn kết}
\chuhoa{C}{hildren} \textbf{in an orphanage felt very cold}.\\
\textit{(Tụi nhóc tỳ sống ở một cô nhi viện đang mếu máo nằm co ro phải cắn răng cảm thấy một cái lạnh đọng cực độ đông thấu.)}

\textbf{People donated square scraps} of \textbf{old clothes}.\\
\textit{(Nhiều nhà dân xung xở hảo tâm đã quyên góp rải rác từng cuộn mảnh vải chắp vá vụn chắp lẻ loi lấy cắt từ các đống nùi quần áo cũ sờn.)}

A \textbf{kind nun sewed them} all \textbf{together}.\\
\textit{(Một bà sơ mụn thánh thiện nhân ái đã tần tảo ngồi chắp mạng và đã kiên nhẫn khâu viền ráp dính tất thảy các mảnh nùi đó lại cùng với nhau liền mảnh.)}

It \textbf{became} a \textbf{huge, warm, colorful blanket}.\\
\textit{(Thành quả một dạt đắp thành chăn nệm cộm xù to sụ khổng lồ sặc sỡ nhấp nhô muôn vàn màu sắc mà lại cực kỳ ủ ấm đã được đắp lên ra đời.)}
\end{truyen}

\begin{baihoc}
  \textbf{Small parts make a great whole.} \textit{(Vô số những thứ dẫu phân mảnh dời rời rạc li ti rốt cuộc sẽ cùng nhau chắp vá tụ thành một đại thể trọn bộ khổng lồ tuyệt luân.)}
\end{baihoc}

\section{Mở Mang}
\begin{truyen}{Mở Mang}{Kinh nghiệm}
\chuhoa{A}{ } \textbf{farmer always grew the best corn} in the \textbf{village}.\\
\textit{(Một bác nông dân quanh năm suốt tháng bao mùa cày luôn vinh dự ẵm giải cao thủ trồng được hạng loại những bắp ngô xuất sắc hạng ngon đỉnh cấp nhất làng.)}

He \textbf{always shared his top seeds} with his \textbf{neighbors}.\\
\textit{(Thế nhưng bác lại luôn không hề hà tự dưng miễn phí vung tay chia sẻ hết thảy số lô vốc hạt hạt ngô nhân giống mầm siêu phẩm đó cho những nhà tá điền hàng xóm.)}

A \textbf{reporter asked} why \textbf{he helped his competitors}.\\
\textit{(Một anh gã nhà báo tò mò thắc mắc truy hỏi dò la về cái lý do mà bác lại rộng rãi hào sảng xả láng đi giúp đỡ cho cả nùi những đối thủ kỳ kiên cừ kia của mình dại đến thế.)}

He \textbf{replied}: \textit{"If \textbf{their corn is sick}, the \textbf{wind will blow illness to my farm}. We \textbf{must all be good}."}\\
\textit{(Ông bác lão bình phàm điềm tính chất phác đĩnh trịch đáp vỗ thẳng: "Bởi vì một lẽ nếu như cánh đồng ngô của bọn họ mà bị mắc bệnh sâu thối rữa nát ốm bủng, thì cơn gió kiểu gì ắt cũng sẽ tàn nhẫn phả luồng lây lan thổi bùng thổi nọc lan tỏa dội tuốt luốt cái vi trùng chết lây đó cuốn vô sang tận trang ấp rẫy bắp của nhà tôi thôi. Vì vậy tất thảy thảy tụi tôi đều phải cần cùng tốt lây.")}
\end{truyen}

\begin{baihoc}
  \textbf{Helping neighbors is helping yourself.} \textit{(Thứ công án giúp đỡ đùm nhường cứu vớt láng giềng sát vách hóa ra âu quy vào cũng chính là một sự tự bảo vệ trợ lực giúp kéo ích lợi cho chính bạn.)}
\end{baihoc}

\section{Khoản Đáp Lại}
\begin{truyen}{Khoản Đáp Lại}{Đáp nghĩa}
\chuhoa{A}{ } \textbf{poor man collected a basket} of \textbf{fresh fruits}.\\
\textit{(Một gã nhà nghèo đói hèn rách khố đã hì hụi lượm lặt nhặt nhạnh chắt được vun vút một thúng giỏ mây đầy ắp trái nhồi đầy những quả dâu quả cây hoa quả mọng tươi non.)}

He \textbf{gave it to a rich but unhappy neighbor}.\\
\textit{(Anh đã cầm xách nó nhấc cả mả hên ấy đem đặng biếu không không tặng đem cho gã ông nhà xóm cạnh một tên hàng xóm tuy chất kho tải lắm tiền rủng rĩnh của nhưng lại là kẻ vô cùng vô ưu phiền sầu não nề chẳng có mấy giọt vui vẻ.)}

The \textbf{rich man mocked} him and \textbf{returned the basket full of dirt}.\\
\textit{(Gã phú ông xấc xược ấy đã cười khẩy mỉa mai chế nhạo khinh miệt vào mặt anh tiều phu rồi sau đó đổ bực đá tống lại trả cho anh một bọc cái vành giỏ đó nhưng lại chứa dựng đầy phé một một đám phân sình đất bùn lút ngập dơ bẩn.)}

The \textbf{poor man quietly dumped the dirt} and \textbf{filled it with flowers}.\\
\textit{(Trái lại đáp trả thay, người ở dơ nghèo khổ ấy đã lẳng lặng hiền hòa không cãi đi đổ sụp một đống loại vứt xổ số đất phân thải cặn đi và rồi từ tốn tỉ mẩn bốc trám cắm lại điền lấp đầy cái thùng rổ lại bằng một bịch đầy cả những nụ đóa bông hoa tươi ngát thơm phức.)}

\textit{"We \textbf{give what we have in our hearts},"} he \textbf{said}.\\
\textit{(Và rồi anh cất giọng thủ thỉ nói: "Chúng ta sẽ chỉ cho đi chia ra phóng ném ra được những thứ mà chúng ta đang ngậm giấu chứa có ở hiện hữu chứa được từ phía tận bên sâu trong cõi lồng màng trái tim phàm mỏng của chính tâm chúng ta mà thôi.")}
\end{truyen}

\begin{baihoc}
  \textbf{Give kindness even when receiving dirt.} \textit{(Hãy có tấm lòng trao ban rắc tưới trải đi một hạt sự tử tế dịu từ ngay cả trong một lúc bạn đang vô duyên bị chuốc nhét nhận phải một mớ thứ đất cục bùn rác cặn dơ phế bẩn xú.)}
\end{baihoc}

\section{Giúp Đỡ}
\begin{truyen}{Giúp Đỡ}{Trường lớp}
\chuhoa{L}{isa} \textbf{forgot her set of coloring pencils}.\\
\textit{(Cô trò nhỏ Lisa lại não cá vàng xí lộn lỡ nhen lại quên béng lỡ xách lôi theo bỏ dẹp xó không thèm xách theo cái hộp hộp bộ bút hộp viết túi màu bút màu của mình ở cữ chớ chít nhà.)}

She \textbf{started to cry quietly}.\\
\textit{(Tủi hờn tủi trào em bé nín lự em bắt đầu nín lặng lẽ sụt thút sùi chóp lén rơm rụt xụt sùi khóc nắp nấc khóc len nghẻn nức khóc tỉ thầm lén trong nhút nhát một góc mình.)}

\textbf{Tom snapped all his pencils} in \textbf{half}.\\
\textit{(Phát hiện được nhỏ bạn, nhóc nhẵn nhóc lém lỉnh thằng cu Tom đã ngay lập tức thẳng cánh không nao bẻ một phát rụng gãy lặn đôi tách vỡ tách hụt mủn vỡ xoảng dứt phát dẹp tất đành ngắt gãy vụn làm ranh hai nửa toàn bộ những mớ chiếc cái đồ bút đi mầu của nhóc con cu cậu bé con nít ranh chính ra ấy cho một cú gập luôn.)}

\textit{"Now \textbf{we both have colors},"} he \textbf{smiled}.\\
\textit{(Cậu nhỏ cu nhóc nhoẻn nhoẻn nhoẻn hàm môi hớn hàm lém nhếch khoé đắc cười hàm tếu nhe hàm cười tươi nắc tít: "Ngó nè coi chừ bây giờ thì tát rốt cả hai gộp chúng tớ ta đều đã coi như mỗi người cũng đã có sở giữ một lô đục phần phần màu tô để vẽ rồi nhé chưa nhóc kia.")}
\end{truyen}

\begin{baihoc}
  \textbf{Sharing brings immediate joy.} \textit{(Việc chủ ý sự việc rành cố chia sẻ đồ sự san tớt sẻ đôi khi sự ấy thực sẽ đi kéo đẩy kèm dắt gọi lôi theo tạo sinh đem dẫn ngay tới liền cho chính ta lại tạo một loại hình nỗi thứ kiểu loại niềm khoái thụ hạnh hỉ hoan vui thích sung vui liền tại chỗ rất chóng nhanh rụp tức thì phất trốc.)}
\end{baihoc}

\section{Khoảng Khắc}
\begin{truyen}{Khoảng Khắc}{Gia đình}
\chuhoa{I}{t} \textbf{was a hot summer night} with \textbf{no electricity}.\\
\textit{(Hôm khắc bấy khi ấy bấy thời khi thời khắc ấy trời vào bấy dạo ấy đó thật lúc đúng ngay đang là cả lúc một đêm tiết mùa vụ phiên mùa hè vãng tối oi ực đổ bức ực rất hè lại còn hên trùng rớt vô ngay phốc một bận khốn khổ xui trận xui bị quất sụm mất dập cúp toi chập ngắt sấp cả điện đóm điếc nhà trơn tắt thui tối om mù.)}

A \textbf{mother fanned her sleeping child} continuously.\\
\textit{(Một bà người me mẹ bần hiền lam đang phải tốn dồn dùng hao vận động nỗ hao nỗ gắng phải quạt gượng xõa tay phẩy quạt đùn liên liếm phẩy quạt gật giũ mát mãi không ngừng ngớt liên lọi quạt phẩy thốc liên xoành luôn tay xua luồn mớ luồng luồng hơi thở xoay để cho đứa ngáo oắt nhỏ nhóc con ngủ tít ru thiu của cô mẹ chị.)}

Her \textbf{own face was covered in sweat}, but \textbf{she felt at peace}.\\
\textit{(Mà dẫu cho rằng cái dọi vùng phần cái diện bộ mặt mày khuôn mặt của bà cô mụ mạ cô ả thân ấy có cũng đang nhấp nhễ rỉu ướt nhầy nhũa lã lã lã lọt lấm nhem túa nhễ tứa nhễ rịn đẫm rệu ướt tắm đổ nhễ đổ màng nước toát giọt đẫm hồ đẫm dợn nhỏ mồ hôi đầm túa mẹ hồ đìa cả dấp nhem nhớp mồ đọng túa hôi thì bà má ấy vẫn tĩnh yên lòng thanh thỏa thoả yên cảm tợ thanh nhiên sướng yên lòng tại thanh thư thấy tận lòng vui thái thái tột tại một sự vui an tột yên lòng tĩnh nhiên tột tịnh lòng thanh.)}
\end{truyen}

\begin{baihoc}
  \textbf{We share comfort over own struggle.} \textit{(Ta thà chịu nhận chuốc để dành nhường sự gồng sự khổ gánh nhọc ráng ôm vào thọc ôm người để có giành đem xẻ phát phần để xẻ trọn đem chia dư sức mang gánh ban phát dư mạn san giãi nhường trao nhượng lại được lại được hết cái sự những phần phần nhàn sự cảm thái thoải nệm dư sự dễ thoả chịu nhàn nhã ưu khoái ưu dâng bọc đem cho kẻ gửi bạn trao đổi đến tay tận tay dâng người thụ phần nhân hưởng tiếp đối nhận hưởng kẻ khác.)}
\end{baihoc}

\section{Giữ Lại Hơi Ấm}
