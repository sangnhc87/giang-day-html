\chapter{Crush Và Đợi Chờ}
\chapterintro{Đợi chờ là hạnh phúc, nhưng đợi bảng điểm thì là cực hình.}{Crush chưa đổ mà tiền đồ đã lung lay.}

\begin{tutuyetbox}{Đứng cổng chờ mong}{Waiting at the Gate}
Đứng cổng trường chờ bóng dáng ai
Mà sao bài học chẳng thèm mài
Bảng đen phấn trắng trôi vùn vụt
Nhục nhã thi xong khóc dông dài.
\tcblower
Waiting at the gate for a shadow to see 
But studying lessons isn't for me 
Blackboard and chalk are flying away 
After the test, it's a long, sad day.
\end{tutuyetbox}

\begin{tutuyetbox}{Thất bại toàn tập}{Total Failure}
Crush đã yêu người chẳng chọn ta
Bài thi lại rớt thảm thê nhà
Thân tàn lụn bại lòng tan nát
Khóc hận nghìn thu tủi xót xa.
\tcblower
Crush loves another, they didn't choose me 
The exam is a failure, for all to see 
A broken spirit and a shattered heart 
Crying forever, falling apart.
\end{tutuyetbox}

\begin{ghichu}
Đừng làm "hòn vọng crush" mà quên mất mình đang là học sinh. Tình cảm là duyên, còn điểm số là do mình tự tạo nghiệp.
\end{ghichu}
