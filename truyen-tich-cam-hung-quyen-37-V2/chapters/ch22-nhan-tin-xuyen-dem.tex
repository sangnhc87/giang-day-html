\chapter{Nhắn Tin Xuyên Đêm}
\chapterintro{Đêm là lúc tâm hồn rạo rực, ngày là lúc cơ thể rã rời.}{Cày game, nhắn tin thì không biết mệt, vào tiết thì ngáp như cá ngáp nước.}

% --- BÀI 241: ĐÊM TRẮNG (Sửa gắt) ---
\begin{tutuyetbox}{Đêm trắng}{White Night}
Thức trắng đêm thâu bấm chạch chành
Tin lòng gửi bạn tít trời xanh
Sáng mai vô lớp đầu như đá
Ngủ gục trên bàn mộng vỡ bành.
\tcblower
Staying up all night, typing away 
Sending your heart, far, far away 
Morning in class, your head is like stone 
Falling asleep, sitting alone.
\end{tutuyetbox}

% --- BÀI 242: TIN NHẮN OAN GIA (Nâng cấp) ---
\begin{tutuyetbox}{Tin nhắn oan gia}{The Accursed Message}
Mắt đọc tin hồng miệng mỉm chi
Ngờ đâu bài tập bỏ quên đi
Thầy tra lý thuyết mồm câm bặt
Điểm liệt mang về khóc lâm li.
\tcblower
Reading pink messages with a sweet smile 
Ignoring the homework for a long while 
Teacher asks questions, you go so still 
Failing the test, against your will.
\end{tutuyetbox}

% --- BÀI 243: MẮT GẤU TRÚC ---
\begin{tutuyetbox}{Nhan sắc tàn phai}{Panda Eyes}
Mắt quầng đen kịt tựa gấu mèo
Vì mải nhắn tin, dáng khẳng khiu
Crush nhìn phát khiếp xua tay chạy
Học dốt mặt sưng... nhục bám theo.
\tcblower
Dark circles under eyes, like a panda bear
Typing all night, a ghost-like air
Crush sees you and runs in a fright
Ugly and dumb, a pitiful sight.
\end{tutuyetbox}

% --- BÀI 244: CÚ ĐÊM VÔ DỤNG ---
\begin{tutuyetbox}{Anh hùng bóng đêm}{Night Hero}
Bấm phím thâu đêm tưởng rất oai
Chuyện trên trời dưới hót dặm dài
Sáng ra kiểm tra mồm ngáp gió
Tương lai xách vữa... nhục bi hài.
\tcblower
Typing all night, feeling so bold
Stories of the world, or so I'm told
In the morning test, you yawn at the sky
A laborer's life, until you die.
\end{tutuyetbox}

% --- BÀI 245: BẮT QUẢ TANG ---
\begin{tutuyetbox}{Ánh sáng tử thần}{The Death Light}

Mẹ cầm cây gậy xông vào hang
Điện thoại lìa tay, đời bế mạc
Bảng điểm đưa về... nhục vẻ vang.
\tcblower
Hiding under blankets, typing so fast
Mom grabs a stick, the peace won't last
The phone is gone, your world is dead
Failing marks are all that's ahead.
\end{tutuyetbox}

% --- BÀI 246: TIN NHẮN "THẢ THÍNH" ---
\begin{tutuyetbox}{Kẻ lụy tình}{The Messenger Simp}
Đợi một hồi âm đến bạc đầu
Người ta chẳng tiếp, chẳng thèm câu
Học hành dốt nát vì mê muội
Mặt dơi mỏ nhọn... nát đời nhau.
\tcblower
Waiting for a reply until your hair turns gray
They don't even care, looking the other way
Ignoring your studies for a foolish dream
A rat-like face, a silent scream.
\end{tutuyetbox}

% --- BÀI 247: SỰ CỐ GẬT GÙ ---
\begin{tutuyetbox}{Gục ngã}{The Collapse}
Đang lúc nhắn tin ngủ lịm đi
Ấn nhầm gửi mẹ chuyện yêu vi (yêu đương)
Sáng ra ăn gậy sưng mông dốt
Sách vở chưa thông... nhục cực kì.
\tcblower
Typing a text then falling asleep
Sent it to Mom, the secrets you keep
In the morning, the cane hits your rear
No knowledge gained, only a tear.
\end{tutuyetbox}

% --- BÀI 248: CHAT XUYÊN LỤC ĐỊA ---
\begin{tutuyetbox}{Thánh chém gió}{Global Chat}
Tiếng Anh bẻ nửa viết chưa xong
Mà đòi chat Mỹ sướng trong lòng
Nó nghe "u-mê" (ngôn ngữ mạng) rồi bỏ chạy
Mặt ngáo mồm hôi... nhục nát tan.
\tcblower
Can't write English, can't even try
But chatting with foreigners, aiming so high
They hear your gibberish and run away fast
A stinking breath, your pride won't last.
\end{tutuyetbox}

% --- BÀI 249: KẾT CỤC CÚ ĐÊM ---
\begin{tutuyetbox}{Ảo mộng tan vỡ}{Shattered Dreams}
Máy nóng tay run mắt lờ đờ
Tương lai phía trước mịt mù mờ
Người yêu đậu đại mình bốc vác
Đứng nhìn theo họ... mặt ngẩn ngơ.
\tcblower
Hot phone, shaking hands, blurry sight
The future ahead is dark as night
Lover passed the test, you're lifting stone
Watching them go, you're all alone.
\end{tutuyetbox}

% --- BÀI 250: LỜI KẾT NHẮN TIN ---
\begin{tutuyetbox}{Tỉnh mộng đi con}{Final Wake-up}
Học chẳng lo lo chuyện nhắn tin
Bán rẻ tương lai lấy cái hình
Mai sau nghèo đói ai thèm nhắn
Nhục nhã nghìn đời... lệ hiển linh.
\tcblower
Ignoring the books for a digital chat
Selling your soul for a photo, at that
When you're poor, who will text you back?
A thousand years of shame on your track.
\end{tutuyetbox}
% --- BÀI 251: KẺ ĐIÊN VÌ SẮC ---

\tcblower

Begging for scraps on the dirty ground.
\end{tutuyetbox}

% --- BÀI 252: CÁI KẾT PHỤ HỒ (Nâng cấp) ---
