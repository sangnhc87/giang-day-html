\markboth{Tưởng Cố Là Sẽ Được Trả Công}{Thinking Hard Work Is Always Rewarded}

\section{Làm Âm Thầm, Không Ai Thấy}
\begin{truyen}{Làm Âm Thầm, Không Ai Thấy}{Working Quietly While No One Notices}
\chuhoa{C}{o nhan vien tre lam rat cham chi nhung it noi, it trinh bay, va cu nghi cap tren se tu thay.}
\textit{(A young worker is diligent but quiet, explains little, and assumes superiors will notice automatically.)}

Ban ay tuong cu co gang deu la ket qua se tu den dung cho.
\textit{(They think steady effort alone will naturally bring the right result.)}

Nhung trong nhieu noi, viec tot neu khong duoc trinh bay dung va de lai dau vet ro cung de tro nen vo hinh.
\textit{(But in many places, good work without clear visibility and trace can still become invisible.)}

Co gang can sach, nhung cung can hoc cach de cong suc khong bi chim mat.
\textit{(Effort must stay clean, but it also needs visibility so it does not sink.)}

\end{truyen}

\begin{baihoc}
Lam tot la nen. Cho nguoi ta thay duoc no la ky nang.
\textit{(Doing good work is the base. Letting people see it is a skill.)}
\end{baihoc}

\begin{ghinhoanh}
\textbf{Short English to remember:}

Doing good work is the base.

Letting people see it is a skill.

\end{ghinhoanh}

\section{Tăng Ca Mãi, Lương Vẫn Đứng}
\begin{truyen}{Tăng Ca Mãi, Lương Vẫn Đứng}{Working Overtime Again and Again While Pay Stays the Same}
\chuhoa{C}{o nguoi o lai den muon rat nhieu dem va tin rang su tan tuy se som duoc dap lai.}
\textit{(Someone stays late many nights and believes such devotion will soon be rewarded.)}

Ho tuong co gang nhieu hon nguoi khac la du de doi doi xu cong bang.
\textit{(They think harder effort than others is enough to earn fair treatment.)}

Nhung neu noi do khong co he thong ghi nhan ro rang, tang ca co the chi tro thanh mot thoi quen duoc mac dinh.
\textit{(But if that place has no clear system of recognition, overtime can become a habit others simply expect.)}

No luc khong duoc dat dung cho se de bien thanh suc lao dong gia re.
\textit{(Effort placed in the wrong structure easily becomes cheap labor.)}

\end{truyen}

\begin{baihoc}
Khong phai cu rat co gang la se duoc tra cong dung. Em con phai nhin xem minh dang co gang o dau.
\textit{(Hard work is not always rewarded fairly. You also need to examine where you are working.)}
\end{baihoc}

\begin{ghinhoanh}
\textbf{Short English to remember:}

Hard work is not always rewarded fairly.

You also need to examine where you are working.

\section{Cố Gắng Sai Hướng}
\begin{truyen}{Cố Gắng Sai Hướng}{Working Hard in the Wrong Direction}

\textit{(They think busyness means progress.)}

\textit{(Mature people do not only ask whether they are trying.)}

\textit{(They ask where the effort is landing.)}

\end{truyen}

\textit{(Diligence is good. Diligence in the right direction creates results.)}

\section{Ở Nơi Không Trọng Người}
\begin{truyen}{Ở Nơi Không Trọng Người}{Staying in a Place That Does Not Respect People}

\textit{(Wisdom is not always staying longer.)}

\textit{(Sometimes it is leaving at the right time.)}

Co gang la dep. Dung co gang de nuoi mot noi khong biet quy suc nguoi.
\textit{(Effort is noble. Do not spend it feeding a place that does not value human effort.)}

\textit{(Endurance is different from stubbornly staying just to save face.)}

\section{Chăm Chỉ Thiếu Chiến Lược}
\begin{truyen}{Chăm Chỉ Thiếu Chiến Lược}{Being Hardworking but Without Strategy}
\chuhoa{C}{o nguoi rat sieu nang, ngay nao cung nhieu gio, nhung nam nao cung dung yen vi khong biet uu tien cai gi.}
\textit{(Some people work many hours every day, yet stay in the same place for years because they do not know what to prioritize.)}

\textit{(They think more effort can compensate for every other weakness.)}

Nhung co gang khong co chien luoc thuong tan ra thanh rat nhieu viec nho ma khong ket tinh thanh buoc nhay lon.
\textit{(But effort without strategy often spreads into many small tasks without forming any major leap.)}

Nguoi tre can hoc chac mot vai huong quan trong hon la lam het moi thu mot chut.
\textit{(Young people need to hold a few important directions instead of doing a little of everything.)}

\end{truyen}

\begin{baihoc}
No luc can duoc dan duong boi uu tien dung.
\textit{(Effort needs to be guided by clear priorities.)}

\section{Kiệt Sức Rồi Mới Biết Giá}
\begin{truyen}{Kiệt Sức Rồi Mới Biết Giá}{Learning the Cost Only After Burning Out}
\chuhoa{C}{o nguoi co gang lien tuc den muc than the va dau oc deu can ma van tu nhac minh phai gang them.}
\textit{(Some people keep pushing until body and mind are empty, yet still tell themselves to push harder.)}

Ho tuong nghi lai la yeu va cham lai la tut hau.
\textit{(They think rest means weakness and slowing down means falling behind.)}

Den khi kiet suc, ho moi thay mot cai gia rat dat: viec van con do, chi co nguoi lam la da hong nhieu hon.
\textit{(Only after burnout do they see the cost: the work is still there, but the person doing it is now damaged.)}

Co gang lau dai can nhan vao suc ben, khong chi vao y chi trong vai tuan nong.
\textit{(Long-term effort must rely on stamina, not only on a few hot weeks of willpower.)}

\end{truyen}

\begin{baihoc}
Muon duoc tra cong dai han, em phai biet giu nguon luc cua minh.
\textit{(If you want long-term reward, you must know how to protect your own energy.)}
\end{baihoc}

\begin{ghinhoanh}
\textbf{Short English to remember:}

If you want long-term reward, you must know how to protect your own energy.

\section{Đổi Cách Cố}
\begin{truyen}{Đổi Cách Cố}{Changing the Way You Try}
\chuhoa{S}{au nhieu lan co gang ma khong duoc dap lai, co nguoi dung lai de xem minh can doi cach lam gi.}
\textit{(After many unrewarded attempts, someone finally stops to examine what must change.)}

Ban ay truoc do tuong chi can tang them suc la du.
\textit{(Before that, they thought more force was enough.)}

Nhung doi khi thu em thieu khong phai no luc, ma la cach chon noi, chon viec, chon nguoi, va chon thoi diem.
\textit{(But sometimes what is missing is not effort, but choice of place, work, people, and timing.)}

Ngay khi cach co gang doi, ket qua bat dau ro hon du no luc khong he giam.
\textit{(Once the method changes, results become clearer even without less effort.)}

\end{truyen}

\begin{baihoc}
Dung chi co gang hon. Hay co gang khon hon.
\textit{(Do not only try harder. Try wiser.)}
\end{baihoc}

\begin{ghinhoanh}
\textbf{Short English to remember:}

Do not only try harder.

Try wiser.

\end{ghinhoanh}
