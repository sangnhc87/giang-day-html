\chapter{Nghe Trống Là Tỉnh}
\chapterintrobi{Đầu giờ mơ màng, cuối giờ nghe trống thì sáng như chưa từng buồn ngủ}{Wake up when the bell rings}{Có những bạn dành trọn nguồn năng lượng tỉnh táo cho đúng hai giây cuối tiết.}

\begin{lucbatbox}{Lục bát: Chuông là phép màu}
Suốt giờ mắt nặng như mây\
Vai em gật gật, hồn bay nửa vòng\
Đến khi nghe tiếng trống xong\
Tự nhiên em tỉnh như không có gì\
Phép màu đến thật diệu kỳ\
Chỉ thương bài giảng còn đi nửa đường
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Năng lượng cuối giờ}
Đầu tiết em như chiếc lá mềm\
Cuối giờ nghe trống bỗng vùng lên\
Nếu đem sức bật vào đầu buổi
Trang nào cũng đã đẹp như tên
\end{tutuyetbox}

\begin{ghichu}
Nghe trống là tỉnh chứng minh một điều thú vị: năng lượng không mất đi, nó chỉ đợi đúng tín hiệu để xuất hiện.
\end{ghichu}

\begin{englishnote}
\textbf{school bell}: tiếng trống trường.\par
\textbf{wake up}: tỉnh lại.\par
\textbf{I am awake now because class is over.}: Giờ em tỉnh vì tiết học đã hết.
\end{englishnote}
