\chapter{Tuổi Học Trò Và Các Cú Ngô Nghê}
\chapterintro{Những lần làm trò con bò tưởng hài hước, ai ngờ trở thành tấu hài thất bại.}{Sau này nhìn lại, muốn đào hố chui xuống vì ngượng.}

\begin{tutuyetbox}{Góc ngô nghê}{The Silly Corner}
Mấy đứa ranh ranh nghịch phá đình
Cài mìn ném phấn tưởng lung linh
Thầy phê kiểm điểm bêu tên móc
Tuổi mộng mơ hoài nhục bẽ mình.
\tcblower
Sneaky kids playing a trick 
Throwing chalk, acting so thick 
Teacher writes a note to shame 
A dreaming youth, with a broken name.
\end{tutuyetbox}

\begin{tutuyetbox}{Đùa quá trớn}{Out of Bounds Joke}
Chọc phá đùa dai hất ngã bành
Cười vui hớn hở vẻ sành sành
Trượt chân té ngã rách đầu tóc
Nhục nhã ngàn thu khóc nát khanh.
\tcblower
Pranking hard, knocking them down 
Laughing loud, playing the clown 
Tripping and falling, a bloody sight 
Eternal shame, crying through the night.
\end{tutuyetbox}

\begin{ghichu}
Giới hạn giữa hài hước và vô duyên nó mỏng manh tựa như số điểm cách nhau giữa đỗ và trượt tốt nghiệp vậy.
\end{ghichu}
