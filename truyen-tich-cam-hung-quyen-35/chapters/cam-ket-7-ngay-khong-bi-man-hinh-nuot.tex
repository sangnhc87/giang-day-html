\begin{truyen}{Không Cần Anh Hùng, Chỉ Cần Giữ Thật}{Seven Days of Honest Resistance}
\chuhoa{K}{hông cần tuyên bố lớn.}
\textit{(No grand declaration is needed.)}

Chỉ cần bảy ngày giữ vài luật nhỏ đủ thật.
\textit{(Only seven days of keeping a few small rules seriously.)}

Nếu không giữ nổi bảy ngày, ít nhất người đọc sẽ biết mình đang lệ thuộc đến mức nào.
\textit{(If the reader cannot even keep seven days, they will at least know how dependent they truly are.)}
\end{truyen}

\begin{tcolorbox}[colback=white,colframe=bookprimary!60,arc=4pt,boxrule=0.8pt,title={Ngày 1. Bàn Ăn},fonttitle=\bfseries]
Không điện thoại trên bàn ăn.
\textit{(No phone on the dining table.)}
\end{tcolorbox}

\begin{tcolorbox}[colback=white,colframe=bookprimary!60,arc=4pt,boxrule=0.8pt,title={Ngày 2. Giờ Học},fonttitle=\bfseries]
Ít nhất một giờ học sâu với điện thoại nằm ngoài tầm tay.
\textit{(At least one deep study hour with the phone out of reach.)}
\end{tcolorbox}

\begin{tcolorbox}[colback=white,colframe=bookprimary!60,arc=4pt,boxrule=0.8pt,title={Ngày 3. Đường Đi},fonttitle=\bfseries]
Không nhìn màn hình khi lái xe, đi xe, hoặc băng qua đường.
\textit{(No looking at the screen while driving, riding, or crossing the street.)}
\end{tcolorbox}

\begin{tcolorbox}[colback=white,colframe=bookprimary!60,arc=4pt,boxrule=0.8pt,title={Ngày 4. Cuộc Nói Chuyện},fonttitle=\bfseries]
Úp điện thoại xuống khi ai đó đang nói chuyện nghiêm túc với mình.
\textit{(Turn the phone face down when someone is speaking seriously to you.)}
\end{tcolorbox}

\begin{tcolorbox}[colback=white,colframe=bookprimary!60,arc=4pt,boxrule=0.8pt,title={Ngày 5. Khoảng Trống},fonttitle=\bfseries]
Giữ ít nhất mười phút chờ đợi mà không mở máy.
\textit{(Keep at least ten waiting minutes without opening the phone.)}
\end{tcolorbox}

\begin{tcolorbox}[colback=white,colframe=bookprimary!60,arc=4pt,boxrule=0.8pt,title={Ngày 6. Giấc Ngủ},fonttitle=\bfseries]
Không mang điện thoại lên giường.
\textit{(Do not bring the phone onto the bed.)}
\end{tcolorbox}

\begin{tcolorbox}[colback=white,colframe=bookprimary!60,arc=4pt,boxrule=0.8pt,title={Ngày 7. Người Khác},fonttitle=\bfseries]
Dành trọn một cuộc gặp hoặc một bữa ăn cho con người trước mặt.
\textit{(Give one full meeting or one meal entirely to the person in front of you.)}
\end{tcolorbox}

\begin{baihoc}
Bảy ngày này không phải để làm màu.
\textit{(These seven days are not for show.)}

Chúng là một phép thử đơn giản.
\textit{(They are a simple test.)}

Nếu giữ được, em bắt đầu lấy lại quyền trên đời mình.
\textit{(If you can keep them, you begin reclaiming authority over your life.)}

Nếu giữ không nổi, ít nhất em không còn quyền giả vờ rằng mình vẫn hoàn toàn làm chủ.
\textit{(If you cannot keep them, then at least you lose the right to pretend you are still completely in control.)}
\end{baihoc}
