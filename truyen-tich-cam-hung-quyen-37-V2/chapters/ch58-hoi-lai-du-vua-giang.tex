\chapter{Hỏi Lại Dù Vừa Giảng}
\chapterintro{Thầy vừa nói câu trước, trò hỏi lại y câu sau.}{Não bộ có chức năng tự động xóa thông tin sau 3 giây.}

\begin{tutuyetbox}{Lỗ hổng không gian}{Spatial Hole}
Vừa cất lời xong giảng mệt đừ
Trò ngoan lại hỏi dốt chần chừ
Thầy cô chết lặng nhìn thoi thóp
Phép tính sai lầm rớt mịt mờ.
\tcblower
Just finished explaining, tired and weak 
The student asks again, acting so meek 
Teacher stands frozen, losing their breath 
Calculation errors, a date with death.
\end{tutuyetbox}

\begin{tutuyetbox}{Trí nhớ cá vàng}{Goldfish Memory}
Chữ đổ lên bàn mất tích ngay
Vừa nghe đã rụng chán nghìn ngày
Thầy cô quát trảo lòi sái bét
Lên lớp năm trường rớt khổ cay.
\tcblower
Words fall on the desk, vanishing fast 
Forgotten instantly, nothing can last 
Teacher shouts out, showing the wrong 
Year after year, where you don't belong.
\end{tutuyetbox}

\begin{ghichu}
Mỗi lần học sinh giơ tay hỏi lại bài vừa giảng xong, tuổi thọ của giáo viên lại bị bào mòn đi một chút.
\end{ghichu}
