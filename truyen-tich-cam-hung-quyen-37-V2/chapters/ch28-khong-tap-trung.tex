\chapter{Không Tập Trung}
\chapterintro{Ngồi trong lớp mà tâm hồn lơ lửng ngọn tre.}{Xác thịt ở lại, nhưng lý trí đã lên cỗ xe bay đi chơi hội.}

\begin{tutuyetbox}{Tâm trên cành cây}{Mind on the Branch}
Lờ đờ ánh mắt mộng chơi vơi
Trí nhớ dạo chơi tận cuối trời
Góc bảng nghìn câu trôi vút mất
Giật mình ngậm đắng rớt tơi bời.
\tcblower
Drowsy eyes, dreaming of the blue 
Memory wandering, far away from you 
Lessons on the board, vanishing so fast 
Staring in shock, a failure at last.
\end{tutuyetbox}

\begin{tutuyetbox}{Hồn lìa khỏi xác}{Soul Leaves the Body}
Tiết lý lời cha mỏi rã rời
Trò ngoan rũ rượi cắn vành môi
Chiều nay quẳng bút gom rác rưởi
Điểm thấp tan tành nhục rã rời.
\tcblower
Teacher's voice is tired and weak 
A lazy student, acting so meek 
Throwing away the pen, collecting the trash 
The low score hits like a lightning flash.
\end{tutuyetbox}

\begin{ghichu}
Sự không tập trung khiến bạn chỉ nghe được đúng 2 chữ duy nhất trong cả tiết học: Xin Chào và Ra Chơi.
\end{ghichu}
