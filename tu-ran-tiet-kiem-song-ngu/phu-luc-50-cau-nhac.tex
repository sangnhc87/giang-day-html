\chapter*{Phụ Lục VI. Năm Mươi Câu Nhắc Buổi Sáng}
\addcontentsline{toc}{chapter}{Phụ Lục VI. Năm Mươi Câu Nhắc Buổi Sáng}
\markboth{Năm Mươi Câu Nhắc Buổi Sáng}{Năm Mươi Câu Nhắc Buổi Sáng}

\begin{truyen}{Đọc Một Câu Mỗi Sáng}{Read One Sentence Each Morning}
\chuhoa{N}{ăm mươi câu dưới đây được viết theo cách rất ngắn để người đọc có thể mở ra vào buổi sáng, đọc một câu, rồi mang câu ấy vào ngày sống của mình.}
\textit{(The fifty lines below are written briefly so that the reader may open them in the morning, read one sentence, and carry it into the day.)}

1. Hôm nay tôi không tiêu tiền để mua cảm giác hơn người.\
\textit{(Today I will not spend money to feel above others.)}

2. Hôm nay tôi nhớ tiền đến từ lao động.\
\textit{(Today I remember that money comes from labor.)}

3. Hôm nay tôi phân biệt rõ điều cần và điều muốn.\
\textit{(Today I clearly separate need from want.)}

4. Hôm nay tôi giữ một khoản nhỏ cho ngày mai.\
\textit{(Today I keep a small amount for tomorrow.)}

5. Hôm nay tôi không tiêu vì buồn.\
\textit{(Today I will not spend because I am sad.)}

6. Hôm nay tôi không tiêu vì mệt.\
\textit{(Today I will not spend because I am tired.)}

7. Hôm nay tôi không tiêu để quên.\
\textit{(Today I will not spend to forget.)}

8. Hôm nay tôi không tiêu để khoe.\
\textit{(Today I will not spend to show off.)}

9. Hôm nay tôi nhớ gia đình cần sự ổn định.\
\textit{(Today I remember that my family needs stability.)}

10. Hôm nay tôi quý sự đơn giản.\
\textit{(Today I value simplicity.)}

11. Hôm nay tôi nhìn kỹ những khoản nhỏ.\
\textit{(Today I watch the small expenses carefully.)}

12. Hôm nay tôi tắt quảng cáo nhanh hơn tắt lý trí.\
\textit{(Today I will turn off advertisements before I turn off reason.)}

13. Hôm nay tôi giữ bình tĩnh trước món giảm giá.\
\textit{(Today I remain calm before discounted items.)}

14. Hôm nay tôi không nhầm giảm giá với cần thiết.\
\textit{(Today I will not confuse a discount with necessity.)}

15. Hôm nay tôi không lấy tiền tương lai trả cho sự yếu lòng.\
\textit{(Today I will not use future money to pay for weakness.)}

16. Hôm nay tôi mở sổ trước khi mở ví.\
\textit{(Today I open my notebook before my wallet.)}

17. Hôm nay tôi nói thật với con số.\
\textit{(Today I will be honest with the numbers.)}

18. Hôm nay tôi chọn bình yên thay vì hình ảnh.\
\textit{(Today I choose peace over image.)}

19. Hôm nay tôi chậm lại một nhịp trước khi mua.\
\textit{(Today I slow down for one breath before buying.)}

20. Hôm nay tôi nhớ một khoản nhỏ cũng là tiền.\
\textit{(Today I remember that a small amount is still money.)}

21. Hôm nay tôi không gọi sĩ diện là nhu cầu.\
\textit{(Today I will not call pride a need.)}

22. Hôm nay tôi không thưởng mình bằng mọi ham muốn.\
\textit{(Today I will not reward myself through every desire.)}

23. Hôm nay tôi để tiền đứng gác cho gia đình.\
\textit{(Today I let money stand guard for the family.)}

24. Hôm nay tôi giữ lời hứa tiết kiệm trước tiên với chính mình.\
\textit{(Today I keep my saving promise first to myself.)}

25. Hôm nay tôi sống vừa sức.\
\textit{(Today I live within my means.)}

26. Hôm nay tôi nhớ con mình đang nhìn cách mình dùng tiền.\
\textit{(Today I remember that my child is watching how I use money.)}

27. Hôm nay tôi ăn vừa đủ, không bỏ thừa.\
\textit{(Today I take only enough food and waste none.)}

28. Hôm nay tôi giữ đồ dùng bền hơn một chút.\
\textit{(Today I preserve my belongings a little more carefully.)}

29. Hôm nay tôi không để cảm xúc lái thẻ tiền.\
\textit{(Today I will not let emotion drive the wallet.)}

30. Hôm nay tôi để dành trước khi chi.\
\textit{(Today I save before I spend.)}

31. Hôm nay tôi không ganh với cuộc sống của người khác.\
\textit{(Today I will not compete with someone else’s life.)}

32. Hôm nay tôi biết đủ ở một điều nhỏ.\
\textit{(Today I practice enoughness in one small area.)}

33. Hôm nay tôi tránh một cuộc mua sắm vô nghĩa.\
\textit{(Today I avoid one meaningless purchase.)}

34. Hôm nay tôi giữ quỹ dự phòng như giữ bình an.\
\textit{(Today I protect the emergency fund as I protect peace.)}

35. Hôm nay tôi không để mệt mỏi biến thành hóa đơn.\
\textit{(Today I will not let exhaustion turn into a bill.)}

36. Hôm nay tôi sửa một thói quen nhỏ.\
\textit{(Today I correct one small habit.)}

37. Hôm nay tôi không chờ hoàn hảo mới bắt đầu.\
\textit{(Today I will not wait for perfection before starting.)}

38. Hôm nay tôi không coi thường sức mạnh của một khoản tiết kiệm nhỏ.\
\textit{(Today I will not underestimate a small saving.)}

39. Hôm nay tôi quý mái nhà yên hơn món đồ đẹp.\
\textit{(Today I value a peaceful home more than a beautiful object.)}

40. Hôm nay tôi sống sao để cuối tháng đỡ hối tiếc.\
\textit{(Today I live in a way that will bring less regret at month’s end.)}

41. Hôm nay tôi để đồng tiền đi đúng đường.\
\textit{(Today I send money along the right path.)}

42. Hôm nay tôi giữ được mình trước một cám dỗ là đã thắng.\
\textit{(Today, if I restrain myself before one temptation, I have already won.)}

43. Hôm nay tôi biết ơn bữa cơm đang có.\
\textit{(Today I am grateful for the meal before me.)}

44. Hôm nay tôi không xem thường chi tiêu lặt vặt.\
\textit{(Today I will not underestimate petty spending.)}

45. Hôm nay tôi không lấy nợ che lỗ rò.\
\textit{(Today I will not use debt to cover a leak.)}

46. Hôm nay tôi nói ít về ước muốn, làm nhiều về kỷ luật.\
\textit{(Today I speak less about wishes and do more about discipline.)}

47. Hôm nay tôi giữ cho ví tiền đi cùng lòng tự trọng.\
\textit{(Today I keep my wallet aligned with self-respect.)}

48. Hôm nay tôi sống để ngày mai đỡ sợ hơn.\
\textit{(Today I live so that tomorrow may feel less fearful.)}

49. Hôm nay tôi nhớ rằng bình yên không tự đến.\
\textit{(Today I remember that peace does not arrive on its own.)}

50. Hôm nay tôi giữ tiền cũng là giữ mình.\
\textit{(Today, to keep money is also to keep myself.)}
\end{truyen}

\begin{baihoc}
Một câu đúng đọc đi đọc lại nhiều ngày có thể mạnh hơn một bài khuyên dài đọc rồi để đó.
\textit{(One right sentence repeated over many days can be stronger than a long piece of advice read once and forgotten.)}
\end{baihoc}

\begin{ghinhoanh}
Repeat one good line each morning.

Repetition trains the mind.
\end{ghinhoanh}
