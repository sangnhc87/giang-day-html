\chapter{Bí quyết nhân sinh}
\markboth{Bí quyết nhân sinh}{Secrets of Life}


\section{Biết đủ}
\begin{truyen}{Biết đủ}{Life lesson}
\chuhoa{A}{nh chạy theo nhiều — mệt. Khi biết đủ, anh thở. Biết đủ.}
\textit{(He chased many things — exhausted. When he knew enough, he breathed. Know enough.)}
\end{truyen}

\begin{baihoc}
Đủ là giàu.
\textit{(Enough is rich.)}
\end{baihoc}

\begin{ghinhoanh}
\textbf{Short English to remember:}

Enough is rich.

\end{ghinhoanh}

\begin{tuhocnhanh}
1. Câu chuyện này gợi cho bạn suy nghĩ gì? \textit{What does this story make you think?}

2. Bạn sẽ nhớ bài học nào nhất? \textit{Which lesson will you remember most?}
\end{tuhocnhanh}
