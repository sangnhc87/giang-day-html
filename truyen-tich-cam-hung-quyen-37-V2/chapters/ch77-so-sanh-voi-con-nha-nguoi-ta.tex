\chapter{So Sánh Với Con Nhà Người Ta}
\chapterintro{Sinh vật kỳ bí và hoàn hảo nhất vũ trụ chính là "Con nhà người ta".}{Chưa ai thấy nó bao giờ, nhưng nó xuất hiện trong mọi bữa cơm gia đình.}

\begin{tutuyetbox}{Bóng khuya ảo ảnh}{Shadowy Apparition}
Cha bần trách móc mắng oai linh
Khen mãi con ngoan cạnh vách đình
Bạn ấy tuyệt trần không tì vết
Ngó về ngu dốt nhục thân mình.
\tcblower
Father's scolding, loud and strong 
Praising the neighbor all along 
That kid is perfect, in every way 
Looking at my failure, a dark day.
\tcblower
\end{tutuyetbox}

\begin{tutuyetbox}{Tột cùng bi đát}{Ultimate Tragedy}
Bữa cơm gia thất khóc thành sông
Má chửi tai trâu dốt nát mồng
Đứa cháu đằng kia ba mớ giỏi
Thân tàn lụn bại rớt đành không.
\tcblower
Family dinner, tears like rain 
Mother's shouting, full of pain 
The nephew over there is so bright 
My broken spirit, out of sight.
\end{tutuyetbox}

\begin{ghichu}
Không có áp lực nào mạnh bằng áp lực vô hình từ đứa con của đồng nghiệp mẹ bạn.
\end{ghichu}
