\chapter{Mơ Mộng Tương Lai}
\chapterintro{Vẽ ra viễn cảnh làm sếp, nhưng bài tập về nhà lại hoàn thành bấp bênh.}{Mơ mộng không sai, nhưng mơ mà không học thì chỉ có mộng... du.}

% --- BÀI 291: MỘNG LÀM SẾP (Sửa gắt) ---
\begin{tutuyetbox}{Mộng làm sếp}{Dreaming of Authority}
Ngồi dòm mây xám mộng làm sang
Ghế tựa vinh quang oai ngỡ ngàng
Ngoảnh lại thầy cô tra lí thuyết
Đầu không một chữ nhục lang bang.
\tcblower
Looking at clouds, dreaming of fame 
An office chair and a powerful name 
Teacher returns with a question for you 
Blank and silent, without a clue.
\end{tutuyetbox}

% --- BÀI 292: TƯƠNG LAI MÙ MỊT (Nâng cấp) ---
\begin{tutuyetbox}{Tương lai mù mịt}{Cloudy Future}
Thế giới bao la tự nhủ lòng
Thành công sẽ tới đợi chờ trông
Hành trang trống rỗng dăm ba chữ
Cuối khóa ra trường trắng nợ nần.
\tcblower
The world is wide, I say to myself 
Success will find me on the shelf 
An empty backpack, a clouded mind 
Leaving the school with nothing to find.
\end{tutuyetbox}

% --- BÀI 293: SẾP SÒNG TIỆM NÉT ---
\begin{tutuyetbox}{Tổng tài vỉa hè}{Sidewalk CEO}
Ngồi lớp mơ làm tổng giám đốc
Xe sang bồ đẹp mặt vênh râu
Sáng mai thi rớt lòi đuôi dốt
Làm sếp... trông xe nhục ngập đầu.
\tcblower
Dreaming of being a CEO grand
Rich and powerful across the land
Failing the test, your secrets are out
A parking guard is your job, no doubt.
\end{tutuyetbox}

% --- BÀI 294: TRIỆU PHÚ "BỐC PHÉT" ---
\begin{tutuyetbox}{Triệu phú nước bọt}{Saliva Millionaire}
Mở miệng bàn điều tỷ tỷ đô
Học hành be bét dáng thô khô
Cha kêu xách gạch xây nhà xí
Mới biết chữ lười... nhục phát rồ.
\tcblower
Talking of billions and global trade
While your school grades are starting to fade
Father makes you build a latrine
A lazy soul in a shameful scene.
\end{tutuyetbox}

% --- BÀI 295: DU HÀNH VŨ TRỤ ---
\begin{tutuyetbox}{Phi hành gia rác}{Trash Astronaut}
Mơ bay vũ trụ ngắm trăng sao
Kiến thức không thông mặt vếnh vào
Ngày kia rớt đại nằm ôm hận
Quét rác đêm ngày... nhục xiết bao.
\tcblower
Dreaming of stars and the galaxy wide
But no bit of science is kept inside
Failing the boards, you're down on your luck
Sweeping the streets, you're stuck in the muck.
\end{tutuyetbox}

% --- BÀI 296: LOGIC KẺ ĐIÊN ---
\begin{tutuyetbox}{Mơ không tốn thuế}{Tax-free Dreams}
Thế gian cặm cụi học thành tài
Riêng mình ngồi mộng sướng hoài ai
Tương lai xách vữa nuôi thiên hạ
Mặt mếu môi sưng... nhục dặm dài.
\tcblower
The world is studying, aiming so high
You're just dreaming under the sky
Hard labor is coming to pay your way
A swollen face on a dark, sad day.
\end{tutuyetbox}

% --- BÀI 297: KHOẢNG CÁCH TƯƠNG LAI ---
\begin{tutuyetbox}{Bạn làm quan, tôi làm công}{Master and Slave}
Bạn học ngày đêm mộng hiển vinh
Mình ngồi rung đùi tưởng thông minh
Mai sau nó gọi mình: "Thằng vác!"
Nhục nhã nghìn đời... nát hiển linh.
\tcblower
Friend studies hard, dreaming of light
You sit and shake, thinking you're bright
He'll call you "Laborer" later on
A thousand years of pride are gone.
\end{tutuyetbox}

% --- BÀI 298: CHIẾC GHẾ CHỦ TỊCH ---
\begin{tutuyetbox}{Chủ tịch... xã}{Village Chairman?}
Vẽ cảnh ngồi cao quát mắng người
Ngờ đâu thi rớt mặt đen thui
Về quê mần ruộng tay bùn đất
Nhục nhã nghìn năm... nát đời tươi.
\tcblower
Dreaming of shouting from a high seat
Waking to find your total defeat
Back to the fields with mud on your hand
A shameful life in a lonely land.
\end{tutuyetbox}

% --- BÀI 299: ẢO TƯỞNG THÀNH SAO ---
\begin{tutuyetbox}{Ca sĩ kẹo kéo}{Street Singer}
Mơ đứng trên sàn rực ánh quang
Học không một chữ mặt ngang tàng
Cuối năm điểm liệt bồ hủy kết (unfriend)
Bán kẹo đêm ngày... lệ chảy hàng.
\tcblower
Dreaming of spotlights and a loud crowd
Dumb and stubborn, acting so proud
Failing the class, your lover is gone
Selling snacks until the break of dawn.
\end{tutuyetbox}

% --- BÀI 300: LỜI KẾT TƯƠNG LAI ---
\begin{tutuyetbox}{Hồi kết}{The Final Verdict}
Mơ mộng cho nhiều nát đời hoa
Học dốt mai sau chỉ làm ma
Bằng cấp không thông tình cũng mất
Đời thành mớ rác... thật xót xa.
\tcblower
Dreaming too much will ruin your soul
Without an education, you lose control
No degree, no love, everything is lost
Your life is trash, and that's the cost.
\end{tutuyetbox}
