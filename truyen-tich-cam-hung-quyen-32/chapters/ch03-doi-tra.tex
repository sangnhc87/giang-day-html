\chapter{Đổi chác âm thầm}
\markboth{Đổi chác âm thầm}{Quiet Trade-Offs}

\section{Có người được tiền nhưng mất bữa cơm gia đình.}
\begin{truyen}{Có người được tiền nhưng mất bữa cơm gia đình.}{Income can replace meals, not moments.}
\chuhoa{M}{ỗi tối anh về rất muộn. Tiền tăng dần, nhưng anh không còn nhớ bữa cơm nhà gần nhất có đủ mặt là khi nào. Đến lúc nhìn con lớn nhanh, anh mới thấy mình đã đổi những giờ rất sống để lấy thứ rất cần nhưng không đủ.}
\textit{(Each night he came home late. Income rose, but he could no longer remember the last full family dinner. Watching his child grow quickly, he saw he had traded deeply living hours for something necessary but not sufficient.)}
\end{truyen}
\begin{baihoc}
Có những thứ tiền mua được, nhưng chỉ bằng cái giá là những khoảnh khắc không mua lại được.
\textit{(There are things money can buy, but only at the cost of moments it cannot buy back.)}
\end{baihoc}
\begin{ghinhoanh}\textbf{Short English to remember:}

Income cannot buy moments back.
\end{ghinhoanh}
\begin{tuhocnhanh}
1. Em đang dùng thời gian cho điều gì nhiều nhất? \textit{(What takes most of your time right now?)}

2. Điều gì đang bị bỏ lại phía sau? \textit{(What is being left behind?)}
\end{tuhocnhanh}

\section{Có người giữ tự trọng nên mất một mối lợi.}
\begin{truyen}{Có người giữ tự trọng nên mất một mối lợi.}{Some losses protect your worth.}
\chuhoa{A}{nh từ chối một cách kiếm tiền mờ ám. Người ngoài bảo dại. Nhưng nhiều năm sau nhìn lại, anh biết có những mất mát chính là thứ giữ cho mình còn ngẩng đầu nổi.}
\textit{(He refused a shady chance to make money. Others called him foolish. Years later, he knew some losses are exactly what keep a person able to hold their head high.)}
\end{truyen}
\begin{baihoc}
Không phải mất nào cũng là thiệt; có cái mất là hàng rào cho phẩm giá.
\textit{(Not every loss is defeat; some are fences that protect dignity.)}
\end{baihoc}
\begin{ghinhoanh}\textbf{Short English to remember:}

Some losses protect your worth.
\end{ghinhoanh}
\begin{tuhocnhanh}
1. Điều gì em không muốn đánh đổi dù có lợi? \textit{(What will you not trade away even for gain?)}

2. Giới hạn đó có đủ rõ chưa? \textit{(Is that boundary clear enough?)}
\end{tuhocnhanh}
\ngancach

\section{Có người được yên ổn nhưng mất cơ hội bứt lên.}
\begin{truyen}{Có người được yên ổn nhưng mất cơ hội bứt lên.}{Comfort can hide a slow loss.}
\chuhoa{C}{họn điều an toàn nhất nhiều lần, anh ít khi thất bại lớn. Nhưng nhiều năm sau, anh cũng ít khi thật sự đi xa. Lúc đó anh mới hiểu có những bình yên chỉ là cái giá của việc không dám sống lớn hơn.}
\textit{(By repeatedly choosing the safest option, he rarely failed big. Years later, he also rarely went far. Then he saw that some peace is just the price of not daring to live larger.)}
\end{truyen}
\begin{baihoc}
Điều an toàn không phải lúc nào cũng là điều tốt nhất cho một đời dài.
\textit{(What is safe is not always what is best for a long life.)}
\end{baihoc}
\begin{ghinhoanh}\textbf{Short English to remember:}

Comfort can hide slow loss.
\end{ghinhoanh}
\begin{tuhocnhanh}
1. Em đang an toàn hay đang chững lại? \textit{(Are you safe or simply stagnating?)}

2. Điều gì đáng để em dám hơn một chút? \textit{(What is worth daring a little more for?)}
\end{tuhocnhanh}

\section{Có người được tiếng tốt nhưng mất tiếng nói thật.}
\begin{truyen}{Có người được tiếng tốt nhưng mất tiếng nói thật.}{Approval can silence truth.}
\chuhoa{V}{ì muốn ai cũng hài lòng, cô nói điều an toàn ở mọi nơi. Dần dần cô không còn biết đâu là tiếng nói thật của mình nữa.}
\textit{(Wanting everyone to be pleased, she said only safe things everywhere. Gradually she no longer knew her own true voice.)}
\end{truyen}
\begin{baihoc}
Được lòng tất cả đôi khi đồng nghĩa với mất mình từng chút một.
\textit{(Being liked by everyone can mean losing yourself little by little.)}
\end{baihoc}
\begin{ghinhoanh}\textbf{Short English to remember:}

Approval can silence truth.
\end{ghinhoanh}
\begin{tuhocnhanh}
1. Em có đang im tiếng để được yên? \textit{(Are you silencing yourself to keep the peace?)}

2. Điều gì em cần nói thật hơn? \textit{(What do you need to say more honestly?)}
\end{tuhocnhanh}

\section{Có người được thắng nhanh nhưng mất lòng tin lâu dài.}
\begin{truyen}{Có người được thắng nhanh nhưng mất lòng tin lâu dài.}{Quick wins can bankrupt trust.}
\chuhoa{B}{ằng mẹo nhỏ, anh qua được một lần khó. Nhưng khi người khác biết chuyện, họ không còn tin anh như cũ. Một lợi ngắn ngủi đã lấy đi thứ tích lũy lâu hơn nhiều.}
\textit{(Through a small trick, he got through one hard moment. But once others knew, their trust changed. A brief gain had taken away something built over much longer time.)}
\end{truyen}
\begin{baihoc}
Không phải thắng nào cũng lời; có thắng rất ngắn nhưng lỗ rất sâu.
\textit{(Not every victory is profit; some are brief gains with deep losses.)}
\end{baihoc}
\begin{ghinhoanh}\textbf{Short English to remember:}

Quick wins can bankrupt trust.
\end{ghinhoanh}
\begin{tuhocnhanh}
1. Em có đang muốn lợi nhanh ở đâu không? \textit{(Are you chasing a quick gain somewhere?)}

2. Nó có làm mòn lòng tin không? \textit{(Would it erode trust?)}
\end{tuhocnhanh}
