\chapter{Chọn Chỗ Gần Quạt}
\chapterintrobi{Ưu tiên vi khí hậu trước khi ưu tiên tầm nhìn bảng}{Sit near the fan}{Một số quyết định chỗ ngồi của học sinh rất thực tế và rất ít liên quan tới học tập.}

\begin{lucbatbox}{Lục bát: Gió là chân lý}
Đầu năm xếp chỗ nghiêm trang\
Em nhìn chẳng phải bảng vàng trước kia\
Điều em cân nhắc rất chia\
Quạt quay tới được hay lìa tóc thôi\
Ngồi đâu mặc chữ đang trôi\
Miễn sao mát cổ là rồi ổn ghê
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Học tập theo hướng gió}
Chỗ nào mát nhất, chỗ em xin\
Tầm nhìn bảng viết tính sau liền\
Nếu kiến thức bay theo làn gió mát\
Chắc em hôm ấy rất thần tiên
\end{tutuyetbox}

\begin{ghichu}
Chọn chỗ gần quạt là nhu cầu chính đáng. Chỉ đừng để việc học trở thành nhu cầu xếp sau luồng gió.
\end{ghichu}

\begin{englishnote}
\textbf{near the fan}: gần quạt.\par
\textbf{seat choice}: chọn chỗ ngồi.\par
\textbf{This seat is cooler.}: Chỗ này mát hơn.
\end{englishnote}
