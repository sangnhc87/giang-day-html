\chapter{Tu lop hoc ra cuoc doi}

\begin{leadbox}
Chuong cuoi khong dong sach lai bang nhung loi khuyen chung chung. No mo rong tu lop hoc ra cuoc doi: nhung gi mot hoc sinh hoc duoc ve ky luat, ton trong, trung thuc va long tot se di theo em vao cong viec, gia dinh va cach doi xu voi nguoi khac.
\end{leadbox}

\storycard{111. Ra truong roi moi hieu}{Only Understood After Graduation}{Nhieu loi nhac tung gay kho chiu sau nay tro thanh diem tua trong nhung nam thang di lam.}
\storycard{112. Gap lai thay cu}{Meeting the Old Teacher Again}{Khoanh khac gap lai la luc nguoi ta nhin ro hon minh da duoc uon nan the nao.}
\storycard{113. Nguoi tro tung gay roi di day lai}{The Troublemaker Who Becomes a Mentor}{Qua khu xau khong cam ai tro thanh nguoi huong dan tot neu ho biet chuyen hoa no.}
\storycard{114. Loi cam on chua tung noi}{The Thank-You Never Said}{Co nhung mon no tinh cam chi duoc nhan ra khi nguoi ta da du truong thanh de goi ten.}
\storycard{115. Vet thuong can duoc sua}{Wounds That Need Repair}{Khong phai moi ky niem hoc duong deu dep; truong thanh cung gom ca viec biet go ten vet dau va han gan no.}
\storycard{116. Day con bang bai hoc da tra gia}{Teaching the Next Generation}{Nguoi lon thuong day tre bang chinh nhung bai hoc minh da tung thieu.}
\storycard{117. Cong viec can ky luat nhu lop hoc}{Work Needs Discipline Too}{Ky nang hoc duong tot thuong tro thanh nang luc nghe nghiep ben vung sau nay.}
\storycard{118. Trung thuc khi khong ai kiem tra}{Honesty Without Supervision}{Ra khoi truong lop, bai thi lon nhat cua nhan cach moi that su bat dau.}
\storycard{119. Tu te voi nguoi yeu hon minh}{Kindness Toward the Weaker}{Gia tri cua giao duc lo ro nhat trong cach con nguoi dung quyen luc voi nguoi thap hon minh.}
\storycard{120. Dinh nghia lai hai chu ngoan va tot}{Redefining Good}{Ket lai, sach huong den mot cach hieu sau hon: ngoan khong phai phuc tung mu quang, tot khong phai vo khuyet, ma la biet sua minh va ton trong nguoi.}
