\chapter{Ky luat, cong bang va gioi han}

\begin{leadbox}
Khong co lop hoc nao ton tai ma khong can ky luat. Van de khong phai co phat hay khong, ma la ky luat do phuc vu nhan cach hay chi phuc vu su tien cho nguoi day.
\end{leadbox}

\storycard{11. Chiec dien thoai bi thu}{The Confiscated Phone}{Thu dien thoai de lap lai trat tu la can thiet, nhung cach hoan tra no se quyet dinh hoc sinh hoc duoc dieu gi.}
\storycard{12. Ban truc nhat vo to}{The Messy Duty Table}{Ky luat tot la khi hoc sinh hieu vi sao can lam, khong chi so vi so bi ghi ten.}
\storycard{13. Hai loi vi pham, hai cach xu}{Two Mistakes, Two Responses}{Su cong bang khong phai luc nao cung la giong nhau, ma la xu ly dua tren hoan canh va muc dich giao duc.}
\storycard{14. Hinh phat truoc ca lop}{Punished in Front of Everyone}{Be mat the dien co the lam hoc sinh ngoan hon tam thoi, nhung thuong de lai hanh vi chong doi ngam.}
\storycard{15. Ban cam ket da nhaum}{The Torn Promise Note}{Ban tuong trinh khong sua duoc nhan cach neu nguoi viet chang tin mot ai doc bang su ton trong.}
\storycard{16. Nhat ky vi pham}{The Discipline Ledger}{Ghi chep giup quan ly, nhung neu chi luu loi sai ma khong luu tien bo, hoc sinh se mac ket trong qua khu.}
\storycard{17. Mot lan bo qua}{The One Time Let-Go}{Tha thu dung luc khong lam mat uy, trai lai co the day hoc sinh vao nang luc tu sua.}
\storycard{18. Noi quy do tro viet}{Rules Written by Students}{Khi hoc sinh cung tham gia lap luat, ky luat bat dau tro thanh cam ket thay vi ap dat.}
\storycard{19. Ban nang dong o cuoi lop}{The Restless Student at the Back}{Co nhung em khong hu, chi la khong the ngoi yen theo cach lop hoc dang ep buoc.}
\storycard{20. Thay doi cho ngoi}{Changing Seats}{Mot dieu chinh nho trong khong gian co the giai quyet mot xung dot ma la rao giang khong cham toi.}
