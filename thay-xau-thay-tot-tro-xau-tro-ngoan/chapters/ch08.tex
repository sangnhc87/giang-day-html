\chapter{That bai, ky luat va co hoi lam lai}

\begin{leadbox}
Giao duc dung nghia khong xoa that bai, ma day cach song cung no. Chuong nay tap trung vao nhung khoanh khac con nguoi dung truoc hau qua, tu nhan loi, sua sai, va buoc tiep.
\end{leadbox}

\storycard{71. Bai thi diem mot}{The One-Point Test}{Diem rat thap co the la dau cham het, hoac la dau dong cho mot cach hoc moi.}
\storycard{72. Truot doi tuyen}{Missing the Team}{Khong duoc chon la vet dau, nhung cung la luc hoc cach phan biet gia tri ban than va mot ket qua.}
\storycard{73. Lo dien vi quay cop}{Caught Cheating}{Khoanh khac bi phat hien khong chi can hinh phat, ma can mot loi giai thich ve pham gia va niem tin.}
\storycard{74. Tat ca deu that vong}{Everyone Is Disappointed}{Ap luc tu nhieu phia co the de hoc sinh muon tron chay, neu khong co nguoi day cach doi dien.}
\storycard{75. Ban xin duoc lam lai}{Asking for Another Chance}{Can phan biet xin co hoi moi va xin xoa hau qua; giao duc dung la giu ca hai dieu o the can bang.}
\storycard{76. Nguoi thay ke mot lan truot cua minh}{The Teacher's Own Failure}{Chia se that bai that giup hoc sinh hieu rang nga nga khong lam mat gia tri con nguoi.}
\storycard{77. Ban nop bai muon}{Late Submission}{Gioi han can duoc giu, nhung cung can co cach de hoc sinh hoc duoc trach nhiem tu hanh vi cua minh.}
\storycard{78. Sau ky luat la cam ket}{The Promise After Punishment}{Mot hinh phat khong dan toi thay doi neu khong kem mot cam ket cu the de lam lai tu dau.}
\storycard{79. Mat mot hoc ky de tinh lai}{A Semester to Reset}{Co nhung luc giam toc lai la cach nhanh nhat de dung huong.}
\storycard{80. Bai hoc lon hon diem so}{Bigger Than the Grade}{That bai duoc doc dung se tro thanh nguon luc dai han cho mot nhan cach biet tu sua.}
