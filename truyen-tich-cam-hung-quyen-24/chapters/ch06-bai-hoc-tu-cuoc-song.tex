\section{Đừng tin quá nhanh}
\begin{truyen}{Đừng tin quá nhanh}{Life lesson}
\chuhoa{A}{nh từng tin lời hứa đẹp, tin nụ cười. Nhiều lần bị lừa. Đừng tin quá nhanh.}
\textit{(He once trusted pretty promises and smiles. He was deceived many times. Don't trust too quickly.)}
\end{truyen}

\begin{baihoc}
Tin nhưng vẫn quan sát.
\textit{(Trust but still observe.)}
\end{baihoc}

\begin{ghinhoanh}
\textbf{Short English to remember:}

Trust but still observe.

\end{ghinhoanh}

\begin{tuhocnhanh}
1. Câu chuyện này gợi cho bạn suy nghĩ gì? \textit{What does this story make you think?}

2. Bạn sẽ nhớ bài học nào nhất? \textit{Which lesson will you remember most?}
\end{tuhocnhanh}
