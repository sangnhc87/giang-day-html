\chapter{Nhìn Đồng Hồ Liên Tục}
\chapterintrobi{Tiết học chưa hết nhưng ánh mắt đã chuẩn bị ra chơi từ sớm}{Keep watching the clock}{Khi người học nhìn đồng hồ quá nhiều, kim phút thường trở nên chậm một cách đáng ngờ.}

\begin{lucbatbox}{Lục bát: Mỗi phút dài như thơ}
Kim kia mới nhích một phần\
Mắt em đã ngó đến lần thứ năm\
Tiết học chưa cũ bao năm\
Sao nghe kéo dãn như trăm buổi rồi\
Nếu mà tập trung chút thôi\
Có khi đồng hồ đỡ chơi trò chậm
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Mong chuông từng giây}
Đồng hồ treo đó rất vô tư\
Mà mắt em dõi suốt từ đầu giờ\
Người mong ra chơi thường hay thấy chậm\
Vì lòng đã đứng sẵn ngoài cửa
\end{tutuyetbox}

\begin{ghichu}
Nhìn đồng hồ liên tục hiếm khi làm thời gian chạy nhanh hơn. Nó chỉ làm tiết học thêm dài trong cảm giác.
\end{ghichu}

\begin{englishnote}
\textbf{watch the clock}: nhìn đồng hồ.\par
\textbf{break time}: giờ ra chơi.\par
\textbf{How many minutes are left?}: Còn mấy phút nữa ạ?
\end{englishnote}
