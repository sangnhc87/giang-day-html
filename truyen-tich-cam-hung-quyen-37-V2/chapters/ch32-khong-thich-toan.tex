\chapter{Không Thích Toán}
\chapterintro{Có bạn vừa nghe tới Toán là mặt đã đổi màu}{Không thích Toán là một cảm giác phổ biến tới mức gần như truyền đời.}

\begin{lucbatbox}{Lục bát: Thấy số là thở dài}
Toán chưa làm gì tới em\
Mà em đã thấy buồn thêm nửa phần\
Số kia nằm rất hiền lành\
Nhìn qua mắt em hóa thành hiểm nguy\
Bạn bè bảo cứ từ từ\
Em nghe chữ Toán đã hư tinh thần
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Toán vừa xuất hiện}
Toán vừa ló mặt trên trang thôi
Tinh thần em đã lui quân rồi
Nếu mà e ngại đem đi chấm
Tên em chắc đã đầy điểm cao
\end{tutuyetbox}

\begin{ghichu}
Không thích Toán là một tâm trạng thật. Nhưng nhiều học sinh hay nhầm giữa “không thích” và “không thèm thử lại”.
\end{ghichu}

\ngancach

\begin{lucbatbox}{Lục bát: Mở đề là mệt}
Đề vừa lật nửa trang thôi\
Em như đã chạy bộ rồi ba cây\
Tim thì đập, mắt thì cay\
Não chưa làm việc, tay đầy ngại ngần\
Nếu mà bình tĩnh một lần\
Câu kia có dễ cũng gần với em
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Ghét từ cái nhìn đầu}
Em ghét môn ấy tự nhiên ghê
Chưa cần học kỹ đã phiền rồi
Nếu người cũng bị ghét mau thế
Chắc bao hiểu lầm nối nhau dài
\end{tutuyetbox}
