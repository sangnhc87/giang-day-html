\chapter{Không Chép Bài}
\chapterintro{Học như đi tu, mang lòng thanh tịnh đến lớp, để lại chữ nghĩa ở trường.}{Trên bàn không có vở, trong đầu không có chữ.}

\begin{tutuyetbox}{Chuyên gia mượn vở}{The Note-Borrowing Expert}
Tay trơn nhẵn thín chẳng ghi bài
Mượn vở thằng bên chép miệt mài
Tới lúc vô thi nhìn tối mắt
Đành buông giấy trắng lệ tuôn hoài.
\tcblower
Hands are smooth, no ink in sight 
Borrowing notes with all your might 
When the test hits, eyes grow dim 
Leaving it white, life's looking grim.
\end{tutuyetbox}

\begin{tutuyetbox}{Vở trắng tinh khôi}{The Pristine Notebook}
Xé nháp truyền tay tít mít mù
Bài ghi trắng bóc tối trăng lu
Dòm vô sách mới bày lố bịch
Điểm rớt oan gia kiếp ngục tù.
\tcblower
Tearing scrap paper to pass around 
Notes are blank, nowhere to be found 
Looking at new books, acting like a fool 
Failing grades, the nightmare of school.
\end{tutuyetbox}

\begin{ghichu}
Đừng tự hào vì cuối năm vở vẫn còn thơm mùi giấy mới. Đó không phải nề nếp, đó là bằng chứng của việc dối gian!
\end{ghichu}
