\chapter{Bạn Bè, Tổ Trưởng, Cán Sự}
\chapterintro{Một lớp học là cả một xã hội thu nhỏ}{Ai làm cán sự lâu rồi cũng có vài nếp nhăn trong giọng nói.}

\begin{lucbatbox}{Lục bát: Tổ trưởng hụt hơi}
Bạn làm tổ trưởng khổ thay
Gánh bao việc lớp suốt ngày chẳng ngơi
Nhắc tên khản cổ hụt hơi
Mà em vẫn cứ dạo chơi bên lề
Cuối ngày đứng tựa góc đê
Thương thân cán sự trăm bề đắng cay.
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Bạn tốt hay "tố"}
Cán sự thường ngày vốn rất tươi
Vào giờ trực nhật bỗng lười cười
Đi trễ quên bài hay ngủ gật
Vạch tên sổ đỏ chẳng nể lời.
\end{tutuyetbox}

\ngancach

\begin{lucbatbox}{Lục bát: Bạn hiền "nhắc" bài}
Hôm nay em lỡ quên bài
Bạn bên khều nhẹ nhắc vài ý hay
Ơn này em nhớ chẳng phai
Lúc thi hai đứa cùng sai một đường
Học hành chẳng chịu tự vươn
Dựa dẫm cho lắm, lệ vương mắt buồn.
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Đoàn kết mùa thi}
Lúc chơi tình nghĩa tựa biển sâu
Kiểm tra bỗng hóa nhịp chung đầu
Đứa nào chưa thuộc nhìn quanh quất
Bạn hiền lơ mặt... dạ thêm sầu.
\end{tutuyetbox}

\ngancach

\begin{lucbatbox}{Lục bát: Lớp trưởng oai vệ}
Lớp trưởng oai vệ đứng đầu
Ghi tên "chiến sĩ" ngủ sâu trong giờ
Mặc cho chúng nó vật vờ
Lệnh quan đã ký, đừng mơ xóa dòng
Thương thay cái kiếp trông song
Làm dâu trăm họ, lòng vòng thị phi.
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Lớp phó học tập}
Lớp phó cầm bài mặt tỉnh bơ
Hỏi gì cũng biết chẳng đợi chờ
Mà sao bài tập không cho chép
Để bạn ngồi đây mặt thẫn thờ.
\end{tutuyetbox}

\ngancach

\begin{lucbatbox}{Lục bát: Bạn thân "chí cốt"}
Bạn thân là đứa hiểu ta
Lên trường nó bóc phốt nhà cho vui
Thầy cô hỏi tội ngậm ngùi
Nó ngồi nó chỉ: "Tội vùi ở đây"
Tình xanh thắm thiết vợi đầy
Lúc nguy nó đẩy, mình gầy xác xơ.
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Hội bàn cuối}
Bàn cuối anh hùng họp rất đông
Học hành thì ít, quậy thì thông
Bảng đen thầy giảng như gió thổi
Chỉ đợi chuông reo chạy sạch lòng.
\end{tutuyetbox}

\ngancach

\begin{lucbatbox}{Lục bát: Thủ quỹ cầm tiền}
Thủ quỹ mặt sắt đen sì
Tiền quỹ chưa đóng, đừng đi đâu ngoài
Nợ nần nhắc nhở dông dài
Thân em như thể chủ đài đòi nợ
Học hành chẳng thấy ai sợ
Sợ cô thủ quỹ mắng khờ bên tai.
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Cán sự lao động}
Lớp phó lao động miệng thét vang
Tay cầm chổi quét rộn xóm làng
Bạn bè trốn sạch vào góc tối
Để một mình em mặt võ vàng.
\end{tutuyetbox}

\begin{ghichu}
Cán sự lớp khổ nhất là lúc phải đứng giữa tình bạn và trách nhiệm, mà bên nào cũng nhìn mình khó chịu.
\end{ghichu}
