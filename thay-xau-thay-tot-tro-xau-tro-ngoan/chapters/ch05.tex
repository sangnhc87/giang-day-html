\chapter{Luoi hoc, doi pho va quay cop}

\begin{leadbox}
Nguoi lon thuong nhin thay ket qua xau va goi do la luoi. Nhung sau hanh vi doi pho co the la lo lang, thieu ky nang, mat goc, hoac da quen voi cam giac that bai. 10 tinh huong sau mo xeo vao lop vo nay.
\end{leadbox}

\storycard{41. To giay kep trong tay ao}{The Hidden Cheat Sheet}{Quay cop la sai, nhung ly do dan den no moi la noi can giao duc den tan cung.}
\storycard{42. Bai tap chep tu mang}{Homework from the Internet}{Khi cong nghe chi duoc dung de tron viec, hoc sinh mat di co hoi hinh thanh nang luc thuc.}
\storycard{43. Loi hua hoc tu ngay mai}{Starting Tomorrow}{Tri hoan khong chi la thoi quen xau; no thuong la vo giap cua nguoi so bat dau.}
\storycard{44. Di hoc them de ngu}{Sleeping Through Tutoring}{Nhieu gio hoc khong bang mot phuong phap hop, va hoc sinh can hoc cach hoc chu khong chi hoc them.}
\storycard{45. Kiem tra 15 phut bat ngo}{The Surprise Quiz}{Mot bai kiem tra ngan co the lo ra lo hong kien thuc, nhung cung lo ra ca cach lop dang hoc vi diem.}
\storycard{46. Ve bai nhung khong hieu}{Copying the Answer}{Hoc sinh co the lam xong bai tap ma van khong hoc duoc gi neu chi di theo dau vet cua nguoi khac.}
\storycard{47. Mat goc mon hoc}{Losing Interest in a Subject}{Khi hoc sinh khong con thay y nghia, dong luc hoc tap suy giam rat nhanh.}
\storycard{48. Ban hoc kem nhung co ky luat}{The Struggling but Steady Student}{Tien bo ben vung thuong khong bat dau tu thong minh, ma tu nhip deu va su kien nhan.}
\storycard{49. Cai gia cua diem so}{The Cost of Grades}{Neu diem tro thanh muc tieu duy nhat, moi thu de thuong tu trung thuc den niem vui hoc tap deu bi dem ra doi.}
\storycard{50. Mot bai kiem tra duoc lam lai}{A Second Chance Test}{Cho lam lai khong phai ha chuan; do co the la cach day trach nhiem va kha nang sua sai.}
