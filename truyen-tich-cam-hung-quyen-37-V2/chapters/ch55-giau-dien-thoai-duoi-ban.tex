\chapter{Giấu Điện Thoại Dưới Bàn}
\chapterintrobi{Bàn học đôi lúc bị dùng như một trung tâm điều hành bí mật}{Hide the phone under the desk}{Khi màn hình sáng hơn bài giảng, sự tập trung thường bắt đầu rời lớp.}

\begin{lucbatbox}{Lục bát: Tin nhắn dưới mép bàn}
Trên kia thầy giảng rất đều\
Dưới này điện thoại phát yêu một đời\
Em nghiêng góc mắt vừa thôi\
Ngón tay lướt nhẹ như người vô can\
Bài đi một nửa rất ngoan\
Tin thì đọc đủ, trang đang bỏ quên
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Màn hình bí mật}
Điện thoại nằm thấp dưới bàn thôi\
Nhưng kéo chú ý lệch xa rồi\
Nếu em giữ bài như giữ máy
Điểm đâu cần phải khóc than hoài
\end{tutuyetbox}

\begin{ghichu}
Giấu điện thoại dưới bàn tạo cảm giác mình rất kín. Thực ra cả lớp thường biết, chỉ riêng phần học là không biết đi đâu.
\end{ghichu}

\begin{englishnote}
\textbf{under the desk}: dưới bàn.\par
\textbf{check messages}: xem tin nhắn.\par
\textbf{I was only checking for a second.}: Em chỉ xem một chút thôi.
\end{englishnote}
