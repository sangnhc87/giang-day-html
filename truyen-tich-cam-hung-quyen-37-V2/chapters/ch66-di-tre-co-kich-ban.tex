\chapter{Đi Trễ Có Kịch Bản}
\chapterintrobi{Tới muộn nhưng phần giải trình thì luôn chuẩn bị kỹ}{Arrive late with a story}{Một số bạn không chỉ đi trễ, mà còn xây dựng hẳn một bối cảnh cho việc đó.}

\begin{lucbatbox}{Lục bát: Lý do rất điện ảnh}
Em vào chậm mấy phút thôi\
Nhưng câu giải thích hẳn dài hai trang\
Xe hư, mưa tạt, đường ngang\
Chó hàng xóm đuổi, bác sang hỏi nhờ\
Nghe thì sống động như thơ\
Chỉ mong đồng hồ đừng chờ thật lâu
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Muộn mà rất mượt}
Đi trễ nhưng lời lại rất hay\
Nghe qua tưởng cảnh phim nào đây\
Nếu em giữ nếp đúng giờ chăm thế\
Đâu cần kể chuyện mỗi ban ngày
\end{tutuyetbox}

\begin{ghichu}
Đi trễ có kịch bản cho thấy năng lực kể chuyện không tệ. Chỉ tiếc nó thường được dùng để trang trí cho một thói quen cũ.
\end{ghichu}

\begin{englishnote}
\textbf{late}: đi trễ.\par
\textbf{excuse}: lý do.\par
\textbf{I have a long story for being late.}: Em có cả một câu chuyện dài về việc đi trễ.
\end{englishnote}
