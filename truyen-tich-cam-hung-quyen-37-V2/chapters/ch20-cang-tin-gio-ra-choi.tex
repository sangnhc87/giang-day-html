\chapter{Căng Tin Giờ Ra Chơi}
\chapterintro{Trống vừa đánh 1 nhịp là cõi lòng đã lao vào nồi mỳ cay.}{Trốn tiết đi ăn là niềm đam mê mãnh liệt.}

\begin{tutuyetbox}{Tranh ăn bấp bênh}{Snack Scramble}
Mười giờ trống nện nhảy nhào mong
Quán chặt chen thân chịu mất phòng
Đứa giựt đồ no nhai hóp háo
Vừa vô vô lớp ngủ say mòng.
\tcblower
Ten o'clock bell, jumping with glee 
Crammed in the shop, nowhere to be 
Grabbing your snacks, chewing in haste 
Back in the class, sleep is to taste.
\end{tutuyetbox}

\begin{tutuyetbox}{Nước đi sai lầm}{The Wrong Move}
Nhịn bữa ăn trưa để nướng đài
Ngồi chơi điện tử sướng hoài ai
Chiều vô trốn lớp luồn theo vách
Đến lúc lòi tên nhục chí trai.
\tcblower
Skipping lunch for a gaming spree 
Digital glory is all you see 
Sneaking past walls in the afternoon 
Caught by the teacher, shame comes soon.
\end{tutuyetbox}

\begin{ghichu}
Ngốn đồ ăn thật nhanh ở căng tin rồi lết về lớp ngồi ngáp, đó là quy trình hoàn chỉnh của lười biếng.
\end{ghichu}
