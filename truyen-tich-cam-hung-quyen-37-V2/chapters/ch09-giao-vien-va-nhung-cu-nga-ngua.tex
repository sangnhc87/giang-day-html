\chapter{Giáo Viên Và Những Cú Ngã Ngửa}
\chapterintro{Có những pha học sinh làm thầy cô đứng hình trong vài giây rất dài}{Không phải cú ngã nào cũng nằm trên sân, nhiều cú ngã nằm ngay ở phần trả lời.}

\begin{lucbatbox}{Lục bát: Trả lời rất tự tin}
Thầy vừa hỏi một câu thôi
Em đứng rất vững đáp lời thật nhanh
Giọng nghe quả quyết rành rành
Chỉ riêng nội dung thì thành hướng xa
Tự tin vốn rất đẹp mà
Giá thêm đúng ý nữa là tuyệt ghê
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Cô nghe mà lặng}
Cô nhìn em rất hiền từ thôi
Nghe câu trả lời bỗng lặng người
Chắc cô đang nghĩ trong im lặng
Sao em đi lạc tới đây rồi
\end{tutuyetbox}

\begin{ghichu}
Điều khiến thầy cô bất ngờ nhất nhiều khi không phải lỗi sai, mà là cách học sinh sai với một niềm tin rất lớn.
\end{ghichu}

\ngancach

\begin{lucbatbox}{Lục bát: Gặp câu hỏi bất ngờ}
Em ngồi tưởng sẽ bình yên
Ai ngờ bị gọi đúng phiên lên lời
Từ khi đứng dậy tới nơi
Não em kịp tắt mất rồi nửa công
Nếu mà ôn kỹ trong lòng
Câu bất ngờ hẳn bớt làm mình run
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Lớp cười rất nhẹ}
Em nói xong rồi lớp khẽ rung
Không ai ác ý chỉ vui thôi
Mai này nhớ lại phen hôm ấy
Biết đâu lại thấy cũng thương rồi
\end{tutuyetbox}
