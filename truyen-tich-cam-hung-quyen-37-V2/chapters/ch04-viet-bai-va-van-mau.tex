\chapter{Viết Bài Và Văn Mẫu}
\chapterintro{Có những bài văn rất tròn, chỉ tiếc là không thấy người viết ở trong}{Văn mẫu giúp đỡ ban đầu, nhưng lạm dụng thì chữ nghĩa mất giọng riêng.}

\begin{lucbatbox}{Lục bát: Nhớ mẫu hơn nhớ ý}
Đề ra mắt bỗng lờ đờ
Tay nhanh tìm kiếm cậy nhờ máy thưa
Văn chương sáo rỗng lọc lừa
Chép nguyên một bản để vừa lòng ta
Mới hay khéo léo thật thà
Học mà chỉ biết tra ra mỗi dòng.
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Bài văn na ná}
Chữ nghĩa mượn vay của thế gian
Vào bài lai láng viết tràn lan
Thầy nhìn nét bút cười ngao ngán
Mẫu cũ từ xưa đã cũ càng.
\end{tutuyetbox}

\ngancach

\begin{lucbatbox}{Lục bát: Mở bài rất kêu}
Mở bài em viết tựa thơ
Câu nào cũng đẹp cũng mơ cũng đầy
Đến phần ý chính cần xây
Tự nhiên chữ nghĩa đứng ngây giữa đường
Hóa ra phần khó khôn lường
Chẳng nằm ở đoạn phô trương ban đầu.
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Kết bài bay cao}
Kết bài em chốt tận mây xanh
Nhân ái bao la tỏa ngát lành
Khổ nỗi thân bài đang nấc nghẹn
Đầu đuôi xa cách chạy loanh quanh.
\end{tutuyetbox}

\ngancach

\begin{lucbatbox}{Lục bát: Văn chương vay mượn}
Văn em mượt tựa dải nhung
Hóa ra chữ nghĩa dùng chung mọi người
Đọc xong thầy chỉ biết cười
Tiếng em chẳng thấy, thấy mười người dưng
Học hành sao cứ lửng lơ
Mượn hồn thiên hạ dệt thơ làm gì.
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Tác giả vắng mặt}
Bút múa lời tuôn tựa suối nguồn
Mượn văn thiên hạ chép luôn tuồn
Thầy tìm nét riêng trong câu chữ
Chỉ thấy văn chương vẻ héo buồn.
\end{tutuyetbox}

\begin{ghichu}
Viết theo văn mẫu quá tay dễ làm bài bớt dở, nhưng cũng làm người viết biến mất rất nhanh.
\end{ghichu}
