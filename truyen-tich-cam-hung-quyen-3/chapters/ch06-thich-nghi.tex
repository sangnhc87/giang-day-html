\chapter{Bai Hoc: Thich Nghi De Song Con (Adapt to Survive)}
\begin{center}
  \textit{\color{truyenblue}Khong phai ke manh nhat ton tai ma la ke thich nghi tot nhat voi su thay doi.}\\
  \textit{(It is not the strongest who survive, but those most adaptable to change.)}\\[4pt]
  \small\color{truyengold}--- Thien nhien day ta ---
\end{center}
\ngancach

\section{Ca Thu Thuc Dong}
\begin{truyen}{Ca Thu Thuc Dong}{Ca Thu}
\chuhoa{K}{}K hi mat ho dong bang, ca thu chuyem sang trao doi chat cham den muc gan nhu ngung han.\\
\textit{(When the lake freezes over, the crucian carp shifts its metabolism to near standstill.)}

No ngung an, ngung boi, tim xuong day bun lan sau buoc vao giac ngu dong sau.\\
\textit{(It stops eating, stops swimming, sinks to the muddy bottom and enters deep winter sleep.)}

Tim dap chi con vai nhip moi phut, nhiet do co the ha xuong gan bang diem dong.\\
\textit{(Heart beats only a few times per minute, body temperature drops near freezing point.)}

Khi mua xuan toi no thuc day khoe manh, bat dau an va lai vui song nhu khong co gi xay ra.\\
\textit{(When spring comes it wakes healthy, starts eating and swims joyfully as if nothing happened.)}

Kha nang thich nghi hoat dong cua co the la su song con thong minh nhat ma sinh vat biet.\\
\textit{(The ability to adapt bodily functions is the smartest survival skill any organism possesses.)}

\end{truyen}
\begin{baihoc}
  Doi khi dung lai, lam cham, hoac nguoi than truoc bao to chinh la dang thich nghi khon ngoan nhat.\\
  \textit{(Sometimes stopping, slowing down, or going quiet before a storm is the wisest form of adaptation.)}
\end{baihoc}

\section{Tac Ke Doi Mau Da}
\begin{truyen}{Tac Ke Doi Mau Da}{Tac Ke}
\chuhoa{T}{}T ren tang da lich su tac ke doi mau tu xanh la cay sang tim tim nau nau chi trong vai giay.\\
\textit{(On a mossy rock the gecko changes color from green to purple-brown in just a few seconds.)}

Khong phai no tu mau nao no muon ma no phan anh chinh xac mau nen phia sau no.\\
\textit{(It does not choose any color it wants but exactly reflects the color of the background behind it.)}

He thong sac to trong te bao da no tich hop tin hieu mat tu mat bong va tu nao.\\
\textit{(The pigment system in its skin cells integrates light signals from its eyes and brain simultaneously.)}

Ke thu nhin thang vao no ma khong thay gi vi bao mat mau sac cua no hoan hao.\\
\textit{(Predators stare directly at it and see nothing because its color camouflage is perfect.)}

Thich nghi khong phai thay doi ban than ma la hoa minh vao hoan canh mot cach tu nhien.\\
\textit{(Adaptation is not changing yourself but naturally blending into circumstances.)}

\end{truyen}
\begin{baihoc}
  Nghe thuat hoa minh khong co nghia la mat di ban sac; do la biet bieu hien dung dieu dung luc.\\
  \textit{(The art of blending in does not mean losing identity; it means knowing to show the right thing at the right time.)}
\end{baihoc}

\section{Trau Bien Song Tren Bai Nong}
\begin{truyen}{Trau Bien Song Tren Bai Nong}{Trau Bien}
\chuhoa{N}{}N oi nui lua am tham phun nham chay ra bien, da lava mat lanh nhanh thanh nhung dao nui nho.\\
\textit{(Where volcanoes quietly erupt into the sea, lava quickly cools into small rocky islands.)}

Chi trong nam muoi nam dau tien khong co gi song duoc tren nhung tang da trang tron.\\
\textit{(In the first fifty years nothing can live on those barren bare rock surfaces.)}

Roi den loai trau bien: chung bam chat vao da lan nhat xua do, song nho nuoc bien va anh sang.\\
\textit{(Then sea lettuce arrives: it clings tightly to the now-cooled rock, living off seawater and sunlight.)}

Chung chet di de lai lop mun huu co mong manh dau tien tren mat da lava lạnh nguoi.\\
\textit{(They die leaving the first thin organic layer on the cold lava surface.)}

Tren lop mun do, sau muoi nam, cac loai co va hoa nho bat dau moc len xanh tuoi.\\
\textit{(On that layer, after ten more years, small grasses and flowers begin growing fresh and green.)}

\end{truyen}
\begin{baihoc}
  Moi truong ky nhat van co the tro nen song duoc neu ai do dam la nguoi di tien phong dau tien.\\
  \textit{(Even the most hostile environment can become livable if someone dares to be the first pioneer.)}
\end{baihoc}

\section{Chim Di Cu Thay Duong Bay}
\begin{truyen}{Chim Di Cu Thay Duong Bay}{Chim Di Cu}
\chuhoa{K}{}K hi bieu bien khi hau thay doi khien dong vat den muon hon moi nam khi loai chim cho doi.\\
\textit{(When climate change makes insects arrive later each year than the birds waiting for them.)}

Nhung con chim co dau oc linh hoat thay doi lich trinh bay cua minh theo.\\
\textit{(The more flexible-minded birds change their own flight schedules accordingly.)}

Mot so loai thay doi han noi dem trung tu khu vuc truyen thong sang cac vi tri moi mat.\\
\textit{(Some species completely shift nesting sites from traditional areas to new locations.)}

Nhung con bam vao tap quan cu thi dan dan tuyet chung trong khi loai linh hoat phat trien.\\
\textit{(Those clinging to old habits gradually go extinct while flexible species thrive.)}

Thich nghi khong phai phan boi qua khu; do la ton trong qua khu va tien buoc vao tuong lai.\\
\textit{(Adapting is not betraying the past; it is honoring the past while stepping into the future.)}

\end{truyen}
\begin{baihoc}
  Thay lich trinh, thay dia ban, thay chien thuat khi hoan canh doi, nhung khong bao gio thay muc tieu.\\
  \textit{(Change the schedule, change the territory, change the tactics when circumstances change, but never change the goal.)}
\end{baihoc}

\section{Rang San Ho Trang Hoa}
\begin{truyen}{Rang San Ho Trang Hoa}{San Ho}
\chuhoa{R}{}R ang san ho tuoi dep nhat the gioi bi tang nhiet va axit hoa bien lam trang hon toan bo.\\
\textit{(One of the world's most beautiful coral reefs was completely bleached by rising temperature and ocean acidification.)}

Nha khoa hoc nghi no da chet nhung sau chin thang mot so polip bat dau thich nghi.\\
\textit{(Scientists thought it was dead but after nine months some polyps began adapting.)}

Chung bat dau chap nhan loai tao co kha nang chiu nhiet cao hon len song trong mo.\\
\textit{(They began accepting heat-resistant algae species to live within their tissues.)}

Khong dep bang truoc nhung chung song, sinh san va dan dan xay lai rao chan theo cach moi.\\
\textit{(Not as beautiful as before but alive, reproducing, and gradually rebuilding the barrier in a new way.)}

Hoi phuc sau ton thuong la dang thich nghi cao quy nhat va can long can dam nhat.\\
\textit{(Recovery after injury is the noblest form of adaptation and requires the most courage.)}

\end{truyen}
\begin{baihoc}
  Khong can phai hoan la hoa da; can la song sot, phat trien va xay dung lai sau tham hoa.\\
  \textit{(No need to be perfectly pristine; what counts is surviving, growing, and rebuilding after disaster.)}
\end{baihoc}

\section{Loai Ech Dong Bang Va Tan Chay}
\begin{truyen}{Loai Ech Dong Bang Va Tan Chay}{Con Ech}
\chuhoa{V}{}V ung Alaska mua dong xuong am 40 do C, loai ech go tim each vuot qua mot cach ky dieu.\\
\textit{(In Alaska when winter drops to minus 40 C, the wood frog overcomes it in a remarkable way.)}

No de nhan the gan nhu dong bang hoan toan, tim ngung dap, phoi ngung tho binh thuong.\\
\textit{(It allows its body to nearly completely freeze, heart stopping, lungs ceasing normal breathing.)}

Nuoc duong va glycoprotein trong mau no ngan can cac tinh the bang pha vo te bao.\\
\textit{(Sugar and glycoproteins in its blood prevent ice crystals from rupturing cells.)}

Mua xuan ve nhiet do tang, tim ech dap lai, phoi thu khi oxy va no nhay di kieu nhu binh thuong.\\
\textit{(Spring comes, temperature rises, the frog's heart beats again, lungs take in oxygen and it hops off normally.)}

No da che thach cau tru sinh va vat ly theo cach khoa hoc chua ai ngu o duoc.\\
\textit{(It has mastered biology and physics in ways science has not yet fully understood.)}

\end{truyen}
\begin{baihoc}
  Doi khi chi can ton tai qua thoi diem kiet suc nhat la mot thanh cong vi dai; hay truong thu nhu con ech.\\
  \textit{(Sometimes merely surviving through the most exhausted moment is a great achievement; hibernate like the wood frog.)}
\end{baihoc}

\section{Gau Bac Cuc Dieu Chinh Che Do An}
\begin{truyen}{Gau Bac Cuc Dieu Chinh Che Do An}{Gau Bac Cuc}
\chuhoa{K}{}K hi bien bang tan chay, gau bac cuc mat nguon ca la thuc an chinh cua minh.\\
\textit{(When sea ice melts, polar bears lose their seal hunting grounds, their primary food source.)}

Cac nha nghien cuu phat hien mot so con gau bat dau an trung chim, tao bien va qua mam.\\
\textit{(Researchers found some bears began eating bird eggs, seaweed, and berries.)}

Loai di ran than tran suot muon nam gio chap nhan che do an thuc vat linh dong hon.\\
\textit{(The species long labeled carnivore now accepts a more flexible plant-heavy diet.)}

Chung khong chet nhu tu tung nghi; chung dieu chinh va tim ra cach song trong the gioi moi.\\
\textit{(They did not die as once predicted; they adjusted and found ways to live in the new world.)}

Khi nguon sinh ke cu mat di, se lo lang qua lau la mat thu gian quy, hay hanh dong ngay.\\
\textit{(When your old livelihood disappears, worrying too long is wasting precious time; act now.)}

\end{truyen}
\begin{baihoc}
  Khong ai bi buoc phai tuyet chung boi viec bien doi hoan canh; thay doi la lua chon cua ban.\\
  \textit{(No one is forced to extinction by changing circumstances; changing with them is your choice.)}
\end{baihoc}

\section{Cay Rung Ngap Man}
\begin{truyen}{Cay Rung Ngap Man}{Cay Man}
\chuhoa{C}{}C ay man song trong nuoc man co nong do muoi cao gap ba lan man nuoc bien binh thuong.\\
\textit{(Mangroves live in salt water three times saltier than average ocean water.)}

Chung phat trien he thong re ky la nho ra khoi mat dat de lay khong khi ngap truc tiep.\\
\textit{(They develop strange root systems sticking up from the ground to breathe air directly.)}

La cay tiet muoi ra ngoai qua cac te bao dac biet de bao ve he thong noi tai.\\
\textit{(Leaves excrete salt through specialized cells to protect the internal system.)}

Chung song o vung ma khong loai cay nao khac chiu duoc, tho dat tho bien lam tam thuong.\\
\textit{(They live where no other trees can survive, claiming estuary land where sea meets land as ordinary.)}

Kha nang ton tai o ranh gioi giua hai the gioi la suc manh hiem co cua nhung linh hon doc dao.\\
\textit{(The ability to survive on the boundary between two worlds is the rare strength of truly unique souls.)}

\end{truyen}
\begin{baihoc}
  Nhung nguoi manh nhat khong song trong moi truong de chiu ma trong moi truong thach thuc.\\
  \textit{(The strongest people do not live in easy environments but in challenging ones.)}
\end{baihoc}

\section{Ca Vang Nho Tao Ra Hang Soa}
\begin{truyen}{Ca Vang Nho Tao Ra Hang Soa}{Ca Vang}
\chuhoa{K}{}K hi nuoc troi can kho do han han, ca vang song trong ao nho bat dau duoc long dat.\\
\textit{(When water evaporates and dries in drought, goldfish living in small ponds began burrowing into soil.)}

Chung vui lon vao lop bun day nhat phat la da am, nam thu minh cho den khi mua den.\\
\textit{(They push deep into the thickest mud layer still moist, curl up and wait until the rain comes.)}

Tim dap rat cham, ho hap bang da va lac chat dinh duong du tru tu dong.\\
\textit{(Heart beats very slowly, breathing through skin and consuming stored nutrients automatically.)}

Sau muoi tun it nuoc roi xuong, chung thoat khoi bun va boi troi ra binh thuong.\\
\textit{(After ten weeks when water falls, they emerge from mud and swim out normally.)}

Co the cua chung bien ao ho thanh noi tru an de song sot qua han ha, khong co gi tu nhien hon.\\
\textit{(Their bodies transform the pond mud into a shelter to survive drought; nothing is more natural.)}

\end{truyen}
\begin{baihoc}
  Khi moi truong thay doi, hay tim rao can moi trong chinh hoan canh kho khan cua ban.\\
  \textit{(When your environment changes, find new shelter within your very difficult circumstances.)}
\end{baihoc}

\section{Loai Qua Hoc Biet Mo Hop}
\begin{truyen}{Loai Qua Hoc Biet Mo Hop}{Con Qua}
\chuhoa{Q}{}Q ua den cua thanh pho Tokyo hoc duoc cach mo hop ca hop thep bang cach tha tu tren cao.\\
\textit{(Tokyo city crows learned to open canned food by dropping it from height onto hard surfaces.)}

Chung cung hoc lot vo hat oc bang cach tha xuong duong va cho xe chay qua no.\\
\textit{(They also learned to peel walnut shells by dropping them on roads and waiting for cars to run over them.)}

The he qua sau hoc lai hanh vi moi tu the he truoc, tao thanh mot nen van hoa thi cu tang dan.\\
\textit{(Subsequent crow generations learn new behaviors from previous ones, creating a culture that accumulates.)}

Day la mot trong nhung bang chung dau tien ve truyen dat van hoa phi con nguoi trong tu nhien.\\
\textit{(This is one of the first pieces of evidence of non-human cultural transmission in nature.)}

Loai co the hoc va day lai cho the he sau se luon thich nghi va ton tai vuot troi.\\
\textit{(Species that can learn and pass on to the next generation will always adapt and survive superbly.)}

\end{truyen}
\begin{baihoc}
  Kha nang hoc, di truyen va thich nghi voi hoan canh moi la di san quy bau nhat ban co the trao con chau.\\
  \textit{(The ability to learn, transmit, and adapt to new circumstances is the most precious legacy you can pass to descendants.)}
\end{baihoc}
