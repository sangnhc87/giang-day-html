\section{Nghe Lợi Nhuận Cao Là Muốn Vào}
\begin{truyen}{Nghe Lợi Nhuận Cao Là Muốn Vào}{Hearing High Profit and Wanting In Immediately}
\chuhoa{C}{o nguoi tre chi can nghe mot con so loi dep la tim da chay nhanh hon ly tri.}
\textit{(Some young people hear an attractive profit number and their heart outruns their judgment.)}

Ho bi hut boi cam giac minh sap di duong tat toi tu do tai chinh.
\textit{(They are drawn to the feeling of taking a shortcut to financial freedom.)}

Nhung cau chuyen tien nhanh thuong chi ke phan sang, con phan mat mat duoc giau rat ky.
\textit{(Fast-money stories usually show the bright side and hide the losses.)}

Khi ham muon di truoc hieu biet, vi tien thuong la noi phai tra hoc phi.
\textit{(When desire moves ahead of understanding, the wallet usually pays the tuition.)}

\end{truyen}

\begin{baihoc}
Tien den nhanh trong loi ke chua chac den nhanh trong doi that.
\textit{(Money that comes fast in stories does not necessarily come fast in real life.)}
\end{baihoc}

\begin{ghinhoanh}
\textbf{Short English to remember:}

Money that comes fast in stories

does not necessarily come fast in real life.

\end{ghinhoanh}
