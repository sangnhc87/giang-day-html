\section{Từ câu 801 đến câu 820}
\begin{enumerate}[leftmargin=*]
\item \hsqa{Em không nói chuyện, em đang trao đổi học thuật.}{Học thuật gì mà cả hai cười như họp chợ?}
\item \hsqa{Em ngủ nhưng em vẫn nghe bài.}{Nghe bằng tâm linh hả em?}
\item \hsqa{Em đi trễ vì đồng hồ hại em.}{Đồng hồ em chắc bị bắt nạt hơi nhiều.}
\item \hsqa{Em không quay bài, em chỉ tham khảo nhanh.}{Tham khảo trong giờ thi nghe rất có nghề.}
\item \hsqa{Em không cãi, em đang bảo vệ quan điểm.}{Quan điểm tốt thường không cần thái độ mạnh tới vậy.}
\item \hsqa{Em không lười, em đang dưỡng sức.}{Sức em dưỡng lâu hơn pin dự phòng luôn.}
\item \hsqa{Em không hỗn, em chỉ thẳng tính.}{Thẳng mà người nghe muốn xỉu thì coi lại hướng thẳng.}
\item \hsqa{Cả lớp làm mà sao chỉ nhắc em?}{Vì em đứng hơi sáng trong phần sai đó.}
\item \hsqa{Em quên mang não hôm nay.}{Mai nhớ mang đủ bộ, đừng chỉ mang xác.}
\item \hsqa{Em không chép bài vì em hiểu rồi.}{Hiểu rồi thì tay em đang thất nghiệp hơi lâu.}
\item \hsqa{Thầy nhìn em là em hiểu rồi.}{Hiểu rồi thì lên bảng chứng minh cho ánh mắt có giá trị đi.}
\item \hsqa{Em không học bài vì muốn giữ đầu óc tự nhiên.}{Tự nhiên kiểu này điểm cũng sống hoang dã lắm.}
\item \hsqa{Em không làm bài để thử lòng thầy.}{Tim thầy ổn, điểm em mới đáng lo.}
\item \hsqa{Em không sai, em khác biệt.}{Khác biệt mà trật đáp án thì vẫn là sai.}
\item \hsqa{Em nói riêng thôi nên không tính.}{Riêng của em nhập lại thành công cộng của lớp rồi.}
\item \hsqa{Em làm sai vì em sáng tạo.}{Sáng tạo mà đề không mời thì hơi lạc chỗ.}
\item \hsqa{Em không bỏ tiết, em chỉ đến muộn lâu.}{Định nghĩa mới nghe rất có tiềm năng gây họa.}
\item \hsqa{Em không vô lễ, em đang phản biện.}{Phản biện mà thiếu tôn trọng thì thành phản ứng.}
\item \hsqa{Em không nhìn bài bạn, mắt em lỡ trôi.}{Mắt em trôi đúng đáp án hay ghê.}
\item \hsqa{Em không làm biếng, em đang để não nghỉ hồi phục.}{Não em phục hồi theo mùa à?}
\end{enumerate}
