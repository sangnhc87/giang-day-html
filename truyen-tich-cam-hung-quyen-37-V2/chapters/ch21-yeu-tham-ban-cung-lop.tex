\chapter{Yêu Thầm Bạn Cùng Lớp}
\chapterintro{Yêu thầm là một loại gia vị, nhưng nêm quá tay thì bài vở mặn chát.}{Ngồi cạnh crush thì tim đập nhanh, óc chạy chậm.}

\begin{tutuyetbox}{Liếc nhìn trộm}{Stealing Glances}
Ngồi cạnh bàn bên mắt láo liên
Dòm người ta học ngỡ thần tiên
Quên luôn bài giảng thầy đang viết
Điểm rớt ngang hông nhục triền miên.
\tcblower
Sitting together, eyes wandering wide 
Looking at them, with nothing to hide 
Forgetting the lecture on the white board 
Grades drop low, a bitter reward.
\end{tutuyetbox}

\begin{tutuyetbox}{Tình thư dang dở}{Unfinished Love Letter}
Viết vội dòng thư nấp gầm bàn
Định trao cho bạn lúc tan hàng
Ngờ đâu giám thị lôi ra đọc
Nhục nhã bêu tên khóc nát mang.
\tcblower
Writing a letter under the desk 
To give it to them, acting grotesque 
The proctor finds it and reads it out loud 
Shameful name, in front of the crowd.
\end{tutuyetbox}

\begin{ghichu}
Đừng để tình thầm làm mờ mắt. Crush có thể thích một người giỏi, chứ ít ai thích một "chiến thần lưu ban".
\end{ghichu}
