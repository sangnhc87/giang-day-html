\chapter{Nội Quy Giấy Kiểm Điểm}
\chapterintro{Viết bảng kiểm điểm dạt dào cảm xúc hơn cả thi môn Văn.}{Hứa ngàn lần rồi cũng chừa ra lần nghìn lẻ một.}

\begin{tutuyetbox}{Nỗi oán thư rơi}{The Sorrowful Confession}
Mười trang kiểm điểm dọn trình bày
Nét chữ ăn năn viết ngó hay
Khóc lóc van lơn rơi rớt lệ
Năm sau bệnh cũ lại xé cày.
\tcblower
Ten pages of penance, displayed so fine 
With apologetic words in every line 
Crying and begging with fake remorse 
Next year you'll follow the same old course.
\end{tutuyetbox}

\begin{tutuyetbox}{Hứa Cuội}{The Empty Vow}
Lỗi phạm muôn trùng tuôn chuỗi dài
Thề rằng nỗ lực lệ tha lai
Cầm đơn vạch áo lòi gian trá
Hạ kiểm đành mang tức thở dài.
\tcblower
A long list of sins, a mountain of guilt 
Vowing to change the life you built 
Holding the paper, your lies are shown 
Conduct points fall, you're on your own.
\end{tutuyetbox}

\begin{ghichu}
Chữ ký dưới bản kiểm điểm luôn có sức nặng, cho đến khi bị gọi lên văn phòng lần thứ hai.
\end{ghichu}
