\chapter{Nhắn Tin Xuyên Đêm}
\chapterintro{Đêm là lúc tâm hồn rạo rực, ngày là lúc cơ thể rã rời.}{Cày game, nhắn tin thì không biết mệt, vào tiết thì ngáp như cá ngáp nước.}

\begin{tutuyetbox}{Đêm trắng}{White Night}
Thức trắng đêm thâu bấm chạch chạch
Tin lòng gửi bạn tít trời xanh
Sáng mai vô lớp đầu như đá
Ngủ gục trên bàn mộng vỡ bành.
\tcblower
Staying up all night, typing away 
Sending your heart, far, far away 
Morning in class, your head is like stone 
Falling asleep, sitting alone.
\end{tutuyetbox}

\begin{tutuyetbox}{Tin nhắn oan gia}{The Accursed Message}
Mắt đọc tin hồng miệng mỉm chi
Ngờ đâu bài tập bỏ quên đi
Thầy tra lý thuyết mồm câm bặt
Điểm liệt mang về khóc lâm li.
\tcblower
Reading pink messages with a sweet smile 
Ignoring the homework for a long while 
Teacher asks questions, you go so still 
Failing the test, against your will.
\end{tutuyetbox}

\end{ghichu}
