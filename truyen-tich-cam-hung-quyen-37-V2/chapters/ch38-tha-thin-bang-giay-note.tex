\chapter{Thả Thính Bằng Giấy Note}
\chapterintro{Trong lớp học, có những mảnh giấy nhỏ mang nhiều rung động hơn cả vở bài}{Cách tỏ ý này vừa kín, vừa run, vừa rất tuổi học trò.}

\begin{lucbatbox}{Lục bát: Gấp giấy thật vuông}
Em ngồi gấp mảnh giấy xinh\
Viết đôi ba chữ lung linh vừa đủ\
Nhờ bạn chuyền hộ một đường\
Tim đi trước giấy, mồ hôi theo sau\
Mảnh note bé xíu thôi nào\
Mà nghe như gánh biết bao bồi hồi
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Chuyền tay rất khẽ}
Tờ note đi hết mấy bàn\
Nhỏ thôi mà khiến tim gan rộn ràng\
Nếu mà bài tập được sang\
Chăm như mảnh giấy, vẻ vang mấy phần
\end{tutuyetbox}

\begin{ghichu}
Thả thính bằng giấy note đáng yêu ở chỗ rất nhỏ mà rất to trong lòng người viết lẫn người chờ nhận.
\end{ghichu}

\ngancach

\begin{lucbatbox}{Lục bát: Viết rồi lại xóa}
Một câu em viết rất lâu\
Xóa đi sửa lại qua cầu mấy phen\
Sợ nhiều, lại sợ ít trên\
Sợ nghe quá lộ, sợ mềm quá xa\
Cuối cùng gửi chữ hiền hòa\
Mà tim đập cứ như là thi xong
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Nhận được là run}
Người kia vừa mở giấy ra\
Mặt không đổi mấy, tim ta đổi liền\
Tuổi học trò thật dịu hiền\
Một mảnh note nhỏ cũng nghiêng nửa ngày
\end{tutuyetbox}
