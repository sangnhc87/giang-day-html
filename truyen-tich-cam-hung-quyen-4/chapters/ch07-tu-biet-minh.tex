\chapter{Bai Hoc: Tu Biet Minh --- Tri Tue Bat Dau Tu Su Khiem Ton (Self-Knowledge --- Wisdom Begins with Humility)}
\begin{center}
  \textit{\color{truyengold}``Tu biet minh la khoi dau cua moi tri tue.''}\\[4pt]
  \textit{(Knowing yourself is the beginning of all wisdom.)}\\[4pt]
  \small\color{truyenblue}--- Co Kim Dong Tay ---
\end{center}
\ngancach

\section{Socrates --- Ta Biet Ta Khong Biet Gi}
\begin{truyen}{Socrates --- Ta Biet Ta Khong Biet Gi}{Socrates --- Hy Lap TK 5 TCN}
\chuhoa{N}{}N am no, than truyen dat Delphi tuyen bo Socrates la nguoi khon ngoan nhat Athens.\\
\textit{(One year, the Oracle at Delphi proclaimed Socrates to be the wisest man in Athens.)}

Socrates boi roi --- ong tu nhan minh la khong co tri tue gi dac biet.\\
\textit{(Socrates was puzzled --- he considered himself to have no special wisdom.)}

Ong quyet dinh kiem tra: ong gap tung nguoi bi cho la khon ngoan va dat cau hoi.\\
\textit{(He decided to verify this: he visited each person reputed to be wise and questioned them.)}

Ket qua: moi nguoi deu nghĩ ho biet nhung thuc ra ho khong biet; con Socrates --- ong biet ong khong biet.\\
\textit{(The result: each person thought they knew things but actually did not; while Socrates --- he knew that he did not know.)}

Do chinh la tri khon cua ong: biet rat ro ranh gioi kien thuc cua minh --- dieu ma nhung nguoi bi goi la khon ngoan khac khong lam duoc.\\
\textit{(That was precisely his wisdom: knowing clearly the limits of his own knowledge --- the very thing others called 'wise' could not do.)}

\end{truyen}
\begin{baihoc}
  Khoi diem cua moi tri tue la biet ro nhung gi minh khong biet --- dieu nay kho hon nhieu so viec biet nhung gi minh biet.\\
  \textit{(The starting point of all wisdom is knowing clearly what you do not know --- this is far harder than knowing what you do know.)}
\end{baihoc}

\section{Benjamin Franklin Va Nhat Ky Tu Hoan Thien}
\begin{truyen}{Benjamin Franklin Va Nhat Ky Tu Hoan Thien}{Benjamin Franklin --- Hoa Ky 1726}
\chuhoa{O}{}O tuoi 20, Benjamin Franklin lap ra mot he thong tu kiem soat ban than chinh minh --- ong gui 13 duc tinh vao mot cuon so tay.\\
\textit{(At age 20, Benjamin Franklin created a system of self-monitoring --- he placed 13 virtues into a small notebook.)}

Moi tuan, ong tap trung chiu nguong luyen mot duc tinh --- va ghi diem do o hen chu buoi toi.\\
\textit{(Each week he focused on practicing one virtue --- and recorded how he had done each evening.)}

Ong lam dieu nay trong suot cuoc doi --- cuon so tay bi rach boc, duoc don chep va tiep tuc.\\
\textit{(He did this throughout his life --- the notebook wore out, was recopied, and continued.)}

Ong viet: 'Toi khong bao gio dat duoc su hoan hao --- nhung noi luc toi co gang hon, toi biet ro hon minh dang o dau.'\\
\textit{(He wrote: 'I never achieved perfection --- but each time I tried harder, I knew more clearly where I stood.')}

Chinh he thong nay giup Franklin tro thanh thanh tich: nha van, nha khoa hoc, nha chinh khach, nha ngoai giao.\\
\textit{(This very system helped Franklin achieve his breadth: writer, scientist, statesman, diplomat.)}

\end{truyen}
\begin{baihoc}
  Tu kiem soat kien tri --- khong phai de tro nen hoan hao --- la cach lam cho ban dan dan tro nen nguoi ban muon tro thanh.\\
  \textit{(Persistent self-monitoring --- not to become perfect --- is how you gradually become the person you want to be.)}
\end{baihoc}

\section{Marcus Aurelius --- Hoang De Tu Van Minh}
\begin{truyen}{Marcus Aurelius --- Hoang De Tu Van Minh}{Marcus Aurelius --- La Ma TK 2}
\chuhoa{M}{}M arcus Aurelius la mot trong nhung vi hoang de la ma quyen luc nhat --- nhung cuon sach ong de lai la mot cuon nhat ky ca nhan.\\
\textit{(Marcus Aurelius was one of Rome's most powerful emperors --- but the book he left behind is a personal diary.)}

Ong viet trong cuon 'Suy Nghi' khong phai de xuat ban ma de chinh ong doc lai --- mot ban ghi nho de tu kiem diem.\\
\textit{(He wrote in 'Meditations' not for publication but for his own rereading --- a memo for self-examination.)}

Ong viet: 'Nguoi khac han ta khi ho lam ta kho chiu. Dieu duy nhat ta kiem soat duoc la phan ung cua ta.'\\
\textit{(He wrote: 'Others are beyond my control when they annoy me. The only thing I control is my own reaction.')}

Ong cung viet: 'Moi buoi sang ta nhac lai: khong co ly do nao de khong han che voi tat ca moi nguoi.'\\
\textit{(He also wrote: 'Every morning I remind myself: there is no reason not to be restrained with everyone.')}

De che La Ma du' canh chien tranh lien mien, song ben vung them hai the ky phan lan vi tri tue tu bien chiet nay.\\
\textit{(The Roman Empire, despite constant warfare, endured stably two more centuries partly because of this philosophy of continuous self-examination.)}

\end{truyen}
\begin{baihoc}
  Lanh dao lon nhat trong lich su khong la nguoi khong sai lam --- ma la nguoi con buoc di lien tuc tu kiem diem ban than.\\
  \textit{(The greatest leaders in history are not those who never err --- but those who continuously walk the path of honest self-examination.)}
\end{baihoc}

\section{Confucius --- Moi Ngay Tu Kiem Ba Dieu}
\begin{truyen}{Confucius --- Moi Ngay Tu Kiem Ba Dieu}{Khong Tu --- Trung Hoa TK 5 TCN}
\chuhoa{C}{}C uoc Thuoc --- mot trong nhung hoc tro cu nhat cua Khong Tu --- co thoi quen moi ngay tu kiem ba dieu.\\
\textit{(Zeng Shen --- one of Confucius's oldest students --- had a daily habit of examining himself on three things.)}

Thu nhat: 'Ta co lam het long voi nguoi khac khong?' Thu hai: 'Ta co giu chu tin voi ban be khong?'\\
\textit{(First: 'Have I been wholehearted with others?' Second: 'Have I kept my word to friends?')}

Thu ba: 'Ta co hoc thuc su va thuc hanh nhung gi thay day khong?'\\
\textit{(Third: 'Have I truly studied and practiced what my teacher taught?')}

Ong khong bat buoc minh dat cac cau hoi phuc tap hon --- ba cau hoi re nay, lam hang ngay, tao nen mot nghia vu kho chap nhan.\\
\textit{(He didn't impose more complex questions on himself --- these three simple questions, done daily, created an obligation hard to meet.)}

Khong Tu khen ong la nguoi tien bo nhanh nhat trong so cac mon de --- khong phai vi ong thong minh hon, ma vi ong tu kiem diem kien nhan hon.\\
\textit{(Confucius praised him as the fastest-progressing student --- not because he was smarter, but because he self-examined more consistently.)}

\end{truyen}
\begin{baihoc}
  Khong can ket thuc ngay voi nhung cau hoi lon --- chi can ba cau hoi nho nhung lam kien tri moi ngay la du.\\
  \textit{(You need not end the day with great questions --- just three simple ones, asked persistently every day, is enough.)}
\end{baihoc}

\section{Gandhi --- Tu Thuong Luong Voi Ban Than}
\begin{truyen}{Gandhi --- Tu Thuong Luong Voi Ban Than}{Mahatma Gandhi --- An Do TK 20}
\chuhoa{G}{}G andhi co mot nguyen tac rat la thuc te: moi chieu toi ong ngoi lai viet ra nhung gi ong da noi, da lam, da nghi trong ngay.\\
\textit{(Gandhi had a very practical principle: every evening he sat and wrote out what he had said, done, and thought during the day.)}

Day khong phai la viec ke su --- do la viec doi chieu: minh da song dung dam me minh chua?\\
\textit{(This was not journaling --- it was comparison: had he lived in accordance with his convictions?)}

Ong viet: 'Suc manh den tu viec lam dung dieu minh biet la dung --- khong phai tu nhung gi nguoi khac mong muon.'\\
\textit{(He wrote: 'Strength comes from doing what you know to be right --- not from what others expect.')}

Moi cuoc thay doi xa hoi to lon phai bat dau tu mot ca nhan tu thay doi --- va ca nhan do phai tu biet ban than minh truoc.\\
\textit{(Every great social change must begin with one person changing themselves --- and that person must first know themselves honestly.)}

'Hay la su thay doi ban muon thay tren doi' --- khong the noi cau noi nay neu ban chua biet ban dang o dau.\\
\textit{('Be the change you wish to see in the world' --- this cannot be said if you don't first know where you yourself stand.)}

\end{truyen}
\begin{baihoc}
  Truoc khi thay doi the gioi, ban phai thay doi ban than --- va truoc dieu do, ban phai hieu ban than bang su trung thuc tam su.\\
  \textit{(Before changing the world you must change yourself --- and before that, you must understand yourself with complete honesty.)}
\end{baihoc}

\section{Albert Einstein --- Biet Dieu Minh Khong Gioi}
\begin{truyen}{Albert Einstein --- Biet Dieu Minh Khong Gioi}{Albert Einstein --- Duc/My TK 20}
\chuhoa{E}{}E instein la bieu tuong cua thien tai hi dai --- nhung it nguoi biet rang chinh ong thuong noi ve su kem coi cua minh.\\
\textit{(Einstein was the symbol of extraordinary genius --- but few knew that he himself often spoke of his own limitations.)}

Ong viet: 'Toi khong co tai nang dac biet --- toi chi la nguoi co su hieu ky dat biet.'\\
\textit{(He wrote: 'I have no special talent --- I am only passionately curious.')}

Ong bat gioi ve nhiều mat: ngheo o xa hoi, hay quen ten nguoi, khong biet cach lam viec theo nhom.\\
\textit{(He was poor in many ways: socially awkward, forgetful of names, unable to work well in groups.)}

Nhung ong biet ro nhung kho khan nay va khong de chung can tra dieu ong gioi nhat: suy nghi doc lap.\\
\textit{(But he knew clearly these difficulties and did not let them obstruct what he did best: independent thinking.)}

'Tri tue la kha nang thich ung voi su thay doi' --- va thich ung bat dau tu biet minh can thay doi dieu gi.\\
\textit{('Intelligence is the ability to adapt to change' --- and adaptation begins with knowing what in yourself needs to change.)}

\end{truyen}
\begin{baihoc}
  Thien tai thiet gua vi ho biet ro dieu gi minh khong biet --- va tap trung het nang luong vao dieu du minh gioi.\\
  \textit{(Geniuses thrive because they know clearly what they don't know --- and focus all their energy on what they are already good at.)}
\end{baihoc}

\section{Tolstoy --- Nha Van Xet Xu Chinh Minh}
\begin{truyen}{Tolstoy --- Nha Van Xet Xu Chinh Minh}{Leo Tolstoy --- Nga TK 19}
\chuhoa{L}{}L eo Tolstoy la tac gia cua Chien Tranh Va Hoa Binh --- mot trong nhung tieu thuyet vi dai nhat nhan loai.\\
\textit{(Leo Tolstoy was the author of War and Peace --- one of humanity's greatest novels.)}

Nhung cuoi doi, ong goc bo tat ca su nghiep van chuong va tro thanh mot nha tu tuong den gian.\\
\textit{(But in old age, he abandoned his literary career and became a simple moral philosopher.)}

Ong soat xet cuoc doi minh va nhan thay: ong da song xa hoa trong khi nong dan lam ruong chet doi.\\
\textit{(He reviewed his life and saw: he had lived in luxury while the peasants who farmed his land starved.)}

Ong bat dau tu thu truuc: hoa cho nong dan, chia dat, tu cam lay cay va lam viec tren dong.\\
\textit{(He began to reform himself: freeing his peasants, dividing the land, picking up a plow and working the fields himself.)}

Vo ong that vong, cac con khong hieu --- nhung ong noi: 'Toi khong the viet ve su that ma khong song theo su that do.'\\
\textit{(His wife was distraught, his children didn't understand --- but he said: 'I cannot write about truth without living by that truth.')}

\end{truyen}
\begin{baihoc}
  Tu biet minh khong du neu khong di kem voi hanh dong thay doi --- biet ma khong lam thi biet cung la nof co nhua.\\
  \textit{(Self-knowledge is not enough unless paired with the action to change --- knowledge without action is just another way of standing still.)}
\end{baihoc}

\section{King Le Thanh Tong --- Vua Hoc Cung Dan}
\begin{truyen}{King Le Thanh Tong --- Vua Hoc Cung Dan}{Le Thanh Tong --- Viet Nam TK 15}
\chuhoa{V}{}V ua Le Thanh Tong la mot trong nhung vi vua hoc rong, thanh tich nhieu nhat trong lich su Viet Nam.\\
\textit{(King Le Thanh Tong was one of the most learned and accomplished rulers in Vietnamese history.)}

Ong chu tri Hoi Tao Dan --- mot to chuc van chuong --- va tu minh viet tho cung voi cac tri thuc.\\
\textit{(He presided over the Tao Dan Club --- a literary society --- and himself composed poetry alongside scholars.)}

Ong bieu hien hay hoi cac quan chuc: 'Dan dan dang song ra sao --- cac nguoi co biet khong?'\\
\textit{(He repeatedly asked officials: 'How are the people actually living --- do you know?')}

Mot lan ong nghe bao cao rang mua mang tot --- nhung roi trac tiep di tham lang va thay nguoc lai.\\
\textit{(Once he received a report that harvests were good --- but he personally visited villages and found the opposite.)}

Sau chuyen di do, ong canh bao: 'Biet ve quan ly ma khong biet ve dan la biet nua voi --- nua con lai moi la quan trong.'\\
\textit{(After that trip he warned: 'Knowing governance without knowing the people is knowing only half --- the other half is what truly matters.')}

\end{truyen}
\begin{baihoc}
  Tri tue quoc gia bat dau tu vua biet minh con nhieu dieu chua biet --- va tu do di tim hieu chu khong cho bao cao may moc.\\
  \textit{(National wisdom begins with a ruler who knows they still have much to learn --- and who goes to find out themselves rather than waiting for sanitized reports.)}
\end{baihoc}

\section{Confucius --- 70 Tuoi Moi Theo Long Muon}
\begin{truyen}{Confucius --- 70 Tuoi Moi Theo Long Muon}{Khong Tu --- Trung Hoa TK 5 TCN}
\chuhoa{K}{}K hong Tu ke lai hanh trinh phat trien ban than trong suot cuoc doi minh bang ngo, ngay trong chuong dau Luan Ngu.\\
\textit{(Confucius recounted his lifetime of self-development in a few words, right in the opening chapter of the Analects.)}

'15 tuoi, ta chi am hoc. 30 tuoi, ta dung vung. 40 tuoi, ta khong nghi ngo.'\\
\textit{('At 15 I set my heart on learning. At 30 I stood firm. At 40 I had no doubts.')}

'50 tuoi, ta biet menh troi. 60 tuoi, tai ta thuan theo loi noi. 70 tuoi, ta theo long muon khong pha khuon phep.'\\
\textit{('At 50 I understood heaven's decree. At 60 my ear was attuned to truth. At 70 I followed my heart without transgressing what was right.')}

Cau cuoi la tat ca tri hue cua ong --- o tuoi 70, tam hon hanh dong hoa lam mot: khong con phai co gang.\\
\textit{(The last sentence contains all his wisdom --- at 70, mind and action became one: no longer needing effort.)}

Moi co the dat duoc dieu nay neu biet minh dang o giai doan nao --- va khong bo cac giai doan can thiet.\\
\textit{(Anyone can reach this if they know what stage they are at --- and do not skip the necessary stages.)}

\end{truyen}
\begin{baihoc}
  Tu biet minh la hanh trinh ca doi --- bay gio ban o giai doan nao la la cau hoi de bi bo qua nhung quan trong nhat.\\
  \textit{(Self-knowledge is a lifelong journey --- knowing what stage you are at now is the most easily skipped yet most important question.)}
\end{baihoc}

\section{Diogenes --- Nguoi Tim Nguoi Trung Thuc}
\begin{truyen}{Diogenes --- Nguoi Tim Nguoi Trung Thuc}{Diogenes --- Hy Lap TK 4 TCN}
\chuhoa{D}{}D iogenes co mot thi nghiem noi tieng: ong di giua cho ban ngay voi cay den trong tay.\\
\textit{(Diogenes had a famous experiment: he walked through the marketplace at noon with a lantern in hand.)}

Nguoi ta hoi: 'Sao ong cam den ban ngay?' Ong tra loi: 'Ta dang tim mot nguoi trung thuc.'\\
\textit{(People asked: 'Why do you carry a lantern in daytime?' He replied: 'I am looking for an honest person.')}

Ong khong noi ong dang tim nguoi khon --- ong dang tim nguoi trung thuc ---va y noi rang hai dieu nay hiem khi song cung nhau.\\
\textit{(He wasn't looking for a smart person --- he was looking for an honest one --- implying these two qualities rarely coexist.)}

Bai hoc sau nhat cua Diogenes: truoc khi tu biet con nguoi khac ban phai hoi --- 'ta co trung thuc voi chinh minh khong?'\\
\textit{(Diogenes's deepest lesson: before knowing other people you must ask --- 'am I honest with myself?')}

Vi tu tra loi that su cau hoi nay con kho hon bat ky cuoc thi triet hoc nao.\\
\textit{(Because truly answering this question is harder than any philosophical examination.)}

\end{truyen}
\begin{baihoc}
  Tu biet minh bat dau tu su trung thuc voi chinh minh --- va day la loai trung thuc kho khan nhat cua tat ca.\\
  \textit{(Self-knowledge begins with honesty toward yourself --- and this is the hardest honesty of all.)}
\end{baihoc}
