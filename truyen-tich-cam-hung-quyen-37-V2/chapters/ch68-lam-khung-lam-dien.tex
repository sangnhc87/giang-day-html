\chapter{Làm Khùng Làm Điên}
\chapterintro{Lớp học luôn có một thằng đóng vai trò hề để giữ bầu không khí vui nhộn.}{Và nó cũng là thằng bị bế lên văn phòng giám thị nhiều nhất.}

\begin{tutuyetbox}{Rạp Xiếc Học Đường}
Bục giảng chưa khô tiếng trống rung
Trò đùa nghịch phá giỡn tưng bừng
Thầy lân trên lầu quăng dép tới
Rớt thẳng sổ đen khóc rụng từng.
\tcblower
The board is still dry, the bell hasn't rung 
Pranking around, with a song that's sung 
Teacher descends with a slipper in hand 
On the blacklist you'll certainly land.
\end{tutuyetbox}

\begin{tutuyetbox}{Thằng Hề Khóc Ròng}
Cợt nhả đua mồm nổ túa lưa
Bạn hiền cười mỉm hứng say thưa
Đình chỉ tuần thi khóc đổ mưa
Nhục nhã bêu tên nát mộng hờ.
\tcblower
Joking and chatting, words fly about 
Friends are smiling, without a doubt 
Suspended from class, crying for days 
A shameful name in a foggy haze.
\end{tutuyetbox}

\begin{ghichu}
Tấu hài là một nghệ thuật, nhưng tấu hài vô tri trong giờ Sinh Hoạt thì là bán mạng cho hội đồng kỹ luật.
\end{ghichu}

{\normalsize\bfseries\color{titlecolor} Bài Đọc Thêm: Làm Lớp Vui Quá Đà}
\end{center}
\chapterintrobi{Có duyên là tốt, nhưng vui quá tay thì tiết học dễ trượt nhịp}{Be too funny in class}{Một vài bạn sinh ra để làm lớp cười, rồi vô tình làm bài học hụt hơi theo.}

\begin{lucbatbox}{Vui đến lố nhịp bài}
Em vừa chen một câu thôi\
Cả lớp cười ngặt, tiết trôi mất đà\
Nhiều khi ý tốt thật mà\
Muốn cho không khí đỡ là nặng ghê\
Chỉ là quá mức một bề\
Tiếng cười đi trước, bài về sau lưng
\tcblower
One line you dropped into the class 
Everyone laughed, time came to pass 
Good intentions were in your mind 
To make the mood a bit more kind 
But you went just a step too far 
Laughter is here, but the lesson is a star.
\end{lucbatbox}

\begin{tutuyetbox}{Duyên hơn kế hoạch}
Em rất có duyên giữa lớp này\
Nói câu nào trúng tiếng cười ngay\
Biết giữ vừa đúng lúc nữa thôi
Tiết học vừa vui chẳng chao lay
\tcblower
You've got a charm in this room today 
Whatever you say, laughter finds a way 
Just learn to stop at the right time 
The class is fun, and the mood is prime.
\end{tutuyetbox}

\begin{ghichu}
Làm lớp vui là một năng lực tốt. Cái cần học thêm là biết dừng đúng lúc để tiếng cười không nuốt mất trọng tâm.
\end{ghichu}
