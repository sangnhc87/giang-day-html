\chapter{Đổi Chỗ Để Tám}
\chapterintro{Mục đích duy nhất của việc xin đổi chỗ là để tìm đối tác buôn chuyện hợp cạ nhất.}{Hợp tác cùng phát triển... khả năng tấu hài.}

\begin{tutuyetbox}{Hội nghị du mục}{Nomadic Conference}
Lén lút dời xa gốc học hành
Ghé tai bạn tốt tám cho nhanh
Thầy cô liếc xéo dòm sái bét
Nhục nhã bêu tên nát mộng lành.
\tcblower
Sneaking away from the study zone 
To chat with a friend, never alone 
Teacher glares at the silly play 
Shameful name, ruined is the day.
\end{tutuyetbox}

\begin{tutuyetbox}{Sự kết hợp tai hại}{The Accursed Duo}
Hai đứa xúm vô nói miên dài
Trách phạt bao phen vẫn mải bài
Đến lúc lên thi lòi dốt nát
Điểm liệt ôm sầu khóc lai rai.
\tcblower
The two of you chatting out loud 
Punished many times, still in the crowd 
When the test hits, showing you're a fool 
Failing grades, the nightmare of school.
\end{tutuyetbox}

\begin{ghichu}
Đổi chỗ để tám chuyện là một nước đi sáng suốt để kết thúc năm học bằng một tấm giấy báo "thi lại".
\end{ghichu}
