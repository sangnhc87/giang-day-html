\chapter{Không Chép Bài}
\chapterintro{Có những em tin vào trí nhớ nhiều hơn vào cây bút}{Và rồi trí nhớ cũng bỏ các em đi rất đúng lúc.}

\begin{lucbatbox}{Lục bát: Chắc con nhớ được mà}
Thầy cô bảo chép ý vào\
Em cười con nhớ, khỏi hao sức người\
Lúc ấy khí thế ngời ngời\
Tới khi hỏi lại thì vơi một nửa\
Thêm vài hôm nữa mới vừa\
Ý đâu mất hết, chỉ chừa niềm tin
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Vở sạch quá mức}
Vở em hôm ấy trắng tinh khôi
Nhìn như chưa sống buổi nào rồi
Nếu kiến thức tìm nơi rộng quá
Chắc nhìn một lát cũng quay thôi
\end{tutuyetbox}

\begin{ghichu}
Không chép bài vì lười khác với không chép vì biết tự học thật. Phần lớn học sinh hay chọn nhầm loại thứ nhất rồi đặt cho nó một cái tên đẹp hơn.
\end{ghichu}

\begin{lucbatbox}{Lục bát: Mượn vở chép sau}
Giờ nào em cũng rất yên\
Không chép vì đã tính liền phương sau\
Tan trường mượn vở bạn mau\
Mà bạn hôm ấy lại đau giống mình\
Thế là kiến thức lặng thinh\
Người thì về ngủ, bài thành mây trôi
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Chụp bảng thay chép}
Em chụp bảng xong tưởng ổn rồi
Kho ảnh học tập trông đông vui
Tối về mở được chừng vài phút
Rồi bao quyết tâm lại bay thôi
\end{tutuyetbox}
