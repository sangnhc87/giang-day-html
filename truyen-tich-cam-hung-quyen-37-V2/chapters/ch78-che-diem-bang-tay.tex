\chapter{Che Điểm Bằng Tay}
\chapterintrobi{Điểm đã chấm rồi nhưng lòng vẫn cố kéo rèm}{Cover the score with a hand}{Thói quen này thường xuất hiện vào đúng lúc bài kiểm tra quay trở lại bàn.}

\begin{lucbatbox}{Lục bát: Giấu nhẹ mà đau nhẹ đâu}
Bài kia trả tới tay em\
Ngón tay tự động phủ lên góc đầu\
Bạn nhìn, em kéo rất mau\
Như che được hết nỗi sầu đang lên\
Điểm kia vẫn đó nguyên tên\
Chỉ là mình bớt để thêm người nhìn
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Phản xạ tự nhiên}
Điểm vừa ló mặt, tay che liền\
Như muốn giữ riêng nỗi ưu phiền\
Nếu lần sau học kỹ hơn một chút\
Có khi khỏi phải diễn cảnh hiền
\end{tutuyetbox}

\begin{ghichu}
Che điểm bằng tay không làm con số đổi khác. Nó chỉ là một chiếc áo khoác mỏng cho sự tự ái lúc mới nhận bài.
\end{ghichu}

\begin{englishnote}
\textbf{cover the score}: che điểm.\par
\textbf{test paper}: bài kiểm tra.\par
\textbf{Do not look at my score yet.}: Đừng xem điểm của mình vội.
\end{englishnote}
