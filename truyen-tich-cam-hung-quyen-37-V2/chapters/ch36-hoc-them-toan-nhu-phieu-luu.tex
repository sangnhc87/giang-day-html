\chapter{Học Thêm Toán Như Phiêu Lưu}
\chapterintro{Có bạn đi học thêm Toán với tâm thế như sắp bước vào một vùng đất chưa có bản đồ}{Mỗi buổi học thêm là một lần vừa hy vọng, vừa nơm nớp sợ không theo nổi.}

\begin{lucbatbox}{Lục bát: Đi học thêm như đi xa}
Tối nay lại học thêm Toán\
Em mang theo đủ bút, vở, niềm lo\
Bước vào lớp đã thấy cho\
Bạn kia hiểu trước, còn mình ngồi im\
Buổi nào cũng tự nhủ tìm\
Biết đâu lần tới mình êm hơn nhiều
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Hy vọng đầu buổi}
Đầu giờ em thấy tự tin ghê
Nghĩ rằng hôm ấy nhìn ra ngay
Giữa giờ công thức chen kín đầu
Tự tin lúc ấy hao đi nửa
\end{tutuyetbox}

\begin{ghichu}
Học thêm Toán như phiêu lưu không phải vì môn học thích dọa ai, mà vì người học bước vào khi nền tảng còn chỗ lỏng chỗ chùng.
\end{ghichu}

\ngancach

\begin{lucbatbox}{Lục bát: Về nhà vẫn mơ hồ}
Ở lớp nghe tưởng hiểu rồi\
Về nhà mở lại thấy trời nhiều sương\
Nét ghi thì vẫn thẳng hàng\
Mà sao bản chất còn đang đứng ngoài\
Nếu mà hỏi lại một vài\
Có khi chuyến ấy bớt dài hơn xưa
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Phiêu lưu có bản đồ}
Đi học thêm đỡ mông lung hơn
Nếu em chịu hỏi tới cùng luôn
Sợ thì vẫn sợ như xưa thôi
Nhưng đường sẽ rõ dần từng hôm
\end{tutuyetbox}
