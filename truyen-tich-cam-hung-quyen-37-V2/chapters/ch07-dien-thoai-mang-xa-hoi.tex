\chapter{Điện Thoại Và Mạng Xã Hội}
\chapterintro{Gần mặt không tỏ, cách mạng ảo ngỡ thấu muôn đời.}{Đạo lý chém trên mạng thì nhiều, điểm số ngoài đời thì thấp lẹt đẹt.}

\begin{tutuyetbox}{Lén lút gầm bàn}{Sneaking Under the Desk}
Bàn tay lén lút dưới gầm bàn
Mắt liếc trông thầy dáng thật ngoan
Đạo lý trên phây tuôn xối xả
Nhìn qua điểm số lệ tuôn tràn.
\tcblower
Hands typing sneaky under the desk
Looking at the teacher, acting grotesque
Preaching online with a holy soul
Seeing your grades, you lose control.
\end{tutuyetbox}

\begin{tutuyetbox}{Đăng hình sống ảo}{Virtual Living}
Hình sang chảnh chọe pốt tràn lan
Chút xíu u sầu léo nhéo than
Tới lúc thầy cô tra lí thuyết
Đầu không chữ nghĩa khóc tàn tàn.
\tcblower
Fancy photos posted everywhere
Whining about a single gray hair
When the teacher asks for a theory
Your brain is empty and your eyes are teary.
\end{tutuyetbox}

\begin{ghichu}
Bạn có thể thao túng tâm lý cả cõi không gian mạng, nhưng thầy cô thì chỉ nhìn vào tờ giấy kiểm tra.
\end{ghichu}
