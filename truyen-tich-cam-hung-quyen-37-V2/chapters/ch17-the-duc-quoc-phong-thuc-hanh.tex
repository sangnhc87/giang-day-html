\chapter{Thể Dục Quốc Phòng Thực Hành}
\chapterintro{Ngồi máy lạnh quen rồi, ra sân nắng tự nhiên muốn ngất xỉu.}{Chưa chạy vòng nào đã râm ran đòi về y tế.}

\begin{tutuyetbox}{Trốn chạy mệt mỏi}{Lazy Escape}
Môn này chạy nhảy đứt lòi hơi
Tới tiết mò ra rũ rượi rời
Báo ốm nằm òm mưu xó khuất
Mùa thi rớt sạch oán than trời.
\tcblower
Running and jumping, out of breath 
Approaching the field like a date with death 
Faking an illness, hiding in the dark 
Failing the course, you lose your spark.
\end{tutuyetbox}

\begin{tutuyetbox}{Đội nắng thao trường}{Under the Hot Sun}
Cởi mũ xa hiên trút gió tà
Cầm cây súng gỗ rớt kêu la
Chuồn nhanh dưới bóng nằm che lấp
Đến lúc hô danh nhục đắng xa.
\tcblower
Take off the hat, embrace the sun 
Drop the wooden gun, thinking it's fun 
Sneak to the shade, hidden from sight 
When names are called, it's a shameful night.
\end{tutuyetbox}

\begin{ghichu}
Trốn nắng dưới gốc cây là một nghệ thuật, nhưng đi thi lại môn Quốc phòng vào giữa trưa hè là một thảm họa.
\end{ghichu}
