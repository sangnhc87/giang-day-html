\chapter{Điện Thoại Và Mạng Xã Hội}
\chapterintro{Cái màn hình nhỏ thường kéo được sự chú ý lớn}{Điện thoại không xấu, chỉ là nhiều bạn để nó làm lớp trưởng của thời gian mình.}

\begin{lucbatbox}{Lục bát: Cầm lên là mất giờ}
Em cầm điện thoại một khi
Định xem năm phút trôi đi nửa chiều
Bảng bài nằm đó đìu hiu
Tin vui clip ngắn thì nhiều vô biên
Nếu mà biết ngắt đúng phiên
Đời đi học bớt đảo điên hơn nhiều
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Lướt rất chuyên tâm}
Lúc lướt mạng em chăm vô cùng
Mỗi bài mỗi ảnh nhớ như in
Giá mà giữ sức bền như thế
Trang vở lâu nay sáng đẹp nhìn
\end{tutuyetbox}

\begin{ghichu}
Điện thoại giỏi nhất ở việc nuốt từng chút thời gian mà người dùng không thấy đau ngay lúc đó.
\end{ghichu}

\ngancach

\begin{lucbatbox}{Lục bát: Chụp bảng cho có lệ}
Em chăm chụp bảng từng dòng
Máy đầy kiến thức đầu trống mênh mông
Bạn bè tưởng rất có công
Ai ngờ chỉ để tối không mở vào
Kho ảnh học tập rất ngầu
Riêng phần ghi nhớ thì hầu như không
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Thông báo quá mạnh}
Máy rung một cái rất khẽ thôi
Tâm trí em đi tận cuối trời
Chỉ một thông báo nhỏ như không
Đủ kéo bài giảng lệch nửa đời
\end{tutuyetbox}
