\chapter{Học Thêm Và Học Nhóm}
\chapterintro{Danh nghĩa là học, thực tế nhiều bữa là xã giao có đề cương}{Nhóm học chỉ hiệu quả khi người ta sợ bài hơn mê tám chuyện.}

\begin{lucbatbox}{Lục bát: Lập nhóm rất nhanh}
Lập nhóm học tập rất mau
Vừa ngồi đã hỏi quán nào trà chanh
Vở bày chẳng thấy học hành
Chỉ nghe tiếng nói vòng quanh chuyện đời
Ba giờ tổng kết thảnh thơi
Bài thì vẫn trống, ảnh tươi đầy hình.
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Học cho yên dạ}
Chiều chiều cắp sách đến lớp thôi  
Nghe giảng gật gù tưởng hiểu rồi  
Về mở trang kia vừa học đó  
Chữ theo cánh gió mất tăm hơi
\end{tutuyetbox}
\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Học cho yên dạ}
Chiều nào sách vở cũng theo vai
Nghe giảng trong đầu tưởng nhớ hoài
Về đến bàn học xem trang cũ
Chữ nghĩa theo mây gió mất rồi
\end{tutuyetbox}


\ngancach

\begin{lucbatbox}{Lục bát: Chia việc rất đều}
Bạn kia chụp ảnh đề cương
Bạn này góp chuyện trên đường rất vui
Bạn kia bấm máy liên hồi
Bạn tôi làm hết, đứng ngồi không yên
Nộp bài mặt mũi tươi liền
Nhóm mình cố gắng, chẳng phiền một ai.
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Học nhóm bằng đồ ăn}
Bánh tráng trà chanh sẵn cạnh mình
Vở bày ngay ngắn trông xinh xinh
Miệng nhai rôm rả quên bài tập
Kiến thức ngồi nhìn cũng lặng thinh
\end{tutuyetbox}

\ngancach

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Học thêm vắng hồn}
Thân ngồi lớp học buổi chiều tà  
Lời thầy như gió thoảng bay xa  
Gật gù chợp mắt hồn phiêu lãng  
Giật mình bảng phấn kín trang qua
\end{tutuyetbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Cái giá của việc lười}
Mải mê tám chuyện ở bàn trà
Bài tập giao về chẳng ngó qua
Đến buổi nộp bài tim đập mạnh
Mượn hồn thiên hạ giả làm ta.
\end{tutuyetbox}

\ngancach

\begin{lucbatbox}{Lục bát: Trốn học nghệ thuật}
Mẹ khen em học thật chuyên cần
Chiều nào cũng thấy bước ra sân
Nào ngờ quán nét dừng chân đứng
Luyện đạo làm game phí vạn lần
Tương lai rồi sẽ về đâu
Khi mà con chữ chẳng bầu bạn thân.
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Hội nghị trà chanh}
Chia nhóm học hành để tiến xa
Vừa vào đã hỏi có quà nha
Hết giờ bài tập còn dang dở
Chỉ thấy hình xinh đăng kín nhà.
\end{tutuyetbox}

\ngancach

\begin{lucbatbox}{Lục bát: Học thêm hay đi dạo}
Em đi học tối mải mê vui
Ghé quán ghé hàng dạo ngược xuôi
Về nhà mẹ hỏi bài xong chưa
Gãi đầu gãi tai miệng mỉm cười
Học hành mà cứ lửng lơ
Điểm mười chỉ có trong mơ hiện về.
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Kết quả mông mênh}
Học nhóm cùng nhau gắng sức bền
Vậy mà con chữ chẳng hề lên
Tại vì mồm miệng bận thi thố
Kiến thức rơi đâu chớ có đền.
\end{tutuyetbox}

\begin{ghichu}
Học nhóm thất bại không vì đông người, mà vì phần thích nói thường lấn phần thật sự muốn hiểu.
\end{ghichu}
