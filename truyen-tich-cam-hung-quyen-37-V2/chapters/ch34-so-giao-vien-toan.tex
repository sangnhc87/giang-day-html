\chapter{Sợ Giáo Viên Toán}
\chapterintro{Có bạn chưa thấy đề đã run, chỉ vì thấy thầy cô Toán bước vào}{Nỗi sợ này phần vì áp lực, phần vì tự mình nợ bài quá nhiều.}

\begin{lucbatbox}{Lục bát: Vừa thấy thầy là rén}
Thầy vừa bước tới cửa thôi\
Tim em đã đánh ba hồi liên miên\
Không hẳn thầy dữ hay nghiêm\
Chỉ vì em nợ quá nhiều bài xưa\
Người ngay đâu phải sớm trưa\
Giật mình mỗi lúc thầy vừa tới nơi
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Sợ từ ánh nhìn}
Ánh nhìn thầy chẳng nói gì\
Mà em đã thấy chân đi chậm dần\
Thật ra nỗi sợ trong lòng\
Là bao công thức còn không chịu làm
\end{tutuyetbox}

\begin{ghichu}
Sợ giáo viên Toán nhiều khi không hoàn toàn vì giáo viên, mà vì cảm giác mình luôn ở thế chưa chuẩn bị kịp.
\end{ghichu}

\begin{lucbatbox}{Lục bát: Gọi lên bảng là run}
Thầy vừa đọc đúng tên em\
Mặt em đổi sắc thêm mềm một phen\
Phấn chưa chạm tới bảng đen\
Não như đã trốn về bên cuối trời\
Muốn bớt hoảng sợ một đời\
Chắc đành học thật hơn lời cầu may
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Nghiêm mà không dữ}
Thầy nghiêm chứ chẳng hề hung\
Chỉ là bài cũ trong lòng em đau\
Nếu mà ôn sớm từ đầu\
Gặp thầy chắc cũng bớt màu tái xanh
\end{tutuyetbox}
