\chapter{Loi noi cua thay, tieng noi cua tro}

\begin{leadbox}
Nhieu vet thuong hoc duong sinh ra khong phai tu don phat, ma tu cau noi. Chuong nay xoay quanh suc nang va suc cong pha cua ngôn tu trong moi quan he giao duc.
\end{leadbox}

\storycard{21. Cau noi: Em se chang bao gio kha len}{You Will Never Improve}{Mot nhan xet tuyet doi hoa co the tro thanh loi tien tri xau ma hoc sinh mang theo rat lau.}
\storycard{22. Khi hoc sinh noi: Em met roi}{When a Student Says I Am Tired}{Lang nghe khong phai nuong chieu; do la cach nhan ra dau hieu can giup do truoc khi qua muon.}
\storycard{23. Loi xin loi truoc bang den}{An Apology Before the Blackboard}{Khi giao vien dam nhan sai, lop hoc hoc duoc mot bai hoc lon hon ca kien thuc mon hoc.}
\storycard{24. Cau tra loi coc can}{The Abrupt Reply}{Mot cau tra loi cat ngang co the khien hoc sinh im lang, nhung cung dong luon kha nang hoi that.}
\storycard{25. Tieng cuoi cua ca lop}{The Class Laughs}{Tieng cuoi khong dung chon co the bien sai sot nho thanh vet nhuc kho quen.}
\storycard{26. Thu gop y khong ky ten}{The Anonymous Note}{Khi hoc sinh chi dam noi qua mot manh giay, do la luc can xem lai khong khi doi thoai trong lop.}
\storycard{27. Loi cam on sau gio hoc}{Thanks After Class}{Nhung moi quan he dep bat dau tu nhung cau rat ngan nhung chan thanh.}
\storycard{28. Tin nhan luc nua dem}{The Midnight Message}{Ky nguyen so mo rong kenh lien lac, dong thoi doi hoi gioi han ro rang giua quan tam va xam lan.}
\storycard{29. Cau hoi ma thay ne tranh}{The Question Avoided}{Khi nguoi day so mot cau hoi cua hoc sinh, bai hoc ve trung thuc tri tue bat dau lo dien.}
\storycard{30. Mot lan cho tro noi den het}{Letting the Student Finish}{Ton trong doi khi bat dau tu viec khong ngat loi.}
