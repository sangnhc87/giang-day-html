\chapter{Hứa Học Từ Thứ Hai}
\chapterintrobi{Lời hứa nổi tiếng nhất đời học sinh}{I will start on Monday}{Kế hoạch học tập thường rất đẹp vào tối Chủ nhật và rất mỏng manh khi sang sáng Thứ hai.}

\begin{lucbatbox}{Lục bát: Lại hẹn đầu tuần}
Chủ nhật em viết rất dài\
Thứ hai đổi mới, miệt mài, chăm hơn\
Sáng ra chuông báo dập dờn\
Ý chí nằm lại cô đơn đầu giường\
Chiều về em lại dễ thương\
Hẹn sang tuần tới, con đường lại xa
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Lịch mới, nếp cũ}
Thứ hai nghe rất có tương lai\
Nhưng tới nơi rồi lại kéo dài\
Kế hoạch đẹp như tranh treo tường vậy\
Ngắm thì thấy thích, làm thì phai
\end{tutuyetbox}

\begin{ghichu}
Học từ Thứ hai là một lời hứa tử tế. Vấn đề là học sinh thường yêu lời hứa đó hơn phần thực hiện nó.
\end{ghichu}

\begin{englishnote}
\textbf{start on Monday}: bắt đầu từ Thứ hai.\par
\textbf{new routine}: nếp mới.\par
\textbf{I will do it next week.}: Tuần sau em sẽ làm thật.
\end{englishnote}
