\chapter{Học Thêm Toán Như Phiêu Lưu}
\chapterintro{Đi học thêm ngỡ là nạp tri thức, ai dè nạp toàn drama 100 chữ.}{Góp mặt cho đủ sĩ số lấy tiền ủng hộ cơ sở vật chất nhà thầy.}

\begin{tutuyetbox}{Cuộc dạo chơi mỏi mệt}{The Tiring Journey}
Đóng mớ tiền vô học chẳng vào
Bạn bè ríu rít nói lao xao
Mải vuốt phôn hoài bôi lếch thếch
Lên thi rớt thảm khóc rêu rao.
\tcblower
Paid the fees but nothing stays 
Friends are chatting in a haze 
Scrolling phone, a messy sight 
Failing tests in the morning light.
\end{tutuyetbox}

\begin{tutuyetbox}{Đầu rơi não tàn}{Brain Drain}
Học riết mà thi rớt não nề
Bao nhiêu công sức đứt thỏa thuê
Trong giờ gục ngã mặt tái mét
Điểm số vùi sâu nhục lối về.
\tcblower
Study hard but fail again 
Effort wasted, all in vain 
Fainting in class, face so white 
Low scores bury the soul's light.
\end{tutuyetbox}
