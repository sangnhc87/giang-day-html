\chapter{Thả Thính Bằng Giấy Note}
\chapterintro{Ngôn tình luôn được viết trên những tờ nháp bỏ đi.}{Thả thính một người chưa chắc dính, nhưng ném mẩu giấy trong lớp thì giám thị thu chắc chắn trúng.}

\begin{tutuyetbox}{Bức thư tay}{The Hand-written Note}
Giấy nháp truyền tay thả thính nồng
Trò lười dệt chữ mộng màu hồng
Ngờ đâu giáo chủ lòi soi móc
Chép phạt trăm lần xót rỗng không.
\tcblower
Passing notes with a spicy crush 
Lazy student, dreaming in a rush 
But the teacher catches the play 
Writing lines for a whole long day.
\end{tutuyetbox}

\begin{tutuyetbox}{Mẩu giấy rơi}{The Fallen Scrap}
Thơ thẩn buông lơi viết hão huyền
Tay chuyền góc lớp lỡ tay duyên
Giám thị đi tuần lên hốt trọn
Khóc nhục bêu tên nát mộng hiền.
\tcblower
Poetry written on a whim 
Passing notes, the light is dim 
Proctor catches the secret prize 
Shameful names and teary eyes.
\end{tutuyetbox}

\begin{ghichu}
Đừng lụy những câu thả thính nắn nót trên giấy note, vì có khi ngày mai chủ nhân của nó còn không chép nổi bài văn trên bảng.
\end{ghichu}
