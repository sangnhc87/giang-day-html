\chapter{Quên Mang Sách}
\chapterintrobi{Đi học đúng môn mà sách ở nhà nghỉ phép}{Forget the textbook}{Không ít bạn thuộc mặt đường tới lớp nhưng không thuộc lịch để mang đúng sách.}

\begin{lucbatbox}{Lục bát: Sách còn nằm ở bàn học}
Sáng nay em bước rất nhanh\
Tới trường mới biết thiếu anh sách rồi\
Bạn thì lật, chữ thì trôi\
Em ngồi ghé mép như ngồi xem phim\
Nếu mà nhớ sách cho yên\
Tiết kia đỡ phải nghiêng nghiêng mượn cùng
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Đi học tay không}
Em đến lớp mà dáng rất hiền\
Chỉ là sách vở lại chưa đi liền\
Kiến thức hôm nay ngồi xa quá đỗi\
Vì khoảng cách từ nhà tới bàn bên
\end{tutuyetbox}

\begin{ghichu}
Quên mang sách thường đi kèm câu quen thuộc: “Hôm nay con tưởng không học bài đó.” Vấn đề là bài đó lại đang nằm ngay trên thời khóa biểu.
\end{ghichu}

\begin{englishnote}
\textbf{textbook}: sách giáo khoa.\par
\textbf{leave at home}: để quên ở nhà.\par
\textbf{I left my book at home.}: Em để quên sách ở nhà.
\end{englishnote}
