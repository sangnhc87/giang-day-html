\chapter{Khoanh Bừa Cầu May}
\chapterintrobi{Khi kiến thức tạm rời sân, niềm tin vào vận may được gọi vào thay}{Guess and hope}{Đây là khoảnh khắc học sinh gửi gắm tương lai cho xác suất.}

\begin{lucbatbox}{Lục bát: Gửi đời cho đáp án A}
Đề kia em ngẫm một hồi\
Thấy bốn đáp án như ngồi nhìn nhau\
Não chưa chọn được lối nào\
Tim bèn chỉ nhẹ: cứ vào A thôi\
Khoanh xong em thấy yên rồi\
Dẫu cho khoa học đang ngồi thở than
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Tin vào nhân phẩm}
Khi chẳng còn nhớ một điều chi\
Em gửi niềm tin vào vận thì\
Nếu điểm tính bằng lòng dũng cảm
Hôm nay em đã quá uy nghi
\end{tutuyetbox}

\begin{ghichu}
Khoanh bừa không phải chiến lược học tập. Nó chỉ là cách lịch sự để thừa nhận mình chưa chuẩn bị đủ.
\end{ghichu}

\begin{englishnote}
\textbf{guess}: đoán.\par
\textbf{multiple choice}: trắc nghiệm.\par
\textbf{I just picked one and hoped.}: Em chọn đại rồi hy vọng.
\end{englishnote}
