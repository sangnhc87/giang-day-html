\section{Người thân đôi khi làm ta đau nhất}
\begin{truyen}{Người thân đôi khi làm ta đau nhất}{Life lesson}
\chuhoa{Đ}{ó là chuyện nhiều người từng trải. Đau nhưng là sự thật. Người thân đôi khi làm ta đau nhất.}
\textit{(That is what many have been through. It hurts but it is true. Người thân đôi khi làm ta đau nhất.)}
\end{truyen}

\begin{baihoc}
Sự thật đôi khi đau — nhìn thẳng để bước tiếp.
\textit{(The truth sometimes hurts — face it to move on.)}
\end{baihoc}

\begin{ghinhoanh}
\textbf{Short English to remember:}

The truth sometimes hurts — face it to move on.

\end{ghinhoanh}

\begin{tuhocnhanh}
