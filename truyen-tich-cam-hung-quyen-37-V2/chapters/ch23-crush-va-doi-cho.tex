\chapter{Crush Và Đòi Đổi Chỗ}
\chapterintro{Có những cuộc đổi chỗ nghe rất học tập nhưng lý do thì ngồi ngay ở dãy bên}{Tình cảm tuổi học trò thường bắt đầu bằng một đề nghị rất vòng vo.}

\begin{lucbatbox}{Lục bát: Xin đổi chỗ rất ngoan}
Em xin đổi chỗ nhẹ nhàng\
Nói là để dễ học hành hơn thôi\
Nhưng khi ngồi đúng chỗ ngồi\
Mắt em lại ngó một người hơi lâu\
Bạn bè hiểu chuyện từ đầu\
Chỉ thầy cô tưởng mình cầu tiến hơn
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Lý do học tập}
Miệng em trình bày nghe rất xuôi
Rằng chỗ cũ khó tập trung ghê
Chỉ là chỗ mới ngồi thôi ấy
Lại gần ngay một người em thương
\end{tutuyetbox}

\begin{ghichu}
Đổi chỗ vì crush là chiến lược giàu động lực. Chỉ tiếc động lực ấy thường không chảy nhiều vào bài học.
\end{ghichu}

\ngancach

\begin{lucbatbox}{Lục bát: Gần hơn mà vẫn xa}
Ngồi gần được một hàng ghế\
Em vui hơn cả lúc đề dễ bất ngờ\
Chỉ là từ sáng đến trưa\
Vẫn chưa dám nói dư thừa câu chi\
Hóa ra đổi chỗ làm gì\
Nếu tim thì mạnh, miệng đi rất vòng
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Chỗ đẹp, điểm xấu}
Chỗ ngồi nay đã gần hơn rồi
Mắt em vui lắm vở buồn thôi
Nếu em chăm học bằng chừng ấy
Bàn đầu bảng chắc cũng mừng rồi
\end{tutuyetbox}
