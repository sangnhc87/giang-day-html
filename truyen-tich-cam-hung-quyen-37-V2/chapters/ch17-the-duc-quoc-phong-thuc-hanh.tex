\chapter{Thể Dục, Quốc Phòng Và Thực Hành}
\chapterintro{Có những môn không quá khó, chỉ khó đúng vào lúc phải vận động thật}{Đây là vùng đất màu mỡ của mọi kiểu chống chế học đường.}

\begin{lucbatbox}{Lục bát: Thể dục đầu tuần}
Nghe tin có tiết vận động
Mặt em đổi sắc như đông ghé về
Chạy chưa hết một vòng sân
Em đã thở kiểu như cần viết thư
Bạn bè còn cố bon bon
Em xin dừng lại để thương lấy mình
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Quốc phòng rất oai}
Em đi quốc phòng trông rất oai
Nắng lên một chút đã mềm vai
Tinh thần kỷ luật còn nguyên đó
Chỉ có thể lực chạy sai hoài
\end{tutuyetbox}

\begin{ghichu}
Những môn cần vận động luôn bóc trần rất nhanh khoảng cách giữa lời hứa “em khỏe mà” và sự thật ngoài sân.
\end{ghichu}

\ngancach

\begin{lucbatbox}{Lục bát: Thực hành nhưng sợ bẩn}
Giờ thực hành tới rất nhanh
Em mang găng mới dáng thành chuyên gia
Đụng vô thì lại kêu xa
Sợ dơ sợ rối sợ là sai luôn
Muốn mà kết quả vuông tròn
Ít ra cũng phải động đòn chút đi
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Chạy bền và nhân sinh}
Chạy bền mới thấy sức lòng em
Nửa vòng sân đã muốn mềm chân
Nếu ai sống mãi bằng ngại ngần
Thì môn bền chí chắc vắng dân
\end{tutuyetbox}
