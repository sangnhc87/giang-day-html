\chapter{Tự Kỷ Khi Rớt Tiết}
\chapterintro{Ngồi trong góc lớp, nhìn mưa rơi và tưởng tượng bản thân là nhân vật chính trong một bộ web drama bi đát.}{Thế giới sụp đổ không phải vì tận thế, mà vì bảng điểm dưới 5.}

\begin{tutuyetbox}{Góc khuất sầu bi}{The Sorrowful Corner}
Góc phòng câm lặng núp im lìm
Điểm số te tua buốt vỡ chìm
Nhìn chúng vui đùa cười ha hả
Một mình rơi lệ khóc từ tim.
\tcblower
Silent corner, hiding in the room 
Exam scores bring a sudden gloom 
Watching others laugh and play 
Crying alone, at the end of the day.
\end{tutuyetbox}

\begin{tutuyetbox}{Kịch bản drama}{Drama Script}
Nhìn mưa lất phất dáng u buồn
Tưởng chừng rớt môn nhục đắng luồn
Tự họa thân tàn như diễn xuất
Trình bày kịch bản khóc thành tuôn.
\tcblower
Looking at the rain with a sad face 
Failing subjects in this lonely place 
Acting like a tragic movie star 
Crying tears, near and far.
\end{tutuyetbox}

\begin{ghichu}
Khóc lóc một mình không làm thay đổi điểm số đâu, nhưng nó giúp bạn luyện nét diễn xuất để khi về nhà khóc cho má tin.
\end{ghichu}
