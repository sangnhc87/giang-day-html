\chapter{Điện Thoại Và Mạng Xã Hội}
\chapterintro{Cái màn hình nhỏ thường kéo được sự chú ý lớn}{Điện thoại không xấu, chỉ là nhiều bạn để nó làm lớp trưởng của thời gian mình.}

\begin{lucbatbox}{Lục bát: Thức đêm lướt máy}
Tối về lướt máy thâu đêm
Mắt quầng môi tái bỏ quên học bài
Lên trường mệt mỏi dông dài
Lời thầy như gió thoảng ngoài tai em
Học hành chẳng chịu ngó xem
Chỉ lo "hóng biến" mà quên tương lai.
\end{lucbatbox}


\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Tâm hồn mạng ảo}
Thân ngồi lớp học dạ phương nao
Điện thoại xa tay ngỡ chiêm bao
Chỉ đợi người kia vừa thả thích
Một dòng hồi đáp dậy lòng xao
\end{tutuyetbox}


\begin{ghichu}
Nỗi khổ của thầy không phải là thu điện thoại của các em, mà là thấy các em mang một "cơ thể không linh hồn" vào lớp vì thức đêm cày phim, lướt mạng.
\end{ghichu}

\begin{lucbatbox}{Lục bát: Anh hùng bàn phím}
Trên mạng em luận như thần
Chữ tuôn dào dạt chẳng ngần ngại chi
Thế mà đến tiết kiểm tra
Em ngồi ngơ ngác như là người dưng
Học hành thì cứ lửng lừng
Chỉ lo gõ phím nửa chừng hư vô.
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Nỗi nhớ không tên}
Dù tay không giữ cái màn hình
Nhưng dạ tương tư ảnh bóng hình
"Tóp Tóp" đêm qua còn dang dở
Nên giờ con chữ cứ lặng thinh.
\end{tutuyetbox}

\ngancach

\begin{lucbatbox}{Lục bát: Nghệ thuật thức khuya}
Đêm khuya điện thoại sáng ngời
Học bài em bảo mệt rồi thầy ơi
Sáng ra lớp học thảnh thơi
Gục đầu một giấc kệ đời trôi ngang
Bằng xinh vẫy gọi bên đường
Chào em "chiến sĩ" thiên đường mạng xanh.
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Cái giá của pin đầy}
Sạc pin cho máy thật là nhanh
Để thức thâu đêm lướt cho đành
Kiến thức trong đầu vơi vạch đỏ
Điểm thi chót vót nét mong manh.
\end{tutuyetbox}

\ngancach

\begin{lucbatbox}{Lục bát: Đam mê sống ảo}
Check-in ảnh đẹp lung linh
Ngỡ đâu kiến thức cũng hình như hoa
Nào ngờ thi cử lộ ra
Chữ đi mượn tạm, người ta làm giùm
Học mà cứ thích um sùm
Trên mạng rạng rỡ, trong mùng lệ rơi.
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Cơn nghiện không lời}
Máy thu dạ bỗng thấy bâng khuâng
Ngồi học mà tâm cứ tần ngần
Thông báo giờ này ai nhắn nhỉ?
Để rồi con chữ chẳng hề gần.
\end{tutuyetbox}

\ngancach

\begin{lucbatbox}{Lục bát: Tương lai mạng ảo}
Ước làm "Ai-đồ" (Idol) cao sang
Mà sao con chữ dở dang thế này
Học hành chẳng chịu miệt mài
Chỉ lo tính chuyện dông dài "lai-stream" (Livestream)
Mai sau vất vả đi tìm
Cái nghề nuôi sống, lại chìm vực sâu.
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Nhắn nhủ chân tình}
Mạng ảo vui chơi chốc lát thôi
Đừng để thời gian cứ thế trôi
Học hành mới thực là gốc rễ
Để đời tươi sáng rạng nụ cười.
\end{tutuyetbox}
