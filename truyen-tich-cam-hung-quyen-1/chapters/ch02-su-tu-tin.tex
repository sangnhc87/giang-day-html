\chapter{Sự Tự Tin (Confidence)}

\begin{center}

\ngayDay{}
\section{Giọng Hát Riêng}
\begin{truyen}{Giọng Hát Riêng}{Truyện loài vật}
\chuhoa{A}{ } \textbf{small bird} was \textbf{afraid} to sing.\\
\textit{(Một chú chim nhỏ rất sợ hãi việc phải cất tiếng hót.)}

She \textbf{thought} her \textbf{voice} was ugly.\\
\textit{(Cô nghĩ rằng giọng hát của mình vô cùng chua chát xấu xí.)}

One \textbf{day}, a \textbf{wise owl told} her to just \textbf{try}.\\
\textit{(Một ngày nọ, một bác cú già tinh khôn đã khuyên cô hãy cứ thử một lần xem sao.)}

She \textbf{closed} her \textbf{eyes} and \textbf{sang loudly}.\\
\textit{(Cô chim nhắm nghiền hai mắt lại và cất cao tiếng hót vang dội.)}

The \textbf{whole forest stopped} to \textbf{listen} to her \textbf{beautiful song}.\\
\textit{(Cả một khu rừng bỗng chốc im bặt để lắng nghe được âm vọng tuyệt đẹp của cô.)}
\end{truyen}

\begin{baihoc}
  \textbf{Do not hide your talents.} \textit{(Đừng bao giờ giấu giếm đi những tài năng thật sự của bản thân.)}\\
  \textbf{Trust your voice.} \textit{(Hãy có niềm tin vào chất giọng của riêng mình.)}
\end{baihoc}

\ngayDay{}
\section{Chấp Nhận Bản Thân}
\begin{truyen}{Chấp Nhận Bản Thân}{Chuyện cuộc sống}
\chuhoa{A}{ } \textbf{boy had} a bright \textbf{red coat}.\\
\textit{(Một cậu học trò sở hữu một chiếc áo khoác màu đỏ chói.)}

He \textbf{was shy} because it \textbf{was different}.\\
\textit{(Cậu vô cùng xấu hổ và rụt rè chỉ vì nó mang một diện mạo khác biệt.)}

He \textbf{wore} it to \textbf{school} anyway.\\
\textit{(Dù vậy cậu vẫn quyết định mặc nó đến trường học.)}

Everyone \textbf{loved} the \textbf{color} and \textbf{praised} him.\\
\textit{(Thế nhưng tất thảy mọi người lại đều yêu thích tông màu rực rỡ đó và dành những lời khen ngợi cho cậu.)}

He \textbf{felt happy} and \textbf{confident} all \textbf{day}.\\
\textit{(Cậu bé thầm cảm thấy rất vui vẻ và tràn trề sự tự tin suốt dọc cả một ngày trời.)}
\end{truyen}

\begin{baihoc}
  \textbf{Being different is a strength.} \textit{(Việc bạn mang một màu sắc khác biệt chính là điểm mạnh của bạn.)}
\end{baihoc}

\section{Vượt Qua Sợ Hãi}
\begin{truyen}{Vượt Qua Sợ Hãi}{Truyện ngụ ngôn}
\chuhoa{A}{ } \textbf{seed lay deep} in the \textbf{dark soil}.\\
\textit{(Một hạt mầm nằm im lìm chôn vùi lấp sâu dưới một mô đất đen kịt.)}

It \textbf{was scared} of the \textbf{sunlight}.\\
\textit{(Nó ôm trong mình một nỗi khiếp sợ với ánh nắng của mặt trời.)}

But it \textbf{pushed through} the \textbf{dirt}.\\
\textit{(Thế nhưng rồi nó cũng đã lấy được can đảm để đâm chồi đội lớp đất mùn mọc vươn lên.)}

It \textbf{became} a \textbf{beautiful flower}.\\
\textit{(Và rồi nó đã trở thành một đóa hoa vô cùng rực rỡ lộng lẫy.)}
\end{truyen}

\begin{baihoc}
  \textbf{Growth requires stepping out of the dark.} \textit{(Sự phát triển luôn luôn đòi hỏi bạn phải dốc sức bước ra khỏi vùng tối mù mịt.)}
\end{baihoc}

\section{Sai Lầm Nhỏ}
\begin{truyen}{Sai Lầm Nhỏ}{Chuyện hằng ngày}
\chuhoa{A}{n} \textbf{artist painted} a \textbf{picture} slightly \textbf{off center}.\\
\textit{(Một người họa sĩ đã vẽ nên một bức tranh nghệ thuật bị lệch nhẹ ra khỏi khung ở tâm.)}

He \textbf{wanted} to \textbf{throw} it away.\\
\textit{(Ông ta đã toan vứt xó nó đi ngay lập tức.)}

A \textbf{buyer saw} it and \textbf{called} it a \textbf{masterpiece}.\\
\textit{(Một vị khách mua hàng đã tình cờ nhìn thấy nó và gọi tác phẩm này là một tuyệt tác.)}

\textbf{Mistakes} can be \textbf{art}.\\
\textit{(Những sai lầm đôi khi cũng có thể là một nét nghệ thuật.)}
\end{truyen}

\begin{baihoc}
  \textbf{Confidence means owning your flaws.} \textit{(Tự tin có nghĩa là biết làm chủ và chấp nhận cả những khiếm khuyết của bản thân.)}
\end{baihoc}

\section{Bước Chân Đầu Tiên}
\begin{truyen}{Bước Chân Đầu Tiên}{Chuyện tuổi thơ}
\chuhoa{T}{om} \textbf{refused} to \textbf{dance} at the \textbf{party}.\\
\textit{(Tom đã từ chối giãy nảy để không phải tham gia việc khiêu vũ tại một bữa tiệc.)}

He \textbf{felt clumsy}.\\
\textit{(Cậu ấy cảm thấy bản thân mình thực sự rất vụng về xốc xếch.)}

His \textbf{friends pulled} him \textbf{in}.\\
\textit{(Những người bạn của cậu đã cố sức kéo tuột cậu vào sàn nhảy.)}

He \textbf{started to move} and \textbf{realized} it \textbf{was fun}.\\
\textit{(Cậu bắt đầu cử động các bước chân và từ từ nhận ra rằng việc này lại vui vẻ đến lạ lùng.)}
\end{truyen}

\begin{baihoc}
  \textbf{Stop worrying and start living.} \textit{(Hãy dừng việc lo lắng tột độ đi và bắt đầu sống chậm lại trọn vẹn hơn.)}
\end{baihoc}

\section{Lời Động Viên}
\begin{truyen}{Lời Động Viên}{Truyện loài vật}
\chuhoa{A}{ } \textbf{frog fell} into a \textbf{deep hole}.\\
\textit{(Một chú ếch vô cớ bị rơi thỏm xuống một cái hố rất sâu.)}

Due to \textbf{bad dirt}, \textbf{his eyes were blurry}.\\
\textit{(Do lớp bụi bẩn tệ hại, hai mắt của cậu đã bị che mờ mịt mù.)}

He \textbf{thought} the others \textbf{were cheering} him \textbf{up}.\\
\textit{(Cậu ta đã tưởng lầm rằng những người bạn trên bờ đang vỗ tay nhiệt tình cổ vũ động viên cậu.)}

He \textbf{jumped out} with \textbf{great power}.\\
\textit{(Cậu rướn mình phóng nhảy thoát ra khỏi nơi tối tăm bằng một sức mạnh vĩ đại.)}

Actually, they \textbf{had been telling} him to \textbf{give up}.\\
\textit{(Sự thật phũ phàng là, lũ chúng nó trên kia đang thi nhau khuyên cậu hãy từ bỏ hi vọng ngỏm đi cho xong.)}
\end{truyen}

\begin{baihoc}
  \textbf{Ignore negative forces.} \textit{(Hãy tập lờ đi và bỏ ngoài tai những thế lực quá đỗi tiêu cực xung quanh.)}
\end{baihoc}

Others \textbf{said} it \textbf{would sink quickly}.\\

\textit{(Kỳ diệu rằng nó lại có thể trôi bềnh bồng đi xa và dai dang dẳng hơn bất cứ chiếc thuyền bằng nhựa cứng nào.)}

  \textbf{Have faith in what you build.} \textit{(Hãy có giữ lấy một tấm đức tin bất diệt vào những thứ do chính tay bạn xây dựng hình thành nên.)}
\end{baihoc}

\section{Sáng Tạo Vô Hạn}
\begin{truyen}{Sáng Tạo Vô Hạn}{Nghệ thuật}
\chuhoa{A}{ } \textbf{writer stared} at a \textbf{blank white page}.\\
\textit{(Một nhà văn ngồi thẫn thờ nhìn chằm chằm đầy bất lực vào một tờ giấy trắng tinh trơn tuột.)}

He \textbf{felt} he \textbf{had nothing} to \textbf{write}.\\
\textit{(Ông nhụt chí cảm thấy dường như mình đã hoàn toàn cạn kiệt không còn dính một chữ nào để viết.)}

He \textbf{just wrote one word}.\\
\textit{(Ông đã lấy hết sức bình sinh chỉ viết duy nhất ra một chữ cái đầu tên.)}

Soon, \textbf{an entire book followed}.\\
\textit{(Chẳng mấy chốc sau đó, cả một cuốn truyện dày dặn cứ liên hồi tuôn chảy nối tiếp bủa vây.)}
\end{truyen}

\begin{baihoc}
  \textbf{Confidence builds from action.} \textit{(Sự tự tin nảy nở được gầy dựng chặt chẽ từ chính hành động kiên quyết của bạn.)}
\end{baihoc}

\section{Tỏa Sáng}

\textit{(Có một ngôi sao kích cỡ nhỏ bé tí tẹo thảm hại hơn hẳn những người bạn đồng hành.)}

It \textbf{wanted to shine}.\\
\textit{(Nó vô cùng khát khao mãnh liệt muốn được chớp lóe tự mình tỏa sáng thật rạng rỡ.)}

It \textbf{burned steadily} all \textbf{night long}.\\
\textit{(Nó cứ bền bỉ cháy rực kiên định suốt dọc một đêm đen.)}

Sailors \textbf{used it} to \textbf{find their way home}.\\
\textit{(Đến tận mức những thủy thủ đoàn phương xa đã ưu ái tin dùng chính nó để vạch lối dò tìm đường về nhà.)}
\end{truyen}

\begin{baihoc}
  \textbf{Size does not measure your worth.} \textit{(Kích cỡ hữu hình chưa bao giờ là thước đo để đo đạc giá trị cốt lõi của riêng bạn cả.)}
\end{baihoc}
