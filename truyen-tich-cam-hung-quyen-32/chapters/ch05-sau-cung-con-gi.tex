\chapter{Sau cùng còn gì}
\markboth{Sau cùng còn gì}{What Remains}

\section{Một người thắng cả đời nhưng tuổi già rất cô độc.}
\begin{truyen}{Một người thắng cả đời nhưng tuổi già rất cô độc.}{Winning everything can still leave a quiet house.}
\chuhoa{O}{ng từng là người ai cũng nể, mọi cuộc cạnh tranh đều muốn vượt lên. Nhưng khi về già, nhà rộng và yên đến lạnh. Lúc ấy ông mới hiểu có những cái thắng gom được rất nhiều thứ, trừ người ở lại bên mình.}
\textit{(He was once respected and determined to outdo everyone. In old age, his large house felt cold and silent. He then understood that some victories gather many things, except the people who remain beside you.)}
\end{truyen}
\begin{baihoc}
Nếu cuối cùng không còn ai muốn ngồi cạnh, nhiều cái thắng trước đó sẽ bỗng nhỏ lại.
\textit{(If in the end no one wants to sit beside you, many earlier victories suddenly become small.)}
\end{baihoc}
\begin{ghinhoanh}\textbf{Short English to remember:}

Winning everything can still leave a quiet house.
\end{ghinhoanh}
\begin{tuhocnhanh}
1. Cách sống hôm nay của em có giữ người lại hay đẩy người ra xa? \textit{(Does your way of living keep people close or push them away?)}

2. Cuối đời, em muốn còn lại điều gì nhất? \textit{(At the end of life, what do you most want to remain?)}
\end{tuhocnhanh}

\section{Một người thua vài cuộc nhưng giữ được danh dự.}
\begin{truyen}{Một người thua vài cuộc nhưng giữ được danh dự.}{Integrity can outlast defeat.}
\chuhoa{A}{nh không chấp nhận lách luật để thắng nhanh. Vì vậy anh thua vài dịp trước mắt. Nhưng lâu dần, điều còn lại với anh là sự tin cậy của người khác. Có những phần thua ngắn mà đỡ cho ta một phần nhục dài.}
\textit{(He refused to bend rules for quick wins, so he lost some immediate chances. Over time, what remained was the trust of others. Some short losses protect us from a long shame.)}
\end{truyen}
\begin{baihoc}
Danh dự không phải cái giúp ta thắng nhanh, nhưng là cái giúp ta còn đứng vững lâu.
\textit{(Integrity may not help you win quickly, but it helps you stand for a long time.)}
\end{baihoc}
\begin{ghinhoanh}\textbf{Short English to remember:}

Integrity can outlast defeat.
\end{ghinhoanh}
\begin{tuhocnhanh}
1. Có chiến thắng nào em sẽ từ chối nếu nó làm bẩn mình không? \textit{(Is there any victory you would reject if it stained your character?)}

2. Em muốn người khác tin mình ở điểm nào nhất? \textit{(What do you most want others to trust about you?)}
\end{tuhocnhanh}
\ngancach

\section{Một người thua tiền nhưng thắng được bài học.}
\begin{truyen}{Một người thua tiền nhưng thắng được bài học.}{A painful lesson can still be profit.}
\chuhoa{C}{ú đầu tư sai làm anh mất một khoản lớn. Đau thì rất đau, nhưng từ đó anh bớt tham, bớt tin dễ và bớt tưởng mình hiểu hết. Số tiền mất đi không quay lại, nhưng cách sống sau đó trưởng thành hơn hẳn.}
\textit{(A wrong investment cost him a large amount. It hurt deeply, yet from then on he became less greedy, less gullible, and less certain of himself. The money did not return, but his way of living became more mature.)}
\end{truyen}
\begin{baihoc}
Thua mà biết rút bài học còn hơn thắng rồi giữ nguyên thói dại.
\textit{(Losing and learning is better than winning and keeping the same foolish habits.)}
\end{baihoc}
\begin{ghinhoanh}\textbf{Short English to remember:}

A painful lesson can still be profit.
\end{ghinhoanh}
\begin{tuhocnhanh}
1. Cái giá nào em từng trả đã dạy em điều đáng tiền? \textit{(What price have you paid that taught you something valuable?)}

2. Bài học ấy có đang thật sự được em sống theo không? \textit{(Are you truly living by that lesson now?)}
\end{tuhocnhanh}

\section{Một người không hơn ai nữa và thấy nhẹ.}
\begin{truyen}{Một người không hơn ai nữa và thấy nhẹ.}{Dropping the race can feel like freedom.}
\chuhoa{S}{au nhiều năm tự ép mình phải nổi hơn, anh thử sống chậm, làm vừa đủ tốt, bớt nhìn sang người khác. Điều lạ là anh không tụt xuống như sợ hãi trước kia; anh chỉ nhẹ đi rất nhiều.}
\textit{(After years of forcing himself to stand out, he tried slowing down, doing his work well enough, and looking less at others. Strangely, he did not fall apart as he had feared; he simply became much lighter.)}
\end{truyen}
\begin{baihoc}
Có những cuộc đua không cần về nhất; chỉ cần dừng lại là đã thắng lại được đời mình.
\textit{(Some races do not need first place; simply stopping them is a way to win your life back.)}
\end{baihoc}
\begin{ghinhoanh}\textbf{Short English to remember:}

Dropping the race can feel like freedom.
\end{ghinhoanh}
\begin{tuhocnhanh}
1. Cuộc đua nào em có thể bước ra mà vẫn sống tốt? \textit{(Which race could you step out of and still live well?)}

2. Em đang mệt vì mục tiêu thật hay vì so sánh? \textit{(Are you tired because of a real goal or because of comparison?)}
\end{tuhocnhanh}

\section{Người già nhìn lại và chỉ tiếc những lần thắng không đáng.}
\begin{truyen}{Người già nhìn lại và chỉ tiếc những lần thắng không đáng.}{Regret often follows the wrong victories.}
\chuhoa{K}{hi đã đi gần hết một đời, ông không tiếc những lần mình bị lép vế. Ông chỉ tiếc những lần mình cố thắng cho bằng được rồi làm tổn thương người khác, làm cứng lòng mình, làm mối quan hệ không quay lại như cũ.}
\textit{(Near the end of life, he did not regret the times he was outdone. He regretted the times he insisted on winning and hurt others, hardened himself, and changed relationships beyond repair.)}
\end{truyen}
\begin{baihoc}
Điều đáng sợ không phải là từng thua vài lần, mà là thắng sai quá nhiều lần.
\textit{(The frightening thing is not having lost a few times, but having won wrongly too many times.)}
\end{baihoc}
\begin{ghinhoanh}\textbf{Short English to remember:}

Regret often follows the wrong victories.
\end{ghinhoanh}
\begin{tuhocnhanh}
1. Có cái thắng nào của em nhìn lại đã không còn đáng tự hào? \textit{(Is there any victory in your life that no longer feels worth being proud of?)}

2. Từ nay, em muốn chọn thắng theo cách nào? \textit{(From now on, how do you want to choose your victories?)}
\end{tuhocnhanh}
