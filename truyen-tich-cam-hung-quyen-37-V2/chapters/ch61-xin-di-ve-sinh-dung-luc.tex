\chapter{Xin Đi Vệ Sinh Đúng Lúc}
\chapterintrobi{Cứ tới đoạn khó là nhu cầu sinh hoạt bỗng trở nên cấp thiết}{Ask to leave at the perfect moment}{Nhiều bạn không hẳn muốn trốn học, chỉ muốn né đúng phút mình sắp bị gọi tên.}

\begin{lucbatbox}{Lục bát: Đúng câu khó thì xin ra}
Đến khi cô gọi gần bàn\
Em nghe nhu cầu bỗng tràn rất nhanh\
Xin ra một lát cho lành\
Vừa hay đoạn khó cũng đành trôi qua\
Bạn nhìn hiểu ý quá mà\
Chỉ thương câu hỏi ở nhà một mình
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Kỹ năng canh giờ}
Câu khó vừa lên, em đứng liền\
Xin ra ngoài chút, mặt còn hiền\
Nếu canh thời điểm cho việc học
Thành tích đâu cần đợi phép tiên
\end{tutuyetbox}

\begin{ghichu}
Xin đi đúng lúc là kỹ năng quan sát tốt. Chỉ tiếc kỹ năng ấy thường được dùng để né hơn là để học.
\end{ghichu}

\begin{englishnote}
\textbf{leave the classroom}: ra ngoài lớp.\par
\textbf{perfect timing}: đúng thời điểm.\par
\textbf{May I go out for a minute?}: Em xin ra ngoài một phút được không ạ?
\end{englishnote}
