\chapter{Đồng Phục Và Nề Nếp}
\chapterintro{Có những bạn mặc đồng phục như đang thương lượng với nội quy}{Nề nếp học đường luôn sinh ra thơ cà khịa rất tự nhiên.}

\begin{lucbatbox}{Lục bát: Áo khoác che lỗi}
Áo em kéo khóa lên cao
Che luôn phù hiệu lẫn bao lỗi lầm
Thầy nhìn chưa kịp hỏi thăm
Em thưa hôm ấy con nhằm tránh gió
Gió thì chắc cũng không to
Chỉ là nội quy đang lo cho mình
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Đồng phục biến tấu}
Đồng phục em sửa cũng khéo tay
Ngắn thêm một chút bó vai ngay
Nội quy nhìn chắc còn phân vân
Chỉ gương vẫn thấy đẹp hôm nay
\end{tutuyetbox}

\begin{ghichu}
Khịa đồng phục không phải khịa chuyện mặc đẹp, mà khịa cái kiểu mỗi sáng đều thử xem nội quy chịu được mình tới đâu.
\end{ghichu}

\ngancach

\begin{lucbatbox}{Lục bát: Phù hiệu đi vắng}
Tên em ghi đủ trong sổ
Phù hiệu hôm ấy lại trốn biệt tăm
Bạn bè hỏi có mang không
Em cười con để trong cơn ngủ chiều
Nếu mà nhớ nó đôi điều
Nề nếp hẳn bớt lênh đênh hơn nhiều
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Dép lạ vào trường}
Hôm nay em bước rất êm rồi
Nhìn chân mới biết giày quên rồi
Dép kia tuy cũng hiền hòa lắm
Chỉ e bảo vệ lắc đầu thôi
\end{tutuyetbox}
