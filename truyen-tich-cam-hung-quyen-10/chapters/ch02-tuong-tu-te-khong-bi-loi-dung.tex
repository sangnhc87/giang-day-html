\section{Lúc Nào Cũng Trực Hộ}
\begin{truyen}{Lúc Nào Cũng Trực Hộ}{Always Covering Other People}
\chuhoa{T}{rong nhom, co ban tre luc nao cung nhan phan truc ho vi so moi quan he xau di.}
\textit{(In a group, one young person keeps taking extra turns because they fear hurting the relationship.)}

Ban ay tuong nhan them nhu vay la song tinh cam va tu te.
\textit{(They think taking more like that is kindness and loyalty.)}

Nhung khi viec chung thanh thoi quen mot nguoi canh va nhieu nguoi dung, long tot da bi dung sai.
\textit{(But when group work becomes one person carrying and many people using, kindness is being misused.)}

Den mot luc, ban ay met va am uc voi chinh su hien lanh cua minh.
\textit{(At some point, the person becomes tired and resentful of their own softness.)}

\end{truyen}

\begin{baihoc}
Tu te khong co nghia la nhan het phan nguoi khac bo lai.
\textit{(Kindness does not mean taking all the work others leave behind.)}
\end{baihoc}

\begin{ghinhoanh}
\textbf{Short English to remember:}

Kindness does not mean taking all the work others leave behind.

\end{ghinhoanh}

Nhung tien bac mo nhat la noi long tot khong co ranh gioi rat de bi thu thach.
\textit{(But money is exactly where kindness without boundaries is tested hardest.)}

Neu em khong hoc cach noi ro, em vua mat tien vua mat luon su ton trong.
\textit{(If you do not learn to speak clearly, you may lose both money and respect.)}

\end{truyen}

\begin{baihoc}
Long tot can di cung quy tac ro rang, nhat la voi tien.
\textit{(Kindness needs clear rules, especially around money.)}
\end{baihoc}

\section{Làm Thêm Phần Việc Người Khác}
\begin{truyen}{Làm Thêm Phần Việc Người Khác}{Doing the Part That Belongs to Someone Else}
\chuhoa{T}{rong van phong, co nhan vien tre thuong xuyen sua bai, boi file, va chua loi cho nguoi khac.}
\textit{(In an office, a young worker often edits files, fills gaps, and repairs mistakes for others.)}

Ban ay tuong minh dang de tap the duoc viec va khong nen tinh toan.
\textit{(They think they are serving the team and should not calculate too much.)}

\textit{(But when rescue becomes constant, other people may grow comfortable living irresponsibly on your effort.)}

\textit{(Support with limits lasts.)}

Gan thay thi lam em can nhanh.
\textit{(Carrying for others drains you fast.)}

\end{truyen}

\begin{baihoc}
Em co the giup, nhung dung bien minh thanh bo phan chong lung mien phi.
\textit{(You can help, but do not turn yourself into free spare support.)}

\section{Nhịn Nhịn Để Hòa Khí}
\begin{truyen}{Nhịn Nhịn Để Hòa Khí}{Staying Silent for the Sake of Peace}
\chuhoa{C}{o ban tre thay minh bi chen viec nhung van cuoi tru vi so bi goi la kho tinh.}
\textit{(A young person sees others pushing work onto them, but still smiles because they fear being called difficult.)}

Ban ay tuong im lang la cach giu hoa khi tot nhat.
\textit{(They think silence is the best way to keep peace.)}

Thuc ra, su im lang qua lau thuong de lai mot cai gia lon hon, do la su am uc va mot thoi quen bi xem nhe.
\textit{(In reality, silence held too long often creates a bigger cost: resentment and a habit of being undervalued.)}

Noi mot cau ro rang som thuong nhe hon de den luc no thanh mot vet thuong lon.
\textit{(One clear sentence early is usually lighter than waiting until it becomes a deep wound.)}

\end{truyen}

\begin{baihoc}
Hoa khi that khong duoc xay tren viec mot nguoi nuot het phan khong cong bang.
\textit{(Real peace is not built by one person swallowing all unfairness.)}
\end{baihoc}

\begin{ghinhoanh}
\textbf{Short English to remember:}

\section{Đêm Nào Cũng Nghe Than}

\begin{truyen}{Lúc Nào Cũng Cứu Hỏa}{Being the Emergency Person All the Time}

\textit{(People with self-respect do not absorb every crisis others create.)}

Duoc can la tot. Bi mac dinh la nguoi chua chay cho tat ca thi khong con tot nua.
\textit{(Being needed is good. Becoming the default firefighter for everyone is not.)}

\section{Đồng Ý Vì Sợ Mất Lòng}

Noi khong dung luc co the giu duoc su ton trong tot hon mot loi dong y mien cuong.
\textit{(A timely no can preserve more respect than a forced yes.)}

\end{truyen}

\begin{baihoc}
Tu choi trong sach con hon nhan roi lam do hoac lam trong am uc.
\textit{(A clean refusal is better than acceptance followed by poor work or resentment.)}

\textit{(Respecting family and respecting yourself are not opposite things.)}

\section{Bị Dẫn Dây Bằng Cảm Giác Tội Lỗi}
\begin{truyen}{Bị Dẫn Dây Bằng Cảm Giác Tội Lỗi}{Being Pulled by Guilt}
\chuhoa{M}{oi lan em tu choi, co nguoi lien noi rang em da thay doi, ich ky, hay khong con tinh cam.}
\textit{(Whenever you refuse, someone says you have changed, become selfish, or lost affection.)}

Em tuong minh co loi that nen quay lai giup du khong con muon.
\textit{(You assume the guilt is real, so you go back and help even when you no longer want to.)}

Do la cach rat nhieu nguoi bi dan day bang cam xuc thay vi duoc ton trong bang ly le ro rang.
\textit{(This is how many people are pulled by emotion instead of being respected through clear reason.)}

Neu em khong nhin ra, cam giac co loi se thanh cai day xich rat mem ma rat chac.
\textit{(If you do not see it, guilt becomes a soft but strong chain.)}

\end{truyen}

\begin{baihoc}
Long tot can tinh tao. Neu khong, em se cho di vi bi day, khong phai vi tu nguyen.
\textit{(Kindness needs awareness. Otherwise you give because you are pushed, not because you choose.)}
\end{baihoc}

\begin{ghinhoanh}
\textbf{Short English to remember:}

Kindness needs awareness.

Otherwise you give because you are pushed, not because you choose.

\section{Học Nói Không}
\begin{truyen}{Học Nói Không}{Learning to Say No}
\chuhoa{S}{au nhieu lan met va am uc, co ban tre tap noi mot cau ngan gon rang minh khong lam duoc viec do.}
\textit{(After many tired and resentful days, a young person starts saying one short sentence: I cannot do that.)}

Ban ay so cau noi do se lam minh xau di trong mat nguoi khac.
\textit{(They fear that sentence will make them look worse in other people’s eyes.)}

Nhung dieu xay ra la nhung ai ton trong minh van o lai, con nhung ai chi quen dung minh thi bat dau kho chiu.
\textit{(What happens is that respectful people stay, while those who only used them become uncomfortable.)}

Cau no ngan, nhung no mo ra mot doi song nhe hon va sach hon.
\textit{(The sentence is short, but it opens a lighter and cleaner life.)}

\end{truyen}

\begin{baihoc}
Noi khong la mot ky nang giu long tot song lau, khong phai mot dau hieu lanh lung.
\textit{(Saying no is a skill that helps kindness survive, not a sign of coldness.)}
\end{baihoc}

\begin{ghinhoanh}
\textbf{Short English to remember:}

Saying no is a skill that helps kindness survive, not a sign of coldness.
