\chapter{Bạn Cùng Bàn Đổi Chỗ}
\chapterintro{Ngồi cạnh nhau thì cấu xé, lúc bị tách ra lại khóc như mưa.}{Bàn cũ người đâu, chỉ thấy điểm số chia lìa.}

\begin{tutuyetbox}{Cuộc chia ly đắng}{The Bitter Farewell}
Thầy bắt dời xa lệ não nề
Đứa trước bàn trên mặt tái tê
Đứt mất phao thi tiêu điểm số
Nhìn bài rớt điểm khóc chua chề.
\tcblower
Teacher moves us, we're drowning in tears 
Facing the front with sudden fears 
Lost my cheat buddy, my grades will slide 
Looking at the fail, I've nowhere to hide.
\end{tutuyetbox}

\begin{tutuyetbox}{Kẻ thù cùng bàn}{The Benchmate Enemy}
Bạn mới qua phân gét đắng đay
Tình nghĩa thờ ơ hất hất mày
Tẩy thước chia ngăn phần rạch rọi
Ngồi chung một tiết bực chua cay.
\tcblower
A new benchmate with a bitter face 
Ignoring me in this tiny space 
Pens and erasers mark the border line 
Sitting together, nothing is fine.
\end{tutuyetbox}

\begin{ghichu}
Tưởng đổi chỗ sẽ học tốt hơn, nhưng hóa ra chỉ là đổi từ "buôn chuyện" sang chiến tranh lạnh.
\end{ghichu}
