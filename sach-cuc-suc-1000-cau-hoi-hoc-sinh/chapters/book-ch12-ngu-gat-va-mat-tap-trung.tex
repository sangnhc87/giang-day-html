\chapter{Ngủ gật, mất tập trung và những pha hồn lìa khỏi tiết học}
\markboth{Ngủ gật, mất tập trung và những pha hồn lìa khỏi tiết học}{Ngủ gật, mất tập trung và những pha hồn lìa khỏi tiết học}

\section{Ngủ Trong Giờ Nhưng Lý Do Rất Điện Ảnh}
\begin{enumerate}[leftmargin=*]
\item \hsqa{Em chỉ nhắm mắt suy nghĩ thôi.}{Suy nghĩ gì mà đầu em rơi theo nhịp vậy?}{I was just closing my eyes to think.}{What kind of thinking makes your head drop in rhythm?}
\item \hsqa{Em ngủ có năm phút à.}{Năm phút đó đi đúng lúc kiến thức ghé lớp.}{I only slept for five minutes.}{Those five minutes happened precisely when knowledge visited.}
\item \hsqa{Tối qua em học khuya nên giờ em đuối.}{Học khuya hay thức khuya để tin mình chăm?}{I stayed up late studying, so I'm exhausted now.}{Studying late, or staying up late to feel hardworking?}
\item \hsqa{Em gục đầu vì đau đầu thật.}{Đau đầu thật thì nghỉ đúng cách, chứ không nghỉ kiểu mất bài.}{I put my head down because I really had a headache.}{If it's real, rest properly. Not in a way that loses the lesson.}
\item \hsqa{Em nghe bằng tiềm thức mà.}{Tiềm thức em nếu làm được bài thì thầy xin lỗi.}{I'm listening subconsciously.}{If your subconscious can solve the exercises, I apologize.}
\item \hsqa{Tiết này buồn ngủ quá thầy.}{Buồn ngủ của em chăm chỉ xuất hiện đúng môn khó ghê.}{This class makes me so sleepy, teacher.}{Your sleepiness appears very faithfully in difficult subjects.}
\item \hsqa{Em ngủ mà vẫn nghe tiếng thầy.}{Nghe tiếng khác với hiểu ý và giữ lại.}{I can sleep and still hear your voice.}{Hearing a sound is not the same as keeping its meaning.}
\item \hsqa{Em thiếu ngủ do bài nhiều quá.}{Bài nhiều một phần, điện thoại chen giờ ngủ mười phần.}{I lack sleep because there is too much homework.}{Homework plays a part. Your phone steals the other ten parts.}
\item \hsqa{Em định chợp mắt tí thôi.}{Định nghĩa của em về 'tí' khá dài bằng nửa tiết.}{I just meant to take a quick nap.}{Your definition of 'quick' lasts about half a period.}
\item \hsqa{Em tỉnh mà mắt em mỏi.}{Mắt mỏi mà tay em ôm điện thoại đêm nào cũng khỏe.}{I'm awake, only my eyes are tired.}{Interesting. Those same eyes survive late-night phone use just fine.}
\end{enumerate}

\section{Mất Tập Trung Xong Đổ Cho Hoàn Cảnh}
\begin{enumerate}[leftmargin=*,resume]
\item \hsqa{Lớp ồn quá em không tập trung nổi.}{Lớp ồn là thật, nhưng ý thức em cũng không nên mỏng vậy.}{The class is too noisy, I can't focus.}{The class is noisy, yes. Your discipline should not be that fragile.}
\item \hsqa{Em ngồi gần cửa sổ nên hay bay não.}{Cửa sổ mở ra ngoài trời, không mở hộ lý do cho em đâu.}{I sit near the window, so my mind drifts.}{The window opens to the world, not to your excuse.}
\item \hsqa{Em bị bạn kế bên làm phân tâm.}{Bạn là cái cớ gần nhất, chưa chắc là gốc sâu nhất.}{My deskmate distracts me.}{Your deskmate is the nearest excuse, not always the deepest cause.}
\item \hsqa{Em tập trung được lúc đầu thôi.}{Vậy em đang luyện sự chú ý hay chỉ thuê nó theo giờ?}{I can only focus at the beginning.}{So are you training attention, or renting it by the hour?}
\item \hsqa{Em nghe một lúc là đầu trôi.}{Đầu em trôi vì nó chưa có mỏ neo nề nếp.}{After a while, my mind just floats away.}{Your mind floats because it lacks an anchor of discipline.}
\item \hsqa{Em hay mơ mộng nên khó học.}{Mơ thì tốt, miễn đừng mơ luôn phần trách nhiệm.}{I daydream a lot, so studying is hard.}{Dreaming is fine, as long as you don't dream your responsibilities away too.}
\item \hsqa{Em đang nghĩ chuyện nhà nên không vô bài.}{Chuyện nhà cần được hiểu, nhưng bài học cũng cần được giữ.}{I'm thinking about family matters, so I can't get into the lesson.}{Family matters deserve care, but the lesson still deserves a place.}
\item \hsqa{Em vô lớp mà đầu em nặng.}{Vậy mình tìm cách nhẹ đầu, không phải nhẹ luôn việc học.}{I enter class with a heavy head.}{Then we find ways to lighten the mind, not lighten study into nothing.}
\item \hsqa{Em không tập trung lâu được, tính em vậy.}{Tính cách không nên bị lôi ra làm ô trú ẩn cho thói quen yếu.}{I just can't focus long. That's my personality.}{Personality should not become a shelter for weak habits.}
\item \hsqa{Em đang cố mà não không hợp tác.}{Não nghe lời người rèn nó đều, không nghe lời hứa bốc đồng.}{I'm trying, but my brain won't cooperate.}{Brains cooperate with routine, not with dramatic promises.}
\end{enumerate}

\section{Chuông Cứu Rỗi Và Những Pha Tỉnh Rất Đúng Lúc}
\begin{enumerate}[leftmargin=*,resume]
\item \hsqa{Nghe trống là em tỉnh liền.}{Tỉnh đúng lúc tan học thì hơi phản giáo dục.}{The moment the bell rings, I wake right up.}{Waking up exactly at dismissal is educationally suspicious.}
\item \hsqa{Ra chơi em sung lại ngay.}{Năng lượng của em rất biết chọn chỗ không cần kiến thức.}{I'm energetic again during break.}{Your energy has excellent instincts for avoiding knowledge.}
\item \hsqa{Sao tới giờ làm bài em mới buồn ngủ.}{Cơ thể em có vẻ trung thành với sự né tránh.}{Why do I only get sleepy when it's time to work?}{Your body seems deeply loyal to avoidance.}
\item \hsqa{Em tỉnh nhất lúc nói chuyện với bạn.}{Chứng tỏ pin em không yếu, chỉ đang dùng sai ứng dụng.}{I'm most alert when talking to friends.}{So your battery is not weak, it's just running the wrong app.}
\item \hsqa{Thầy gọi tên là em tỉnh hẳn.}{Ừ, vì nguy cơ luôn là caffein mạnh nhất.}{When you call my name, I'm instantly awake.}{Yes. Threat is often the strongest caffeine.}
\item \hsqa{Em mà lên bảng là hết ngủ ngay.}{Vậy hóa ra em ngủ được vì cứ nghĩ mình an toàn.}{If you call me to the board, my sleepiness disappears.}{So you only sleep because you keep assuming you're safe.}
\item \hsqa{Em đang dưỡng sức cho phần sau.}{Em dưỡng hơi quá nên phần trước chết oan.}{I'm conserving energy for later.}{You're conserving so much energy that the first half died unfairly.}
\item \hsqa{Em học kiểu nước rút, thầy thông cảm.}{Nước rút hợp chạy đua, không hợp nuôi não dài hạn.}{I study in sprints, teacher. Please understand.}{Sprints suit races, not brains that need long-term training.}
\item \hsqa{Em sẽ bù sau.}{Sau của em lúc nào cũng đông việc hơn hiện tại.}{I'll make up for it later.}{Your 'later' is always more crowded than your present.}
\item \hsqa{Em tiếc vì vừa ngủ lỡ mất đoạn hay.}{Ít ra em còn biết tiếc, giờ biến tiếc thành sửa đi.}{I regret sleeping through the useful part.}{At least you know regret. Now turn regret into correction.}
\end{enumerate}

\section{Muốn Tỉnh Trong Giờ Thì Phải Tử Tế Với Đêm Trước}
\begin{enumerate}[leftmargin=*,resume]
\item \hsqa{Thầy ơi em muốn đỡ gật thì làm sao?}{Bắt đầu bằng đừng ôm điện thoại tới giờ khuya như ôm số mệnh.}{Teacher, how do I stop nodding off?}{Start by not hugging your phone at midnight like destiny itself.}
\item \hsqa{Em ngủ sớm một hôm có ăn thua không?}{Một hôm cứu ít, nhưng một nếp cứu rất nhiều.}{Does sleeping early just once really help?}{One night helps a little. A habit helps a lot.}
\item \hsqa{Em cần động lực để giữ giờ ngủ.}{Không cần động lực nhiều, cần bớt thương cái tiện trước mắt.}{I need motivation to keep a sleep schedule.}{You don't need more motivation. You need less attachment to short-term convenience.}
\item \hsqa{Em sợ ngủ sớm rồi học không kịp.}{Ngủ đàng hoàng để mai đầu em làm việc được mới là khôn.}{I'm afraid if I sleep early I won't finish studying.}{Sleeping properly so your brain works tomorrow is the wiser move.}
\item \hsqa{Em hay hứa mai sẽ khác.}{Mai rất mệt vì phải gánh quá nhiều lời hứa của em rồi.}{I keep telling myself tomorrow will be different.}{Tomorrow is exhausted from carrying so many of your promises already.}
\item \hsqa{Em có cứu nổi nề nếp không thầy?}{Cứu được, miễn em chịu sửa những thứ nhìn nhỏ mà phá rất to.}{Can I still save my routine?}{Yes, if you're willing to fix the small things that cause huge damage.}
\item \hsqa{Em nên nhớ câu gì nhất?}{Đêm sống bừa thì ngày học trả giá.}{What should I remember most?}{When the night is chaotic, the school day pays the bill.}
\item \hsqa{Em cần ai nhắc em ngủ.}{Nhắc ngoài tốt, nhưng nhắc trong mới bền.}{I need someone to remind me to sleep.}{External reminders help. Internal ones last.}
\item \hsqa{Em muốn tỉnh cả người lẫn đầu.}{Vậy sống đừng để lịch nghỉ và lịch học thù nhau nữa.}{I want both my body and mind to feel awake.}{Then stop making your sleep schedule and your school schedule enemies.}
\item \hsqa{Thầy chốt em một câu.}{Muốn tỉnh trong tiết thì phải tử tế với đêm trước.}{Give me one final line.}{If you want to stay awake in class, be decent to the night before.}
\end{enumerate}
