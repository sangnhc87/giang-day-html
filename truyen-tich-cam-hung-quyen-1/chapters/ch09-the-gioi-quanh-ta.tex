\section{Đóa Hoa Của Người Mù}
\begin{truyen}{Đóa Hoa Mù}{Cách Nhìn Đời}
\chuhoa{A}{ } \textbf{blind man} was \textbf{sitting on a park bench}, enjoying the warm sun.\\
\textit{(Một người đàn ông mù lòa đang ngồi trĩu trên một chiếc ghế đá công viên, tận hưởng ánh nắng mặt trời ấm áp.)}

A \textbf{little girl} came up to him and \textbf{handed him a flower}.\\
\textit{(Một bé gái nhỏ xíu sải bước chạy lại gần ông và tự trao cho ông một bông hoa rực rỡ.)}

\textit{"It is a \textbf{beautiful red rose},"} she \textbf{said cheerfully}.\\
\textit{("Đó là một bông hồng đỏ tuyệt đẹp đấy thưa ông," cô bé vui vẻ hào hứng cất lời.)}

The old man \textbf{could not see} the color, but he \textbf{brought it to his nose}.\\
\textit{(Cụ ông thực chất không thể nhìn soi thấy được màu sắc nọ, nhưng ông ngay lập tức cầm đưa nó rúc lên sát mũi mũi mình.)}

He \textbf{inhaled deeply} and the \textbf{sweet scent filled his mind}.\\
\textit{(Ông từ tốn hít trút một hơi thật sâu thẳm, và cái thứ hương thơm ngọt lịm kia nhanh chóng tràn bay ngạt ngập tâm trí ông.)}

\textit{"You are right,"} he \textbf{smiled warmly}, \textit{"I \textbf{cannot see its red color}, but I \textbf{can feel its beautiful soul}."}\\
\textit{("Cháu nói xuất sắc," ông cụ hiền hòa mỉm cười ấm áp, "Ông mù lòa vĩnh viễn chẳng thể dỏng mắt nhìn thấu thấy sắc đỏ của nó, nhưng ngược lại ông xài tim để hân hoan cảm nhận được thần thái trọn vẹn cái linh hồn rực rỡ tuyệt mỹ của nó cháu ạ.")}

The \textbf{little girl clapped} and \textbf{ran away happily}.\\
\textit{(Cô bé ngoan ngoãn vỗ hai tay đập bồm bộp đầy rạng mừng rỡ, cười tung tăng chân sáo và đâm xoài người chạy thoăn thoắt đi dạt chơi đùa tiếp.)}
\end{truyen}

\begin{baihoc}
  \textbf{Beauty is not always seen} with the \textbf{eyes}.\\
  \textit{(Cái định nghĩa vẻ mỹ tuyệt đẹp đâu phải khi nào trớ trêu cũng được con người dán mắt ngắm dòm nhìn thẳng bằng phẳng bởi mỗi đôi mắt. )}\\
  \textbf{Learn to feel the world} with your \textbf{open heart}.\\
  \textit{(Hãy học cách cảm nhận thế giới bằng cả một trái tim rộng mở.)}
\end{baihoc}
