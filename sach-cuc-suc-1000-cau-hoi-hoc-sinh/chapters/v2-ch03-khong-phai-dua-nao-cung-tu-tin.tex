\chapter{Không phải đứa nào cũng tự tin sẵn}
\markboth{Không phải đứa nào cũng tự tin sẵn}{Không phải đứa nào cũng tự tin sẵn}

\section{Em không ngu, em chỉ vô bài chậm}
\begin{truyen}{Em không ngu, em chỉ vô bài chậm}{Classroom Talk}
\chuhoa{M}{ột bạn nói thẳng: ``Thầy đừng nhìn em kiểu bất lực. Em không ngu. Em chỉ vô bài chậm hơn tụi kia thôi.''}

Thầy đáp ngay: ``Nói vậy là đúng hơn nhiều đứa chỉ biết tự chửi mình. Vào chậm không đáng sợ bằng tưởng mình hết cứu.''

Bạn ấy hỏi: ``Nhưng chậm riết em cũng nản.''

Thầy nói: ``Vậy mình tìm cách học hợp với nhịp của em, chứ không lấy nhịp người khác đập lên đầu em nữa.''

\textit{(A slower learner does not need insults. They need a method that matches their pace and keeps them moving.)}
\end{truyen}

\begin{baihoc}
Tự nhận mình chậm còn sửa được. Tự kết án mình ngu là đang tự khóa cửa.
\textit{(Admitting you are slower can be worked with. Labeling yourself stupid shuts the door.)}
\end{baihoc}

\begin{ghinhoanh}
\textbf{Short English to remember:}

Slow is not hopeless.

Wrong labels damage real progress.
\end{ghinhoanh}

\begin{tuhocnhanh}
1. Em đang thiếu khả năng hay thiếu cách học hợp mình? \textit{(Are you lacking ability, or the right learning method?)}

2. Nhịp học nào khiến em đỡ bị ngợp nhất? \textit{(What learning pace helps you feel less overwhelmed?)}
\end{tuhocnhanh}

\section{Bạn kia có nền sẵn rồi, sao so với em được}
\begin{truyen}{Bạn kia có nền sẵn rồi, sao so với em được}{Classroom Talk}
\chuhoa{C}{ó bạn cằn nhằn: ``Bạn đó học thêm từ năm nào rồi, về nhà còn có người kèm. Còn em tự bơi. Sao cứ lôi em ra so?''}

Thầy gật đầu: ``Em nói đúng một nửa. Xuất phát điểm khác nhau thật. Nhưng nếu cứ giữ nửa đúng đó để khỏi phải cố, em sẽ tự thua luôn nửa còn lại.''

Bạn hỏi: ``Nửa còn lại là gì?''

Thầy đáp: ``Là dù chậm hơn, em vẫn phải chạy phần đường của em. Không ai chạy hộ.''

\textit{(Different starting points are real, but they cannot become an excuse to stop moving.)}
\end{truyen}

\begin{baihoc}
Đừng phủ nhận bất công. Nhưng cũng đừng dùng bất công làm cớ để đứng yên.
\textit{(Do not deny unfairness, but do not let it become a reason to remain still.)}
\end{baihoc}

\begin{ghinhoanh}
\textbf{Short English to remember:}

Different starts are real.

Your path is still yours to run.
\end{ghinhoanh}

\begin{tuhocnhanh}
1. Em hay than về xuất phát điểm đến mức nào? \textit{(How often do you dwell on your starting point?)}

2. Phần việc nào vẫn là trách nhiệm riêng của em? \textit{(What part of the work is still your responsibility?)}
\end{tuhocnhanh}

\section{Em cứ bị đơ khi lên bảng}
\begin{truyen}{Em cứ bị đơ khi lên bảng}{Classroom Talk}
\chuhoa{M}{ột bạn thú nhận: ``Ngồi dưới em còn nghĩ được. Lên bảng là đầu em trắng trơn. Nhìn cả lớp là em đơ.''}

Thầy hỏi: ``Em sợ sai hay sợ bị nhìn thấy mình sai?''

Bạn ấy cười gượng: ``Chắc sợ cái thứ hai nhiều hơn.''

Thầy nói: ``Vậy vấn đề không chỉ là kiến thức. Em còn đang bị sĩ diện bóp cổ. Mình phải tập chịu quê một chút mới khá lên được.''

\textit{(Freezing in public is often not just about knowledge. Pride and fear of embarrassment can paralyze performance.)}
\end{truyen}

\begin{baihoc}
Muốn đỡ run, không chỉ học bài. Còn phải học chịu cảm giác mình chưa giỏi trước mặt người khác.
\textit{(To become less anxious, you must train not only knowledge but also your tolerance for visible imperfection.)}
\end{baihoc}

\begin{ghinhoanh}
\textbf{Short English to remember:}

Fear of looking foolish is powerful.

Growth needs some public discomfort.
\end{ghinhoanh}

\begin{tuhocnhanh}
1. Em sợ sai hay sợ mất mặt hơn? \textit{(Do you fear being wrong, or losing face more?)}

2. Em có thể tập nói trước một nhóm nhỏ trước không? \textit{(Can you practice speaking before a smaller group first?)}
\end{tuhocnhanh}

\section{Cứ bị nói lười riết em cũng chán cố}
\begin{truyen}{Cứ bị nói lười riết em cũng chán cố}{Classroom Talk}
\chuhoa{C}{ó bạn buông một câu rất mệt: ``Em biết em chưa tốt. Nhưng nghe bị gắn mác lười suốt ngày, riết em cũng muốn mặc kệ luôn.''}

Thầy ngừng một lúc rồi nói: ``Người lớn nhiều khi nóng miệng. Nhưng em cũng phải tự hỏi: mình đang bị oan hoàn toàn, hay mình có phần để người ta nói?''

Bạn ấy đáp: ``Có phần thật. Nhưng nghe hoài khó chịu lắm.''

Thầy gật: ``Khó chịu là thật. Nên mình cần hai việc: bớt cái khiến người ta nói, và đừng lấy lời nói đó làm lý do hỏng tiếp.''

\textit{(Repeated criticism can discourage, but letting it define you only deepens the problem.)}
\end{truyen}

\begin{baihoc}
Không phải lời chê nào cũng đúng. Nhưng nếu lời chê làm mình buông luôn, phần thiệt lại về mình.
\textit{(Not every criticism is fair, but if it makes you quit, you still lose.)}
\end{baihoc}

\begin{ghinhoanh}
\textbf{Short English to remember:}

Labels hurt.

Do not let them become your excuse.
\end{ghinhoanh}

\begin{tuhocnhanh}
1. Em bị dán nhãn gì nhiều nhất? \textit{(What label do people place on you most often?)}

2. Em cần sửa điều gì thật sự để tự gỡ nhãn đó? \textit{(What real change would help remove that label?)}
\end{tuhocnhanh}
