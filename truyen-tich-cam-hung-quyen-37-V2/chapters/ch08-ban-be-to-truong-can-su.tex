\chapter{Bạn Bè, Tổ Trưởng, Cán Sự}
\chapterintro{Một lớp học là cả một xã hội thu nhỏ}{Ai làm cán sự lâu rồi cũng có vài nếp nhăn trong giọng nói.}

\begin{lucbatbox}{Lục bát: Tổ trưởng mòn lời}
Bạn kia làm tổ trưởng thôi
Mà như gánh cả một trời việc chung
Nhắc bài nhắc vở nhắc luôn
Nhắc xong lớp vẫn ung dung như thường
Cuối ngày thở nhẹ bên tường
Làm cán sự lớp cũng thương ghê người
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Bạn tốt nhưng hay tố}
Bạn em rất tốt mọi đường thôi
Chỉ khi ghi sổ lại nghiêm rồi
Đi trễ quên vở hay ngủ gật
Tên ai cũng dễ bước cùng thôi
\end{tutuyetbox}

\begin{ghichu}
Cán sự lớp khổ nhất là lúc phải đứng giữa tình bạn và trách nhiệm, mà bên nào cũng nhìn mình khó chịu.
\end{ghichu}

\ngancach

\begin{lucbatbox}{Lục bát: Bạn cùng bàn cứu nguy}
Hôm nay em lỡ quên bài
Bạn bên khều nhẹ chỉ vài ý thôi
Ơn này em nhớ suốt đời
Tất nhiên điều nhớ nhất thời là ơn
Chỉ mong mai bớt dại khờ
Đừng để người khác cứu hoài phần ta
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Cả lớp rất đoàn kết}
Khi vui thì lớp thương nhau lắm
Khi bị kiểm tra mặt rất co
Đứa nào chưa thuộc đều nhìn bạn
Tưởng như chung một nhịp tim lo
\end{tutuyetbox}
