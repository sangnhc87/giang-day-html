
\chapter{Huyền Thoại Những Giờ Học}
\chapterintro{Lớp học là nơi hội tụ của những thiên tài ngủ gật và các nhà bình luận học.}{Học là chuyện cả đời, nhưng ngủ là chuyện không thể đợi đến ngày mai.}

\begin{lucbatbox}{Lục bát: Họa hổ tàng long}
Thân ngồi tại lớp thảnh thơi
Hồn bay tận cuối khung trời mộng mơ
Bảng đen bụi phấn hững hờ
Em đang bận dệt giấc mơ huy hoàng
Tỉnh ra điểm kém ngàng hàng
Mới hay chữ nghĩa sang trang mất rồi.
\end{lucbatbox}
\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Họa hổ tàng long}
Hổ ngự bàn sau mộng chẳng rời
Mặc thầy giảng giải phí bao lời
Bảng vàng dệt mộng đầy vinh hiển
Tỉnh giấc nhìn đề lệ mãi rơi.
\end{tutuyetbox}
\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Bí kíp ngủ thiền}
Lời giảng thầy vang tựa kinh thiền
Hổ con say giấc ngủ triền miên
Bảng đen bụi phấn sầu nghiêng bóng
Mới biết tương lai khó vẹn nguyên.
\end{tutuyetbox}

\begin{lucbatbox}{Lục bát: Triết gia bàn cuối}
Thầy đang giảng giải nhiệt tình
Dưới này em mải luận bình nhân sinh
Miệng mồm liến thoắng linh tinh
Ngỡ như cả lớp của mình em thôi
Mai này con chữ phai phôi
Tiếc công đèn sách một thời lãng quên.
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Kẻ lười chép bài}
Bút nằm im lặng giữa trang tờ
Chủ mải nhìn quanh mặt thẫn thờ
Sách trắng tinh khôi như tờ lịch
Học hành thế ấy thực như mơ.
\end{tutuyetbox}

\chapter{Hội Những Người ``Mất Tích''}
\chapterintro{Cái điện thoại là vật bất ly thân, còn cuốn sách là vật... bất khả xâm phạm.}{Thức đêm không phải để học, mà để xem thế gian đang biến động ra sao.}

\begin{lucbatbox}{Lục bát: Nỗi khổ cầm điện thoại}
Học bài em bảo đau đầu
Cầm phone lướt suốt đêm thâu chẳng rời
Mắt hoa nhưng vẫn ham chơi
Lên trường lại ngủ kệ đời trôi ngang
Thầy nhìn vừa giận vừa thương
Cái bằng tốt nghiệp bên đường vẫy tay.
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Nghệ thuật đi muộn}
Lần nào đến muộn cũng thanh minh
Sợ trễ sao không bớt ngủ kinh
Giải thích nghìn câu nghe phát mệt
Thôi đừng diễn kịch nữa u mình.
\end{tutuyetbox}

\ngancach

\begin{lucbatbox}{Lục bát: Người khách vãng lai}
Thân đi tới lớp thảnh thơi
Sách bài chẳng có như người dạo chơi
Cặp kia nhẹ hẫng bên vai
Mặc thầy đứng đó thở dài nhìn em
Tương lai rồi sẽ tối đen
Nếu không chịu học, ai khen cho cùng.
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Quên vở kinh điển}
Sách vở đem quên ở xó nhà
Tới trường tay trắng tựa hào ca
Thầy hỏi bài cũ cười tươi rói
Kiến thức trong đầu bỗng hóa xa.
\end{tutuyetbox}

\begin{ghichu}
Các em có thể lừa thầy bằng những lý do rất thuyết phục, nhưng không thể lừa được tờ giấy thi khi nó trắng tinh như tờ lịch mới.
\end{ghichu}
