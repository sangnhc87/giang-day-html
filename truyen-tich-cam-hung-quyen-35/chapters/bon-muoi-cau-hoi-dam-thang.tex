\chapter*{40 Câu Hỏi Đâm Thẳng}
\addcontentsline{toc}{chapter}{40 Câu Hỏi Đâm Thẳng}
\markboth{40 Câu Hỏi Đâm Thẳng}{40 Câu Hỏi Đâm Thẳng}

\begin{tcolorbox}[colback=bookbg,colframe=booksecondary,arc=4pt,boxrule=0.8pt,enhanced,breakable]
\begin{enumerate}[leftmargin=0.9cm]
\item Em có đang sợ im lặng hơn sợ lãng phí đời mình không?
\item Bữa ăn nào gần nhất em đã để màn hình ngồi vào giữa bàn?
\item Em cầm điện thoại vì cần, hay vì tay em đã thành nô lệ của phản xạ?
\item Em có đang lấy vài phút học thật để bào chữa cho hàng giờ lướt rác không?
\item Đầu óc em còn chịu nổi một trang sách dài chưa?
\item Em còn nhớ cảm giác đi bộ mà đầu óc không bị kéo đi đâu không?
\item Có người nào đang đau vì em luôn liếc màn hình khi họ nói chuyện không?
\item Em có đang gọi mệt là áp lực trong khi một phần mệt đó do chính mình nuôi ra?
\item Điện thoại đang giúp em mạnh hơn ở đâu?
\item Điện thoại đang làm em yếu đi ở đâu?
\item Em đang tra cứu để hiểu hơn, hay để khỏi phải nhớ?
\item Em còn giữ được bao nhiêu tri thức khi không có mạng?
\item Em có đang kết nối nhiều nhưng gặp thật quá ít không?
\item Người thân nào ngồi cạnh em nhưng lâu rồi em chưa nhìn họ trọn vẹn?
\item Em đã từng xem điện thoại khi chạy xe chưa?
\item Nếu có, em lấy quyền gì để đánh cược mạng người khác?
\item Em có đang chán đến mức nào cũng phải lấp bằng nội dung không?
\item Một ngày của em có bao nhiêu khoảng trống thật sự?
\item Em có đang dùng điện thoại trong toilet chỉ vì không chịu nổi việc không kích thích không?
\item Đêm của em mất vì bài vở thật, hay vì ánh sáng xanh?
\item Trong lớp, mắt em ở trên bảng hay dưới ngăn bàn?
\item Em có đang ngồi học hay chỉ đang đóng vai người học?
\item Nhà em có quy tắc điện thoại công bằng cho mọi người không?
\item Người lớn quanh em đang dạy bằng lời hay bằng nếp sống?
\item Em có đang tha thứ cho mình quá dễ chỉ vì ai cũng đang dùng điện thoại như thế?
\item Em đã bao giờ để một thông báo thắng một con người trước mặt mình chưa?
\item Em có còn chịu được việc đợi mà không mở máy không?
\item Em đang sống cùng công nghệ hay bị nó cưỡi lên đầu?
\item Nếu cắt mạng một ngày, em mất công cụ hay mất luôn bản thân?
\item Em có đang biến mọi ký ức thành nội dung đăng tải không?
\item Em đang cần thông tin, hay đang nghiện cảm giác được kéo đi liên tục?
\item Cái gì trong đời em đang bị điện thoại ăn mòn rõ nhất: học, ngủ, lễ phép, an toàn, hay tình thân?
\item Người nào em nợ một cuộc nói chuyện không có màn hình xen vào?
\item Một đứa trẻ nhìn cách em dùng điện thoại sẽ học được bài gì?
\item Nếu em là học sinh của chính mình, em có tin lời em đang dạy không?
\item Em có đủ dũng khí đặt ra một quy tắc nhỏ rồi giữ nó thật không?
\item Em muốn thắng màn hình ở chỗ nào trước tiên?
\item Em có đang bận thật, hay chỉ đang phân tán thật?
\item Em muốn đời mình được nhớ bằng những cuộc gặp thật, hay bằng lịch sử lướt vô tận?
\item Nếu không sửa từ bây giờ, năm năm nữa em nghĩ cái giá sẽ nhẹ đi hay nặng thêm?
\end{enumerate}
\end{tcolorbox}
