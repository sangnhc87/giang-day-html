%========== PHẦN I: CẬU GHẸO TỚ ==========
\part{Cậu Ghẹo Tớ}

\section*{Bài 1: Qua Cửa Lớp}

Cậu qua cửa lớp nghiêng trời nắng,
Tớ chép bài thôi vẫn ngỡ ngàng.
Mực tím vô tình sa trật nhịp,
Viết nhầm một chữ ở tên nàng.

\vspace{1.5cm}\noindent
\textit{(Một thoáng đi ngang là đủ để chữ nghĩa mất trật tự.)}

\vspace{2cm}

\section*{Bài 2: Áo Trắng}

Áo cậu hong mùi nắng,
Gió cứ níu tà bay.
Tớ nhìn mà quên sách,
Thầy gọi mới giật mình.

\vspace{1.5cm}\noindent
\textit{(Có những hôm áo trắng còn sáng hơn cả bảng phấn.)}

\vspace{2cm}

\section*{Bài 3: Cây Bút}

Cậu mượn tớ cây bút,
Rồi quên trả nguyên ngày.
Tớ đâu cần lấy lại,
Chỉ cần cậu nhớ tên.

\vspace{1.5cm}\noindent
\textit{(Thứ giữ lại không phải cây bút, mà là cái cớ.)}

\vspace{2cm}

\section*{Bài 4: Cuối Hành Lang}

Cuối hành lang gió lộng,
Cậu đứng buộc dây giày.
Tớ đi qua làm ngơ,
Tim thì quay lại trước.

\vspace{1.5cm}\noindent
\textit{(Làm ngơ ngoài mặt là chuyện thường; quay lại trong lòng mới khó.)}

\newpage

%========== PHẦN II: TỚ GHẸO CẬU ==========
\part{Tớ Ghẹo Cậu}

\section*{Bài 5: Chép Giùm}

Tớ bảo cậu chép hộ,
Có ba dòng Văn thôi.
Cậu ngồi nắn từng chữ,
Nghiêm như viết thư tình.

\vspace{1.5cm}\noindent
\textit{(Có người càng tử tế càng dễ bị ghẹo.)}

\vspace{2cm}

\section*{Bài 6: Cậu Màu Xanh}

Cậu mặc áo xanh nhạt,
Nhìn hiền như chiều mây.
Tớ ghẹo: ``Ai mà thích?''
Cậu cười: ``Chắc là cậu.''

\vspace{1.5cm}\noindent
\textit{(Một câu trả lời thật thà đôi khi làm người ghẹo đứng hình.)}

\vspace{2cm}

\section*{Bài 7: Sân Bóng}

Cậu từ sân bóng lại,
Trán còn dính mồ hôi.
Tớ đưa chai nước mát,
Rồi quay đi đỏ tai.

\vspace{1.5cm}\noindent
\textit{(Ghẹo là một chuyện; quan tâm lại là chuyện khác.)}

\vspace{2cm}

\section*{Bài 8: Bài Toán}

Cậu giỏi riêng môn Toán,
Tớ giỏi khoản nhìn cậu.
Đề hôm nay nhiều ẩn,
Mà tớ chỉ thấy cười.

\vspace{1.5cm}\noindent
\textit{(Có những bài không giải bằng công thức.)}

\newpage

%========== PHẦN III: TỨC CẢNH SINH TÌNH ==========
\part{Tức Cảnh Sinh Tình}

\section*{Bài 9: Mưa Sân Trường}

Mưa nghiêng qua sân gạch,
Cậu nép dưới hiên dài.
Tớ chìa chung nửa ô,
Thấy trời thương tụi mình.

\vspace{1.5cm}\noindent
\textit{(Mưa có khi chỉ là cái cớ để hai người đứng gần.)}

\vspace{2cm}

\section*{Bài 10: Hoa Phượng}

Phượng rơi lên vai áo,
Đỏ một góc trời trưa.
Cậu phủi hoài không xuống,
Tớ nhìn hoài không thôi.

\vspace{1.5cm}\noindent
\textit{(Có thứ đỏ vì hoa, có thứ đỏ vì tim.)}

\vspace{2cm}

\section*{Bài 11: Gió Tháng Ba}

Gió tháng ba lật vở,
Lật luôn cả mắt nhìn.
Cậu đưa tay giữ hộ,
Tớ giữ một buổi chiều.

\vspace{1.5cm}\noindent
\textit{(Đôi khi một bàn tay giữ trang giấy, mà lòng giữ luôn người.)}

\vspace{2cm}

\section*{Bài 12: Xe Đạp}

Xe cậu qua cổng nhỏ,
Chuông kêu một tiếng trong.
Tớ nghe như ai gọi,
Tên mình ở giữa chiều.

\vspace{1.5cm}\noindent
\textit{(Tiếng chuông xe đạp đôi khi vang hơn cả lời chào.)}

\newpage

%========== PHẦN IV: THẢ THÍNH NHẸ ==========
\part{Thả Thính Nhẹ}

\section*{Bài 13: Hỏi Bâng Quơ}

Tớ hỏi cậu vu vơ:
``Mai tan học ai đón?''
Cậu nhìn trời rồi bảo:
``Nếu cậu rảnh, thì cậu.''

\vspace{1.5cm}\noindent
\textit{(Thả một câu nhẹ, dính nguyên buổi tối.)}

\vspace{2cm}

\section*{Bài 14: Không Cố Ý}

Cậu bảo là không cố,
Mà cứ ngồi cạnh tớ.
Không cố gì lạ vậy,
Ngày nào cũng trùng ghế.

\vspace{1.5cm}\noindent
\textit{(Sự trùng hợp lặp nhiều lần thường không còn là trùng hợp.)}

\vspace{2cm}

\section*{Bài 15: Trà Sữa}

Cậu mua hai ống hút,
Bảo lỡ lấy dư rồi.
Tớ cầm ly mà biết,
Dư gì, toàn cố ý.

\vspace{1.5cm}\noindent
\textit{(Người khéo thả thính thường giả vờ vụng.)}

\vspace{2cm}

\section*{Bài 16: Tin Nhắn Đêm}

Mười một giờ còn hỏi:
``Ngủ chưa, học tới đâu?''
Tớ cười trong chăn mỏng,
Biết cậu nhớ hơn bài.

\vspace{1.5cm}\noindent
\textit{(Có những tin nhắn học bài rất ít học bài.)}

\newpage

%========== PHẦN V: NAM THÍCH NỮ ==========
\part{Nam Thích Nữ}

\section*{Bài 17: Bạn Nữ Bàn Trên}

Em ngồi bàn trên nhé,
Anh ngồi bàn sau thôi.
Ngày nào anh mở vở,
Mà nhớ tóc em thôi.

\vspace{1.5cm}\noindent
\textit{(Khoảng cách từ bàn dưới lên bàn trên là một loại thương nhớ.)}

\vspace{2cm}

\section*{Bài 18: Khi Em Cười}

Em cười cùng lũ bạn,
Anh giả bộ nhìn mây.
Thật ra anh đang đếm,
Mấy cái lúm trên môi.

\vspace{1.5cm}\noindent
\textit{(Con trai hay giả vờ bình tĩnh vào đúng lúc rối nhất.)}

\vspace{2cm}

\section*{Bài 19: Mượn Vở}

Anh mượn em quyển vở,
Đâu phải vì quên bài.
Chỉ vì trong trang cuối,
Có mùi tay em cầm.

\vspace{1.5cm}\noindent
\textit{(Yêu thầm thường bắt đầu bằng những lý do rất không học thuật.)}

\vspace{2cm}

\section*{Bài 20: Cổng Trường}

Anh đứng chờ cổng trường,
Nói là đợi bạn đến.
Bạn nào thì anh rõ,
Chỉ là chưa dám nhận.

\vspace{1.5cm}\noindent
\textit{(Cái khó của con trai không phải thích, mà là nhận.)}

\newpage

%========== PHẦN VI: NỮ THÍCH NAM ==========
\part{Nữ Thích Nam}

\section*{Bài 21: Cậu Ở Tổ Bốn}

Cậu ngồi bên tổ bốn,
Tớ ở tận tổ hai.
Thế mà hễ tan học,
Mắt lại gặp đúng người.

\vspace{1.5cm}\noindent
\textit{(Con gái thích ai thường rất biết cách giả bộ vô tình.)}

\vspace{2cm}

\section*{Bài 22: Đồng Phục Thứ Hai}

Thứ hai cậu chỉnh tề,
Cổ áo thẳng như gió.
Tớ chê trông nghiêm quá,
Rồi nhớ suốt cả tuần.

\vspace{1.5cm}\noindent
\textit{(Miệng chê đôi khi chỉ để lòng nhớ được hợp lý.)}

\vspace{2cm}

\section*{Bài 23: Bài Hát Giờ Ra Chơi}

Loa trường vang bài cũ,
Cậu gõ nhịp lên bàn.
Tớ ngồi im ngoài cửa,
Mà vui như trúng thưởng.

\vspace{1.5cm}\noindent
\textit{(Thích một người là thấy họ làm gì cũng có nhịp.)}

\vspace{2cm}

\section*{Bài 24: Tớ Cũng Thích}

Cậu ghẹo tớ một chút,
Tớ ghẹo lại một câu.
Sân trường nhiều cây lá,
Mà lòng xanh nhất cậu.

\vspace{1.5cm}\noindent
\textit{(Đến cuối cùng, ghẹo qua ghẹo lại cũng chỉ vì thích.)}

\begin{center}
\Large\bfseries\color{edublue}
LỜI KẾT
\end{center}

\vspace{2cm}

\noindent
\itshape
Những bài trong quyển này không cố làm lớn chuyện tình yêu học trò.

Nó chỉ giữ lại những cái rất nhỏ: một ánh nhìn, một câu ghẹo, một nửa chiếc ô, một tiếng chuông xe, một cánh phượng rơi đúng vai áo.

Thích ai đó ở tuổi học trò nhiều khi không cần bi kịch, cũng chưa chắc cần lời hứa. Chỉ cần đọc lên nghe vui, thấy lòng sáng hơn, là đủ.

Nếu dùng những bài này để đọc cho học sinh nghe, mong các em nhận ra rằng cảm mến ai đó là điều tự nhiên; điều đẹp nhất vẫn là sự hồn nhiên, tử tế và chừng mực.

Chúc những rung động đi qua sân trường luôn trong trẻo như nắng đầu giờ.
