\chapter{Góc Lớp, Hành Lang, Mùa Nắng}
\chapterintro{Chủ đề đời thường}{Chương này giữ những tình huống rất lớp học: học nhóm, trực nhật, dã ngoại, lớp online, sân bóng, báo tường. Chất ghẹo nhau tăng hơn một chút nhưng vẫn nhẹ và có duyên.}

\poem{Bài 21}{Ngũ ngôn tứ tuyệt}{Em ngang qua hành lang}{Gió đi theo tà áo}{Tôi đi sau giả tỉnh}{Thật ra lạc vì sao}
\poem{Bài 22}{Ngũ ngôn tứ tuyệt}{Mưa treo ngoài cửa lớp}{Tôi treo mắt ngoài mưa}{Em treo ô bên ghế}{Treo luôn nhớ buổi trưa}
\poem{Bài 23}{Ngũ ngôn tứ tuyệt}{Học nhóm nhà lớp trưởng}{Em đến muộn chút thôi}{Tôi mừng như trúng thưởng}{Vì được ngồi cạnh thôi}
\poem{Bài 24}{Ngũ ngôn tứ tuyệt}{Tiết Anh cô gọi đọc}{Em đọc sáng như chuông}{Tôi lên quên mất chữ}{Chỉ nhớ má em hồng}
\poem{Bài 25}{Ngũ ngôn tứ tuyệt}{Xe tôi non hơi bánh}{Em đứng đợi đầu đường}{Tôi bơm mà tay vụng}{Chỉ mong em đừng chuồn}
\poem{Bài 26}{Ngũ ngôn tứ tuyệt}{Sân bóng chiều đầy nắng}{Em ngồi buộc dây giày}{Tôi ném trật ba quả}{Bạn bảo bị say ai}
\poem{Bài 27}{Ngũ ngôn tứ tuyệt}{Báo tường vừa dán xong}{Em ghim bài thơ nhỏ}{Tôi ghim thêm tên giả}{Cả lớp đọc ra to}
\poem{Bài 28}{Ngũ ngôn tứ tuyệt}{Trại xuân đêm lửa đỏ}{Em chuyền cốc trà gừng}{Tôi cầm chưa kịp uống}{Đã ấm tới lưng chừng}
\poem{Bài 29}{Ngũ ngôn tứ tuyệt}{Giờ hóa mùi cồn nhẹ}{Em mang kính nhìn ngầu}{Tôi nhìn quên thí nghiệm}{Chỉ còn nhớ đôi mày}
\poem{Bài 30}{Ngũ ngôn tứ tuyệt}{Ngày nhận phần thưởng tốt}{Em bước lên đầu hàng}{Tôi vỗ tay rất mạnh}{Mạnh hơn cả trống trường}
\poem{Bài 31}{Ngũ ngôn tứ tuyệt}{Tờ lịch treo góc lớp}{Tôi gạch mỗi giờ trôi}{Không mong nhanh nghỉ lễ}{Chỉ mong tan học thôi}
\poem{Bài 32}{Ngũ ngôn tứ tuyệt}{Sân khấu lên đèn mới}{Em hát giọng như tơ}{Tôi quên luôn vỗ nhịp}{Chỉ nghe tim lơ mơ}
\poem{Bài 33}{Ngũ ngôn tứ tuyệt}{Dã ngoại xe lên dốc}{Em ngồi cạnh cửa sổ}{Tôi chuyền gói bim bim}{Và được cười thật to}
\poem{Bài 34}{Ngũ ngôn tứ tuyệt}{Giờ trực nhật cuối}{Em xách chậu nước trong}{Tôi xin phần quét lớp}{Để gần em thêm vòng}
\poem{Bài 35}{Ngũ ngôn tứ tuyệt}{Lớp online mạng lag}{Em hiện rồi em mất}{Tôi vừa kịp vui chút}{Màn hình đã biết trêu}
\poem{Bài 36}{Ngũ ngôn tứ tuyệt}{Hành lang đầy hoa giấy}{Em đứng giữa nắng hanh}{Tôi giơ máy chụp ảnh}{Tay run hơn lá cành}
\poem{Bài 37}{Ngũ ngôn tứ tuyệt}{Đề cương dày như núi}{Em bảo chia từng đêm}{Tôi nghe mà thấy núi}{Cũng mềm như tên em}
\poem{Bài 38}{Ngũ ngôn tứ tuyệt}{Tháng ba ve mới tập}{Em ghim chiếc kẹp hồng}{Tôi chép sai bài tập}{Vì mắt cứ sang hồng}
\poem{Bài 39}{Ngũ ngôn tứ tuyệt}{Cuối buổi học thêm muộn}{Em nói chuyện bên đường}{Tôi ước con đường ấy}{Dài thêm vài phố phường}
\poem{Bài 40}{Ngũ ngôn tứ tuyệt}{Ve vừa rung sân vắng}{Em bước ngang hiên gầy}{Tôi mang bao bài kiểm}{Tự dưng thấy nhẹ tay}
