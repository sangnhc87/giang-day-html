\chapter{Nói Nhiều Xuyên Tiết}
\chapterintro{Có những bạn xem mỗi tiết học là một cơ hội kéo dài câu chuyện hôm qua}{Nói nhiều không xấu, nói đúng lúc mới là chuyện khó.}

\begin{lucbatbox}{Lục bát: Chuyện không bao giờ hết}
Vừa vào em đã mở lời\
Từ chuyện sáng sớm tới trời chiều qua\
Thầy cô giảng tới đâu xa\
Bạn bên vẫn phải nghe ta đủ điều\
Nếu chia bớt miệng đôi chiều\
Chắc phần bài học cũng nhiều hơn nay
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Tám suốt một giờ}
Em có tài nói rất bền\
Một giờ chưa đủ cho nên dứt lời\
Chỉ mong lúc chuyển sang bài\
Năng lượng ấy cũng thành tài học thêm
\end{tutuyetbox}

\begin{ghichu}
Nói nhiều xuyên tiết thường không bắt đầu bằng ý xấu. Nó bắt đầu bằng một câu nhỏ rồi kéo theo cả chuỗi mất tập trung.
\end{ghichu}

\ngancach

\begin{lucbatbox}{Lục bát: Bạn bên chịu trận}
Bạn bên nghe mãi thành quen\
Mặt cười nhưng mắt đã đèn kéo quân\
Thỉnh thoảng còn phải gật gù\
Như đang tham dự một khu vui buồn\
Chỉ thương bài học trên đường\
Đi ngang qua bạn mà thường bị quên
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Im thì không nỡ}
Em bảo im cũng khó ghê\
Ý nào trong bụng muốn về thành câu\
Nếu mà biết giữ nhịp đầu\
Tiết kia vừa học vừa sâu vẫn vui
\end{tutuyetbox}
