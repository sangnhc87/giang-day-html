\section{Trông Rất Ổn, Bên Trong Rất Lệch}
\begin{truyen}{Trông Rất Ổn, Bên Trong Rất Lệch}{Looking Fine While Falling Apart Inside}
\chuhoa{C}{o nguoi tre di dau cung xuat hien gon gang, noi chuyen binh tinh, va de lai mot hinh anh rat on.}
\textit{(Some young people always appear neat, calm, and stable.)}

Ai nhin vao cung tuong cuoc song cua ho dang vao nep rat dep.
\textit{(Everyone assumes their life is moving in a good order.)}

Nhung ben duoi la nhieu dem mat ngu, nhieu noi so, va nhieu dieu khong dam noi that.
\textit{(But underneath are sleepless nights, hidden fears, and truths they do not dare say.)}

Vo boc cang dep, nguoi ta cang kho thua nhan minh dang can duoc giup.
\textit{(The prettier the surface, the harder it becomes to admit the need for help.)}

\end{truyen}

\begin{baihoc}
Vo boc dep khong tu dong la suc manh. Co khi no chi la lop son che di mot noi met.
\textit{(A polished surface is not automatically strength. Sometimes it is only paint over exhaustion.)}
\end{baihoc}

\begin{ghinhoanh}
\textbf{Short English to remember:}

A polished surface is not automatically strength.

Sometimes it is only paint over exhaustion.

\end{ghinhoanh}

\begin{tuhocnhanh}
\textbf{Câu hỏi tự học / Self-study questions}

\end{tuhocnhanh}

not from always looking occupied.

\section{Hạ Vỏ Bọc Xuống}
\begin{truyen}{Hạ Vỏ Bọc Xuống}{Putting the Surface Down}
\chuhoa{S}{au nhieu va cham, co nguoi tre bat dau bot co trong on moi luc.}
\textit{(After many collisions, a young person stops trying to look fine all the time.)}

Ho noi that hon ve kha nang cua minh, ve tinh trang cua minh, va ve nhung gi minh chua lam duoc.
\textit{(They become more honest about ability, condition, and what is still unfinished.)}

Ky la la khi bot dung hinh anh, ho lai song nhe hon va nhan duoc nhung ket noi that hon.
\textit{(Strangely, when they stop constructing image, life becomes lighter and more real connections appear.)}

Vo boc khong bien mat hoan toan, nhung no khong con ngoi len phan that nua.
\textit{(The surface does not disappear completely, but it no longer sits on top of the real self.)}

\end{truyen}

\begin{baihoc}
Song that khong lam em nho di. No giup em do ton suc de gia on.
\textit{(Living honestly does not make you smaller. It saves the energy spent pretending to be fine.)}
\end{baihoc}

\begin{ghinhoanh}
\textbf{Short English to remember:}

Living honestly does not make you smaller.

It saves the energy spent pretending to be fine.
