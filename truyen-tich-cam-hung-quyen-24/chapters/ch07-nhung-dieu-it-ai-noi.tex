\section{Thành công có thể cô đơn}
\begin{truyen}{Thành công có thể cô đơn}{Life lesson}
\chuhoa{K}{hi anh lên đỉnh, ít người gọi. Thành công có thể cô đơn.}
\textit{(When he reached the top, few called. Success can be lonely.)}
\end{truyen}

\begin{baihoc}
Chuẩn bị tâm lý cho đỉnh cao.
\textit{(Prepare mentally for the top.)}
\end{baihoc}

\begin{ghinhoanh}
\textbf{Short English to remember:}

Prepare mentally for the top.

\end{ghinhoanh}
