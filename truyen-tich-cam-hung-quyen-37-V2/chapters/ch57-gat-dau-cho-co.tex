\chapter{Gật Đầu Cho Có}
\chapterintro{Đầu gật lia lịa nhưng não thì chưa kịp load xong.}{Giáo viên cứ tưởng đạt đồng thuận hóa ra chỉ là phản xạ có điều kiện.}

\begin{tutuyetbox}{Máy gật cơ học}{Mechanical Nodder}
Hỏi thử lời khuyên hiểu chớ trò
Gật đầu lia lịa dối co ro
Nhưng khi khảo sát mù tăm tối
Chết dở thi tài trượt đắng phô.
\tcblower
Asking if you understand the rule 
Nodding like a puppet, acting like a fool 
But when tested, you are in the dark 
Failing the trial, losing your spark.
\end{tutuyetbox}

\begin{tutuyetbox}{Sự hiểu lầm tai hại}{Fatal Misunderstanding}
Thầy giáo tận tình rọi sáng soi
Đầu kia gật gật dáng hời hòi
Kiểm tra ngớ ngẩn lòi trang trắng
Điểm rớt rành rành nhục bấy toi.
\tcblower
Teacher explains with a guiding light 
You keep nodding, out of sight 
Exam starts, showing a blank sheet 
Failing grades, a bitter defeat.
\end{tutuyetbox}

\begin{ghichu}
Cái gật đầu của tuổi trẻ rất lừa tình. Đừng tin học sinh khi chúng nói "em hiểu rồi".
\end{ghichu}
