\chapter{Phu huynh, gia dinh va cach day con}

\begin{leadbox}
Lop hoc khong bao gio ton tai tach khoi gia dinh. Co nhung dieu giao vien thay tren lop la am vang cua mot can nha, mot cuoc cai va, mot ky vong qua lon, hay mot su bo mac khong noi thanh loi.
\end{leadbox}

\storycard{61. Cuoc goi chi hoi diem}{The Call About Grades Only}{Khi phu huynh chi hoi ket qua, hoc sinh hoc cach gan gia tri ban than vao bang diem.}
\storycard{62. Con toi ngoan lam}{My Child Is Never Wrong}{Benh vuc mu quang co the lay mat co hoi giao duc trung thuc nhat.}
\storycard{63. Buoi hop phu huynh im lang}{Silent Parent Meeting}{Su im lang cua mot nguoi cha me doi khi tiet lo nhieu hon rat nhieu loi giai thich.}
\storycard{64. Em tu cham em nho}{Taking Care of a Younger Sibling}{Co nhung hoc sinh den truong da met tu truoc gio vao lop, vi em dang song vai cua nguoi lon qua som.}
\storycard{65. Ban an toi vang bo}{The Missing Dinner Table}{Tre can mot noi de ke chuyen trong ngay, neu khong lop hoc phai ganh them ca phan viec cua gia dinh.}
\storycard{66. Ky vong vao top lop}{Must Be Top of the Class}{Ky vong khong sai, nhung ky vong khong co cho tho se de hoc sinh song trong ap luc vo hinh.}
\storycard{67. Phu huynh xin ho diem}{Asking for Extra Points}{Khi nguoi lon thoa hiep voi su that, hoc sinh se hoc rat nhanh cach doi chuan muc.}
\storycard{68. Me xin thay dung mang con}{Please Do Not Scold My Child}{Bao boc qua muc co the cat mat kha nang cham vao hau qua de lon len.}
\storycard{69. Cha xuat hien muon mang}{The Father Who Arrives Late}{Mot buoi gap ngan cung co the la diem dau tien de mot gia dinh va nha truong bat dau lam viec that voi nhau.}
\storycard{70. Nha la noi ve truoc khi hoc}{Home Before School}{Neu muon doi mot hoc sinh, doi khi phai bat dau bang cach giup em co mot noi ve an toan hon.}
