\chapter{Guong mat nguoi thay trong lop hoc}

\begin{leadbox}
Chuong mo dau dat cau hoi nen tang: mot giao vien duoc hoc sinh nho den vi dieu gi? Vi su nghiem khac, su cong bang, hay vi vet thuong ma ho de lai? 10 tinh huong sau nham phac hoa hai mat cua hinh anh nguoi thay.
\end{leadbox}

\storycard{1. Cay but muc do}{The Red Ink Pen}{Mot bai kiem tra day vet do khien hoc sinh nghi minh vo dung; bai hoc la phe binh can di cung duong sua.}
\storycard{2. Tieng go thuoc len ban}{The Ruler on the Desk}{Suc manh ky luat co the tao trat tu, nhung neu dung sai no se bien lop hoc thanh noi so hai.}
\storycard{3. Loi khen dung luc}{Praise at the Right Moment}{Mot loi khen ngan cua thay co the mo ra niem tin ma hoc sinh da mat tu lau.}
\storycard{4. Gio hoc chi doc chep}{The Copying Hour}{Khi giao vien chi truyen lenh, hoc sinh hoc cach doi pho hon la cach tu duy.}
\storycard{5. Nguoi thay nho ten tung tro}{Knowing Every Name}{Su ton trong bat dau tu viec goi dung ten va nhin thay tung ca tinh trong lop.}
\storycard{6. Loi so sanh doc hai}{The Harmful Comparison}{So sanh hoc sinh voi ban khac co the tao ganh dua, nhung thuong de lai vet nut trong long tu trong.}
\storycard{7. Bai giang co cau chuyen that}{A Lesson with a Real Story}{Khi thay ke mot that bai cua chinh minh, hoc sinh bat dau tin vao gia tri cua viec dung day lai.}
\storycard{8. Su thien vi vo hinh}{Invisible Favoritism}{Mot giao vien co the khong co y, nhung cach uu ai lap di lap lai se tao ra chia re trong lop.}
\storycard{9. Cua phong giao vien}{The Door to the Staff Room}{Khi mot hoc sinh duoc moi vao noi chuyen rieng, do co the la hinh phat hoac co the la co hoi duoc cuu.}
\storycard{10. Chet yeu trong nghe day}{Losing the Fire}{Nguoi thay met moi de tro thanh cay bong mo, va hoc sinh thuong nhan ra dieu do som hon nguoi lon tuong.}
