\chapter{Hỏi Lại Dù Vừa Giảng}
\chapterintrobi{Tai nghe đó nhưng sự chú ý vừa đi vắng}{Ask again right after the explanation}{Đây là thói quen khiến người giảng và người nghe cùng ngượng trong vài giây ngắn ngủi.}

\begin{lucbatbox}{Lục bát: Mới nói xong mà quên}
Thầy vừa chốt ý bên trên\
Em liền hỏi lại như quên tức thì\
Bạn bên quay mặt cười khì\
Chắc hồn em đã kịp đi dạo rồi\
Nếu mà neo lại một hồi\
Câu kia khỏi phải hỏi trôi lần hai
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Thông tin vừa rơi}
Điều cô vừa nói mới đây thôi\
Em hỏi lại liền rất đúng thời\
Nếu sự chú ý ngồi yên được chút\
Chắc câu trả lời chẳng chạy rồi
\end{tutuyetbox}

\begin{ghichu}
Hỏi lại không xấu. Cái buồn cười nằm ở chỗ phần trả lời vừa xuất hiện cách đó chưa đầy một phút.
\end{ghichu}

\begin{englishnote}
\textbf{ask again}: hỏi lại.\par
\textbf{pay attention}: chú ý.\par
\textbf{Could you explain that one more time?}: Thầy cô có thể giải thích lại lần nữa không ạ?
\end{englishnote}
