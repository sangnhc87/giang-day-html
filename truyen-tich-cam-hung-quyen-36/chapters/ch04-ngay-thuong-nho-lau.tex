\chapter{Ngày Thường Mà Nhớ Lâu}
\chapterintro{Chủ đề lưu giữ}{Đây là phần gom những ngày rất thường mà thành ký ức mạnh: tin nhắn buổi sớm, giờ đọc sách, pháo hoa, bậc thềm trơn, buổi nhóm bài. Tông vui vẫn còn, nhưng nhớ đã đậm hơn.}

\poem{Bài 61}{Ngũ ngôn tứ tuyệt}{Tin nhắn chào buổi sớm}{Chỉ có hai chữ thôi}{Mà cà phê tôi nhạt}{Cũng hóa ngọt đầu môi}
\poem{Bài 62}{Ngũ ngôn tứ tuyệt}{Sân bóng đông cổ vũ}{Em cột tóc trên cao}{Tôi sút hỏng hai quả}{Vì mắt cứ tìm nhau}
\poem{Bài 63}{Ngũ ngôn tứ tuyệt}{Giờ vẽ chân dung bạn}{Tôi pha nắng pha mây}{Tô xong màu mái tóc}{Mới hay là người này}
\poem{Bài 64}{Ngũ ngôn tứ tuyệt}{Mặt bàn khắc tên nhỏ}{Tôi viết rồi tôi xoa}{Sợ ai nhìn ra được}{Điều vừa nở trong ta}
\poem{Bài 65}{Ngũ ngôn tứ tuyệt}{Ngày hội sách thơm giấy}{Em chọn truyện rất vui}{Tôi đứng bên cười ké}{Tưởng mình cũng thành vui}
\poem{Bài 66}{Ngũ ngôn tứ tuyệt}{Hành lang trưa vắng bước}{Em mở hộp cơm xinh}{Tôi chia nửa quả quýt}{Ngọt tới tận bình minh}
\poem{Bài 67}{Ngũ ngôn tứ tuyệt}{Đầu giờ trời oi quá}{Em quạt tập cho sau}{Gió lạc sang bàn trước}{Cứu luôn kẻ ngồi sầu}
\poem{Bài 68}{Ngũ ngôn tứ tuyệt}{Ngày thi nghe trống gấp}{Em cho mượn cây bút}{Tôi cầm như lá bùa}{Viết chữ ngoan hơn chút}
\poem{Bài 69}{Ngũ ngôn tứ tuyệt}{Tháng sáu ve gọi mãi}{Em ép cánh phượng hồng}{Tôi xin một cánh nhỏ}{Giấu cùng mấy nhớ mong}
\poem{Bài 70}{Ngũ ngôn tứ tuyệt}{Học thêm về chập tối}{Em nói giọng mềm ghê}{Tôi ước đường dài mãi}{Cho mình đi thật chậm}
\poem{Bài 71}{Ngũ ngôn tứ tuyệt}{Giờ sinh nhìn kính hiển}{Em gọi tôi xem cùng}{Tôi chẳng nhìn cánh mỏng}{Chỉ thấy khoảng gần rung}
\poem{Bài 72}{Ngũ ngôn tứ tuyệt}{Cuối tuần làm dự án}{Em rủ họp công viên}{Tôi mang nhiều ý tưởng}{Gặp em quên sạch liền}
\poem{Bài 73}{Ngũ ngôn tứ tuyệt}{Bụi phấn vương vai áo}{Em phủi giúp một tay}{Tôi về ngồi trên xe}{Tim còn rung nguyên ngày}
\poem{Bài 74}{Ngũ ngôn tứ tuyệt}{Trời lạnh trong giờ văn}{Em ho hai tiếng nhỏ}{Tôi lùa viên kẹo gừng}{Qua bàn nghe dễ đỏ}
\poem{Bài 75}{Ngũ ngôn tứ tuyệt}{Bảng tin treo ảnh lớp}{Em cười hàng đầu tiên}{Tôi nhìn hơi quá lâu}{Sợ mai quên mất hiền}
\poem{Bài 76}{Ngũ ngôn tứ tuyệt}{Buổi học nhóm đông đủ}{Em ngồi cạnh cửa nắng}{Tôi giải sai một bước}{Vì nhìn em quá thẳng}
\poem{Bài 77}{Ngũ ngôn tứ tuyệt}{Thể dục vòng số tám}{Em vượt tôi nửa sân}{Tôi thua mà vẫn sướng}{Vì chung một nhịp chân}
\poem{Bài 78}{Ngũ ngôn tứ tuyệt}{Đêm hội trường pháo sáng}{Em reo khẽ bên vai}{Trời rồi sẽ tắt lửa}{Còn tôi nhớ một đời}
\poem{Bài 79}{Ngũ ngôn tứ tuyệt}{Mùa mưa bậc thềm trượt}{Em suýt ngã đầu hiên}{Tôi đưa tay đỡ kịp}{Và đứng khờ nguyên chiều}
\poem{Bài 80}{Ngũ ngôn tứ tuyệt}{Giờ công dân nói mộng}{Em muốn giúp nhiều người}{Tôi nghe mà tự nhủ}{Phải lớn cho bằng người}
