\chapter{Đứng dậy}
\markboth{Đứng dậy}{Get Up}


\section{Nản một lúc không sao — đứng dậy mới quan trọng.}
\begin{truyen}{Nản một lúc không sao — đứng dậy mới quan trọng.}{Support when down.}
\chuhoa{C}{âu thầy nói nhẹ nhàng, không trách. Em ấy ngẩng đầu lên — như nghe được điều cần nghe.}
\textit{(The teacher said it gently, without blame. That student looked up — as if hearing what they needed.)}
\end{truyen}

\begin{baihoc}
Nản một lúc không sao; đứng dậy mới quan trọng.
\textit{(Feeling down for a while is okay; getting back up is what matters.)}
\end{baihoc}

\begin{ghinhoanh}
\textbf{Short English to remember:}

Support when they're down, not blame.

\end{ghinhoanh}

\begin{tuhocnhanh}
1. Câu này giúp em đứng dậy thế nào? \textit{How does this help you get up?}

2. Em muốn nhớ câu này khi nào? \textit{When do you want to remember this?}
\end{tuhocnhanh}
