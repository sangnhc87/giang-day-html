\chapter{Kiểm Tra Và Thi Cử}
\chapterintro{Ngày thường bình yên, tới hôm kiểm tra thì ai cũng thành người khác}{Thi cử là nơi trí nhớ, lòng tin và vận may đụng nhau rõ nhất.}

\begin{lucbatbox}{Lục bát: Trước giờ mới mở vở}
Đến ngày kiểm tra mới lo
Đêm qua mới mở từng tờ vội xem
Chữ thì dày, mắt thì choáng
Tim nghe gõ nhịp như trống đầu sân
Giá ôn sớm tự bao buổi
Đâu cần sát nút ngồi lời thở than
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Vào phòng là quên}
Điều em học tối còn đây nữa
Vào phòng bỗng hóa mây bay ngay
Chỉ còn nhịp thở gấp gáp thôi
Và mong đề giấy đừng sai nữa
\end{tutuyetbox}

\begin{ghichu}
Thi cử không làm kiến thức biến mất. Nó chỉ bóc rất nhanh lớp tự tin mỏng khi việc ôn tập chưa tới nơi.
\end{ghichu}

\ngancach

\begin{lucbatbox}{Lục bát: Học tủ rồi run}
Em ôn đúng một phần thôi
Tin rằng đề sẽ rơi nơi ấy mà
Ai ngờ giấy mở ra xa
Câu quen đứng ít, câu lạ đứng đầy
Mới hay cược cửa vào may
Trong thi cử ấy đắng cay đủ điều
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Khoanh bừa cầu phúc}
Khi chẳng còn nhớ điều chi nữa
Em đành ký gửi chút may thôi
Nếu gặp người chấm lòng thương quá
Thì em còn cửa ngẩng đầu rồi
\end{tutuyetbox}
