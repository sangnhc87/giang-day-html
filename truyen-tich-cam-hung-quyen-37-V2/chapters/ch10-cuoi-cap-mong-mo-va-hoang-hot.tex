\chapter{Cuối Cấp, Mộng Mơ Và Hoảng Hốt}
\chapterintro{Tuổi 17, 18 vừa muốn lớn nhanh vừa sợ ngày lớn thật}{Cuối cấp là quãng thời gian cái gì cũng căng hơn một nhịp.}

\begin{lucbatbox}{Lục bát: Chọn ngành theo cảm xúc}
Hôm nay em thích ngành Luật
Mai mê ngành Y, chiều chuộng kiến trúc
Bạn bè hỏi chốt chưa nào
Em rằng để trái tim thôi lựa đường
Tim quen đi lối phiêu du
Đề cương trước mặt vẫn mù vài trang
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Cuối cấp nhiều lo}
Mười tám tuổi lo chuyện gần xa
Lo ngành, lo điểm, lo tên khoa
Việc gần nhất là ngồi vào học
Lại dời từ sớm đến chiều qua
\end{tutuyetbox}

\begin{ghichu}
Cuối cấp buồn cười ở chỗ ai cũng nói chuyện tương lai rất xa, nhưng việc khó nhất nhiều khi chỉ là ngồi yên học hết một buổi.
\end{ghichu}

\ngancach

\begin{lucbatbox}{Lục bát: Chụp kỷ yếu rất chăm}
Đến mùa chụp ảnh cuối năm\
Em chăm da dẻ hơn chăm sách vở\
Tóc tai đầu tư từng giờ\
Biểu cảm luyện kỹ hơn là môn thi\
Ảnh ra đẹp thật diệu kỳ\
Chỉ mong điểm số đừng đi ngược chiều
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Lo tương lai}
Em lo tương lai suốt sáng đêm
Lo trường lo nguyện vọng triền miên
Chỉ thiếu một việc gần nhất thôi
Là ngồi xuống học bớt than thân
\end{tutuyetbox}
