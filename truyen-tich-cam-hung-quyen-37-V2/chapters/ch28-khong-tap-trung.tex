\chapter{Không Tập Trung}
\chapterintro{Ngồi đủ một tiết nhưng sự chú ý chỉ có mặt lác đác}{Đây là thói quen âm thầm mà hậu quả thì rất rõ.}

\begin{lucbatbox}{Lục bát: Mắt nhìn, đầu vắng}
Mắt em vẫn hướng lên trên\
Nhưng phần tiếp nhận lại quên ngồi cùng\
Lời thầy đi hết mấy vùng\
Tới tai thì ít, tới lòng càng thưa\
Đến khi hỏi lại mới vừa\
Hay mình có mặt mà chưa học gì
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Chú ý mỏng}
Em không nghịch phá gì đâu nhé
Chỉ là chú ý đi đâu mất
Tiết qua nhẹ tựa mây đang trôi
Kiến thức còn đứng gọi tên thôi
\end{tutuyetbox}

\begin{ghichu}
Không tập trung là kiểu trượt bài rất yên lặng. Nó không ồn ào, nhưng gom lại thì hụt rất nhiều.
\end{ghichu}

\begin{lucbatbox}{Lục bát: Câu trước chưa kịp vào}
Câu trước chưa kịp đi qua\
Câu sau đã tới, em ra giữa đường\
Nghe như hiểu hết bình thường\
Đến khi cần nhắc lại dường trắng không\
Hóa ra cái thiếu trong lòng\
Là neo chú ý cho xong một hàng
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Trôi theo tiết học}
Không tập trung chẳng ai hay đâu
Trừ khi kiểm lại lộ chỗ hổng
Một giờ tưởng nhẹ như không vậy
Mà mang về một khoảng trống dài
\end{tutuyetbox}
