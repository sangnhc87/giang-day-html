\chapter{Chuyên Cần Và Đi Muộn}
\chapterintro{Có bạn đến lớp đều, chỉ tiếc là đều muộn đôi phút}{Đúng giờ là chuyện nhỏ, lặp lại mãi thì thành nếp khó thương.}

\begin{lucbatbox}{Lục bát: Muộn mà vẫn thản nhiên}
Chuông vào lớp đã vang rồi
Em còn đi tới như người dạo chơi
Miệng cười dáng vẻ thảnh thơi
Như chưa có chuyện cả nơi đợi chờ
Giá mà sớm được đôi giờ
Tên trong sổ muộn đỡ nhơ trang ghi
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Đi muộn có lý do}
Đi muộn mà lời vẫn rất êm
Lần nào giải thích cũng thêm mềm
Chỉ cần đúng giờ thêm một chút
Tên đâu dính mãi ở trang xem
\end{tutuyetbox}

\begin{ghichu}
Đi học muộn một lần là sự cố. Đi muộn nhiều lần rồi kể khéo, đó mới là phong cách.
\end{ghichu}

\ngancach

\begin{lucbatbox}{Lục bát: Hôm nào cũng sát giờ}
Sáng nào em cũng rất gần
Gần chuông gần trễ gần phần nhắc tên
Bạn vào từ lúc sân yên
Em thì bước đúng khi tên gọi rồi
Giá mà sớm được đôi hồi
Khỏi mang cảm giác bồn chồn đầu hôm
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Chuyên cần nửa vời}
Điểm danh không thấy tên em ta
Chỉ tiếc giờ giấc cứ xô qua
Nếu chịu chuyên cần thêm đúng lúc
Bảng vàng tên tuổi rạng như hoa.
\end{tutuyetbox}
\chapter{Hành Trang Và Ý Thức}
\chapterintro{Sách vở không chỉ là giấy mực, đó là tấm gương soi sự chuẩn bị của mỗi người.}{Quên một cuốn sách là đánh rơi một cơ hội hiểu bài.}

\begin{lucbatbox}{Lục bát: Giấc mộng bình minh}
Sáng mai nắng đã lên cao
Em còn say giấc chiêm bao mơ màng
Tiếc chi giấc mộng dở dang
Để thời gian mãi trôi sang muộn giờ
Đường đi học vẫn đợi chờ
Dậy mau kẻo trễ, đừng ngơ ngẩn lòng.
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Nhắc chuyện quên bài}
Sách vở đem quên dạ thẫn thờ
Do chân bước vội thật đâu ngờ
Chuyên cần vốn tại tâm rèn luyện
Gói ghém hành trang chớ đợi chờ.
\end{tutuyetbox}

\begin{ghichu}
Đừng đổ lỗi cho cái đồng hồ báo thức hay chiếc cặp sách quá nhỏ. Sự chuẩn bị kỹ lưỡng từ đêm trước chính là bí kíp của những người "nhàn hạ" vào sáng hôm sau.
\end{ghichu}

\ngancach

\begin{lucbatbox}{Lục bát: Tâm hồn treo ngọn cây}
Thân ngồi lớp, ý phương nào
Mắt nhìn lên bảng mà chao chát hồn
Chữ thầy như gió dập dồn
Trôi qua kẽ tóc chẳng còn đọng chi
Gắng công tập trung tức thì
Kẻo khi thi cử lệ tì nhòe trang.
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Ý thức học đường}
Trống giục vang rền khắp vách tường
Người người rảo bước tới trường cương
Chuyên cần giữ trọn niềm tin ấy
Chớ để tên rơi giữa dọc đường.
\end{tutuyetbox}
