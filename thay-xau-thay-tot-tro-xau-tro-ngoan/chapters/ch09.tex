\chapter{Tu hoc, tu quan ly va truong thanh}

\begin{leadbox}
Tu mot hoc sinh phu thuoc den mot nguoi co the tu hoc la ca mot qua trinh. Chuong nay huong ve nhung nang luc be ma co suc tao khac biet lon: lap ke hoach, giu loi hua voi chinh minh, va biet tu kiem diem.
\end{leadbox}

\storycard{81. To ke hoach dau tien}{The First Study Plan}{Ke hoach duoc viet ra giup hoc sinh thay rang tien bo co hinh dang, khong chi la cam xuc.}
\storycard{82. Mot gio khong dien thoai}{One Hour Without a Phone}{Tu chu khong bat dau bang nhung loi the lon, ma bang khung thoi gian nho duoc giu tron ven.}
\storycard{83. Hoc nhom ma van hieu bai}{Studying Together Properly}{Hoc nhom dung cach la chia viec, tranh lan man, va buoc moi nguoi tu tu duy.}
\storycard{84. So tay sai lam}{The Mistake Notebook}{Ghi lai loi sai bien that bai thanh tai lieu hoc tap co gia tri lau dai.}
\storycard{85. Day som hon 15 phut}{Fifteen Minutes Earlier}{Ky luat ca nhan rat hay bat dau tu mot thoi quen nho nhung ben.}
\storycard{86. Dat muc tieu thuc te}{A Realistic Goal}{Muc tieu dung lam hoc sinh thay co duong di, khong thay mot buc tuong qua cao.}
\storycard{87. Tu danh gia sau moi tuan}{Weekly Self-Review}{Nguoi hoc biet nhin lai minh se it can nguoi khac keo day hon.}
\storycard{88. Tim mot ban dong hanh}{An Accountability Partner}{Dong luc ca nhan ben hon khi co mot nguoi cung theo doi tien bo mot cach tu te.}
\storycard{89. Hoc vi y nghia}{Studying for Meaning}{Khi hoc sinh biet minh hoc de lam gi, su kien tri tang len ro ret.}
\storycard{90. Tu quan ly la tu do}{Self-Management Is Freedom}{Ky luat tu than khong phai giong xich; do la cach mo rong lua chon cho tuong lai.}
