\chapter{Mùa Thi, Hoa Phượng, Lời Tạm Biệt}
\chapterintro{Chủ đề khép lại}{Phần cuối nhiều phượng đỏ, giấy khen, tổng kết, pháo hoa, ngoái nhìn sau cổng trường. Vẫn giữ nét dí dỏm nhưng cảm xúc sâu hơn để chốt sách cho có dư vị.}

\poem{Bài 81}{Ngũ ngôn tứ tuyệt}{Thứ hai cờ trước gió}{Em đứng hàng rất nghiêm}{Tôi cũng nghiêm như thể}{Chỉ mắt chạy qua em}
\poem{Bài 82}{Ngũ ngôn tứ tuyệt}{Loa nghe Anh rè nhẹ}{Em ghé bảo nghe này}{Khoảng gần hơn từ vựng}{Làm tim nhớ cả ngày}
\poem{Bài 83}{Ngũ ngôn tứ tuyệt}{Ngày tổng vệ sinh lớp}{Em buộc khăn thật cao}{Tôi bưng chậu đi mãi}{Vẫn nghe lòng xôn xao}
\poem{Bài 84}{Ngũ ngôn tứ tuyệt}{Tiết đọc sách im phắc}{Em lật giấy rất khô}{Tôi ngồi bên giả đọc}{Thật ra đọc má hồng}
\poem{Bài 85}{Ngũ ngôn tứ tuyệt}{Lớp thêm xa cuối phố}{Em nhắn kèm bản đồ}{Tôi lần theo từng chấm}{Thấy lòng gần thêm thôi}
\poem{Bài 86}{Ngũ ngôn tứ tuyệt}{Mưa tan cầu vồng hiện}{Em chỉ mắt long lanh}{Tôi đứng bên nghe rõ}{Tuổi này đẹp thật lành}
\poem{Bài 87}{Ngũ ngôn tứ tuyệt}{Giờ hóa hành thí nghiệm}{Em nhắc đừng quá tay}{Một câu nghe thường thế}{Mà dạy tôi bớt say}
\poem{Bài 88}{Ngũ ngôn tứ tuyệt}{Câu lạc bộ ảnh}{Em chọn khung xanh trời}{Tôi treo hình sân vắng}{Trong đó có em cười}
\poem{Bài 89}{Ngũ ngôn tứ tuyệt}{Ngày thi cuối cùng tới}{Em chúc tôi bình tĩnh}{Tôi vào phòng nghe máu}{Cũng ngoan như lời xinh}
\poem{Bài 90}{Ngũ ngôn tứ tuyệt}{Mùa đông tan học sớm}{Em đội mũ len nâu}{Tôi đạp xe theo sau}{Sợ gọi rồi thành đau}
\poem{Bài 91}{Ngũ ngôn tứ tuyệt}{Báo tường vừa đạt giải}{Em ôm bó hoa tươi}{Tôi đứng hàng thật cuối}{Vẫn vui hơn mọi người}
\poem{Bài 92}{Ngũ ngôn tứ tuyệt}{Gió sớm làm tay lạnh}{Em đưa gói trà gừng}{Tôi ôm bằng hai tay}{Ấm từ ngoài tới ngực}
\poem{Bài 93}{Ngũ ngôn tứ tuyệt}{Tiết văn phân thơ cũ}{Em nói mấy ý sâu}{Tôi bỗng hiểu một chuyện}{Mình thích đã từ lâu}
\poem{Bài 94}{Ngũ ngôn tứ tuyệt}{Tháng tư sân hoa bướm}{Em cười dưới nắng trưa}{Tôi chụp bằng đôi mắt}{Sợ máy không giữ vừa}
\poem{Bài 95}{Ngũ ngôn tứ tuyệt}{Họp phụ huynh trường vắng}{Em nhắn đừng lo nha}{Một câu như cái neo}{Giữ chiều đỡ đi qua}
\poem{Bài 96}{Ngũ ngôn tứ tuyệt}{Tập nhảy chân tôi vụng}{Em đếm một hai ba}{Tôi theo không cho kịp}{Mà nghe như tiếng nhà}
\poem{Bài 97}{Ngũ ngôn tứ tuyệt}{Chia nhóm làm dự án}{Em gọi đúng tên tôi}{Tôi gật nhanh hơn não}{Tim thì chậm mất rồi}
\poem{Bài 98}{Ngũ ngôn tứ tuyệt}{Ngày tổng kết cuối khóa}{Em nhận giấy khen đầu}{Tôi vỗ tay như lớp}{Mà vui riêng một màu}
\poem{Bài 99}{Ngũ ngôn tứ tuyệt}{Xe qua cổng rẽ nhỏ}{Em ngoái lại một lần}{Tôi giữ nguyên khoảnh ấy}{Đốt ấm bao ngày gần}
\poem{Bài 100}{Ngũ ngôn tứ tuyệt}{Phượng hồng vừa khép nắng}{Lưu bút chạm tay run}{Tưởng trang thơ đã khép}{Còn tiếng cười sau cùng}
