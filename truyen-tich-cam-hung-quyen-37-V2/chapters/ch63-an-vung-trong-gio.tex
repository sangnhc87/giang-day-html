\chapter{Ăn Vụng Trong Giờ}
\chapterintrobi{Bài giảng đang chạy, bao bánh cũng đang chạy}{Snack during class}{Một số bạn rất giỏi chia sự tập trung giữa kiến thức và tiếng sột soạt.}

\begin{lucbatbox}{Lục bát: Mở bánh thật khẽ}
Tiết kia đang rất nghiêm trang\
Em nghe bụng bảo cần sang phần quà\
Mở cho thật khẽ vậy mà\
Cả bàn bên cạnh nhận ra tức thì\
Thầy cô chưa chắc đã vì\
Nhưng mùi thơm đó có đi đâu nào
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Tiếng giấy khó giấu}
Em nghĩ gói snack mở rất êm\
Ai ngờ cả lớp đã nghe thêm\
Nếu phần tập trung dày như mùi ấy\
Điểm số hôm nay chắc vượt thềm
\end{tutuyetbox}

\begin{ghichu}
Ăn vụng trong giờ không chỉ làm bụng vui. Nó còn làm tiết học trở nên nhiều âm thanh phụ hơn dự kiến.
\end{ghichu}

\begin{englishnote}
\textbf{snack}: đồ ăn vặt.\par
\textbf{during class}: trong giờ học.\par
\textbf{I was only having a quick snack.}: Em chỉ ăn tạm chút thôi.
\end{englishnote}
