\chapter{Đọc Đề Lướt}
\chapterintrobi{Đề chưa đọc kỹ nhưng lòng tin thì rất mạnh}{Skim the question}{Không ít lỗi sai sinh ra chỉ vì mắt đi nhanh hơn não.}

\begin{lucbatbox}{Lục bát: Mới nhìn đã tưởng hiểu}
Đề kia em lướt rất tài\
Vừa xem mấy chữ đã khai bút liền\
Làm xong mới thấy ưu phiền\
Người ta hỏi khác, mình nghiêng hướng nào\
Nếu chậm thêm chút ban đầu\
Điểm kia chưa chắc rủ nhau đi xa
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Sai từ cái nhìn đầu}
Đề nằm trước mặt vẫn hiền khô\
Chỉ tại em xem quá vội thôi\
Mắt đọc một đường tay làm một nẻo\
Nên câu dễ nhất cũng mồ côi
\end{tutuyetbox}

\begin{ghichu}
Đọc đề lướt là tài năng đặc biệt: biến câu dễ thành câu sai bằng tốc độ rất ấn tượng.
\end{ghichu}

\end{englishnote}
