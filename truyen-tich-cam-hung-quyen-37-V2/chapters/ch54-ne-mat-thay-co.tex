\chapter{Né Ánh Mắt Thầy Cô}
\chapterintro{Trò chơi trốn tìm gay cấn nhất là lúc thầy đang cầm sổ điểm danh.}{Tàng hình là kỹ năng sinh tồn bắt buộc của học sinh chưa thuộc bài.}

\begin{tutuyetbox}{Vị thần né tránh}{The God of Avoidance}
Thầy nhìn đảo mắt khắp xung quanh
Em cúi đầu sâu nấp thật nhanh
Mắt lùi né tránh tia dò xét
Trúng số lên bàn khóc nín thinh.
\tcblower
Teacher's glaring, the class is in fear 
The good student bows, avoiding the pier 
Eyes retreat from the searching light 
Called to the board, a silent night.
\end{tutuyetbox}

\begin{tutuyetbox}{Kẻ tàng hình hụt hơi}{The Out-of-breath Invisible}
Thầy nhìn xuống góc lủi gầm bàn
Chữ nghĩa nghèo nàn sợ oán than
Tới lúc gọi tên trò rớt mật
Thân tàn tơi tả điểm tan hoang.
\tcblower
Teacher looks down, hiding under the desk 
Knowledge is poor, feeling grotesque 
Terrified when the name is called 
A broken spirit, completely appalled.
\end{tutuyetbox}

\begin{ghichu}
Đừng nghĩ bạn né được ánh mắt thầy cô, họ chỉ đang chờ thời cơ lúc bạn lơ là nhất để gọi lên bảng thôi.
\end{ghichu}
