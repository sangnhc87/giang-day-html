\section{Từ câu 501 đến câu 520}
\begin{enumerate}[leftmargin=*]
\item \hsqa{Thầy ơi bài này có thi không?}{Có thể có, nhưng cái đáng thi nhất là trí nhớ em đó.}
\item \hsqa{Nếu không thi thì em khỏi học được không?}{Được, rồi kiến thức cũng khỏi vào em luôn.}
\item \hsqa{Thầy ơi ôn bao nhiêu là đủ?}{Đủ là lúc em bớt hỏi câu này trong hoảng loạn.}
\item \hsqa{Em học trúng tủ được không?}{Được, nếu đề thương em như người nhà.}
\item \hsqa{Em chỉ học phần dễ thôi.}{Đề thường không ký cam kết hiền như vậy.}
\item \hsqa{Thi xong em quên sạch có sao không?}{Sao chứ, não em đâu phải phòng trọ theo ngày.}
\item \hsqa{Em ôn từ đêm trước có kịp không?}{Kịp lo, còn kịp giỏi thì khó.}
\item \hsqa{Thầy đoán đề giúp em đi.}{Thầy dạy học, không hành nghề bói bài thi.}
\item \hsqa{Em làm đề cũ là đủ chưa?}{Đủ để biết mình còn thiếu bao nhiêu.}
\item \hsqa{Thi thử thấp quá em nản.}{Thi thử mà nản thì thi thật em tính làm thơ à?}
\item \hsqa{Thầy ơi em run quá.}{Run là bình thường, miễn tay em đừng run tới mức quên não.}
\item \hsqa{Em học mà vào phòng thi là trắng đầu.}{Vậy em ôn kiến thức ít, ôn bản lĩnh còn ít hơn.}
\item \hsqa{Sao càng gần thi em càng muốn ngủ.}{Cơ thể em đang phản đối kiểu ôn nước rút đó.}
\item \hsqa{Em làm đề mà điểm cứ dậm chân.}{Vì em đang luyện đề, chưa chắc đã luyện lỗi.}
\item \hsqa{Em chỉ cần qua môn thôi.}{Qua kiểu này thì kiến thức cũng lết qua cùng em.}
\item \hsqa{Em thi xong mới học được không?}{Học sau thi nghe giống bơi sau khi chìm.}
\item \hsqa{Em sợ không đủ thời gian làm bài.}{Vậy lúc ôn em tập làm đúng giờ đi, đừng tập than.}
\item \hsqa{Thi trắc nghiệm chắc dễ hơn.}{Dễ với người biết, khó với người đoán bằng niềm tin.}
\item \hsqa{Em học thuộc hết là ổn chứ thầy?}{Thuộc mà không hiểu thì đề đổi tí là em tắt.}
\item \hsqa{Thầy ơi em ôn mà không thấy vào.}{Vì em đang quét mắt, chưa chắc đã động não.}
\end{enumerate}

\section{Từ câu 521 đến câu 540}
\begin{enumerate}[leftmargin=*,resume]
\item \hsqa{Em nên ôn từ chương nào trước?}{Từ chỗ em yếu nhất, đừng từ chỗ em thích nhất.}
\item \hsqa{Em có nên thức trắng ôn thi không?}{Có thể, nếu em muốn hôm sau thi bằng linh hồn.}
\item \hsqa{Em học nhóm mà toàn nói chuyện.}{Nhóm em đang ôn thi hay ôn tình bạn?}
\item \hsqa{Em xem tóm tắt thôi được không?}{Được, miễn điểm em cũng thích sống tóm tắt.}
\item \hsqa{Em không biết ôn kiểu gì cho nhớ.}{Lặp lại, làm lại, sửa lại; chứ không phải cầu lại.}
\item \hsqa{Em còn ít thời gian quá.}{Ít còn hơn không, đừng phí luôn phần ít đó.}
\item \hsqa{Em làm đề sai nhiều nên ngại làm tiếp.}{Sai nhiều lúc ôn còn rẻ hơn sai ít lúc thi.}
\item \hsqa{Thầy ơi em có số thi cử không?}{Có, số lần em chịu ngồi xuống ôn đó.}
\item \hsqa{Em hay học xong quên lúc đi ngủ.}{Vậy chắc hôm sau em phải thức dậy ôn lại.}
\item \hsqa{Em thấy bạn em chưa học mà bình tĩnh lắm.}{Có khi bạn em gan lớn, chứ chưa chắc não lớn.}
\item \hsqa{Em đọc nhiều mà không thuộc.}{Đọc nhiều khác với đọc có mục tiêu.}
\item \hsqa{Thi mà trúng câu khó thì sao thầy?}{Thì em làm câu dễ trước, chứ đừng ngồi nhìn đời trôi.}
\item \hsqa{Em nên ôn dài hay ôn ngắn?}{Ôn thật thì kiểu nào cũng được, ôn giả thì kiểu nào cũng hỏng.}
\item \hsqa{Em sợ mở đề ra thấy lạ.}{Lạ là đề làm việc, không phải đề phản bội em.}
\item \hsqa{Em làm sai một câu là mất tinh thần.}{Tinh thần em mong manh hơn đáp án chuẩn luôn.}
\item \hsqa{Em nên học một môn cả ngày không?}{Nếu não em chịu nổi thì được, còn thường nó phản ứng bằng cách lơ mơ.}
\item \hsqa{Em chỉ học phần giáo viên nhấn mạnh.}{Nhấn mạnh không có nghĩa phần còn lại tự biến mất.}
\item \hsqa{Thầy ơi em ôn mà thấy vô vọng.}{Vô vọng thật là lúc em ngừng ôn.}
\item \hsqa{Em nên xem lời giải trước hay tự làm trước?}{Tự vật lộn trước, kẻo não em quen ngồi xem phim.}
\item \hsqa{Em thấy đề mẫu dễ hơn đề thầy cho.}{Đời cũng hay vậy, bản demo thường hiền hơn bản chính thức.}
\end{enumerate}
