\chapter{Thề Đổi Thói Sau Điểm Kém}
\chapterintrobi{Khoảnh khắc hứa hẹn mạnh mẽ nhất thường đến ngay sau một con điểm buồn}{Promise to change after a low score}{Rất nhiều quyết tâm học tập nở rộ đúng lúc cảm xúc còn nóng.}

\begin{lucbatbox}{Lục bát: Từ mai con khác}
Điểm vừa thấp xuống hôm nay\
Em liền hứa sẽ đổi thay từ giờ\
Về nhà viết cả trang thơ\
Lịch học, giờ ngủ, từng bờ kỷ cương\
Ba hôm khí thế còn vương\
Ngày thứ tư lại nhường đường thói quen
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Quyết tâm nóng}
Sau điểm kém em quyết chí bền\
Kế hoạch dựng dậy rất là nên\
Chỉ tiếc quyết tâm cần sức bám
Chứ không chỉ rực đúng vài đêm
\end{tutuyetbox}

\begin{ghichu}
Thề đổi thói sau điểm kém là phản ứng lành mạnh. Điều khó nhất không phải hứa, mà là giữ lời khi cảm xúc đã nguội.
\end{ghichu}

\begin{englishnote}
\textbf{low score}: điểm kém.\par
\textbf{change my habits}: đổi thói quen.\par
\textbf{I will do better next time.}: Lần sau em sẽ làm tốt hơn.
\end{englishnote}
