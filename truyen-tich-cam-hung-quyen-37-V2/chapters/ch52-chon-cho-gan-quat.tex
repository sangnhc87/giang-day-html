\chapter{Chọn Chỗ Gần Quạt}
\chapterintro{Học tập là phụ, hưởng thụ gió mát là chính.}{Chiến thắng trong cuộc đua giành "ghế phong thủy" là niềm tự hào lớn nhất đầu giờ.}

\begin{tutuyetbox}{Ghế phong thủy}{The Feng Shui Seat}
Xắng xái tìm nơi mát rợi mình
Dưới lồng quạt máy mộng lung linh
Thầy cô giảng mỏi mồm mưu trí
Trò ngáp li bì ngủ mất hình.
\tcblower
Searching for a cool and breezy spot 
Under the fan, dreaming quite a lot 
Teacher lectures with a brilliant mind 
You stay yawning, leaving life behind.
\end{tutuyetbox}

\begin{tutuyetbox}{Cơn gió tử thần}{The Death Wind}
Gió thổi hiu hiu dẫn lối buồn
Đầu lười chữ rụng trút muôn phương
Vào thi đáp án mù mây khói
Rớt thảm lưu ban tủi đêm trường.
\tcblower
A gentle breeze leads to a sad path 
Lazy mind drops knowledge in its wrath 
In the exam, answers are like smoke 
Failing the grade, life is but a joke.
\end{tutuyetbox}

\begin{ghichu}
Chỗ ngồi gần quạt giúp bạn mát mẻ lúc học, nhưng sẽ làm bạn "lạnh sống lưng" lúc nhìn thấy tờ đề thi.
\end{ghichu}
