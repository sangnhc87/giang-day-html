\chapter{Học Thêm Học Nhóm}
\chapterintro{Học nhóm là danh nghĩa, tám chuyện là thực tế.}{Tốn tiền xăng xe nhưng chỉ thu về được dăm ba câu chuyện drama.}

\begin{tutuyetbox}{Hội nghị bàn tròn}{The round table meeting}
Rủ nhau học nhóm tại nhà chung
Kiến thức xa vời chém nổ đùng
Vở sách phơi ra làm cảnh tượng
Tới giờ kiểm định khóc ung dung.
\tcblower
Gathering for a group study night 
Knowledge is distant, gossip is bright 
Books are displayed for a showy display 
Failing in the morning, a long, sad day.
\end{tutuyetbox}

\begin{tutuyetbox}{Lò luyện linh hồn}{The soul furnace}
Lò luyện đêm ngày nát óc si
Kiến thức tiêu điều trượt từ vi
Bạn hỡi cùng đi thi rớt hết
Thân tàn lụn bại lệ sầu ghi.
\tcblower
Studying hard, morning and night 
Knowledge vanishes, out of sight 
Every friend fails the test with me 
A broken spirit, for all to see.
\end{tutuyetbox}

\begin{ghichu}
Học thêm không phải là đi đóng tiền, mà là đi thu hoạch chất xám. Đừng để túi tiền của bố mẹ vơi đi mà não bộ vẫn nhẹ tênh.
\end{ghichu}
