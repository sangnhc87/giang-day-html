\chapter{Sống Cùng Màn Hình Mọi Nơi}
\markboth{Sống Cùng Màn Hình Mọi Nơi}{Sống Cùng Màn Hình Mọi Nơi}

\begin{center}
	\textit{\color{bookprimary}``Nếu đi vệ sinh cũng phải xem, đi bộ cũng phải xem, tập thể dục cũng phải xem, thì điện thoại không còn là công cụ nữa. Nó đã thành dây xích.''}

	\textit{\color{bookprimary}``If you must watch while using the toilet, walking, and exercising, then the phone is no longer a tool. It has become a chain.''}
\end{center}

\ngancach

\section{Ngay Cả Lúc Cơ Thể Đang Cần Thở}
\begin{truyen}{Ngay Cả Lúc Cơ Thể Đang Cần Thở}{Even When the Body Needs Air}
\chuhoa{C}{ó bạn vừa đi bộ thể dục quanh sân trường vừa dán mắt vào điện thoại.}
\textit{(A student walks around the school yard for exercise while staring at the phone.)}

Thầy hỏi: ``Em đang tập cái gì?''
\textit{(The teacher asks: ``What exactly are you training?'')}

Bạn ấy cười: ``Dạ em vừa tập vừa thư giãn mà thầy.''
\textit{(The student laughs: ``I am exercising and relaxing at the same time.'')}

\teacherpair{Không. Em đang làm cả hai thứ cùng dở đi. Tập thể dục là lúc máu phải chạy, hơi thở phải sâu, đầu óc phải có cơ hội rời bớt kích thích. Còn em lại kéo thêm một cái màn hình vào. Cơ thể em đang vận động, nhưng đầu óc em vẫn bị buộc bằng một sợi dây rất ngắn.}{``No. You are doing both things more poorly. Exercise is when blood should move, breathing should deepen, and the mind should get a chance to step away from stimulation. But you drag a screen right into it. Your body is moving, while your mind remains tied by a very short leash.''}

Thầy nói tiếp: ``Nhiều người bây giờ không hề được nghỉ thật. Họ chỉ thay đổi tư thế trong khi vẫn tiếp tục tiêu thụ kích thích. Thế nên cơ thể có thể mệt hơn mà đầu óc vẫn không hề nhẹ hơn.''
\textit{(The teacher continues: ``Many people today never truly rest. They only change posture while continuing to consume stimulation. That is why the body may become more tired while the mind does not feel lighter at all.'')}
\end{truyen}

\section{Trong Toilet Cũng Không Buông}
\begin{truyen}{Trong Toilet Cũng Không Buông}{Still Not Letting Go Even in the Toilet}
\chuhoa{C}{ó học sinh thú nhận rằng vào toilet mà không cầm điện thoại thì thấy rất trống trải.}
\textit{(A student admits that entering the toilet without a phone feels strangely empty.)}

Đó là lúc lệ thuộc đã đi vào cả những chỗ riêng tư và cơ bản nhất.
\textit{(That is when dependence has entered even the most private and basic spaces.)}

\teacherpair{Nếu một người không còn chịu nổi vài phút rất riêng của thân thể mà không cần màn hình, thì họ không chỉ đang dùng công cụ. Họ đang bị công cụ bám vào cấu trúc đời sống.}{``If a person can no longer endure even a few very private bodily minutes without a screen, they are not merely using a tool. The tool has started attaching itself to the structure of life.''}
\end{truyen}

\section{Đi Qua Đường Mà Mắt Không Nhìn Đường}
\begin{truyen}{Đi Qua Đường Mà Mắt Không Nhìn Đường}{Crossing the Road Without Looking at the Road}
\chuhoa{C}{ó người vừa đi bộ qua cổng trường vừa cúi xuống nhắn tin, như thể xe cộ có trách nhiệm tránh mình.}
\textit{(Someone walks across the school gate while texting, as if traffic has the duty to avoid them.)}

Đây không còn là chuyện lịch sự hay học tập nữa.
\textit{(This is no longer merely about manners or study.)}

Nó là chuyện đầu óc bị kéo mạnh đến mức mất luôn bản năng tự bảo vệ.
\textit{(It is about the mind being pulled so strongly that even self-protection dulls.)}

\teacherpair{Có những kiểu phân tán không chỉ làm em học kém. Nó làm em nguy hiểm với chính mình. Một màn hình đáng sợ nhất là màn hình khiến mắt còn mở mà sự cảnh giác đã ngủ.}{``Some kinds of distraction do not just make you learn poorly. They make you dangerous to yourself. The most frightening screen is the one that leaves your eyes open while your alertness falls asleep.''}
\end{truyen}

\section{Tập Thể Dục Nhưng Vẫn Nuốt Rác}
\begin{truyen}{Tập Thể Dục Nhưng Vẫn Nuốt Rác}{Exercising While Still Swallowing Junk}
\chuhoa{C}{ó bạn chạy bộ trên máy nhưng mắt vẫn hút vào những đoạn video vụn vặt.}
\textit{(A student runs on a treadmill while the eyes keep absorbing trivial videos.)}

Cơ thể vận động, nhưng đầu óc vẫn tiêu thụ thứ làm nó mệt.
\textit{(The body moves, but the mind continues consuming what exhausts it.)}

\teacherpair{Tập thể dục tốt không chỉ vì cơ bắp. Nó còn vì đó là lúc hệ thần kinh được đổi môi trường. Nếu em vẫn nhét rác vào đầu trong suốt lúc tập, em đang bỏ phí một cơ hội hồi sức hiếm hoi.}{``Exercise is not good only for muscles. It also offers the nervous system a different environment. If you keep feeding junk into the mind while exercising, you waste a rare chance to recover.''}
\end{truyen}

\section{Đi Chơi Để Chụp Chứ Không Để Sống}
\begin{truyen}{Đi Chơi Để Chụp Chứ Không Để Sống}{Going Out to Capture Instead of to Live}
\chuhoa{C}{ó nhóm bạn đi dã ngoại nhưng điều bận nhất lại là chọn góc, chỉnh ảnh và đăng ngay trong lúc chuyến đi còn chưa thấm.}
\textit{(A group goes on an outing, but their main concern is angles, edits, and posting before the outing has even sunk in.)}

Khi đó trải nghiệm thật bị rút bớt để nuôi phiên bản trưng bày.
\textit{(Real experience is then reduced to feed its display version.)}

\teacherpair{Không phải mọi tấm ảnh đều xấu. Nhưng nếu em sống một nơi chỉ để nhanh chóng biến nó thành nội dung, em đã đổi mất phần ngon nhất của việc có mặt thật.}{``Not every photo is bad. But if you inhabit a place only to quickly turn it into content, you trade away the richest part of real presence.''}
\end{truyen}

\section{Chờ Bạn Năm Phút Cũng Không Rời Màn Hình}
\begin{truyen}{Chờ Bạn Năm Phút Cũng Không Rời Màn Hình}{Unable to Leave the Screen While Waiting for a Friend}
\chuhoa{C}{ó người đến điểm hẹn sớm vài phút nhưng chẳng quan sát xung quanh, chỉ cắm đầu vào điện thoại cho tới khi bạn vỗ vai.}
\textit{(Someone arrives at a meeting point early yet notices nothing around them, staying bent into the phone until a friend taps their shoulder.)}

Một thành phố, một sân trường, một buổi chiều đều có thể đi ngang qua đầu óc như không hề tồn tại.
\textit{(A city, a schoolyard, an afternoon can pass through the mind as if they never existed.)}

\teacherpair{Sống mà cứ để màn hình lấp mọi khoảng chờ thì em sẽ ngày càng ít thấy thế giới thực. Không phải vì thế giới nghèo đi, mà vì mắt mình bị thuê mất rồi.}{``If you let screens fill every waiting moment, you will see less and less of the real world. Not because the world has grown poorer, but because your eyes have already been rented out.''}
\end{truyen}

\section{Sáng Mở Mắt Là Mở Máy}
\begin{truyen}{Sáng Mở Mắt Là Mở Máy}{Opening the Phone the Moment You Wake}
\chuhoa{C}{ó học sinh chưa rửa mặt, chưa ngồi dậy hẳn đã với điện thoại để xem đêm qua có gì mới.}
\textit{(A student reaches for the phone before washing up or even fully sitting up, just to see what came in overnight.)}

Ngày mới vì thế không bắt đầu bằng ý thức của chính mình.
\textit{(The new day therefore does not begin with one's own awareness.)}

Nó bắt đầu bằng luồng kích thích từ bên ngoài.
\textit{(It begins with outside stimulation.)}

\teacherpair{Buổi sáng là lúc rất quý để đầu óc vào ngày mới cho sạch. Nếu em trao ngay khoảnh khắc đầu tiên đó cho màn hình, em đang để thuật toán viết nét đầu của ngày mình.}{``Morning is precious because it lets the mind enter the day cleanly. If you hand that first moment to the screen, you let algorithms write the opening line of your day.''}
\end{truyen}

\section{Muốn Thở Thật Phải Có Nơi Không Sóng}
\begin{truyen}{Muốn Thở Thật Phải Có Nơi Không Sóng}{To Breathe for Real, You Need Some Places Without Signal}
\chuhoa{C}{ó bạn bắt đầu để điện thoại ngoài cửa khi tập thể dục, ngoài bàn khi ăn, và xa giường khi ngủ.}
\textit{(A student begins leaving the phone outside the exercise area, off the dining table, and away from the bed at night.)}

Chỉ vài thay đổi như vậy đã làm đời sống đỡ bị xé nhỏ hẳn.
\textit{(Just those few changes already make life feel less fragmented.)}

\teacherpair{Con người không yếu đi khi thiếu sóng vài lúc. Nhiều khi họ mạnh lên. Vì cuối cùng cũng có vài nơi trong ngày mà thân thể và đầu óc không còn phải tranh nhau với một cái máy.}{``People do not grow weaker when they go without signal for a while. Often they grow stronger. At last there are a few places in the day where body and mind no longer have to compete with a device.''}
\end{truyen}

\begin{baihoc}
Có những hoạt động đáng lẽ phải cứu đầu óc khỏi màn hình.
\textit{(Some activities are supposed to rescue the mind from the screen.)}

Nếu em cũng nhét màn hình vào nốt những lúc đó, em đang tự cắt mất đường hồi sức của chính mình.
\textit{(If you also stuff the screen into those moments, you are cutting off your own path to recovery.)}
\end{baihoc}

\begin{tuhocnhanh}
1. Em có đang mang màn hình vào cả những lúc cơ thể cần được tự do không? \textit{(Are you bringing the screen even into moments when the body needs freedom?)}

2. Hoạt động nào trong ngày của em đáng lẽ nên không có điện thoại nhất? \textit{(Which daily activity of yours most deserves to be phone-free?)}

3. Em có còn nhớ cảm giác đi bộ mà đầu óc không bị kéo đi đâu nữa không? \textit{(Do you still remember what it feels like to walk without your mind being pulled away?)}
\end{tuhocnhanh}

\begin{tongketchuong}
Đem màn hình vào mọi nơi không phải là sống hiện đại hơn.
\textit{(Dragging the screen into every place is not a more modern life.)}

Nhiều khi đó chỉ là một kiểu lệ thuộc được bọc trong vẻ tiện lợi.
\textit{(Often it is only dependence wrapped in convenience.)}
\end{tongketchuong}
