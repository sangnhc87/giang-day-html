\chapter{Quyết Tâm Một Đêm}
\chapterintrobi{Có những cuộc cải tổ học tập chỉ kéo dài tới sáng hôm sau}{One-night determination}{Đây là thói quen rất rực rỡ trong khoảnh khắc và rất mong manh khi phải lặp lại.}

\begin{lucbatbox}{Lục bát: Đêm nay đổi đời}
Đêm nay em quyết thật nhiều
Sách vở xếp thẳng, mục tiêu viết đầy
Tưởng từ sáng sớm mai đây
Một con người mới dựng xây lâu bền
Ai ngờ ngủ dậy bình yên
Thói quen cũ lại bước lên chào mình
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Rực rồi tắt nhanh}
Một đêm lòng sáng rực niềm tin
Kế hoạch nhìn đâu cũng rất xinh
Nhưng đổi đời cần nhiều ngày nữa
Đâu chỉ nhờ vài phút hăng tình
\end{tutuyetbox}

\begin{ghichu}
Quyết tâm một đêm không phải vô ích. Nó chỉ cần được nối thành nhiều ngày thường bình tĩnh hơn thì mới thành nếp.
\end{ghichu}

\begin{englishnote}
\textbf{determination}: quyết tâm.\par
\textbf{overnight change}: thay đổi trong một đêm.\par
\textbf{Real change takes more than one night.}: Thay đổi thật cần nhiều hơn một đêm.
\end{englishnote}
