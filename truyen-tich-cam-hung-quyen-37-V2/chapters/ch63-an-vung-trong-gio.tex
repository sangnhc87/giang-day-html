\chapter{Ăn Vụng Trong Giờ}
\chapterintro{Mùi bánh tráng trộn có khả năng xuyên thấu mọi hàng rào kỷ luật.}{Mỗi miếng nhai là một lần tim đập thon thót khi thầy ngoảnh mặt lại.}

\begin{tutuyetbox}{Bữa tiệc gầm bàn}{Under-desk Feast}
Cúi mặt luồn tay mở bịch quà
Mùi thơm nồng nặc bốc gần xa
Thầy cô đảo mắt dòm lố bịch
Kiểm điểm cầm tay nhục thiết tha.
\tcblower
Bowing your head, opening the snack 
The scent is strong, there's no going back 
Teacher glares at the silly play 
Holding a report, ruined is the day.
\end{tutuyetbox}

\begin{tutuyetbox}{Kẻ trộm thời gian}{The Time Thief}
Mồm nhai chóp chép mắt dòm quanh
Cứ tưởng qua tay thật quỷ bành
Ngờ đâu phấn rớt ngay đầu gối
Bêu bảng ghi danh nhục nát khanh.
\tcblower
Chewing away, looking all around 
Thinking you're sneaky, safely bound 
But chalk falls right on your knee 
Shameful name, for all to see.
\end{tutuyetbox}

\begin{ghichu}
Ăn vụng không làm bạn no hơn, nó chỉ làm bạn giỏi hơn trong việc nhai kẹo cao su mà không cử động cơ hàm.
\end{ghichu}
