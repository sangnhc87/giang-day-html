\chapter{Học Lệch Và Môn Tủ}
\chapterintro{Thương môn mình thích là chuyện thường, bỏ luôn môn khác là chuyện đáng khịa}{Học lệch là bản sắc của nhiều học sinh, nhưng cũng là nguồn cười không nhỏ.}

\begin{lucbatbox}{Lục bát: Môn tủ rất sáng}
Môn em yêu quý vô cùng
Làm đâu đúng đó nói năng rất trơn
Đến môn không hợp linh hồn
Em như đổi hẳn một con người khác
Bạn bè nhìn cũng thấy lạ
Cùng một bộ não mà xa thế này
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Ngại nhẹ một môn}
Em bảo chỉ ngại môn kia thôi
Nhưng điểm về lại tụt xa rồi
Nếu đó còn xem là nhẹ nữa
Thì e mức độ quá siêu rồi
\end{tutuyetbox}

\begin{ghichu}
Ai cũng có môn hợp và môn không hợp, nhưng bỏ mặc môn yếu rồi cầu may mỗi kỳ thi là một niềm tin khá mạo hiểm.
\end{ghichu}

\begin{lucbatbox}{Lục bát: Yêu văn, sợ toán}
Văn em bay bổng như mây
Toán vừa xuất hiện chân tay rụng rời
Đề nhìn như thể chuyện đời
Mỗi dòng mỗi dấu gọi mời hoang mang
Em rằng chẳng phải em lười
Chỉ là tâm hồn hợp mùi chữ hơn
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Chọn lọc quá tay}
Em chọn môn thương để đầu tư\
Mấy môn còn lại để như sương\
Đến khi bảng điểm hiện cho đủ\
Mới hay đời chấm rất công bằng
\end{tutuyetbox}
