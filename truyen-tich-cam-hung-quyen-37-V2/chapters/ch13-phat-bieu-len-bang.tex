\chapter{Phát Biểu Và Lên Bảng}
\chapterintro{Có bạn nói rất to khi chưa bị gọi, và rất nhỏ khi bị gọi thật}{Khoảnh khắc lên bảng luôn làm lộ rõ mức chuẩn bị của nhân loại học trò.}

\begin{lucbatbox}{Lục bát: Ở dưới rất trơn}
Ở dưới em nói rất trơn
Đến khi cô gọi hóa hơn tượng đồng
Chân tay bỗng hóa mênh mông
Bảng đen rộng tựa cánh đồng tháng ba
Phấn cầm mà dạ xót xa
Biết đâu kiến thức vừa qua lối nào
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Gọi tên là đứng hình}
Thầy vừa đọc đúng tên em đó
Mặt em đổi sắc chỉ trong giây
Bạn bè khẽ hỏi rằng thuộc chưa
Em nghe chữ thuộc cũng run ngay
\end{tutuyetbox}

\begin{ghichu}
Khịa lên bảng là khịa đúng khoảnh khắc mà sự tự tin và sự chuẩn bị phải gặp nhau. Nhiều em chỉ mang theo vế đầu.
\end{ghichu}

\begin{lucbatbox}{Lục bát: Trả lời lệch hướng}
Em vừa đáp rất hăng say
Giọng nghe khí thế hơn tay dẫn chương
Chỉ riêng hướng đi của đường
Lệch bài một đoạn mà thương vô cùng
Nếu thêm đúng ý sau cùng
Câu kia hẳn đã ung dung vào điểm
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Xung phong trên miệng}
Khi chưa bị gọi em nhanh lắm
Khi lên thật lại hóa im re
Mới hay từ miệng sang tới bảng
Cần thêm một quãng bản lĩnh ghê
\end{tutuyetbox}
