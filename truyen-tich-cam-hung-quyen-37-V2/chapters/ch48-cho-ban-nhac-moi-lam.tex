\chapter{Chờ Bạn Nhắc Mới Làm}
\chapterintro{Chiến thuật ngồi im như tượng và dùng thần giao cách cảm kết nối với bàn trên.}{Ánh mắt của bạn 50\% nhìn vào giấy thi, 50\% quét sóng wifi từ bài của đứa bên cạnh.}

\begin{tutuyetbox}{Ký sinh trùng học đường}{Academic Parasite}
Ngồi lì ngạo nghễ đợi muôn phương
Cống hiến thì không mộng ỷ nhường
Mắt liếc sang bên chờ bố thí
Thời gian cạn kiệt khóc đêm trường.
\tcblower
Sitting proudly, waiting for a sign 
No effort made, thinking you're fine 
Glancing at others, begging for a bone 
Time runs out, you're all alone.
\end{tutuyetbox}

\begin{tutuyetbox}{Xúc tu đứt đoạn}{Broken Tentacles}
Nhấm nháp bờ môi mút cán chì
Dò la vở bạn nhích li kì
Thằng khôn đậy kín bôi nát bét
Rớt thẳng cầu ao nhục quá đi.
\tcblower
Biting your lip, sucking on the lead 
Creeping on notes, filling with dread 
The smart kid hides it, leaving you blind 
Falling straight down, leaving success behind.
\end{tutuyetbox}
