\chapter{Trực Nhật Lao Động}
\chapterintro{Trực nhật là cuộc chiến giữa cây chổi và tinh thần lười biếng.}{Nhất quét nhì lau, tam lẩn tứ trốn.}

\begin{tutuyetbox}{Kẻ lười đẩy việc}{The Lazy Shifter}
Lao động tới phiên mặt thở dài
Bao thằng lẩn khuất trốn ra ngoài
Cán sự gom tên nhờ giáo chủ
Vào lớp ôm sầu chửi mệt nhoài.
\tcblower
Duty day comes with a heavy sigh 
Friends disappear like clouds in the sky 
The monitor reports to the master's lair 
Back in class, you curse in despair.
\end{tutuyetbox}

\begin{tutuyetbox}{Gánh team thảm bại}{Failed Team Carrying}
Đứa đẩy cho tao đứa mảy may
Lớp học dơ tanh rác bẩn bừa
Thấy sếp vung roi biên sổ đỏ
Nhục nhã chung bầy khóc tựa mưa.
\tcblower
Everyone pushes the work to me 
A dirty classroom for all to see 
Teacher swings the cane and marks the book 
We all cry together with a shameful look.
\end{tutuyetbox}

\begin{ghichu}
Thà quét luôn cho rảnh nợ, còn hơn là đứng nhìn bạn làm rồi bị ghi tên trừ điểm hạnh kiểm tháng.
\end{ghichu}
