\chapter{Chọn Bạn, Chọn Môi Trường, Chọn Tương Lai}

\begin{leadbox}
Tuổi học trò hiếm khi nghĩ rằng những mối quan hệ hiện tại sẽ kéo dài cái bóng của chúng sang tương lai.
Nhưng thực tế là bạn bè, môi trường, và những lựa chọn rất nhỏ mỗi ngày đang âm thầm định hình con đường về sau.
\end{leadbox}

\storycard{91. Bạn Thân Nhưng Kéo Xuống}{Close Friends Who Pull You Down}{Tình bạn không thể chỉ đo bằng niềm vui trước mắt; nó phải được nhìn bằng ảnh hưởng lâu dài lên con người mình.}
\storycard{92. Chuyển Lớp Hay Ở Lại}{Transfer or Stay}{Có những môi trường không làm ta xấu đi ngay, nhưng cũng không cho ta cơ hội để trở nên tốt hơn.}
\storycard{93. Theo Phong Trào Hay Theo Giá Trị}{Trend or Value}{Người trẻ thường phải chọn giữa hòa nhập thật nhanh và giữ lại phần cốt lõi của nhân cách mình.}
\storycard{94. Lời Mời Đi Chơi Ngay Trước Kỳ Thi}{The Invitation Before Exams}{Không phải lúc nào nói không cũng dễ, nhưng đó là bài tập quan trọng của khả năng ưu tiên.}
\storycard{95. Chọn Người Để Học Hỏi}{Choosing a Role Model}{Một hình mẫu đúng có thể nâng người trẻ lên; một hình mẫu sai có thể bẻ lệch cả hệ tiêu chuẩn sống.}
\storycard{96. Lớp Học Mới, Cách Sống Mới}{A New Class, A New Self}{Nhiều học sinh chỉ thay đổi khi nhận ra mình có quyền viết lại cách người khác nhìn về mình.}
\storycard{97. Lời Rủ Bỏ Tiết}{Skipping Class Together}{Nhiều lựa chọn sai không nảy sinh trong cô độc, nên năng lực từ chối cũng phải được rèn từ sớm.}
\storycard{98. Điểm Rẽ Nửa Cuối Năm}{The End-of-Year Shortcut}{Lối tắt có thể làm nhẹ một buổi, nhưng thường để lại một thói quen nguy hiểm kéo dài rất lâu.}
\storycard{99. Chọn Nghề Vì Áp Lực Hay Vì Phù Hợp}{Career by Pressure or Fit}{Định hướng tương lai chỉ bền khi nó đi cùng năng lực, xu hướng và những giá trị mình thật sự muốn giữ.}
\storycard{100. Tương Lai Bắt Đầu Từ Một Tối Nay}{The Future Starts Tonight}{Những bước ngoặt lớn của đời người thường mở ra từ một quyết định nhỏ, được lặp lại đủ bền bỉ.}
