\chapter{Bạn bè, lôi kéo và tâm lý bầy đàn trong lớp}
\markboth{Bạn bè, lôi kéo và tâm lý bầy đàn trong lớp}{Bạn bè, lôi kéo và tâm lý bầy đàn trong lớp}

\section{Bạn Rủ Một Cái Là Não Xin Nghỉ}
\begin{enumerate}[leftmargin=*]
\item \hsqa{Bạn em rủ nên em đi luôn.}{Chân em đi theo, não em cũng giao khoán luôn à?}{My friends invited me, so I just went.}{Your feet followed. Did your brain get outsourced too?}
\item \hsqa{Tại cả nhóm làm nên em làm theo.}{Cả nhóm sai không tự động hóa đúng cho em.}{The whole group did it, so I followed.}{A group's mistake does not magically become right for you.}
\item \hsqa{Em không muốn bị lạc lõng.}{Hòa nhập không có nghĩa là nộp bộ não cho đám đông.}{I didn't want to feel left out.}{Belonging does not mean handing your brain to the crowd.}
\item \hsqa{Bạn em bảo không sao đâu.}{Bạn em đâu ký giùm hậu quả cho em.}{My friends said it would be fine.}{Your friends are not signing the consequences on your behalf.}
\item \hsqa{Em ngại từ chối tụi nó.}{Ngại một phút đỡ hơn hối một học kỳ.}{I felt awkward saying no.}{One awkward minute beats a whole semester of regret.}
\item \hsqa{Em sợ bị nói là chảnh.}{Tỉnh táo mà bị gọi chảnh còn đỡ hơn ngu mà được hòa đồng.}{I was afraid they'd call me arrogant.}{Being called arrogant for staying smart is better than being warmly foolish.}
\item \hsqa{Bạn bè phải có nhau chứ thầy.}{Có nhau để đi lên, không phải kéo nhau tụt cho đủ team.}{Friends should stick together, right?}{Yes. To rise together, not to drag each other down as a full team.}
\item \hsqa{Em tưởng nghỉ một bữa không sao.}{Một bữa của em thường mở cửa cho nhiều bữa khác vào theo.}{I thought skipping one class wouldn't matter.}{Your 'one time' often leaves the door open for many more.}
\item \hsqa{Tụi nó làm em nao núng.}{Nội lực yếu mà gặp đám đông thì dễ thành lá bay.}{They made me lose my resolve.}{Weak inner strength turns into a leaf in a crowd.}
\item \hsqa{Em chỉ đi theo cho vui thôi.}{Nhiều cái 'cho vui' của tuổi học trò tính phí rất lâu.}{I only joined for fun.}{A lot of student-age 'fun' sends invoices for years.}
\end{enumerate}

\section{Cười Theo Đám Đông Và Trả Giá Một Mình}
\begin{enumerate}[leftmargin=*,resume]
\item \hsqa{Cả lớp cười nên em cười theo.}{Miệng em cười theo được, hậu quả em lãnh riêng nhé.}{The whole class laughed, so I laughed too.}{Your mouth may follow the crowd. Your consequences arrive alone.}
\item \hsqa{Em chỉ hùa một chút thôi.}{Một chút ác từ đám đông vẫn đau như thường.}{I only joined in a little.}{A little cruelty from a crowd still hurts all the same.}
\item \hsqa{Em không nghĩ bạn đó buồn tới vậy.}{Không nghĩ trước là lỗi rất đông người thích phạm.}{I didn't think they'd feel that hurt.}{Failing to think first is a very popular mistake.}
\item \hsqa{Bạn em trêu trước mà.}{Người lớn là người biết chỗ nào nên dừng chứ không phải biết kể thứ tự.}{My friend started it first.}{Maturity is knowing where to stop, not merely who started it.}
\item \hsqa{Em tưởng ai cũng chịu được.}{Khả năng chịu đựng của người khác không phải đồ em được phép thử nghiệm.}{I thought everyone could handle it.}{Other people's tolerance is not your experiment kit.}
\item \hsqa{Em nói theo cho hòa không khí.}{Không khí vui mà lòng người nặng thì đó không phải vui.}{I just played along to keep the mood light.}{If the mood is light but someone's heart gets heavy, that's not real fun.}
\item \hsqa{Em không ác ý đâu.}{Không ác ý mà cứ làm đau đều đều thì cũng nên xem lại.}{I didn't mean any harm.}{If no harm is intended but harm happens repeatedly, review yourself.}
\item \hsqa{Em bị cuốn theo đám đông.}{Vậy em cần cột sống hơn là cần thêm lý do.}{I got swept up by the crowd.}{Then you need a spine more than another excuse.}
\item \hsqa{Em chỉ share lại cho vui.}{Chia sẻ nỗi nhục của người khác không phải tài lẻ đáng tự hào.}{I only reposted it for fun.}{Sharing someone else's humiliation is not a skill to be proud of.}
\item \hsqa{Em thấy mình cũng hơi quá.}{Biết quá rồi thì đừng để lần sau thành thói quen.}{I guess I did go a bit too far.}{If you know you crossed the line, don't let it grow into a habit.}
\end{enumerate}

\section{Bạn Tốt Không Phải Người Giúp Em Trốn Trách Nhiệm}
\begin{enumerate}[leftmargin=*,resume]
\item \hsqa{Bạn thân thì phải che cho nhau chứ.}{Che lỗi để sửa khác với che lỗi để nuôi nó béo.}{Best friends should cover for each other.}{Covering a mistake to repair it differs from feeding it until it grows fat.}
\item \hsqa{Em xin bạn làm hộ vì nó thương em.}{Thương kiểu đó là đang phụ em yếu thêm.}{My friend did it for me because they care.}{That kind of caring helps your weakness grow.}
\item \hsqa{Bạn em cho chép nên em chép.}{Có người đưa đường tắt không có nghĩa đường đó đỡ dở.}{My friend let me copy, so I copied.}{A shortcut offered does not automatically become a wise road.}
\item \hsqa{Em bênh bạn vì tình nghĩa.}{Tình nghĩa mà dắt nhau né đúng sai thì hơi méo.}{I defended my friend out of loyalty.}{Loyalty becomes crooked when it helps people dodge right and wrong.}
\item \hsqa{Em không nỡ để bạn bị phạt một mình.}{Vậy kéo bạn ra khỏi lỗi mới là tử tế, không phải nhảy vào cùng.}{I couldn't let my friend be punished alone.}{Kindness is pulling them out of the mistake, not jumping in beside them.}
\item \hsqa{Bạn em hiền nên em giúp gian lận chút.}{Hiền mà không có nguyên tắc thì thành hiền hại nhau.}{My friend is nice, so I helped them cheat a bit.}{Niceness without principles becomes dangerous softness.}
\item \hsqa{Không giúp thì mất bạn.}{Bạn dễ mất vì em giữ đúng thường không phải loại nên giữ.}{If I don't help, I'll lose the friendship.}{A friendship lost because you chose what is right is rarely worth preserving.}
\item \hsqa{Em nợ bạn nên phải nghe.}{Biết ơn không phải hợp đồng bán não trả góp.}{I owed my friend, so I had to listen.}{Gratitude is not an installment plan for selling off your brain.}
\item \hsqa{Bạn em buồn nên em nghỉ học đi với nó.}{Đi cùng bạn lúc khó tốt, miễn đừng kéo hai đứa cùng tụt.}{My friend was upset, so I skipped class to stay with them.}{Being there for a friend is good, as long as you don't drag both lives down.}
\item \hsqa{Em không muốn làm người xấu trong nhóm.}{Giữ đúng đôi lúc trông xấu với người đang quen sống dễ.}{I didn't want to be the bad guy in the group.}{Doing right sometimes looks bad to people addicted to convenience.}
\end{enumerate}

\section{Muốn Đi Xa Phải Biết Chọn Người Đi Gần}
\begin{enumerate}[leftmargin=*,resume]
\item \hsqa{Em nên chọn bạn kiểu nào?}{Bạn nào làm em đỡ tệ đi, không phải đỡ thấy tệ trong chốc lát.}{What kind of friends should I choose?}{Friends who make you less bad, not just less guilty for a moment.}
\item \hsqa{Em khó cưỡng khi cả đám rủ.}{Vậy tập mạnh trước với một lần từ chối nhỏ đi.}{I find it hard to resist when the whole group insists.}{Then train your strength with one small refusal first.}
\item \hsqa{Em sợ mình không đủ bản lĩnh.}{Bản lĩnh không rơi xuống đầu, nó được tập bằng mấy lần nói không.}{I'm afraid I'm not strong enough.}{Strength doesn't fall from the sky. It's trained through repeated no's.}
\item \hsqa{Em lỡ theo rồi, giờ quay đầu kịp không?}{Kịp. Miễn là quay thật chứ không chỉ quay trong lời nói.}{I already followed them. Is it too late to turn back?}{No. Provided the turn happens in action, not only in words.}
\item \hsqa{Em thấy khó sống khác đám đông.}{Khó là giá mở cửa cho đời đỡ tầm thường.}{It feels hard to live differently from the crowd.}{Hardness is often the price of not becoming ordinary.}
\item \hsqa{Em cần nhớ gì khi bị rủ rê?}{Người chịu hậu quả cuối cùng thường vẫn là em.}{What should I remember when people pressure me?}{The final person paying the consequences is usually still you.}
\item \hsqa{Thầy ơi em có thể tập từ đâu?}{Từ việc nhỏ: ngừng cười theo cái sai chỉ vì sợ lạc bầy.}{Where should I start practicing?}{With a small step: stop laughing along with wrongness out of fear of standing alone.}
\item \hsqa{Em muốn vừa có bạn vừa giữ mình.}{Đó mới là bài khó đáng học.}{I want to keep my friends and keep myself too.}{That is exactly the difficult lesson worth mastering.}
\item \hsqa{Em nên nhớ câu nào nhất?}{Đám đông cho em tiếng ồn, nguyên tắc cho em đường đi.}{Which line should I remember most?}{The crowd gives noise. Principles give direction.}
\item \hsqa{Thầy chốt một câu cuối đi.}{Bạn tốt không phải người dắt em đi trốn lỗi, mà là người dám kéo em về phía đúng.}{Give me one final line.}{A good friend does not guide you into hiding from mistakes, but pulls you back toward what is right.}
\end{enumerate}
