\chapter{Xin Gia Hạn Nộp Bài}
\chapterintrobi{Một lá đơn vô hình rất được học sinh sử dụng}{Ask for an extension}{Có những bạn chưa làm bài, nhưng bài phát biểu xin thêm thời gian thì lại rất trơn tru.}

\begin{lucbatbox}{Lục bát: Thầy cho em thêm ít hôm}
Bài kia em mới nghĩ xong\
Ý còn đang ở giữa dòng mơ thôi\
Nếu thầy nới nhẹ một hồi\
Em xin hứa sẽ sáng ngời hơn nay\
Nói thì tha thiết lắm thay\
Chỉ là gia hạn mấy ngày thành quen
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Đàm phán rất mềm}
Khi chưa làm kịp giọng em hiền\
Lý do nghe thật cũng ưu phiền\
Nếu viết bài giỏi bằng nửa thế
Trang nào cũng đã hóa thần tiên
\end{tutuyetbox}

\begin{ghichu}
Xin gia hạn một lần có thể là bất đắc dĩ. Xin gia hạn nhiều lần thì thường là phong cách sống.
\end{ghichu}

\begin{englishnote}
\textbf{extension}: gia hạn.\par
\textbf{turn in homework}: nộp bài.\par
\textbf{Could I have one more day?}: Em xin thêm một ngày được không ạ?
\end{englishnote}
