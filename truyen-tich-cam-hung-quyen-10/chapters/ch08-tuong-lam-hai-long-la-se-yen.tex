\section{Gật Đầu Cho Khỏi Mất Lòng}
\begin{truyen}{Gật Đầu Cho Khỏi Mất Lòng}{Saying Yes Just to Avoid Discomfort}
\chuhoa{C}{o nguoi tre rat so khong khi tro nen kho xu nen he ai nho gi cung gat dau.}
\textit{(Some young people fear tension so much that they say yes to almost every request.)}

Ban dau ho thay minh de thuong va de chiu trong mat moi nguoi.
\textit{(At first they feel easy to like and easy to be around.)}

Nhung cang nhan nhieu viec, cang hua nhieu dieu, ho cang song trong cam giac bi keo di khap noi.
\textit{(But the more they accept, the more they feel pulled in every direction.)}

Lam hai long nguoi khac bang cach bo mac gioi han cua minh thuong dan toi kiet suc, khong dan toi binh an.
\textit{(Pleasing others by abandoning your limits usually leads to exhaustion, not peace.)}

\end{truyen}

\begin{baihoc}
Khong phai cai gat dau nao cung la long tot. Co cai chi la noi so bi ghet.
\textit{(Not every yes is kindness. Some are simply the fear of being disliked.)}
\end{baihoc}

\begin{ghinhoanh}
\textbf{Short English to remember:}

Not every yes is kindness.

Some are simply the fear of being disliked.

\end{ghinhoanh}

\textbf{Câu hỏi tự học / Self-study questions}

\section{Cố Sống Đúng Ý Người Nhà}
\begin{truyen}{Cố Sống Đúng Ý Người Nhà}{Trying to Live Exactly as Family Wants}
\chuhoa{N}{hieu nguoi tre thuong gia dinh that, nhung roi dan dan song nhu dang thi de duoc chap thuan.}
\textit{(Many young people genuinely love their family, yet slowly begin living for approval.)}

Ho chon nganh, chon viec, chon cach song chi de bot nghe that vong tu nguoi than.
\textit{(They choose study, work, and lifestyle mainly to reduce disappointment from relatives.)}

Lam dung y nha trong mot thoi gian co the giu duoc su yen be ngoai, nhung lai lam ho xa minh dan.
\textit{(Following family wishes may preserve outer calm for a while, but can slowly separate them from themselves.)}

Hieu thao khong co nghia giao luon tay lai cuoc doi cho nguoi khac.
\textit{(Loving your family does not mean handing over the wheel of your life.)}

\end{truyen}

\begin{baihoc}
Song thay cho ky vong cua nguoi khac qua lau se lam minh can dan.
\textit{(Living too long inside other people’s expectations slowly empties you.)}
\end{baihoc}

\section{Tử Tế Đến Mức Tự Xóa Mình}
\begin{truyen}{Tử Tế Đến Mức Tự Xóa Mình}{Being So Nice That You Erase Yourself}
\chuhoa{C}{o nguoi luon nhuong loi, nhuong phan, nhuong thoi gian, va thay ngai neu phai giu dieu minh can.}
\textit{(Some people keep giving up their turn, time, and needs.)}

Su tu te cua ho lam ai cung de chiu, tru chinh ho.
\textit{(Their kindness makes everyone comfortable except themselves.)}

Lau ngay, ho khong con phan biet dau la cho di dep, dau la tu lam minh mo di.
\textit{(Over time, they stop seeing the line between generosity and self-erasure.)}

Mot nguoi khong giu phan chinh dang cua minh rat de bi quen tay bo qua.
\textit{(A person who never protects their rightful place is easy to overlook.)}

\end{truyen}

\begin{baihoc}
Ranh gioi khong lam em bot tu te. No giup long tot cua em ben hon.
\textit{(Boundaries do not make you less kind. They make your kindness sustainable.)}

\section{Sợ Bị Xem Là Ích Kỷ}
\begin{truyen}{Sợ Bị Xem Là Ích Kỷ}{Afraid of Being Seen as Selfish}

\textit{(Some people feel guilty for protecting time, resting, or defending personal choices.)}

Ho nham moi hanh dong tu bao ve voi ich ky.

Ich ky la chi biet minh. Ranh gioi la biet minh cung la mot con nguoi can duoc giu.
\textit{(Selfishness is caring only about yourself. Boundaries mean knowing you also need care.)}

\end{truyen}

Bao ve phan chinh dang cua minh la nen de em con cho di dung cach.

\section{Từ Chối Rồi Tự Cắn Rứt}
\begin{truyen}{Từ Chối Rồi Tự Cắn Rứt}{Saying No and Then Punishing Yourself}
\chuhoa{C}{o nguoi cu moi lan tu choi ai do la ve nha nghi lai rat lau va tu phan xet minh.}
\textit{(Some people say no once and then spend hours blaming themselves.)}

Ho tuong phai thay nhe long moi la tu choi dung.
\textit{(They think a correct refusal should feel completely easy.)}

Nhung cam giac can ray sau mot ranh gioi moi la rat binh thuong vi em dang boi lai thoi quen cu.
\textit{(But guilt after a new boundary is normal because you are undoing an old habit.)}

Neu vi can ray ma quay lai nhan het, em se khong bao gio hoc duoc cach giu minh.
\textit{(If guilt always sends you back, you never learn to protect yourself.)}

\end{truyen}

\begin{baihoc}
Noi khong dung khong phai luc nao cung de chiu, nhung no van co the dung.
\textit{(A correct no is not always comfortable, but it can still be right.)}
\end{baihoc}

\begin{ghinhoanh}
\textbf{Short English to remember:}

A correct no is not always comfortable,

but it can still be right.

\section{Im Lặng Để Giữ Hòa Khí}
\begin{truyen}{Im Lặng Để Giữ Hòa Khí}{Staying Silent to Preserve Peace}
\chuhoa{C}{o nguoi thay minh bi chen viec, bi noi qua, hay bi bo qua, nhung van im lang vi so khong khi se xau di.}
\textit{(Some people stay silent when treated unfairly because they fear tension.)}

Ho tuong im lang la cach giu hoa khi tot nhat.
\textit{(They think silence is the best way to keep peace.)}

Nhung im lang qua lau thuong de lai am uc va mot thoi quen bi xem nhe.
\textit{(But silence held too long creates resentment and a habit of being undervalued.)}

Mot cau noi ro rang som thuong nhe hon de den luc no thanh vet thuong lon.
\textit{(One clear sentence early is lighter than waiting until the hurt becomes large.)}

\end{truyen}

\begin{baihoc}
Hoa khi that khong duoc xay tren viec mot nguoi nuot het phan khong cong bang.
\textit{(Real peace is not built on one person swallowing all unfairness.)}
\end{baihoc}

\begin{ghinhoanh}
\textbf{Short English to remember:}

on one person swallowing all unfairness.

\section{Hẹn Ranh Giới Để Sau}
\begin{truyen}{Hẹn Ranh Giới Để Sau}{Delaying Boundaries Until Later}
\chuhoa{N}{hieu nguoi biet ro minh can noi chuyen nay, can dat gioi han kia, nhung cu noi de de sau cho de hon.}
\textit{(Many people know they need a boundary, but keep delaying it for a “better time”.)}

Ho tuong doi them chut nua se de noi hon.
\textit{(They think waiting will make it easier.)}

Nhung moi lan tri hoan, thoi quen cu cua nguoi khac lai duoc cung co them.
\textit{(But every delay strengthens the other person’s old habit.)}

Ranh gioi dat muon thuong kho hon nhieu vi luc do no khong con la loi de nghi nhe nua, ma da thanh mot viec chinh sai da tich lai.
\textit{(A late boundary is harder because by then it must correct a pattern, not just make a request.)}

\end{truyen}

\begin{baihoc}
Ranh gioi noi som thuong nhe hon ranh gioi de muon.
\textit{(Boundaries spoken early are usually lighter than boundaries spoken late.)}
\end{baihoc}

\begin{ghinhoanh}
\textbf{Short English to remember:}

Boundaries spoken early

are usually lighter than boundaries spoken late.

\section{Học Nói Không Sớm}
\begin{truyen}{Học Nói Không Sớm}{Learning to Say No Earlier}
\chuhoa{S}{au nhieu lan met va am uc, co nguoi tre tap noi mot cau ngan gon rang minh khong lam duoc viec do.}
\textit{(After many tired and resentful days, a young person learns to say one short sentence: I cannot do that.)}

Ban dau ho so cau noi do se lam minh xau di trong mat nguoi khac.
\textit{(At first they fear that sentence will make them look worse.)}

Nhung dieu xay ra la nhung ai ton trong minh van o lai, con nhung ai chi quen dung minh thi bat dau kho chiu.
\textit{(What happens is that respectful people stay, while users become uncomfortable.)}

Cau no ngan, nhung no mo ra mot doi song nhe hon va sach hon.
\textit{(The sentence is short, but it opens a lighter and cleaner life.)}

\end{truyen}

\begin{baihoc}
Noi khong la ky nang giu long tot song lau, khong phai dau hieu lanh lung.
\textit{(Saying no is a skill that helps kindness survive, not a sign of coldness.)}
\end{baihoc}

\begin{ghinhoanh}
\textbf{Short English to remember:}

Saying no is a skill

that helps kindness survive.
