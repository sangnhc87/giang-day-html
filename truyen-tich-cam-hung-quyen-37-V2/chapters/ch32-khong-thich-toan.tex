\chapter{Không Thích Môn Toán}
\chapterintro{Toán học là kẻ thù số một của những tâm hồn bay bổng.}{Nhìn những con số nhảy múa mà chỉ muốn nhảy lầu.}

\begin{tutuyetbox}{Nỗi sợ số học}{Numerophobia}
Toán Lý lời thầy tựa sấm vang
Đầu em bấn loạn chẳng thèm mang
Cộng trừ nhân chia sai bét nhè
Rớt đại đại học lệ hai hàng.
\tcblower
Math and Physics like thunder roar 
My brain is empty, I can't take more 
Add and multiply, it's all a mess 
Failing college, I must confess.
\end{tutuyetbox}

\begin{tutuyetbox}{Kẻ thù của những con số}{The Enemy of Numbers}
Nhìn bảng đen ngầu nghìn dãy số
Mắt hoa đầu nhức nát tâm hồn
Chẳng chịu luyện rèn qua thử thách
Điểm liệt mang về nhục bế môn.
\tcblower
Staring at numbers on the dark board 
Dizzy and sick, my soul is bored 
Refusing to practice, refusing to try 
A failing grade makes the spirit die.
\end{tutuyetbox}

\begin{ghichu}
Ghét Toán không giúp bạn giàu hơn, nhưng giỏi Toán chắc chắn giúp bạn đếm tiền nhanh hơn sau này.
\end{ghichu}
