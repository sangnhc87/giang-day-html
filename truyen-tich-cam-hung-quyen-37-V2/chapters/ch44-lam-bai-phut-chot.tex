\chapter{Làm Bài Phút Chót}
\chapterintrobi{Thời gian càng gấp, tim càng đập, tay càng run}{Last-minute homework}{Có bạn chỉ thật sự tin vào bản thân khi đồng hồ gần chạm vạch đỏ.}

\begin{lucbatbox}{Lục bát: Hai mươi phút hóa anh hùng}
Ba ngày em để rất êm\
Đến khi còn chút mới đem ra làm\
Mắt cay, tóc rối, lòng cam\
Tự nhiên chăm chỉ hơn ngàn buổi xưa\
Xong rồi em thở như vừa\
Thoát qua bão lớn nhưng chưa chừa lần sau
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Sát hạn mới sáng}
Đến phút chót rồi trí mới lên\
Tay em như có điện trong đêm\
Nếu đem khí thế dùng sớm chút
Bài kia đâu phải khóc từng phen
\end{tutuyetbox}

\begin{ghichu}
Làm bài phút chót tạo cảm giác mình bản lĩnh. Thực ra phần lớn chỉ là đang chạy đua với hậu quả.
\end{ghichu}

\begin{englishnote}
\textbf{last minute}: phút chót.\par
\textbf{deadline}: hạn nộp.\par
\textbf{I work fast under pressure.}: Em làm nhanh khi bị ép thời gian.
\end{englishnote}
