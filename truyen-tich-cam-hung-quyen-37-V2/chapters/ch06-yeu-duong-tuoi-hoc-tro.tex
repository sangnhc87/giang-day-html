\chapter{Yêu Đương Tuổi Học Trò}
\chapterintro{Tim rung một cái là thời khóa biểu cũng rung theo}{Tuổi học trò yêu rất vụng, nên vừa đáng khịa vừa đáng thương.}

\begin{lucbatbox}{Lục bát: Nhớ người hơn nhớ bài}
Ngồi trên mà mắt rất ngoan
Trong đầu lại nhớ một hàng tóc bay
Thầy cô giảng đến đoạn hay
Em còn bận đếm hôm nay bạn cười
Hỏi sao bài vở lơi bơi
Tại tim chen chỗ của nơi học hành
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Yêu đúng mùa thi}
Người ta ôn tập rất chuyên cần
Em thì chuyên nhớ mắt ai gần
Nếu mà tương tư đem đi chấm
Tên em hẳn đứng ở đầu sân
\end{tutuyetbox}

\begin{ghichu}
Yêu đương tuổi học trò đáng khịa nhất ở chỗ trí tưởng tượng thường học nhanh hơn miệng.
\end{ghichu}

\ngancach

\begin{lucbatbox}{Lục bát: Online rất đều}
Ban ngày gặp vẫn làm ngơ
Tối về nhắn chữ dày hơn vở bài
Biểu tượng tim gửi không sai
Bài tập thì lại còn dài phía sau
Giá mà chịu khó như nhau
Điểm kia có lẽ bớt sầu vài phen
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Cùng lớp mà xa}
Ngồi gần nhau suốt mấy dãy bàn
Nói câu nghiêm túc lại rất hiếm
Chào nhau chủ yếu bằng ánh mắt
Bồi hồi xong lại lặng im luôn
\end{tutuyetbox}
