\section{Kiến thức là tài sản}
\begin{truyen}{Kiến thức là tài sản}{Life lesson}
\chuhoa{A}{nh mất tiền có thể kiếm lại. Kiến thức đã học không ai lấy được. Kiến thức là tài sản.}
\textit{(He could earn back lost money. Knowledge learned can't be taken. Knowledge is an asset.)}
\end{truyen}

\begin{baihoc}
Đầu tư vào đầu óc.
\textit{(Invest in your mind.)}
\end{baihoc}

\begin{ghinhoanh}
\textbf{Short English to remember:}

Invest in your mind.

\end{ghinhoanh}
