\chapter{Học Theo Mood}
\chapterintrobi{Tinh thần lên thì học, tinh thần xuống thì đợi cảm hứng}{Study by mood}{Đây là kiểu học rất giàu cảm xúc và rất nghèo tính ổn định.}

\begin{lucbatbox}{Lục bát: Hứng lên mới giỏi}
Hôm nào trời đẹp như thơ
Em ngồi học bỗng bất ngờ rất chăm
Hôm nào mây xám ghé thăm
Trang kia chưa mở đã nằm thở than
Nếu mà việc học chờ mong
Mood đi ngang chắc tương lai muộn rồi
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Cảm hứng làm thầy}
Em học tùy theo gió lòng mình
Hôm vui chữ nghĩa chảy thành sông
Mai buồn sách vở thành xa lạ
Kiến thức nhìn theo cũng chạnh lòng
\end{tutuyetbox}

\begin{ghichu}
Học theo mood không sai hoàn toàn, nhưng nếu mood làm hiệu trưởng thì lịch học sẽ rất hỗn loạn.
\end{ghichu}

\begin{englishnote}
\textbf{study by mood}: học theo cảm hứng.\par
\textbf{consistent}: đều đặn.\par
\textbf{I only study when I feel like it.}: Em chỉ học khi có hứng.
\end{englishnote}
