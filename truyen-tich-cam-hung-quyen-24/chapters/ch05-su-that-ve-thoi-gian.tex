\section{Thời gian trôi rất nhanh}
\begin{truyen}{Thời gian trôi rất nhanh}{Life lesson}
\chuhoa{A}{nh còn nhớ ngày vào công ty. Giật mình đã mười năm. Thời gian trôi rất nhanh.}
\textit{(He still remembers his first day at the company. Suddenly ten years had passed. Time flies.)}
\end{truyen}

\begin{baihoc}
Đừng để thời gian trôi mà không sống.
\textit{(Don't let time pass without living.)}
\end{baihoc}

\begin{ghinhoanh}
\textbf{Short English to remember:}

Don't let time pass without living.

\end{ghinhoanh}
