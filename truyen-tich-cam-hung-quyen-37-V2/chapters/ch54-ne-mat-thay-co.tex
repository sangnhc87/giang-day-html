\chapter{Né Mắt Thầy Cô}
\chapterintrobi{Mắt cúi xuống là một chiến thuật cổ điển}{Avoid eye contact with the teacher}{Nhiều bạn tin rằng chỉ cần không nhìn lên thì mình sẽ an toàn trước mọi câu hỏi.}

\begin{lucbatbox}{Lục bát: Cúi xuống là bình an}
Thầy vừa dừng giữa câu thôi\
Em liền nghiên cứu góc ngồi phía dưới\
Vở dù không có chữ tươi\
Nhìn chăm cứ ngỡ người người ham ghi\
Mắt em tránh được tức thì\
Chỉ là câu hỏi bay đi đâu nào
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Chiến thuật cúi đầu}
Cô nhìn quanh lớp, mắt em nghiêng\
Vờ như đang rất mực chăm chuyên\
Nếu né câu hỏi là môn chính khóa\
Em hẳn đầu bảng đã lâu liền
\end{tutuyetbox}

\begin{ghichu}
Né mắt thầy cô đôi khi cứu được một lần lên bảng, nhưng không cứu nổi phần kiến thức mình đang né cùng lúc.
\end{ghichu}

\begin{englishnote}
\textbf{avoid eye contact}: né nhìn thẳng.\par
\textbf{call on a student}: gọi học sinh trả lời.\par
\textbf{Please do not call me.}: Xin đừng gọi em.
\end{englishnote}
