\chapter{Rung Động Đầu Sân Trường}
\chapterintro{Giọng mở đầu}{Những bài đầu giữ chất áo trắng, nụ cười, mưa hiên, giờ ra chơi và các màn trêu nhau rất nhẹ. Toàn chương đi theo nhịp ngũ ngôn thống nhất để cảm xúc gọn, vui và sáng.}

\poem{Bài 01}{Ngũ ngôn tứ tuyệt}{Em ngang qua bàn rồi}{Tôi đỏ nguyên tai rồi}{Bạn cùng bàn cười khẽ}{Bảo khỏi chối, mê rồi}
\poem{Bài 02}{Ngũ ngôn tứ tuyệt}{Áo trắng chạm rèm mây}{Tim tôi trượt một nhịp}{Em quay đầu hỏi vậy}{Học gì nhìn tôi đây}
\poem{Bài 03}{Ngũ ngôn tứ tuyệt}{Chuông reo giờ ra chơi}{Em cười như gió mới}{Tôi vờ đi lấy nước}{Thật ra để nhìn thôi}
\poem{Bài 04}{Ngũ ngôn tứ tuyệt}{Mưa gõ ngoài song lớp}{Em nghiêng ô gọi với}{Tôi đi chậm ba bước}{Để gần thêm một thôi}
\poem{Bài 05}{Ngũ ngôn tứ tuyệt}{Thầy ghi đầy bảng trắng}{Tôi ghi đầy tên em}{Em quay sang mượn tẩy}{Làm tim tôi mềm mềm}
\poem{Bài 06}{Ngũ ngôn tứ tuyệt}{Bút rơi ngay chân ghế}{Em cúi nhặt giùm tôi}{Cả bàn trên bàn dưới}{Nhìn tôi cười hết rồi}
\poem{Bài 07}{Ngũ ngôn tứ tuyệt}{Căng tin thơm bánh mới}{Em bẻ nửa cho tôi}{Tôi ăn mà nghe ngọt}{Hơn cả kẹo đầu môi}
\poem{Bài 08}{Ngũ ngôn tứ tuyệt}{Sân cờ đông như hội}{Tôi đứng cạnh hàng em}{Quốc kỳ bay phía trước}{Tim bay lệch phía em}
\poem{Bài 09}{Ngũ ngôn tứ tuyệt}{Trời chang chang tiết thể}{Em buộc tóc bên sân}{Tôi chạy chưa xong vòng}{Mặt hồng hơn nắng sân}
\poem{Bài 10}{Ngũ ngôn tứ tuyệt}{Lưu bút vừa mở ra}{Em viết dòng hơi ngắn}{Tôi đọc đi đọc lại}{Dài hơn cả nghỉ trưa}
\poem{Bài 11}{Ngũ ngôn tứ tuyệt}{Thư viện rèm xanh mát}{Em ngồi sát ôn văn}{Tôi lật trang địa lý}{Mà lòng trôi sang em}
\poem{Bài 12}{Ngũ ngôn tứ tuyệt}{Bài văn tả bạn tốt}{Tôi tả nụ cười em}{Thầy đọc chưa hết đoạn}{Cả lớp đã hiểu thêm}
\poem{Bài 13}{Ngũ ngôn tứ tuyệt}{Đường về qua cây bàng}{Em dắt xe chầm chậm}{Tôi dắt mơ bên cạnh}{Gặp ổ gà vẫn êm}
\poem{Bài 14}{Ngũ ngôn tứ tuyệt}{Đêm ôn thi buồn ngủ}{Em nhắn nhớ ngủ sớm}{Tôi định nghe cho khỏe}{Mà vui quá thức thêm}
\poem{Bài 15}{Ngũ ngôn tứ tuyệt}{Góc lớp treo ảnh mới}{Em đứng đầu hàng hai}{Tôi đứng xa cuối mép}{Vẫn thấy trời xanh hoài}
\poem{Bài 16}{Ngũ ngôn tứ tuyệt}{Phấn bay qua vai áo}{Em phủi giúp rất nhanh}{Tay chạm chưa đầy giây}{Tim tôi loạn cả trang}
\poem{Bài 17}{Ngũ ngôn tứ tuyệt}{Bạn bè hay ghẹo quá}{Cứ gọi tôi tên em}{Tôi ngoài môi xin chớ}{Trong lòng lại thích thêm}
\poem{Bài 18}{Ngũ ngôn tứ tuyệt}{Có lần em hỏi nhỏ}{Sao cứ né nhìn hoài}{Tôi bảo sợ điểm kém}{Chứ đâu dám khai hoài}
\poem{Bài 19}{Ngũ ngôn tứ tuyệt}{Áo mưa em treo ghế}{Cả lớp thơm mùi chiều}{Tôi ngoan nguyên tiết toán}{Chỉ để ngồi gần thôi}
\poem{Bài 20}{Ngũ ngôn tứ tuyệt}{Hè vừa hong phượng đỏ}{Tim vừa biết lơ ngơ}{Em cười câu tạm biệt}{Làm tháng năm thành thơ}
