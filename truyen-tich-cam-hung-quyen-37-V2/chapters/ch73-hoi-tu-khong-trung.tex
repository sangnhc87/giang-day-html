\chapter{Học Tủ Lệch Đề}
\chapterintro{Cuộc đời là một canh bạc, và học tủ là lúc bạn đặt cược tương lai vào một mớ giấy lộn.}{Trúng tủ là tiên, lệch đề là hố.}

\begin{tutuyetbox}{Canh bạc cuối cùng}{The Final Gamble}
Một mớ đề thi nấp góc gầm
Em thề học tủ luyện tay thâm
Vô phòng đề mục trôi xa thẳm
Khóc hận nghìn thu nhục tối nầm.
\tcblower
A pile of questions under the bed 
I swore to study what's in my head 
The exam paper drifts far away 
Crying forever, a dark, sad day.
\end{tutuyetbox}

\begin{tutuyetbox}{Sự phản bội của may mắn}{The Betrayal of Luck}
Tưởng trúng phóc pho mộng ngất ngây
Tiêu điều rớt thảm lệ trào vây
Thần linh ngoảnh mặt dòm lố bịch
Điểm liệt nằm chờ nhục bủa vây.
\tcblower
Thinking I hit the jackpot prize 
Total failure before my eyes 
The gods turn away, seeing a fool 
Failing grades, the nightmare of school.
\end{tutuyetbox}

\begin{ghichu}
Tủ có thể làm bạn an tâm trước giờ thi, nhưng kết quả thi sẽ là cái hòm chôn vùi niềm tin của bạn.
\end{ghichu}
