\section{Bảng Điểm Đẹp, Bản Thân Rỗng}
\begin{truyen}{Bảng Điểm Đẹp, Bản Thân Rỗng}{A Nice Transcript, But an Empty Working Self}
\chuhoa{C}{o ban tre ra truong voi bang diem dep va mot bo ho so rat sach.}
\textit{(Some students graduate with a nice transcript and a very clean file.)}

Ban ay tuong ket qua hoc tap dep se du de nguoi ta mo cua cho minh.
\textit{(They think good academic results alone are enough to open doors.)}

Phong van xong moi thay nha tuyen dung can dau vet cong viec, cach trinh bay, va dau hieu lam duoc viec that.
\textit{(After the interview, they see that employers want work traces, clear speaking, and proof of real ability.)}

Luc do ban ay moi hieu hoc gioi la mot nen tang tot, nhung no khong the di thay cho nang luc xuat hien trong doi that.
\textit{(Then they understand that good grades are a good base, but they cannot replace real-world presence.)}

\end{truyen}

\begin{baihoc}
Hoc gioi rat quy. Nhung muon di xa, em con phai cho doi thay em lam duoc gi.
\textit{(Studying well matters. But to go far, you must also let life see what you can do.)}
\end{baihoc}

\begin{ghinhoanh}
\textbf{Short English to remember:}

Studying well matters.

But to go far, you must also let life see what you can do.

\end{ghinhoanh}

\textbf{Câu hỏi tự học / Self-study questions}

Nguoi gioi ma khong truyen dat duoc thuong de mat diem ngay khi can hop tac.
\textit{(A capable person who cannot communicate often loses points at the moment teamwork begins.)}

\end{truyen}

\begin{baihoc}
Gioi khong chi la hieu. Gioi con la lam cho nguoi khac hieu duoc minh.

\textit{(A hard piece of feedback can become a major advantage if you correct yourself early.)}

\begin{baihoc}
Hoc de lam ra mot dau vet nho con hon hoc nhieu ma khong de lai gi.
\textit{(Learning for one small real output is better than learning a lot without leaving any trace.)}
