\chapter{Đọc Đề Lướt}
\chapterintro{Làm xong bài sớm nhất phòng thi, ra tới cổng mới nhớ ra đề yêu cầu tính chu vi chứ không phải diện tích.}{Đọc lướt như cơn gió, điểm trượt như bão tuôn.}

\begin{tutuyetbox}{Thiểu năng đọc chữ}{Reading Blindness}
Cầm đề đọc lướt nhảy ngàn sông
Cứ tưởng thời oai nổ trống đồng
Đánh vội khoanh lầm sai bé bét
Nhìn ra đứt đoạn khóc sưng tròng.
\tcblower
Reading the question, jumping through trees 
Thinking you're a hero, living with ease 
Marking in haste, wrong as can be 
Seeing the truth, a tragedy for me.
\end{tutuyetbox}

\begin{tutuyetbox}{Bút sa gà chết}{Pen Down, Fate Sealed}
Thoáng lướt vài giây bút vạch liền
Điền bao đáp án ngỡ phong tiên
Đâu hay cạm bẫy bày xảo quyệt
Khóc hận phòng thi nát não phiền.
\tcblower
Glancing for seconds, pen leaves a trail 
Filling in answers, thinking you'll sail 
Unaware of the traps hidden inside 
Crying in regret, nowhere to hide.
\end{tutuyetbox}

\begin{ghichu}
Sai một ly đi một dặm, đọc nhầm một chữ điểm trừ về không. Tốc độ làm bài càng cao chứng tỏ độ lười đọc đề càng đáng sợ.
\end{ghichu}
