\chapter{Chon ban, chon moi truong, chon tuong lai}

\begin{leadbox}
Tuoi hoc tro hiem khi nghi rang moi quan he hien tai se ve theo tuong lai. Nhung su that la ban be, moi truong va cac lua chon nho moi ngay dang am tham dinh hinh con duong sau nay.
\end{leadbox}

\storycard{91. Ban than nhung keo xuong}{Close Friends Who Pull You Down}{Tinh ban can duoc do bang anh huong lau dai, khong chi bang su vui ve tuc thoi.}
\storycard{92. Chuyen lop hay o lai}{Transfer or Stay}{Moi truong co the cuu mot hoc sinh khi no cho em co co hoi lam lai nhan dang cua minh.}
\storycard{93. Theo phong trao hay theo gia tri}{Trend or Value}{Nguoi tre thuong phai chon giua viec duoc hoa nhap ngay va su ben vung cua nhan cach.}
\storycard{94. Loi moi di choi ngay truoc ky thi}{The Invitation Before Exams}{Khong phai luc nao tu choi cung de, nhung day la bai tap quan trong cua nang luc uu tien.}
\storycard{95. Chon nguoi de hoc hoi}{Choosing a Role Model}{Than tuong dung co the nang mot nguoi len; than tuong sai co the doi ca he tieu chuan song.}
\storycard{96. Lop hoc moi, cach song moi}{A New Class, A New Self}{Nhieu hoc sinh chi thay doi khi biet rang minh co quyen viet lai cach nguoi khac nhin ve minh.}
\storycard{97. Loi ru bo tiet}{Skipping Class Together}{Nhieu quyet dinh xau khong duoc dua ra mot minh, vi vay kha nang noi khong can duoc ren tu som.}
\storycard{98. Diem re nua cuoi nam}{The End-of-Year Shortcut}{Lua chon loi tat co the lam nhe mot buoi, nhung de lai no thoi quen nguy hiem cho ca doi sau.}
\storycard{99. Chon nghe vi ap luc hay vi phu hop}{Career by Pressure or Fit}{Dinh huong tuong lai dung phai ket hop nang luc, xu huong va nhung gia tri muon giu.}
\storycard{100. Tuong lai bat dau tu mot toi nay}{The Future Starts Tonight}{Moi buoc ngoat lon thuong bat dau bang mot quyet dinh nho, duoc lap lai deu dan.}
