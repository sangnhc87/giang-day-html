\chapter{Bạn Cùng Bàn Năm Ấy}
\chapterintro{Thanh xuân ở chiếc bàn gỗ nhỏ}{Cùng một mặt bàn mà đủ chuyện: chia bánh, mượn gôm, nhắc bài, che nhau ngủ gật rồi cuối cùng phải tạm biệt cái chỗ ngồi thân nhất đời học sinh.}

\poem{Bài 21}{Bài 21 - Chia nửa ổ bánh mì}{Half a bun can taste bigger than the whole canteen.}{Bạn cùng bàn năm ấy\\Hay bẻ bánh chia làm đôi\\Tôi ăn mà nghe rất rõ\\Mùi vui thơm suốt một hồi}

\poem{Bài 22}{Bài 22 - Mượn gôm rồi cãi nhau}{Desk mates fight hardest over the tiniest eraser.}{Bạn cùng bàn năm ấy\\Mượn gôm xong giả luôn rồi\\Tôi đòi, cậu lườm một cái\\Chiều về lại nói chuyện thôi}

\poem{Bài 23}{Bài 23 - Che nhau ngủ gật}{True friendship is blocking the teacher's view for a sleepy friend.}{Bạn cùng bàn năm ấy\\Ngủ gật giữa tiết văn dài\\Tôi ngồi nghiêng vai che bớt\\Cả hai cùng sợ cô nhìn}

\poem{Bài 24}{Bài 24 - Lên bảng thì nhắc nhỏ}{A whispered reminder can feel like rescue from the heavens.}{Bạn cùng bàn năm ấy\\Lên bảng run thấy thương ghê\\Tôi ho một cái thật khẽ\\Đủ cho cậu nhớ đường về}

\poem{Bài 25}{Bài 25 - Bàn trống ngày chia tay}{An empty desk can be louder than a whole classroom.}{Bạn cùng bàn năm ấy\\Ngày xa bàn trống đến lạ\\Tôi ngồi nghe chiều rất chậm\\Bụi phấn cũng buồn theo ta}
