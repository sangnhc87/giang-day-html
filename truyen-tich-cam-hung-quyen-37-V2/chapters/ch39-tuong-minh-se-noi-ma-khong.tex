\chapter{Tưởng Mình Sẽ Nổi Mà Không}
\chapterintro{Tự sướng là tốt, nhưng ảo tưởng sức mạnh thì dễ đâm vào ngõ cụt.}{Làm màu làm mè cho cố vào, kết quả trượt vẫn là trượt.}

\begin{tutuyetbox}{Hào quang ảo giác}{Optical Illusion Glory}
Uốn éo làm duyên hất hất cằm
Ngỡ đâu thiên hạ ngó xa xăm
Dòng vô sổ điểm lòi ngu dốt
Hạng bét toàn trường khóc tối rằm.
\tcblower
Posing and acting with a haughty chin 
Thinking everyone wants to win 
Looking at grades shows you're a fool 
Ranked at the bottom of the school.
\end{tutuyetbox}

\begin{tutuyetbox}{Kẻ bám đuôi}{The Tagalong}
Hoang tưởng tài năng vút đỉnh thiêng
Bao phen vấp ngã chẳng muộn phiền
Môn thi rớt thảm nhìn lố bịch
Đứng tựa người khờ rớt giá danh.
\tcblower
Imagining talent on a holy peak 
Failing often, yet you still seek 
Tests are a disaster, looking absurd 
Standing like a fool, not saying a word.
\end{tutuyetbox}

\end{ghichu}
