\chapter{Giả Vờ Ghi Bài}
\chapterintrobi{Tay chạy đều nhưng ý không vào đầu}{Pretend to take notes}{Một số bạn rất giỏi tạo hình ảnh chăm học trước khi thực sự chăm học.}

\begin{lucbatbox}{Lục bát: Bút đi còn não đứng}
Tay em lia bút rất mềm\
Nhìn xa ai tưởng chăm chuyên lắm rồi\
Lại gần mới biết than ôi\
Nét kia quanh quẩn như trôi dọc dòng\
Chữ chưa chắc đã thành công\
Nhưng vai diễn học hành trông mượt mà
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Chăm chỉ bề ngoài}
Em ghi như thể hiểu sâu ghê\
Trang nào cũng kín những hàng rê\
Đến lúc hỏi lại điều vừa giảng tới\
Mới hay ngòi bút viết thay nghe
\end{tutuyetbox}

\begin{ghichu}
Giả vờ ghi bài làm người khác yên tâm, nhưng không làm kiến thức tự chảy vào mình.
\end{ghichu}

\begin{englishnote}
\textbf{take notes}: ghi chép.\par
\textbf{pretend}: giả vờ.\par
\textbf{I wrote it down, but I did not get it.}: Em có ghi lại nhưng chưa hiểu.
\end{englishnote}
