\chapter{Che Điểm Bằng Tay}
\chapterintro{Một bàn tay che lấp cả bầu trời, à không, che lấp một con số có hình dạng đáng sợ.}{Điểm chưa lật lên thì hi vọng vẫn còn sừng sững.}

\begin{tutuyetbox}{Hé lộ kinh hoàng}{Horrifying Reveal}
Tay run vuốt mép giấy thi về
Hé lộ màu son nét não nề
Trồi sụt ba con không chót vót
Gục đầu bật khóc hận trăm bề.
\tcblower
Shaking hands unfolding the sheet 
Red ink marks, a bitter defeat 
Three points only, nothing to show 
Crying in pain, now everywhere I go.
\end{tutuyetbox}

\begin{tutuyetbox}{Đánh lừa thị giác}{Optical Deception}
Che giấu bằng tay chẳng mảy may
Điểm ba lòi tẹt bỗng phơi bày
Ban nãy hô to tài xuất chúng
Giờ nhìn rệu rã khóc chua cay.
\tcblower
Hiding with hands, but it's no use 
A zero appears, no more excuse 
Just now boasting of talent and fame 
Now looking broken, in total shame.
\end{tutuyetbox}

\begin{ghichu}
Dù bạn có che điểm bằng 10 ngón tay đi chăng nữa, thì nó vẫn nằm nguyên trong quyển sổ cái của thầy giáo.
\end{ghichu}
