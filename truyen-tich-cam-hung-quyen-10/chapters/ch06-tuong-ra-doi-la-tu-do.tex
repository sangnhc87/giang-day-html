\section{Lương Về Rồi Hóa Đơn Đến}
\begin{truyen}{Lương Về Rồi Hóa Đơn Đến}{Salary Arrives and Bills Arrive Too}
\chuhoa{N}{gay linh luong dau hay cuoi thang thuong lam nguoi tre thay minh co ve tu do hon.}
\textit{(Payday often makes young people feel more free.)}

Ho tuong co tien rieng la tu dong co quyen song nhe va rong hon.
\textit{(They think having personal income automatically creates easy and wide freedom.)}

Nhung tien vua ve thi tien nha, tien an, di lai, va nhieu khoan khac cung xep hang di vao.
\textit{(But as soon as money comes in, rent, food, transport, and many other costs line up to claim it.)}

Luc do em moi thay tu do cua nguoi lon luon di kem bang tinh.
\textit{(Then you see that adult freedom always walks with arithmetic.)}

\end{truyen}

\begin{baihoc}
Tu do khong den tu viec co tien. No den tu viec biet quan no.
\textit{(Freedom does not come from money alone. It comes from knowing how to manage it.)}
\end{baihoc}

\begin{ghinhoanh}
\textbf{Short English to remember:}

Freedom does not come from money alone.

It comes from knowing how to manage it.

\end{ghinhoanh}
