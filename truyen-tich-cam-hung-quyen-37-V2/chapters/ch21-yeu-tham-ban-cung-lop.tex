\chapter{Yêu Thầm Bạn Cùng Lớp}
\chapterintro{Ngồi gần nhau cả năm mà nói chuyện nghiêm túc chưa được mấy câu}{Yêu thầm cùng lớp là môn học không có đáp án mẫu.}

\begin{lucbatbox}{Lục bát: Nhìn bảng nhưng nhớ người}
Ngồi trên nhìn bảng rất ngoan\
Mà trong tầm mắt lại toàn bóng ai\
Thầy cô giảng đến ý dài\
Em còn bận nhớ một vài nụ cười\
Hỏi sao bài vở lơi bơi\
Tại tim chen chỗ của nơi học hành
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Cùng lớp mà xa}
Ngồi gần nhau suốt cả năm rồi
Nói câu nghiêm túc vẫn quá ít
Chào nhau chủ yếu bằng ánh mắt
Tình chăm thế mà miệng chơi vơi
\end{tutuyetbox}

\begin{ghichu}
Yêu thầm bạn cùng lớp đáng khịa nhất ở chỗ trí tưởng tượng luôn chạy nhanh hơn thực tế.
\end{ghichu}

\ngancach

\begin{lucbatbox}{Lục bát: Nhớ chỗ ngồi hơn nhớ bài}
Bạn kia nghỉ học một ngày\
Góc bàn bên cạnh tựa gây bão lòng\
Tiết nào em cũng ngó trông\
Như đang chờ một cơn hồng quay ngang\
Giá mà nhớ bài dễ dàng\
Được như nhớ ghế, đời sang biết bao
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Tên trong vở nháp}
Vở nháp hôm nay lạ thật ghê
Tên người ta viết kín lề thôi
Nếu công thức được chăm ngang vậy
Điểm em hẳn đã sáng lên rồi
\end{tutuyetbox}
