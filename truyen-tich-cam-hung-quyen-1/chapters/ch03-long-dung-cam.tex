\chapter{Lòng Dũng Cảm (Courage)}

\begin{center}
  \textit{\color{truyenblue}``Courage is not the absence of fear, but the triumph over it.''}\\
  \textit{(Dũng cảm không phải là không e sợ, mà là chiến thắng nỗi sợ đó.)}\\[4pt]
  \small\color{truyengold}--- Nelson Mandela ---
\end{center}

  \textbf{Courage} is taking the first step. Even when you \textbf{are scared}, you \textbf{choose} to move forward.
  \\ \textit{(Lòng dũng cảm là việc bạn dám cất bước đi đầu tiên. Ngay cả khi kinh hãi sợ sệt, bạn vẫn kiên định lựa chọn tiến về phía trước.)}
\end{tcolorbox}

\ngayDay{}
\section{Đôi Cánh}
\begin{truyen}{Đôi Cánh}{Động vật}
\chuhoa{A}{ } \textbf{young eagle stayed} in his \textbf{nest}.\\
\textit{(Một chú đại bàng non nớt cứ nằm bẹp lỳ trong tổ của mình.)}

He \textbf{looked down} and \textbf{was terrified} of the \textbf{height}.\\
\textit{(Nó nhìn xuống dưới vực thẳm và trở nên vô cùng khiếp sợ trước độ cao ấy.)}

His \textbf{mother pushed him} out gently.\\
\textit{(Đại bàng mẹ đã nhẹ nhàng đẩy con mình rơi ra khỏi mép tổ.)}

He \textbf{flapped} his \textbf{wings} and \textbf{realized} he \textbf{could fly}.\\
\textit{(Cậu ta luống cuống vỗ đập đôi cánh và chợt hoảng hốt nhận ra rằng bản thân mình đã có thể bay lượn trên trời cao.)}
\end{truyen}

\begin{baihoc}
  \textbf{You never know what you can do until you try.} \textit{(Bạn sẽ không bao giờ biết được thần lực thực thụ của mình nếu bạn chưa từng chịu thử.)}
\end{baihoc}

\ngayDay{}
\section{Che Chở}
\begin{truyen}{Che Chở}{Tuổi thơ}
\chuhoa{A}{ } \textbf{storm hit} the \textbf{small village} hard.\\
\textit{(Một cơn bão lớn dữ dội ập vùi xuống một ngôi làng xập xệ.)}

A \textbf{little girl walked} home \textbf{alone}.\\
\textit{(Một cô bé lang thang lầm lũi đi bộ về nhà một mình trên cung đường.)}

She \textbf{used her tiny umbrella} to \textbf{protect a wet dog}.\\
\textit{(Cô bé đã dùng chiếc ô bé nhỏ xíu xiu của mình để che chở dẫu cho một con chó hoang đang ướt sũng.)}

She \textbf{was brave} against the \textbf{thunder}.\\
\textit{(Cô đã quá đỗi vô cùng can đảm dũng mãnh trước điệu sấm chớp ầm ĩ kia.)}
\end{truyen}

\begin{baihoc}
  \textbf{Compassion breeds courage.} \textit{(Lòng trắc ẩn sẽ sản sinh và dung dưỡng sự can đảm vô hạn.)}
\end{baihoc}

\vspace{0.8cm}

\ngayDay{}
\section{Đối Mặt}
\begin{truyen}{Đối Mặt}{Thiên nhiên}
\chuhoa{T}{he} \textbf{wind tried} to \textbf{push the river backwards}.\\
\textit{(Ngọn gió hung hãn đã ra sức cố gắng đẩy ép dòng sông chảy ngược lại thềm thác.)}

The \textbf{river refused} to \textbf{stop flowing}.\\
\textit{(Thế nhưng mặt sông vẫn kiên định từ chối ngừng lại dòng chảy êm đềm của mình.)}

It \textbf{carved} a \textbf{path through solid rock}.\\
\textit{(Nó cứ bào mòn từ từ từng rãnh đi xuyên sâu vào qua một khối đá cứng nhắc.)}

It \textbf{proved} its \textbf{unstoppable power}.\\
\textit{(Dòng sông đã chứng tỏ sành sói được sức mạnh không thể ngăn cản từ chính bản thân.)}
\end{truyen}

\begin{baihoc}
  \textbf{Persistence is a brave choice.} \textit{(Sự bền bỉ kiên trì mãi mãi là một sự bấu víu lựa chọn dũng cảm.)}
\end{baihoc}

\ngayDay{}
\section{Trái Tim Thực}
\begin{truyen}{Trái Tim Thực}{Cổ tích}
\chuhoa{A}{ } \textbf{wooden toy knight lacked} a \textbf{sword}.\\
\textit{(Một chàng kỵ sĩ tượng bằng gỗ bị khuyết thiếu đi một thứ vô cùng quan trọng: thanh kiếm.)}

A \textbf{monster toy bullied} the \textbf{other dolls}.\\
\textit{(Một gã đồ chơi quái vật đáng sợ thường xuyên bắt nạt các búp bê khác tủi hờn.)}

The \textbf{knight stood} in \textbf{front} to \textbf{protect} them.\\
\textit{(Thế là chàng kỵ sĩ dẫu cụt ngủn vẫn nghênh ngang đứng chắn phía trước ngực để bảo vệ đàn em thơ.)}

His \textbf{bravery frightened} the \textbf{monster away}.\\
\textit{(Tấm lòng dũng cảm can trường của chàng ta đã khiến tên quái vật kinh hồn và tự tháo chạy trốn đi mất đất.)}
\end{truyen}

\begin{baihoc}
  \textbf{Bravery is defending the weak.} \textit{(Sự dũng cảm cốt yếu vẫn là ý thức bảo vệ trọn vẹn được những kẻ yếu đuối hơn ta.)}
\end{baihoc}

\ngayDay{}
\section{Mở Ra Cơ Hội}
\begin{truyen}{Mở Ra Cơ Hội}{Câu chuyện}
\chuhoa{E}{veryone} \textbf{was terrified} of the \textbf{old heavy door}.\\
\textit{(Mọi người đều vô cùng kinh sợ vỡ mật trước một cánh cửa cũ kỹ nặng nề bí hiểm nọ.)}

They \textbf{thought} a \textbf{ghost lived behind} it.\\
\textit{(Họ phỏng đoán điên cuồng rằng có hẳn một bóng ma ngự trị rình rập ở sâu ngay sau nó.)}

A \textbf{child opened} it \textbf{slowly}.\\
\textit{(Một đứa trẻ lên ba đã mạnh dạn tiến đến và từ tốn đẩy hé mở bừng nó ra một tí.)}

There \textbf{was only a garden} full of \textbf{sunlight} inside.\\
\textit{(Nào ngờ đằng sau cánh cửa đó chỉ là một lối vườn tược tràn ngập những làn ánh sáng mặt trời lộng lẫy vào đến tận bên trong.)}
\end{truyen}

\begin{baihoc}
  \textbf{Fear creates imaginary ghosts.} \textit{(Nỗi lo sợ hãi vốn dĩ là thứ sẽ tự dưng đẻ tạo ra những bóng ma hư ảo giả mạo lừa mị tâm tư.)}
\end{baihoc}

\ngayDay{}
\section{Tiến Bức Chậm Rãi}
\begin{truyen}{Tiến Bức Chậm Rãi}{Động vật}
\chuhoa{A}{ } \textbf{turtle needed} to \textbf{cross} a \textbf{busy busy road}.\\
\textit{(Có một bác rùa biển nhọc nhằn cần trườn để qua mặt một con đường tấp nập phương tiện.)}

\textbf{Cars zoomed fast} from \textbf{every direction}.\\
\textit{(Những chiếc xe ô tô khổng lồ lao vun vút thần tốc phóng xé gió từ đủ mọi các hướng đi.)}

He \textbf{did not retreat} into his \textbf{shell}.\\
\textit{(Bác rùa già khú ấy đã không hề co thụt nhát gan đầu lùi rụt màng vào bên trong mai ruồi dẫu nguy hiểm.}

He \textbf{crawled slowly} until he \textbf{was safe}.\\
\textit{(Bác ta cứ lết bò di hành chậm rãi nhưng vô cùng rắn rỏi qua cho đến khi qua thềm bờ cập bến yên bình vô sự.)}
\end{truyen}

\begin{baihoc}
  \textbf{Courage does not always roar.} \textit{(Lòng can đảm chưa hẳn đôi khi bắt buộc phải đi kèm với một tiếng tru thét gầm vang vọng.)}
\end{baihoc}

\ngayDay{}
\section{Chịu Đựng}
\begin{truyen}{Chịu Đựng}{Tự nhiên}
\chuhoa{A}{ } \textbf{grain of sand entered} an \textbf{oyster shell}.\\
\textit{(Có một hạt vụn cát phù sa cọ xát bị lộn lọt thóp xông bay vào sâu trong thân lớp một con hàu già cằn cỗi.)}

It \textbf{suffered extreme pain} in the \textbf{dark tightness}.\\
\textit{(Hạt cát bị bắt nhốt vào kẹp nén gồng mình gánh chịu cảm giác đau đớn kịch liệt tồi tệ trong một khoảng chân không tối tăm quánh chặt chội ngục tù.)}

It \textbf{bravely endured} the \textbf{pain for years}.\\
\textit{(Dù ngục hằn tủi đắng, hạt ấy đã vô cùng anh dũng nhẫn nhịn và chịu đựng kiên nhẫn vượt qua được sự tấy đau chà xát đọa đày vô độ suốt hàng tá dòng năm dài dằng dặc.)}

Finally, it \textbf{became} a \textbf{stunning shiny pearl}.\\
\textit{(Sau rốt hồi kết, chính nó đã được kết tinh cấu tạo tỏa hóa trở thành một viên trân ngọc trai to chiếu long lanh đắt giá sáng rực rỡ huy hoàng.)}
\end{truyen}

\begin{baihoc}
  \textbf{Endurance is a quiet courage.} \textit{(Sự nhẫn nhục nhẫn nại bền chí vẫn luôn thầm kín chính xác là một kiểu lòng kiên định dũng cảm tĩnh mịch thầm lặng thiêng liêng nhất.)}
\end{baihoc}

\section{Hy Sinh}
\begin{truyen}{Hy Sinh}{Đời sống}
\chuhoa{T}{he} \textbf{room was completely dark} and \textbf{scary}.\\
\textit{(Không gian căn phòng đã bị chìm chìm đắm hoàn toàn bao vây quất mù không khí dở bóng đen khịt và vô vàn đe nạt đáng sợ.)}

A \textbf{tiny candle decided} to \textbf{light up}.\\
\textit{(Một cây nến nhỏ xíu gầy còm đã tự quyết định sẽ lự thân mình châm bén lửa mà đốt sáng lung linh bừng cháy lên.)}

It \textbf{knew} it \textbf{would melt away}.\\
\textit{(Bản thân nó thừa tột hiểu tường tận một hiện tượng rằng mình rốt cục sau cùng sẽ bị tan chảy nung tan thành từng sáp nước nhão bỏng biến mất tan vào hư vô cát bụi.)}

But \textbf{it smiled} as it \textbf{banished the dark}.\\
\textit{(Thế nhưng đổi lấy nó lại khẽ khàng mỉm môi cười mãn nguyện hãnh diện tươi tắn rạng ngời khoảnh khắc kiêu hãnh vì phần công tự thân đã vừa xó tan đày lui dẹp đánh gục mọi tăm tối muộn màng.)}
\end{truyen}

\begin{baihoc}
  \textbf{Courage lights up the world.} \textit{(Dũng khí quả cảm chính là một tia quang đuốc chiếu rọi bật thắp sáng xua tan bầu quỷ đem lại ánh sinh thế giới.)}
\end{baihoc}

\section{Không Ngại Thử Thách}
\begin{truyen}{Không Ngại Thử Thách}{Phiêu lưu}
\chuhoa{A}{ } \textbf{young sheep chose} to \textbf{climb} a \textbf{mountain}.\\
\textit{(Một cô bé cừu măng non tơ trẻ nhí đã mạo mẩn tự dưng rinh bạo chọn lấy một thử thách đầy gay cấn khó xơi vượt trèo leo thoăn thoắt dốc trên cao ngạo cả một ngọn núi non sừng sững trùng điệp.)}

The \textbf{old sheep told} her it \textbf{was dangerous}.\\
\textit{(Mấy bác lũ làng bầy đàn cừu già khú đểu gạc lố đã nhao nhao thi nhau căn dặn gắt gao xúm cảnh cáo đính chính với nó là chốn ấy vô chừng quá khốn ngập ngụa đầy rẫy hiểm nguy bẫy treo tàn độc cắc cớ.)}

She \textbf{faced wild wolves} and \textbf{harsh snow}.\\
\textit{(Cừu non liễu yếu phải cật lực trơ chịu ngửa nghênh tiếp mặt chiến đấu dạn dĩ vất vả cực nhọc với sói bầy lang hoang dại và một bầu lượng không trung vờn ngợp tuyết đông sương lạnh tê buốt giá thấu xương băng hàn.)}

She \textbf{reached the peak} and \textbf{saw the clouds}.\\
\textit{(Sau cùng rồi chung quy cừu dũng mãnh cũng rướn được mạn đạt đặt chân mình đóng dập bước đỉnh cao tít dốc chạm ngoạn mãn nhãn ngắm nghía tận mắt thấy các tảng áng mây phù phiêu du.)}
\end{truyen}

\begin{baihoc}
  \textbf{Great views require tough paths.} \textit{(Để chiêm ngưỡng được những khung quang cảnh sắc tuyệt đỉnh thi vị hoa lệ mỹ cảnh nhất bắt nài buộc bạn phải đi dẫu có tản dốc cày nát chịu vô cùng đường ải khắc nghiệt sỏi sần xóc.)}
\end{baihoc}

\ngayDay{}
\section{Nhận Lỗi}
\begin{truyen}{Nhận Lỗi}{Bạn bè}
\chuhoa{J}{ack} \textbf{accidentally broke} his \textbf{friend's toy}.\\
\textit{(Jack đã xớ rớ vụng bịch lỡ tay tay xun chệch bất ý sơ sảy làm đập làm rơi vỡ hỏng lộn nhào tanh bành tan nát nhoeét món đồ món chơi trò đồ nhôm đúc yêu quý độc đắt cưng sủng xót của người bạn thân ruột mến thiết báu nhất.)}

He \textbf{could have lied}.\\
\textit{(Đứa trẻ cậu thanh ấy bấy giờ có đủ dư khả giả vờ thơn lẻo điêu điêu có mường tưởng nói trí trá mỉm xấc xạo bịp giả chối bay chối biến vấy bỏ che liếm đổ cho tội người khác đổ trói điêu.)}

Instead, he \textbf{confessed} and \textbf{said sorry}.\\
\textit{(Song ở một sự thay thế hoán cự tuyệt đường ngang lẽ trái gạt trái, cậu quyết đứng oai thú tự vạch tỏ bạch ngay khai tội và mở lời thành tâm thủ thì buông miệng rủ tiếng xin lỗi ngỏ tâm lề xin được tha tự nhận mót thứ.)}

It \textbf{takes courage} to \textbf{admit mistakes}.\\
\textit{(Đó quả là lấy một sự gom ráng đặng tốn hít can lấy hết lòng trác chí vô vàn một dũng lớn mãnh quả quyết trượng khí mới thực sự có tự thừa nhận thẳng các sai lỗi sai lầm vô trật tồi mà mình đã làm chắp lỗi.)}
\end{truyen}

\begin{baihoc}
  \textbf{Honesty is an act of bravery.} \textit{(Một tâm thế trân mộc thật thà thực chất cũng chính là một mặt hành cử nhân nghĩa biểu động biểu tượng hiện một biểu hiện cao đẹp nhất đại dũng nhất cho lòng dũng khí anh hùng can trường thực đức anh hào.)}
\end{baihoc}
