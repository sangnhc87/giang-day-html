\chapter{Xin Vía Bạn Giỏi}
\chapterintrobi{Khi ôn bài không đủ yên tâm, nhiều bạn tìm tới năng lượng may mắn}{Ask for good luck from the top student}{Xin vía cho vui thì được, xin vía thay học thì hơi khó hiệu nghiệm.}

\begin{lucbatbox}{Lục bát: Chạm vai lấy hên}
Mai thi em gặp bạn hiền\
Xin cho chút vía học liền vào con\
Bạn cười rất mực vẫn non\
Kiến thức đâu phải hương thơm lan đều\
Học thì vẫn phải tự nhiều\
Vía kia chỉ giúp tinh thần bớt run
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: May mắn có giới hạn}
Em xin vía bạn giỏi đầu hàng\
Mong mai vào thi trí mở mang\
Nếu phần chăm chỉ truyền qua tiếp xúc\
Trường này hẳn đã quá huy hoàng
\end{tutuyetbox}

\begin{ghichu}
Xin vía bạn giỏi có thể giúp mình vui hơn. Điểm số vẫn thường thích đi cùng việc học thật hơn là nghi thức tâm linh học đường.
\end{ghichu}

\begin{englishnote}
\textbf{good luck}: may mắn.\par
\textbf{top student}: bạn học giỏi.\par
\textbf{Please lend me some luck.}: Cho mình xin ít may mắn nhé.
\end{englishnote}
