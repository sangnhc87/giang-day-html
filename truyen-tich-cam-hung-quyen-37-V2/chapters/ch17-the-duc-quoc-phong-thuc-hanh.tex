\chapter{Thể Dục Quốc Phòng Thực Hành}
\chapterintro{Ngồi máy lạnh quen rồi, ra sân nắng tự nhiên muốn ngất xỉu.}{Chưa chạy vòng nào đã râm ran đòi về y tế.}

% --- BÀI 131: TRỐN CHẠY MỆT MỎI (Sửa gắt) ---
\begin{tutuyetbox}{Trốn chạy mệt mỏi}{Lazy Escape}
Môn này chạy nhảy đứt lòi hơi
Tiết tới mò ra rũ rượi rời
Báo ốm nằm òm mưu xó khuất
Mùa thi rớt sạch oán than trời.
\tcblower
Running and jumping, out of breath 
Approaching the field like a date with death 
Faking an illness, hiding in the dark 
Failing the course, you lose your spark.
\end{tutuyetbox}

% --- BÀI 132: ĐỘI NẮNG THAO TRƯỜNG (Nâng cấp) ---
\begin{tutuyetbox}{Đội nắng thao trường}{Under the Hot Sun}
Cởi mũ xa hiên trút gió tà
Cầm cây súng gỗ rớt kêu la
Chuồn nhanh dưới bóng nằm che lấp
Đến lúc hô danh nhục đắng xa.
\tcblower
Take off the hat, embrace the sun 
Drop the wooden gun, thinking it's fun 
Sneak to the shade, hidden from sight 
When names are called, it's a shameful night.
\end{tutuyetbox}

% --- BÀI 133: CÔNG TỬ BỘT ---
\begin{tutuyetbox}{Công tử bột}{Soft Boys}
Hào hoa phong nhã mặt hiền khô
Ra nắng vài giây thở tựa bò
Bốn phía bạn nhìn cười nắc nẻ
Đàn ông yếu đuối nhục như hồ (nhục như hố).
\tcblower
Fancy and handsome, a gentle soul 
A few seconds of sun, you lose control 
Breathing like cattle, friends start to jeer 
Weakness in men is a shameful tear.
\end{tutuyetbox}

% --- BÀI 134: LĂN LÊ BÒ TRƯỜI ---
\begin{tutuyetbox}{Bò trườn hay bò lết}{Crawling Disaster}
Lăn lê lết bết dưới bùn sâu
Súng giả trên vai nặng nhức đầu
Cố sức vươn mình ra đích đến
Thầy kêu: "Dốt quá... cút đi đâu?"
\tcblower
Crawling in mud, a messy sight 
Fake gun on shoulder, losing the fight 
Struggling forward to reach the goal 
Teacher yells: "Stupid! You have no soul!"
\end{tutuyetbox}

% --- BÀI 135: THÁNH XÀ ĐƠN ---
\begin{tutuyetbox}{Treo mình trên giá}{Hanging on the Bar}
Gồng mình nhấc xác bám xà đơn
Mặt đỏ môi thâm uốn dập dờn
Nhảy xuống chưa đầy một cái rưỡi
Thầy ghi: "Bết bát... quá lươn khươn!"
\tcblower
Pulling your weight with a heavy sigh 
Turning dark red, about to cry 
Falling after one pull-up, so weak 
Teacher writes: "Failed! You're a total geek!"
\end{tutuyetbox}

% --- BÀI 136: CHIẾN THẦN "CHẠY BỀN" ---
\begin{tutuyetbox}{Chạy hay lết}{Running or Dragging}
Vừa vung đôi cẳng chạy vòng quanh
Hơi thở hổn hển mặt chẳng lành
Nửa chặng nằm ườn ra bãi cỏ
Phụ hồ vẫy gọi... nát đời xanh.
\tcblower
Swinging your legs to run the track 
Gasping for air, no turning back 
Halfway through, you collapse on the grass 
Labor is waiting, you've failed the class.
\end{tutuyetbox}

% --- BÀI 137: THÁO LẮP SÚNG ---
\begin{tutuyetbox}{Vụng thối vụng nát}{Clumsy Hands}
Cầm súng lóng ngóng rớt lò xo
Lắp mãi không xong mặt tũn tò
Giám thị đi ngang lắc đầu ngán
"Thôi về nhổ cỏ... chớ có lo!"
\tcblower
Fumbling the gun, the spring falls out 
Can't put it back, you want to shout 
The proctor sighs and shakes his head 
"Go pull some weeds, you're better off dead."
\end{tutuyetbox}

% --- BÀI 138: NHẢY XA BẤT HỦ ---
\begin{tutuyetbox}{Nhảy vào hố cát}{Jumping into the Pit}
Vươn mình lấy đà nhảy thật xa
Ngờ đâu vấp ngã mặt như hoa (bấy bá)
Cát bụi đầy mồm nhai lạo xạo
Bạn bè hô hố... sướng hằng hà.
\tcblower
Taking a run-up to jump so wide 
But falling face-first, nowhere to hide 
Mouth full of sand, a crunching treat 
Friends laugh out loud at your defeat.
\end{tutuyetbox}

% --- BÀI 139: TRỐN TIẾT Y TẾ ---
\begin{tutuyetbox}{Giả bệnh}{The Faking Sick}
Giả vờ đau bụng chạy vào trong
Máy lạnh nằm rung thật sướng lòng
Nào ngờ giáo vụ đi tuần tới
Mời luôn phụ huynh... hết đợi mong.
\tcblower
Pretending to hurt, you run inside 
Cool AC room, a place to hide 
But the staff walks in and finds your play 
Calling your parents, a dark, dark day.
\end{tutuyetbox}

% --- BÀI 140: CHỐT HẠ THỂ DỤC ---
\begin{tutuyetbox}{Lời khuyên cuối}{The Last Advice}
Học chữ cho nhiều dáng chẳng xong
Cơ bắp nhão nhẹt giống con ròng (con loăng quăng)
Ra đời gặp nạn ai mà cứu
Nhục nhã nghìn năm... lệ chảy dòng.
\tcblower
Smart in the head, but weak in the frame 
Muscles like jelly, a life of shame 
Who will save you when trouble is near? 
A thousand years of regret and a tear.
\end{tutuyetbox}

\end{ghichu}
