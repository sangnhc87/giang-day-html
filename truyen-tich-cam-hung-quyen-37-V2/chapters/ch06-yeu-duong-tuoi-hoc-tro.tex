\chapter{Yêu Đương Tuổi Học Trò (Tiếp theo)}
\chapterintro{Khi con tim bắt đầu đập loạn nhịp, thì các công thức toán học cũng bắt đầu rủ nhau đi trốn.}{Tình yêu tuổi học trò đẹp nhất là khi hai đứa cùng nắm tay nhau... đi lên văn phòng vì tội trốn học.}

\begin{lucbatbox}{Lục bát: Tình yêu hình học}
Tình em như toán hình không
Vẽ ra thì đẹp mà lòng rối tơ
Góc nhìn em cứ lờ đờ
Cạnh huyền chẳng thấy, chỉ chờ cạnh ai
Mai này thi cử đường dài
Đừng đem "định lý tình sai" vào bài.
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Sin, Cos tương tư}
Sin nằm Cos đứng mải tương tư
Chữ nghĩa thầy trao bỗng hóa thừa
Chỉ thấy bóng người trong ẩn số
Điểm mười giờ đã quá xa đưa.
\end{tutuyetbox}

\ngancach

\begin{lucbatbox}{Lục bát: Hẹn hò quán nước}
Hẹn nhau quán nước sau trường
Trà chanh tám chuyện quên đường về luôn
Vở bài để mặc sầu tuôn
Cặp đôi cùng tiến... về nguồn số không
Tình yêu có thắm như hồng
Cũng không giúp được bài thi xong lòng.
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Nhắn tin thâu đêm}
Thức suốt đêm thâu nhắn nhủ lời
Lên trường lại ngủ mặc đời trôi
Yêu đương mà chẳng lo đèn sách
Hối hận sau này lệ mãi rơi.
\end{tutuyetbox}

\ngancach

\begin{lucbatbox}{Lục bát: Nháp tên người thương}
Trang bài chỉ thấy nháp tên
Công thức chẳng nhớ, lại quên cả đề
Yêu vào cái mặt thấy ghê
Lờ đờ ngơ ngác như mê ngủ ngày
Tỉnh mau kẻo trễ chuyến này
Tương lai không thể dựa vay tình nồng.
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Valentine mùa thi}
Sô-cô-la ngọt nếm môi cười
Môn Toán nhìn vào thấy rã rời
Nếu tặng điểm mười thay đóa thắm
Thì đời em đã thực vui tươi.
\end{tutuyetbox}

\ngancach

\begin{lucbatbox}{Lục bát: Ngồi chung một bàn}
Ngồi chung một dãy bàn dài
Mắt nhìn lên bảng, ý hoài để đâu
Bên này bạn ấy gãi đầu
Bên kia em cũng âu sầu nhìn sang
Yêu đương sao quá vội vàng
Bỏ quên con chữ ngỡ ngàng nằm trơ.
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Chép bài cho nhau}
Thương nhau em chép hộ bài này
Nét chữ dường như cũng ngất ngây
Thầy hỏi tại sao mà giống hệt
Hai đứa nhìn nhau mặt đỏ lây.
\end{tutuyetbox}

\ngancach

\begin{lucbatbox}{Lục bát: Chia tay mùa thi}
Vừa mới chia tay tối qua
Sáng nay lên lớp như hoa héo sầu
Lệ rơi ướt cả trang đầu
Bài thi mặc kệ, u sầu bủa vây
Thôi đừng than thở kiểu này
Yêu mà lụy thế, đắng cay đủ điều.
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Tình yêu gà bông}
Gà bông chớm nở tuổi trăng tròn
Học hành chẳng chịu giữ lòng son
Đến khi bảng điểm buồn chi lạ
Mới biết tình yêu chẳng nuôi con.
\end{tutuyetbox}

\begin{lucbatbox}{Lục bát: Thư tình kẹp vở}
Thư tình em viết rất hay

Mà sao điểm Văn nụ cười
Lại buồn héo hắt, mười mươi không thành
Lo mà chăm chỉ cho nhanh
Kẻo sau hối hận tóc xanh bạc màu.
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Hội những người ế}
Bạn bè đôi lứa bước chung đường
Ta đứng nhìn theo dạ vấn vương
Nào ngờ điểm số ta cao ngất
Mới biết cô đơn thực phi thường.
\end{tutuyetbox}

\begin{lucbatbox}{Lục bát: Crush là ẩn số}
Tìm X thì thấy khó ghê
Tìm tên người ấy mê mê cả ngày
X kia chẳng chịu cầm tay
Người kia lại khiến lòng say lòng nồng
Tính đi tính lại bằng không
Tương lai mịt mịt như sông không bờ.
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Tình yêu và Logarit}

\end{tutuyetbox}

\begin{lucbatbox}{Lục bát: Giờ kiểm tra định mệnh}
Ngồi gần người ấy thật vui
Làm bài thì lại ngậm ngùi cả hai
Hỏi nhau đáp án chẳng sai
Cùng sai một chỗ dông dài như nhau
Thầy nhìn thầy chẳng nói đâu
Ghi vào sổ điểm một câu... buồn cười.
\end{lucbatbox}

Điểm thi chót vót ở... hàng ngang.
\end{tutuyetbox}

\begin{lucbatbox}{Lục bát: Mộng làm cô dâu}
Học hành em thấy quá lâu
Chỉ mong nhanh chóng làm dâu nhà người
Mới mười tám tuổi xuân tươi
Cái bằng chưa có, lấy đời chi nuôi
Gắng mà học tập cho xuôi
Kẻo sau hối hận ngậm ngùi tấm thân.

\end{tutuyetbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Tình yêu hàm số}
