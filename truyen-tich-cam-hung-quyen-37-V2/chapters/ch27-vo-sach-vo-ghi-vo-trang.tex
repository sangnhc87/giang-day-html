\chapter{Vở Sạch Và Chữ Đẹp}
\chapterintro{Chăm chú nắn nót ghi tên tiêu đề, chép được 5 dòng là bắt đầu xé nháp.}{Đầu năm tự hứa chữ đẹp rạng ngời, cuối năm nhìn lại ngỡ quỷ thần vẽ bùa.}

% --- BÀI 311: ĐẦU NON (Sửa gắt) ---
\begin{tutuyetbox}{Đầu non}{The New Beginning}
Đầu năm sắm sửa vở thơm nồng
Gắn nhãn ghi tên dán nốt hồng
Đứt tháng chia tay vùi sọt rác
Trò lười giấu vở ngủ say nồng.
\tcblower
New year's books, smelling so sweet 
Names and labels, looking quite neat 
A month passes, tossed in the bin 
Hiding the notebook, let sleep begin.
\end{tutuyetbox}

% --- BÀI 312: GHI BÀI TUNG HỎA MÙ (Nâng cấp) ---
\begin{tutuyetbox}{Ghi bài tung hỏa mù}{Smokescreen Notes}
Tới lớp tay không vứt cặp lồng
Thầy cô chép mỏi đứt xiềng còng
Mình ta ngó bảng cười lố bịch
Cuối khóa ra trường rỗng não nồng.
\tcblower
Empty-handed, lunchbox cast aside 
Teacher's writing, nowhere to hide 
Staring at the board, acting like a troll 
Leaving the school with an empty soul.
\end{tutuyetbox}

% --- BÀI 313: NGHỆ THUẬT "VẼ BÙA" ---
\begin{tutuyetbox}{Chữ như gà bới}{Chicken Scratch}
Tiêu đề nắn nót vẽ hình hoa
Dưới viết vài dòng tựa quỷ tà
Giám thị cầm lên dòm nhức mắt
Chấm con điểm một nhục hằng hà.
\tcblower
Fancy titles with flowers so bright
Beneath is a mess, a ghostly sight
Teacher looks at it, eyes in pain
A "one" on the page, crying in vain.
\end{tutuyetbox}

% --- BÀI 314: THÁNH XÉ VỞ ---
\begin{tutuyetbox}{Vở rách}{Torn Notebook}
Viết sai một chữ xé bay vèo
Vở mỏng dần đi dáng khẳng khiu
Cuối năm còn lại bìa xơ xác
Học dốt mặt lì... nhục bám theo.
\tcblower
One mistake and the page is gone
Tearing and tossing until the dawn
A thin notebook with a tattered cover
Stupid and stubborn, shame is your lover.
\end{tutuyetbox}

% --- BÀI 315: BÚT MÀU LỪA ĐẢO ---
\begin{tutuyetbox}{Hào quang rác}{Colorful Fraud}
Xanh đỏ tím vàng vẽ lung tung
Ghi bài lẹt quẹt dốt vô cùng
Thầy kêu kiểm tập lòi não rỗng
Cút xuống gầm bàn... nhục mông lung.
\tcblower
Rainbow colors all over the sheet
But your knowledge is incomplete
Teacher checks the book, brain is a void
Under the desk, pride is destroyed.
\end{tutuyetbox}

% --- BÀI 316: CHỮ GẤP KHÚC ---
\begin{tutuyetbox}{Vượn múa rồng bay}{Dancing Monkeys}
Hàng trên hàng dưới chạy xiên ngang
Chữ nghĩa bay vèo mặt ngỡ ngàng
Chó đọc không ra, người chẳng hiểu
Tương lai bốc vác... nhục vẻ vang.
\tcblower
Lines are jumping, words are in flight
A messy script, a terrible sight
Dogs can't read it, humans are blind
A laborer's job is all you'll find.
\end{tutuyetbox}

% --- BÀI 317: BÁT CANH ĐẬU PHỤ ---
\begin{tutuyetbox}{Bãi chiến trường}{War Zone}
Gạch xóa lung tung bẩn nát nhà
Vở sạch hồi đầu biến quỷ ma
Cô bảo: "Về nhà thay tập mới...
Mời luôn phụ huynh... nhục thiết tha!"
\tcblower
Scribbles and scratches, a dirty mess
The clean book is in deep distress
"Get a new notebook!" teacher cries out
Bring your parents, you shameful lout.
\end{tutuyetbox}

% --- BÀI 318: VỞ PHOTO (Nâng cấp gắt) ---
\begin{tutuyetbox}{Vở đi mượn}{Borrowed Life}
Học chẳng chịu ghi thích mượn đời
Chụp hình in lậu sướng cuộc chơi
Đi thi chữ nghĩa bay ra đất
Nhìn bảng điểm về... mặt héo tơi.
\tcblower
Too lazy to write, borrowing lives
Copying others' work, the spirit thrives?
Exam arrives, the brain is a void
The final score, pride is destroyed.
\end{tutuyetbox}

% --- BÀI 319: THÁNH LƯỜI THAY BÚT ---
\begin{tutuyetbox}{Bút hết mực}{Empty Pen}
Bút hết từ lâu mặt thản nhiên
Dùng ngòi đè giấy ngỡ thần tiên
Thầy kêu lên bảng mồm câm bặt
Điểm liệt treo đầu... nhục triền miên.
\tcblower
The pen is dry, you don't even care
Pressing the paper with a blank stare
Called to the board, a silent show
A failing grade, a bitter blow.
\end{tutuyetbox}

% --- BÀI 320: LỜI KẾT VỞ SẠCH ---
\begin{tutuyetbox}{Tỉnh ngộ đi con}{Wake Up Call}
Vở sạch chữ đẹp nết phải ngoan
Đừng bày trò lố chuốc tai nàn
Học dốt mai sau đời xuống hố
Nhục nhã nghìn đời... lệ nát tan.
\tcblower
Clean books need a soul so kind
Stop the fake show, or trouble you'll find
Lazy at school leads to the pit
A thousand years of shame, this is it.
\end{tutuyetbox}

\begin{ghichu}
Bạn có thể tốn mấy chục ngàn mua bút màu vẽ tiêu đề, nhưng rồi thầy cô cũng chỉ tốn 5 giây phê một con "Không" tròn xoe. Chữ đẹp mà não rỗng thì cũng chỉ là "bùa chú" của kẻ lười biếng mà thôi.
\end{ghichu}
