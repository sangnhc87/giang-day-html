\chapter{Xin Điểm Cuối Kỳ}
\chapterintro{Sự trưởng thành của một học sinh bắt đầu từ lúc biết hạ mình xin thầy thêm 0.25 để qua môn.}{Vào sinh ra tử cũng chỉ vì nửa điểm làm tròn.}

\begin{tutuyetbox}{Chiến thần mặc cả}{The Bargaining Hero}
Tới kỳ chốt điểm khóc van nài
Trình vớt mười phân hận chí trai
Xin xỏ qua môn đừng tạch nữa
Thầy cười lạnh lùng rớt miệt mài.
\tcblower
Begging for points at the end of the year 
Asking for a jump, with a sudden fear 
"Please let me pass, don't let me fail" 
Teacher laughs coldly, leaving no trail.
\end{tutuyetbox}

\begin{tutuyetbox}{Níu kéo vô vọng}{Hopeless Clinging}
Nài nỉ ỉ ôi níu mép bàn
Khóc than thảm thiết lệ lan tràn
Xin trọn phân tư lên cõi sáng
Thầy phê rớt cái hận trần gian.
\tcblower
Clinging to the desk, begging for more 
Crying and screaming, down on the floor 
Asking for a quarter point to rise 
Teacher says no, with steel in his eyes.
\end{tutuyetbox}

\begin{ghichu}
Xin 0.25 điểm còn khó hơn xin tiền mẹ đi chơi net. Một khi bút đã sa thì thầy chỉ biết nói: "Hẹn em mùa hè học lại".
\end{ghichu}
