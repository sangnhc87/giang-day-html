\chapter{Bai Hoc: Tri Tue Co Nhan --- Kho Tang Muon Doi Khong Can (Ancient Wisdom --- Timeless Treasure)}
\begin{center}
  \textit{\color{truyengold}``Cang nhieu thu thay doi, chung cang van nhu cu.''}\\[4pt]
  \textit{(The more things change, the more they stay the same.)}\\[4pt]
  \small\color{truyenblue}--- Co Kim Dong Tay ---
\end{center}
\ngancach

\section{Ton Tu --- Biet Nguoi Biet Ta Tram Tran Tram Thang}
\begin{truyen}{Ton Tu --- Biet Nguoi Biet Ta Tram Tran Tram Thang}{Ton Tu --- Trung Hoa TK 5 TCN}
\chuhoa{T}{}T on Tu viet 'Binh Phap Ton Tu' khoang 500 nam TCN --- cuon sach ngan nhat ve chien thuat va von vat sac nhon nhat con ton tai.\\
\textit{(Sun Tzu wrote 'The Art of War' around 500 BC --- the shortest yet sharpest book on strategy still in existence.)}

Cau noi noi tieng nhat: 'Biet nguoi biet ta, tram tran tram thang; khong biet nguoi biet ta, mot thang mot bai; khong biet nguoi khong biet ta, tran tran bai.'\\
\textit{(Its most famous line: 'Know yourself and know your enemy, and you will win a hundred battles; know yourself but not the enemy, one win one loss; know neither, lose every time.')}

Cuon sach khong chi la binh phap --- la triet hoc ve su hieu biet, su kien nhan, va cach chien thang ma khong can danh nhau.\\
\textit{(The book is not just tactics --- it is a philosophy of understanding, patience, and winning without fighting.)}

Chien luoc 'Muon nuoc dat ca' --- tao ra cho doi phuong cuoc gia ma khong tro mat mot binh sy --- la dinh cao.\\
\textit{(The strategy of 'Disturbing the Water to Catch Fish' --- creating situations to win without losing a single soldier --- is the pinnacle.)}

Den ngay nay, cuon sach duoc day trong cac Truong Kinh Doanh lon nhat the gioi --- vi khai niem khong bao gio loi thoi.\\
\textit{(Today the book is taught in the world's top Business Schools --- because its concepts never become outdated.)}

\end{truyen}
\begin{baihoc}
  Tri tue co nhan khong bi lai thoi boi vi no de cap den ban chat con nguoi --- ban chat nay khong thay doi theo cong nghe hay thoi dai.\\
  \textit{(Ancient wisdom never becomes outdated because it addresses human nature --- and human nature does not change with technology or era.)}
\end{baihoc}

\section{Lao Tu --- Nuoc Mem Thang Da Cung}
\begin{truyen}{Lao Tu --- Nuoc Mem Thang Da Cung}{Lao Tu --- Trung Hoa TK 6 TCN}
\chuhoa{L}{}L ao Tu la nguoi sang lap Tu Tuong Dao Giao --- nguoi ta noi ong viet 'Dao De Kinh' trong mot buoi toi khi sap roi Trung Quoc.\\
\textit{(Laotzi was the founder of Taoist thought --- legend says he wrote the 'Tao Te Ching' in one night as he was leaving China.)}

Mot cau noi cua ong da lam say long bao trien hoc gia: 'Thien ha chi nhuoc nho hu nuoc --- ma nuoc chuyen thang da cung.'\\
\textit{(One saying of his has captivated generations of philosophers: 'Nothing in the world is as soft and yielding as water --- yet nothing is better at overcoming the hard and strong.')}

Nguoi cuong cuong se gap canh cuong cuong --- nguoi nhu hoa thich ung voi moi hoan canh nhu nuoc.\\
\textit{(The stubbornly rigid will meet rigid resistance --- those who flow adapt to every circumstance like water.)}

Triet hoc 'Vo Vi' --- hanh dong khong cang nhac, khong phai la khong lam gi --- la tieu bieu nhat cua Lao Tu.\\
\textit{(The philosophy of 'Wu Wei' --- acting without rigidity, not meaning doing nothing --- is most representative of Laotzi.)}

Dao De Kinh chi co 5.000 chu --- nhung da duoc dich sang hon 250 ngon ngu --- chi sau Kinh Thanh.\\
\textit{(The Tao Te Ching has only 5,000 characters --- yet has been translated into over 250 languages --- second only to the Bible.)}

\end{truyen}
\begin{baihoc}
  Suc manh that su khong phai la cung nhac ma la uyen chuyen --- nuoc khong bao gio thua tran, vi no luon tim duoc con duong lach qua.\\
  \textit{(True strength is not rigidity but flexibility --- water never loses, because it always finds its way through.)}
\end{baihoc}

\section{Khong Tu --- Truoc Het Hay Lam Giang}
\begin{truyen}{Khong Tu --- Truoc Het Hay Lam Giang}{Khong Tu --- Trung Hoa TK 5 TCN}
\chuhoa{T}{}T u Cong hoi Khong Tu: 'Neu phai bo bao mot trong ba: luong thuc, quan doi, hay long tin cua dan --- thu nao bo truoc?'\\
\textit{(Zigong asked Confucius: 'If one must give up one of three: food, army, or the people's trust --- which goes first?')}

Khong Tu tra loi: 'Bo quan doi.' Tu Cong hoi: 'Neu van phai bo them?' Thay noi: 'Bo luong thuc.'\\
\textit{(Confucius replied: 'Give up the army.' Zigong asked: 'If you must give up another?' The master said: 'Give up the food.')}

'Nguoi ta vit tu xua den nay deu chet --- nhung neu khong co long tin cua dan, quoc gia khong dung vung duoc mot ngay nao.'\\
\textit{('People have always died since antiquity --- but without the trust of the people, a nation cannot stand for even one day.')}

Cau noi nay --- de len tren ca luong thuc lan an ninh --- la hoc thuyet tri quoc tam tho lon nhat truoc khi Kinh Dich theo cach trin.\\
\textit{(This saying --- placing trust above both food and security --- is the grandest governance philosophy before the Book of Changes in that interpretation.)}

Quoc gia nao mat long tin cua dan --- van de quan su hay kinh te gi cung khong cuu duoc.\\
\textit{(Any nation that loses the trust of its people --- no military or economic measure can save it.)}

\end{truyen}
\begin{baihoc}
  Long tin cua dan la nen tang cua moi chinh quyen --- mat long tin thi tuong vua cung sup do; giu duoc long tin thi dat nuoc ben vung.\\
  \textit{(The trust of the people is every government's foundation --- lose that trust and the tallest tower falls; keep it and the nation endures.)}
\end{baihoc}

\section{Trang Tu --- Chuyen Bay Cua Con Buo'm}
\begin{truyen}{Trang Tu --- Chuyen Bay Cua Con Buo'm}{Trang Tu --- Trung Hoa TK 4 TCN}
\chuhoa{M}{}M ot hom, Trang Tu mo thay ong ta la mot con buom dang bay luon tu do giua vuon hoa.\\
\textit{(One day, Zhuangzi dreamed he was a butterfly flying freely among garden flowers.)}

Roi ong tinh day --- va cau hoi dau tien ong dat ra la: 'Bay gio ta la nguoi nam mo la buom, hay la buom dang mo la nguoi?'\\
\textit{(Then he awoke --- and the first question he asked was: 'Am I now a person who dreamed of being a butterfly, or a butterfly dreaming of being a person?')}

Dieu Trang Tu muon noi la: thuc tai ma ta tin la thuc chua chac la toan bo thuc tai.\\
\textit{(What Zhuangzi was saying is: the reality we believe is real may not be the whole of reality.)}

Su nhan co va mo nho nay buoc nguoi dat lai cau hoi --- cai gi la that su quan trong, cai gi chi la nhung nhan mac toi nghia?\\
\textit{(This gentle blurring of wake and dream forces us to ask --- what is truly important, and what is merely label and meaning we impose?)}

Trang Tu khong day chung ta an tam --- ong day chung ta nghi khac di, nhin rong hon, mot cach vui ve va khong co tinh than bi lich.\\
\textit{(Zhuangzi did not teach us to be complacent --- he taught us to think differently, see more broadly, with joy and without tragedy.)}

\end{truyen}
\begin{baihoc}
  Tri tue khong phai luc nao cung di kem voi su nghiem tuc --- doi khi no an trong mot bai kho va buoi cuoi biet on su thay doi.\\
  \textit{(Wisdom is not always accompanied by solemnity --- sometimes it hides in a poem and in grateful laughter at the possibility of change.)}
\end{baihoc}

\section{Manh Tu --- Tinh Nguoi Von Thien}
\begin{truyen}{Manh Tu --- Tinh Nguoi Von Thien}{Manh Tu --- Trung Hoa TK 4 TCN}
\chuhoa{M}{}M ach Mencius day rang con nguoi khi sinh ra da co bon mam mong thien luong: nhan, nghia, le, tri.\\
\textit{(Mencius taught that humans at birth already possess four moral sprouts: humanity, righteousness, propriety, and wisdom.)}

Ong dung vi du: khi thay mot dua tre sap roi xuong gieng, bat ky ai cung cam thay bong kinh hai muon chay lai cuu.\\
\textit{(He gave an example: seeing a child about to fall into a well, any person feels sudden alarm and wants to rush to help.)}

Cam giac nay khong phai vi so bi che bai hay muon duoc khen --- do la phan ung tu nhien cua tam nguoi.\\
\textit{(This feeling is not from fear of blame or desire for praise --- it is the natural response of the human heart.)}

Ong noi: 'Neu khong co tam biec khac kho --- khong phai nguoi. Neu khong co tam xau ho --- khong phai nguoi.'\\
\textit{(He said: 'Without a heart of shame and discomfort --- not human. Without a heart of compassion --- not human.')}

Mot xa hoi tot phat sinh khi moi ca nhan duoc nuoi duong bon mam mong thien luong nay, khong phai khi bi nen.\\
\textit{(A good society arises when every individual nurtures these four moral sprouts, not when they are suppressed.)}

\end{truyen}
\begin{baihoc}
  Con nguoi ban chat khong xau --- chi la hoan canh da khong nuoi duong dieu tot trong ho; hay tao ra hoan canh tot hon.\\
  \textit{(Human nature is not evil --- only circumstances have failed to nurture the good within; create better circumstances instead.)}
\end{baihoc}

\section{Han Phi Tu --- Phap Tri La Nen Tang Cua Nuoc Manh}
\begin{truyen}{Han Phi Tu --- Phap Tri La Nen Tang Cua Nuoc Manh}{Han Phi Tu --- Trung Hoa TK 3 TCN}
\chuhoa{H}{}H an Phi Tu la nha triet hoc phap gia noi tieng trong lich su Trung Hoa co dai --- nguoi bat ke tinh tot hay xau cua con nguoi.\\
\textit{(Han Fei was the famous Legalist philosopher in ancient Chinese history --- who discarded assumptions about human nature's goodness or evil.)}

Ong lap luan: 'Luat phap tot khong can nguoi tot de thuc thi ---no hoat dong vi ai cung phai theo.'\\
\textit{(He argued: 'Good laws do not need good people to enforce them --- they work because everyone must follow them.')}

Ong phe phan truong phai duc hanh cua Khong Tu: 'Dat nuoc khong the duoc cai tri chi bang duc hanh --- can luat le ro rang va hinh phat cong bang.'\\
\textit{(He criticized Confucius's virtue school: 'A nation cannot be governed by virtue alone --- it needs clear rules and impartial punishment.')}

Tuong Phan trinh vua Tan --- nguoi lap thong nhat Trung Hoa --- dua tren hoc thuyet Han Phi Tu lam nen tang.\\
\textit{(Chief Minister of Qin --- who unified China --- built on Han Fei's political theory as a foundation.)}

Cau noi cua Han Phi Tu van con mang tinh thoi su: 'Co o luat phap roi moi noi ve duc hanh --- khong phai nguoc lai.'\\
\textit{(Han Fei's words remain relevant: 'First establish the rule of law, then speak of virtue --- not the other way around.')}

\end{truyen}
\begin{baihoc}
  Tri tue co nhan khong diem nhau --- Khong Tu va Han Phi deu dung --- ho chi nhin tu cac goc do khac nhau cua cung mot su that.\\
  \textit{(Ancient wisdom does not contradict itself --- Confucius and Han Fei are both right --- they simply look from different angles at the same truth.)}
\end{baihoc}

\section{Aristotle --- Hanh Dong La Tap Quen, Khong Phai Ta Tu Nhien}
\begin{truyen}{Aristotle --- Hanh Dong La Tap Quen, Khong Phai Ta Tu Nhien}{Aristotle --- Hy Lap TK 4 TCN}
\chuhoa{C}{}C hiu anh huong lon tu Plato, Aristotle da phat trien mot nhin nhan rat thuc te ve dao duc.\\
\textit{(Deeply influenced by Plato, Aristotle developed a very practical view of ethics.)}

Ong noi: 'Chua chac ai sinh ra da tot --- dieu tot la tap quen; chua chac ai sinh ra da xau --- dieu xau cung la tap quen.'\\
\textit{(He said: 'No one is born good --- goodness is a habit; no one is born bad --- badness is also a habit.')}

Ong dat ra khai niem 'Arete' -- su xuat sac trong cuoc song -- dat duoc bang cach ren luyen kien tri.\\
\textit{(He put forward the concept of 'Arete' --- excellence in living --- achieved through persistent practice.)}

Nho biet hanh dong dung lan mot tien den dan cuoc song tot; nhu tuong dat nuoc di nuoc lai.\\
\textit{(Knowing how to act rightly leads step by step to a good life; like a swimmer going back and forth in water.)}

'Cuoc song tot la ket qua cua viec lam viec tot --- va viec tot la ket qua cua thoi quen tot.' Day la triet ly thuc te nhat nhan loai tung co.\\
\textit{('The good life is the result of good work --- and good work is the result of good habit.' This is the most practical philosophy humanity has ever had.)}

\end{truyen}
\begin{baihoc}
  Tinh cach khong tao nen hanh dong --- hanh dong tao nen tinh cach. Hay chon thoi quen ngay hom nay vi chung se chon nen ban ngay mai.\\
  \textit{(Character does not create actions --- actions create character. Choose your habits today because they will choose who you are tomorrow.)}
\end{baihoc}

\section{Ibn Sina --- Bac Si Va Triet Gia Cu'u Nhiet}
\begin{truyen}{Ibn Sina --- Bac Si Va Triet Gia Cu'u Nhiet}{Ibn Sina --- Ba Tu TK 10}
\chuhoa{I}{}I bn Sina --- duoc Chau Au goi la Avicenna --- la bac si, triet gia, nha tho lon nhat Hoi Giao co dai.\\
\textit{(Ibn Sina --- called Avicenna by Europe --- was the greatest physician, philosopher, and poet of classical Islam.)}

Cuon sach 'Qui Phap Y Khoa' cua ong la giao trinh chuan cho cac truong y o chau Au suot 600 nam.\\
\textit{(His 'Canon of Medicine' was the standard textbook for European medical schools for 600 years.)}

Ong phat hien ra rang benh truyen qua khong khi va nuoc --- va khuyen dat bo y phuc bi nhiem benh.\\
\textit{(He discovered that disease spreads through air and water --- and advised burning infected clothing.)}

Ong viet vao dem truoc khi mat: 'Tam linh toi da nhac lua cho the xac minh sang --- bay gio no da mo ra canh cua; no da tu di.'\\
\textit{(He wrote on the night before he died: 'My soul has lit the lamp of my body --- now it has opened the gate; it has gone its way.')}

Kien thuc y hoc cua ong vuot truoc thoi dai cua ong nhieu the ky --- vi ong dat cau hoi bang tri tue, khong bang tin dieu.\\
\textit{(His medical knowledge was centuries ahead of his time --- because he asked questions with reason, not faith alone.)}

\end{truyen}
\begin{baihoc}
  Tri tue co nhan song qua hang the ky boi no dat cau hoi bang ly luong khoa hoc --- va su kien nhan de tim cau tra loi tan cung.\\
  \textit{(Ancient wisdom survives through centuries because it asks questions with reasoned inquiry --- and the patience to pursue answers to their end.)}
\end{baihoc}

\section{Plato --- Hang Dong Va Su That}
\begin{truyen}{Plato --- Hang Dong Va Su That}{Plato --- Hy Lap TK 4 TCN}
\chuhoa{P}{}P lato ke cau chuyen ve nhung nguoi bi xich trong mot hang dong, chi nhin duoc cai bong tren vach --- ho tin do la toan bo the gioi.\\
\textit{(Plato told of people chained in a cave, able to see only shadows on the wall --- they believed this was the whole world.)}

Mot nguoi thoat ra --- thay mat troi, thay cay co, thay the gioi that --- roi quay vao noi voi nhung nguoi kia.\\
\textit{(One person escapes --- sees the sun, trees, the real world --- then returns to tell the others.)}

Ho khong tin --- ho chi thay bong; va hoat nghie an --- ho giet nguoi do.\\
\textit{(They don't believe --- they only trust shadows; and out of fear they kill the escaped one.)}

Plato muon noi: tri tue that su luon bi nguoi so hang dong chong lai, khi no dat cau hoi ve thuc tai.\\
\textit{(Plato's message: genuine wisdom is always resisted by those comfortable in the cave, when it questions reality.)}

Cuon 'Cong Hoa' cua ong --- viet 2500 nam truoc --- van la nen tang cua triet hoc chinh tri va giao duc hien dai.\\
\textit{(His 'Republic' --- written 2,500 years ago --- remains the foundation of modern political philosophy and education.)}

\end{truyen}
\begin{baihoc}
  Tri tue that su doi khi bi xa hoi chong lai --- vi no dat cau hoi lam lo rung chuyen nhung su thoai mai cua nen khong bi xao tron.\\
  \textit{(Genuine wisdom is sometimes resisted by society --- because it asks questions that disturb the comfort of those who prefer not to be disturbed.)}
\end{baihoc}

\section{Nguyen Binh Khiem --- Tinh Toan Cu'a Thanh Nhan}
\begin{truyen}{Nguyen Binh Khiem --- Tinh Toan Cu'a Thanh Nhan}{Nguyen Binh Khiem --- Viet Nam TK 16}
\chuhoa{N}{}N guyen Binh Khiem --- Trang Trinh --- la nha tri thuc lon nhat Viet Nam the ky XVI, co tai tien tri huyen bi.\\
\textit{(Nguyen Binh Khiem --- Trang Trinh --- was the greatest Vietnamese intellectual of the 16th century, known for mysterious prophetic abilities.)}

Ong tu ngai khi triều chinh nhieu dang, ong lui ve que song doi gian tien nong dan, viet tho Nam.\\
\textit{(Wearied by factional court politics, he withdrew to his home village to live the simple life of a farmer, composing Nom poetry.)}

Cau tho ong de lai lam chay oc nhieu the he: 'Ta dai nguoi biet ta la ai --- khong cho ai biet ta la ai cung duoc.'\\
\textit{(One verse he left behind has given generations pause: 'I wait for those who know who I am --- those who don't know, that is also fine.')}

Ong goi ra cai goi la 'tri en tri an' --- biet dung luc nao ra, luc nao lui --- triet ly cua con nuoc len xuong.\\
\textit{(He articulated what he called 'knowing times and seasons' --- knowing when to advance, when to withdraw --- a philosophy of tides.)}

Den ngay nay, cac cau sau luu truyen duoc nhan la bat ly tri cua ong --- khon nguoan ai cung biet nhung khong phai ai cung song theo.\\
\textit{(Even today, his sayings are recognized as his unrivaled wisdom --- wisdom everyone acknowledges but few actually live by.)}

\end{truyen}
\begin{baihoc}
  Tri tue co nhan hay nhat he sinh tu nhung nguoi biet khi nao nen tien, khi nao nen lui --- va du can dam de chon chac chan.\\
  \textit{(The finest ancient wisdom is born in those who know when to advance and when to withdraw --- and have the courage to choose deliberately.)}
\end{baihoc}
