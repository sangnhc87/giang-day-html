\chapter{Thất Tình Đúng Mùa Thi}
\chapterintro{Thất tình đã khổ, thất tình mà vướng thêm bài tập toán thì còn khổ hơn cả vạn lần.}{Tim rớt theo điểm, điểm rớt theo tình.}

\begin{tutuyetbox}{Tim vỡ bài nát}{Broken Heart, Ruined Test}
Mới đá nhau xong rụng rã rời
Bao nhiêu kiến thức cũng buông lơi
Bàn thi rớt lệ tuôn xành xạch
Chẳng chịu ôn bài trách hận trời.
\tcblower
Just broken up, completely torn 
Knowledge vanishes, newly born 
Crying at the desk, tears like rain 
Failed to study, blaming the pain.
\end{tutuyetbox}

\begin{tutuyetbox}{Đang sầu thì gặp thi}{Sorrows Meeting Exams}
Tình lỡ tình tan ruột rã rời
Chữ tài chữ phận rụng ngoài khơi
Nhìn qua đề mục mù tăm tối
Rớt đại lưu ban khóc nát đời.
\tcblower
Love is lost, the heart is weak 
Talent and fate, a future bleak 
The exam paper is a foggy sea 
Flunking out, a tragedy for me.
\end{tutuyetbox}

\begin{ghichu}
Khóc vì tình yêu kết thúc còn được vỗ về, chứ khóc vì tờ điểm Hóa đỏ chót thì chỉ nhận được trận đòn roi mà thôi.
\end{ghichu}
