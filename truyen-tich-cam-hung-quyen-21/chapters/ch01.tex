\chapter{Trả giá}
\markboth{Trả giá}{The Price}


\section{Lười hôm nay, trả giá ngày mai.}
\begin{truyen}{Lười hôm nay, trả giá ngày mai.}{Wake up.}
\chuhoa{C}{âu thầy nói thẳng. Một vài em im lặng — không phải vì sợ, mà vì giật mình.}
\textit{(The teacher said it straight. Some fell silent — not from fear, but from being startled.)}
\end{truyen}

\begin{baihoc}
Giật mình một lần có thể đổi hướng.
\textit{(One wake-up moment can change your direction.)}
\end{baihoc}

\begin{ghinhoanh}
\textbf{Short English to remember:}

Wake up before you pay the price.

\end{ghinhoanh}

\begin{tuhocnhanh}
1. Câu này khiến em giật mình thế nào? \textit{How does this startle you?}

2. Em sẽ làm gì khác đi sau khi đọc? \textit{What will you do differently after reading?}
\end{tuhocnhanh}
