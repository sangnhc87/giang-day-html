\chapter{Mộng Tưởng Tương Lai}
\chapterintro{Có bạn vẽ tương lai rất đẹp nhưng hiện tại thì để trắng vài dòng}{Mơ lớn không sai, chỉ cần nhớ đường tới giấc mơ thường đi qua từng bài nhỏ.}

\begin{lucbatbox}{Lục bát: Mai này rực rỡ}
Em mơ mai sẽ thành danh\
Tên ghi thật đẹp trên cành tương lai\
Bạn bè nghe thấy ai ai\
Cũng khen chí lớn, đường dài rất trong\
Chỉ e bài cũ còn không\
Mà mong tới bến thành công hơi sớm
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Giấc mơ cao tầng}
Em lo tương lai rất dài lâu
Mơ nhà mơ việc mơ thành công
Chỉ mong trang vở hôm nay đấy
Đừng còn để trắng giữa dòng sương
\end{tutuyetbox}

\begin{ghichu}
Mộng tưởng tương lai đẹp nhất khi nó kéo người ta ngồi xuống làm việc hôm nay, không chỉ kéo ra vài câu hùng hồn.
\end{ghichu}

\ngancach

\begin{lucbatbox}{Lục bát: Nói lớn, làm chậm}
Bạn bè hỏi chuyện mai sau\
Em bày dự định một màu rất căng\
Nghe như sắp mở trời giăng\
Lộ trình rõ lắm, chỉ rằng chưa đi\
Ước mơ chẳng có lỗi gì\
Chỉ mong bước nhỏ đi kề cùng nhau
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Tương lai bắt đầu}
Mai sau muốn sáng rỡ hơn người
Thì nay đừng sống quá qua loa
Mộng cao vốn rất nên gìn giữ
Nhưng phải bước đều mỗi sớm mai
\end{tutuyetbox}
