\chapter{Bai Hoc: Hy Vong Thap Sang Con Duong (Hope Lights the Path)}
\begin{center}
  \textit{\color{truyenblue}Hy vong la chiec long chim dau trong linh hon va hat khuc ca khong loi.}\\
  \textit{(Hope is the thing with feathers that perches in the soul and sings the tune without the words.)}\\[4pt]
  \small\color{truyengold}--- Thien nhien day ta ---
\end{center}
\ngancach

\section{Chim Phuong Hoang Tai Sinh}
\begin{truyen}{Chim Phuong Hoang Tai Sinh}{Phung Hoang}
\chuhoa{T}{}T rong truyen thuyet, phung hoang song 500 nam roi tu thieu minh trong ngon lua cua chinh minh.\\
\textit{(In legend, the phoenix lives 500 years then burns itself in its own flames.)}

Tu dong tro tan no tai sinh, lon hon sa hon dep hon truoc va vui ve hon.\\
\textit{(From the ashes it is reborn, larger, more beautiful, more radiant, and more joyful than before.)}

Y nghia cua phung hoang khong phai phep mau ma la su that rang sau moi ket thuc co mot bat dau.\\
\textit{(The phoenix's meaning is not magic but the truth that after every ending comes a beginning.)}

Nhieu nen van hoa khac nhau tren the gioi deu co huyen thoai ve sinh vat tai sinh tu tro tan.\\
\textit{(Many different cultures around the world have legends about creatures reborn from ashes.)}

Nhan loai can hy vong nay; no la ban nang sinh ton cuoi cung khi tat ca phuong an khac deu dung.\\
\textit{(Humanity needs this hope; it is the final survival instinct when all other options have failed.)}

\end{truyen}
\begin{baihoc}
  Moi ket thuc trong cuoc doi ban deu la co hoi tro lai moi hon; dung goc nhin la mat mat, hay goc nhin la tai sinh.\\
  \textit{(Every ending in your life is an opportunity to return renewed; don't see it as loss, see it as rebirth.)}
\end{baihoc}

\section{Hat Giong Qua Mua Dong}
\begin{truyen}{Hat Giong Qua Mua Dong}{Hat Giong}
\chuhoa{N}{}N am ngoai mua nan ha khoc liet hat giong corn cat xuong dat kho nut nut song voi hy vong.\\
\textit{(Last year in the harsh drought season corn seeds fall into cracked dry soil and wait with hope.)}

Khong ai bao dam rang mua se roi; khong ai hua rang dat se am; nhung hat giong van chon minh xuong.\\
\textit{(No one guarantees rain will come; no one promises the soil will warm; but the seed still buries itself below.)}

Trong bong toi cua long dat, hat giong co cau truc phuc tap bao ve noi tang su song.\\
\textit{(In the darkness of the earth's interior, the seed has complex structures protecting its inner life.)}

Khi mua lui cham xuong chay vao dat, hat giong khong phoi hop bang ky thuat; no chi mo ra vi no nho.\\
\textit{(When rain slowly seeps into soil, the seed does not coordinate by technique; it simply opens because it remembers.)}

Hy vong sinh hoc la tri nho co dieu kien hoa: khi dieu kien du, su song nop lai.\\
\textit{(Biological hope is conditioned memory: when conditions are sufficient, life resumes.)}

\end{truyen}
\begin{baihoc}
  Hy vong khong phai mo ho; no la nang luc chuan bi tai nguyen sinh luc o the tinh de cho dieu kien thich hop.\\
  \textit{(Hope is not wishful thinking; it is the capacity to store vital energy in dormancy waiting for the right conditions.)}
\end{baihoc}

\section{Ca Heo Khong Bao Gio Tu Bo Hung}
\begin{truyen}{Ca Heo Khong Bao Gio Tu Bo Hung}{Ca Heo}
\chuhoa{K}{}K hi ca heo bi thung luoi va mat trang, no khong nam chet o may no trong nuoc sau mot minh.\\
\textit{(When a dolphin is net-wounded and loses its pod, it does not lie dying in deep water alone.)}

No boi den vung bien nong hon, de lai song, hoi phuc dan dan trong nhieu tuan.\\
\textit{(It swims to warmer shallower waters, stays alive, recovering gradually over many weeks.)}

Ca heo bi thuong thuong tim den cac tau thuyen cua nguoi, keu la nhu tim kiem su giup do.\\
\textit{(Injured dolphins often find human boats, crying out as if seeking help.)}

Nhieu truong hop ghi nhan ca heo duoc dua len bo, cham soc, roi thuc su boi ra bien lai.\\
\textit{(Many recorded cases of dolphins being lifted ashore, cared for, and truly swimming back to sea.)}

Hy vong khong phai tu nhien xuat hien; no la quyet dinh tich cuc di tim noi co anh sang.\\
\textit{(Hope does not appear naturally; it is the active decision to move toward where there is light.)}

\end{truyen}
\begin{baihoc}
  Khi bi thuong, hay boi den tai nguoi co the giup ban; xin giup do khong phai yeu duoi ma la khon ngoan.\\
  \textit{(When wounded, swim toward those who can help you; asking for help is not weakness but wisdom.)}
\end{baihoc}

\section{Chim Non Lan Dau Mo Mat}
\begin{truyen}{Chim Non Lan Dau Mo Mat}{Chim Non}
\chuhoa{T}{}T rong bong toi cua cai trung, chim non khong biet anh sang la gi ngoai soi am nhe qua vo.\\
\textit{(In the darkness of the egg the young bird does not know what light is except a faint warmth through the shell.)}

No dung mo be vo, mot cai go nho be, lien tuc, mot each kien tri uong tranh.\\
\textit{(It uses its beak to break the shell, one small tap, continuously, stubbornly against heavy resistance.)}

Khi vo trung nut no va anh sang lan dau tien tran vao, no la gi hinh anh dau tien no thay.\\
\textit{(When the shell cracks open and light floods in for the first time, that is the first image it ever sees.)}

Tu giay phut do, no bien anh sang la nha, la noi thuoc ve, la thu no luon huong ve.\\
\textit{(From that moment, it makes light its home, its sense of belonging, what it always moves toward.)}

Hy vong la khi con vat trong bong toi van goc go ve phia anh sang ma no chua bao gio thay.\\
\textit{(Hope is when a creature in darkness still chips toward the light it has never yet seen.)}

\end{truyen}
\begin{baihoc}
  Hay tin rang anh sang o phia kia cua bong toi ban dang o trong; no tan tai du ban chua thay duoc.\\
  \textit{(Trust that light is on the other side of the darkness you are in; it exists even if you cannot yet see it.)}
\end{baihoc}

\section{Xa Thanh Vuon Tren Vet Nuoc Kham Pha}
\begin{truyen}{Xa Thanh Vuon Tren Vet Nuoc Kham Pha}{Cay Nao}
\chuhoa{T}{}T o cay nao oc tren toan bo hanh tinh moi sang den nao phat ra tin hieu ky vong va mo ngu.\\
\textit{(The brains of all mammals on the entire planet fire signals of anticipation and dreaming every night.)}

Chim va ca trong giac ngu thuc hien lai hanh trinh truoc thi hay chac van dang hoc moi.\\
\textit{(Birds and fish during sleep replay journeys practiced or skills currently being learned.)}

Nao tao ra ban nha cua nhung gi se xay ra truoc, mo phong kha nang thu quan trong lieu truoc.\\
\textit{(The brain creates a map of what will happen next, simulating possibilities to test-run important scenarios.)}

Hy vong la chuc nang than kinh hoc: nao tao ra tuong lai truoc roi moi goi no lam xuat hien.\\
\textit{(Hope is a neurological function: the brain creates the future first and then calls it into appearance.)}

Nhung sinh vat biet mo ve tuong lai tot hon se song sot tot hon nhung con khong biet mo.\\
\textit{(Creatures that dream of a better future survive better than those that cannot dream.)}

\end{truyen}
\begin{baihoc}
  Mong uoc khong phai lu an; do la he thong tien ich cua nao de tu tao cuoc song tot hon cho ban.\\
  \textit{(Dreaming is not foolishness; it is the brain's advance-planning system for creating a better life for you.)}
\end{baihoc}

\section{Ran Mo Truoc Khi Lay Lan Da Moi}
\begin{truyen}{Ran Mo Truoc Khi Lay Lan Da Moi}{Con Ran}
\chuhoa{L}{}L uc dau con ran khong the lay duoc lan da moi mot minh co mot minh; no can mat ngoai ma suu.\\
\textit{(First the snake cannot shed its old skin alone; it needs an external abrasive surface.)}

No tim mot viha da rao nham, cham mam phia truoc vai ngay truoc va bat dau co day thu.\\
\textit{(It finds a rough rocky surface, rubbing its snout for days in advance and begins slowly pushing through.)}

Lan da cu tu tach ra tu phia dau, la nguyen ca mot cai ao cu bi tut nguoc ra ngoai hon.\\
\textit{(The old skin peels back from the head, the entire old shell turned inside out.)}

Duoi lop da cu la lan da moi sang bong hoan toan, sac net va tre hon hon truoc.\\
\textit{(Under the old skin is perfectly bright new skin, vibrant and younger-looking than before.)}

Hy vong dat cu ve nhung thay doi lon luon bat dau bang viec cho phep buong bo nhung gi da cu khong con phu hop.\\
\textit{(Hope for great changes always begins with allowing yourself to let go of what no longer fits.)}

\end{truyen}
\begin{baihoc}
  Truoc khi than hinh moi xuat hien, lai da cu phai duoc boc dit ra; dung so viec tu bo thu cu.\\
  \textit{(For the new form to emerge, the old skin must be shed completely; don't fear discarding old things.)}
\end{baihoc}

\section{Rung Chay Hoi Sinh Sau Bao Lua}
\begin{truyen}{Rung Chay Hoi Sinh Sau Bao Lua}{Rung Sau Bao Lua}
\chuhoa{N}{}N am 1988 bao lua quet qua 36 phan tram Vuon Quoc Gia Yellowstone that tan tro than.\\
\textit{(In 1988 wildfire swept through 36 percent of Yellowstone National Park, burning everything to ash.)}

Mua xuan nam sau, trong dam tro den, hang tram ngan mam cay moc khap noi ta goi la may man.\\
\textit{(The following spring in the black ash bed, hundreds of thousands of sprouts grew everywhere what we call miraculous.)}

Lua da mo ra dat, diet sau benh va thuc day hat trong cac loai cay co ham het dau.\\
\textit{(The fire opened the soil, killed pests, and stimulated dormant seeds of serotinous species to germinate.)}

Nam muoi sau tran bao lua, Yellowstone co da dang sinh hoc cao hon bat ky thoi diem nao truoc do.\\
\textit{(Fifty years after the wildfire, Yellowstone has higher biodiversity than at any time before.)}

Tran bao lua khoc liet nhat van de lai hy vong; no chi la hy vong minh chia khong giong nhu da nghi.\\
\textit{(The fiercest wildfire still leaves hope; it is just hope manifesting differently than imagined.)}

\end{truyen}
\begin{baihoc}
  Bao gi ban cung co the tim ra dat moi de trong sau bao lua; cuoc song luon tim duong de hoi sinh.\\
  \textit{(You can always find new soil to plant in after a wildfire; life always finds its way to regenerate.)}
\end{baihoc}

\section{Buom Khong Biet Mau Cua Chinh Minh}
\begin{truyen}{Buom Khong Biet Mau Cua Chinh Minh}{Con Buom}
\chuhoa{N}{}N hieu loai buom co mau sac dep tren canh nhung chinh mat no chua bao gio nhin thay mau do.\\
\textit{(Many butterflies have beautiful color patterns on wings but their own eyes have never seen those colors.)}

Canh buom duoc tao ra cho su thu nhan cua mat ke khac, khong phai cua minh.\\
\textit{(Butterfly wings are made for the perception of others' eyes, not one's own.)}

Dep cua buom la de thu hut ban doi, de canh bao ke thu, de hoa vao hoa loi.\\
\textit{(A butterfly's beauty serves to attract mates, warn enemies, blend into flower benefits.)}

No khong can biet minh dep; viec no phat huy ve dep cua minh da mang lai y nghia roi.\\
\textit{(It need not know it is beautiful; the very act of expressing its beauty already brings meaning.)}

Doi khi ly do de song khong phai vi ban thay ban dep; ma vi cuoc song dung ban de dep hon.\\
\textit{(Sometimes the reason to live is not because you see yourself as beautiful, but because life needs you to make it more beautiful.)}

\end{truyen}
\begin{baihoc}
  Ban ton tai de them y nghia vao cuoc song xung quanh ban; du ban co nhan ra hay khong, that su ban.\\
  \textit{(You exist to add meaning to the life around you; whether you realize it or not, that is true.)}
\end{baihoc}

\section{Nha Du Hanh Vu Tru Nhin Trai Dat}
\begin{truyen}{Nha Du Hanh Vu Tru Nhin Trai Dat}{Phi Hanh Gia}
\chuhoa{N}{}N hiing phi hanh gia dau tien nhin Trai Dat tu vu tru mo ta lanh nuoc mat khong the cap duoc.\\
\textit{(The first astronauts looking at Earth from space describe an indescribable chill of tears coming on.)}

Ho thay mot qua cau xanh mong manh bao boc boi lop khi quyen mong te de bi.\\
\textit{(They see a fragile blue sphere wrapped in an achingly thin and easily destroyed atmosphere.)}

Khong co duong vach quoc gia, khong co gianh gioi dia ly, chi co mot Trai Dat thong nhat.\\
\textit{(No national boundary lines, no geographic borders, only one united Earth.)}

Tu tren do ho hieu tai sao chien tranh la phi ly va tai sao bao ve hanh tinh la thien nhiem.\\
\textit{(From up there they understand why war is irrational and why protecting the planet is a sacred mission.)}

Hy vong vi dat la khi chung ta du can dam nhin dat tu xu xa de thay ro rang gia tri cua no.\\
\textit{(Hope for the Earth is when we have enough courage to look at it from far away to see its value clearly.)}

\end{truyen}
\begin{baihoc}
  Hay nhin ban be, gia dinh va cong dong ban tu mot goc nhin xa hoi hon; ban se thay tat ca dang quy biet bao.\\
  \textit{(Look at your friends, family, and community from a higher perspective; you will see how precious everything is.)}
\end{baihoc}

\section{Hoa Bun Sen}
\begin{truyen}{Hoa Bun Sen}{Hoa Sen}
\chuhoa{G}{}G iua dam li bun hoi thoi tanh tuu, hat sen nam duoi lay nguon duong chat tu bun de song.\\
\textit{(In a foul-smelling swamp, the lotus seed lies below drawing nutrients from the mud to survive.)}

Cuong hoa vuon thang len qua meters nuoc toi de tim anh mat troi tren mat nuoc.\\
\textit{(The flower stem stretches straight up through meters of dark water to find sunlight on the surface.)}

Khi hoa no tren mat nuoc, canh hoa trang tinh khiet khong mot vet bun nao bam lai.\\
\textit{(When it blooms on the water's surface, the pure white petals carry not a single trace of mud.)}

Hoa sen biet mot bi mat lon: ta co the truong thanh tu vung bun ma khong can mang bun len.\\
\textit{(The lotus knows one great secret: one can grow from muddy origins without carrying the mud upward.)}

Moi nguoi deu co kha nang tro thanh hoa sen: born in mud, bloom in sunlight, remain pure.\\
\textit{(Every person has the capacity to become a lotus flower: born in mud, bloom in sunlight, remain pure.)}

\end{truyen}
\begin{baihoc}
  Nguon goc khiem ton hay kho khan cua ban khong dinh hinh tuong lai ban; no chi la dat de ban vu ong.\\
  \textit{(Your humble or difficult origin does not define your future; it is only the soil from which you bloom.)}
\end{baihoc}
