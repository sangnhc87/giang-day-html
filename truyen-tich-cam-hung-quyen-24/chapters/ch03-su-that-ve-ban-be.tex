\section{Bạn nhiều chưa chắc tốt}
\begin{truyen}{Bạn nhiều chưa chắc tốt}{Life lesson}
\chuhoa{A}{nh có hàng trăm bạn ảo, nhưng khi anh ốm nằm viện chỉ một người đến. Bạn nhiều chưa chắc tốt.}
\textit{(He had hundreds of online friends, but when he was in hospital only one came. Many friends doesn't mean good.)}
\end{truyen}

\begin{baihoc}
Chất lượng quan trọng hơn số lượng.
\textit{(Quality matters more than quantity.)}
\end{baihoc}

\begin{ghinhoanh}
\textbf{Short English to remember:}

Quality matters more than quantity.

\end{ghinhoanh}
