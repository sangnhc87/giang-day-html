\chapter{Giấu Điện Thoại Dưới Bàn}
\chapterintro{Bàn tay trái cầm bút tạo nét, bàn tay phải thoăn thoắt bấm phím.}{Đáy ngăn bàn là nơi chứa đựng cả một thế giới ảo.}

\begin{tutuyetbox}{Ổ nhền nhện góc bàn}{The Spider Den}
Trí để trên phây rặt một bầy
Dưới ngăn lén lút vuốt hăng say
Mắt lờ lim dim màn chói sáng
Thầy thó phone thu khóc đứt dây.
\tcblower
Mind on Facebook, joining the crowd 
Under the desk, scrolling aloud 
Drowsy eyes in the glowing light 
Teacher grabs the phone, a bitter night.
\end{tutuyetbox}

\begin{tutuyetbox}{Ánh sáng xanh phản chủ}{The Blue Light Betrayal}
Loài người mù mịt lướt trên không
Chói lóa trùm chăn sáng đỏ hồng
Giáo bộ vồ ngay con xế hộp
Vô thi rớt bịch tạch luôn công.
\tcblower
People wandering in the empty air 
Glowing blankets, a bright red flare 
Teacher snatches the precious prize 
Failing the test, before your eyes.
\end{tutuyetbox}

\begin{ghichu}
Trò ma giấu điện thoại đã lỗi thời từ thập kỷ trước rồi. Chỉ là bao giờ thầy cô mới rảnh tay đi thu thôi.
\end{ghichu}
