\chapter{Viết Bài Và Văn Mẫu}
\chapterintro{Có những bài văn rất tròn, chỉ tiếc là không thấy người viết ở trong}{Văn mẫu giúp đỡ ban đầu, nhưng lạm dụng thì chữ nghĩa mất giọng riêng.}

\begin{lucbatbox}{Lục bát: Nhớ mẫu hơn nhớ ý}
Đề ra chưa kịp nghĩ sâu
Trong đầu câu mẫu từ đâu vọng về
Mở bài trôi chảy rất mê
Thân bài lại giống bài kia năm nào
Chỉ riêng tiếng nói người làm
Tìm hoài chẳng thấy ở trong câu nào
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Bài văn na ná}
Bài em mượt lắm chữ như nhung
Đọc lên nghe rất giống người dùng
Hình như đâu đó từng quen lắm
Chỉ chưa thấy rõ tiếng em rung
\end{tutuyetbox}

\begin{ghichu}
Viết theo văn mẫu quá tay dễ làm bài bớt dở, nhưng cũng làm người viết biến mất rất nhanh.
\end{ghichu}

\ngancach

\begin{lucbatbox}{Lục bát: Mở bài rất kêu}
Mở đầu em viết như thơ
Câu nào cũng đẹp cũng mơ cũng đầy
Đến phần ý chính triển khai
Tự nhiên chữ nghĩa đứng ngay giữa đường
Hóa ra phần khó vô cùng
Không nằm ở đoạn mở đầu lung linh
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Kết bài bay cao}
Kết bài em chốt tận mây xanh
Tình thương nhân ái với trăng lành
Chỉ mong thân bài đừng run nữa
Kẻo đầu với cuối lạc loanh quanh
\end{tutuyetbox}
