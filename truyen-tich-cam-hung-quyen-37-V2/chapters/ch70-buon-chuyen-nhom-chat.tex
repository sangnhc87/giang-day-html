\chapter{Buôn Chuyện Nhóm Chat}
\chapterintrobi{Trong lớp im, trong nhóm chat lại rất đông vui}{Chat nonstop in the class group}{Nhiều bạn ngồi cùng một phòng mà vẫn thích nói chuyện qua màn hình hơn.}

\begin{lucbatbox}{Lục bát: Vắng lời ngoài, đông chữ trong}
Ngoài kia cô giảng nhẹ nhàng\
Trong đây nhóm chat đang tràn chuyện vui\
Em nhắn một chữ, bạn cười\
Tin qua tin lại thành mười phút bay\
Lớp thì tưởng vẫn rất ngay\
Mà phần chú ý đã thay chỗ rồi
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Nói bằng bàn phím}
Ngồi cùng lớp vẫn thích nhắn tin hơn\
Màn hình nối chuyện rất thân quen\
Nếu em trao đổi bài bằng cường độ ấy\
Có khi kiến thức lại lên men
\end{tutuyetbox}

\begin{ghichu}
Buôn chuyện nhóm chat trong giờ tạo cảm giác kín đáo. Thực ra nó chỉ đổi tiếng ồn ngoài miệng thành tiếng ồn trong đầu.
\end{ghichu}

\begin{englishnote}
\textbf{group chat}: nhóm chat.\par
\textbf{text in class}: nhắn tin trong giờ.\par
\textbf{The chat was more active than the lesson.}: Nhóm chat còn sôi nổi hơn cả tiết học.
\end{englishnote}
