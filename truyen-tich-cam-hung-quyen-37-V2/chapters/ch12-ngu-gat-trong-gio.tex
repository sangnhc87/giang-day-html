\chapter{Ngủ Gật Trong Giờ}
\chapterintro{Ngủ trong lớp là nghệ thuật nguy hiểm vì người ta rất tưởng mình không bị thấy}{Cơn buồn ngủ ở THPT là đối thủ ngang hàng với bài giảng dài.}

\begin{lucbatbox}{Lục bát: Gật đầu rất đúng nhịp}
Thầy cô giảng một câu dài
Đầu em gật xuống như bài tán đồng
Nhìn xa tưởng rất cảm thông
Nhìn gần mới biết hồn không ở nhà
Bạn bên cạnh gọi khẽ là
Ê đừng ngủ nữa, kẻo ra bảng giờ
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Mắt mở, hồn tắt}
Mắt em còn mở rất đàng hoàng
Nhưng ý thức đã sang cuối trời
Thầy cô vừa hỏi một câu đó
Em nhìn khoảng trống để trả lời
\end{tutuyetbox}

\begin{ghichu}
Ngủ gật đáng khịa nhất là lúc người ta vẫn cố diễn vai đang nghe rất chăm.
\end{ghichu}

\ngancach

\begin{lucbatbox}{Lục bát: Tiết chiều dễ ngủ}
Chuông chiều vừa dứt xong thôi
Lớp em như thể cả trời mây trôi
Ai ai mắt cũng lưng lưng
Mỗi em ngủ gật một từng nấc riêng
Thầy cô gọi mãi liên miên
Cứu được vài bạn, vài miền còn mơ
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Chống chế rất mềm}
Thầy hỏi vì sao mắt nặng ghê
Em thưa con chỉ nghĩ nhiều ghê
Nếu mà suy nghĩ đến nghiêng ngả
Tư duy em hẳn hạng sang ghê
\end{tutuyetbox}
