\chapter{Long tu trong, xau ho va kha nang dung day}

\begin{leadbox}
Nhieu hoc sinh khong xau, chi dang bi xau ho dan dat. Khi con nguoi cam thay minh kem coi qua lau, ho de chon vach mat, pha phach, hoac bo cuoc hon la doi dien. Chuong nay di vao mien tam ly mong manh do.
\end{leadbox}

\storycard{51. Bai van bi doc to tien lop}{Read Aloud by Force}{Khi mot bai viet rieng tu bi phoi ra dung cach, cam giac xau ho co the dong cua tam hon.}
\storycard{52. Em khong gio tay nua}{Never Raising a Hand Again}{Chi mot lan tra loi sai bi che cuoi cung du de mot hoc sinh im lang ca hoc ky.}
\storycard{53. Chiec ao dong phuc cu}{The Worn Uniform}{Ngheo kho khong nen tro thanh vat can de hoc sinh mat quyen duoc ton trong.}
\storycard{54. Mot tam bang khen muon}{A Late Certificate}{Duoc ghi nhan muon con hon khong bao gio duoc ghi nhan.}
\storycard{55. Loi noi: Em vo duyen qua}{You Are Hopeless}{Gan nhan vao con nguoi thuong de lai vet ham sau hon phe binh mot hanh vi.}
\storycard{56. Nho ban phat bieu ho}{Speaking Through a Friend}{Co nhung em chi dam xuat hien qua giong noi cua nguoi khac truoc khi hoc cach noi bang giong cua minh.}
\storycard{57. So mot mon hoc}{Fear of One Subject}{Noi so hoc tap thuong khong den tu kien thuc, ma den tu nhung lan cam thay minh ngu ngan.}
\storycard{58. Ban nhac viec ben hanh lang}{The Hallway Reminder}{Sua sai rieng tu co khi huu ich hon phe binh giua dam dong.}
\storycard{59. Buoi gap lai sau loi lam}{Meeting After a Mistake}{Cho hoc sinh mot cuoc doi thoai sau sai pham la trao lai cho em co hoi giu pham gia.}
\storycard{60. Tu mot lan nga thanh mot cach dung}{Turning a Fall into Form}{Nguoi truong thanh khong phai nguoi khong nga, ma la nguoi biet dung day co phuong phap.}
