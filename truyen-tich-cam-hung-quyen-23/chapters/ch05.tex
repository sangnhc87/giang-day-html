\section{Em có thể không còn ngồi trong lớp thầy — nhưng thầy vẫn mong em ngồi vững trên đường đời.}
\begin{truyen}{Em có thể không còn ngồi trong lớp thầy — nhưng thầy vẫn mon...}{Year-end words.}
\chuhoa{C}{âu thầy nói trước khi cả lớp ra về. Nhiều em im lặng, có em rơi nước mắt.}
\textit{(The teacher said it before everyone left. Many were silent; some had tears.)}
\end{truyen}

\begin{baihoc}
Những lời cuối năm ở lại lâu hơn điểm số.
\textit{(Year-end words stay longer than grades.)}
\end{baihoc}

\begin{ghinhoanh}
\textbf{Short English to remember:}

Year-end words touch the heart.

\end{ghinhoanh}

\begin{tuhocnhanh}
1. Câu này em muốn giữ lại thế nào? \textit{How do you want to keep this?}

2. Em sẽ nhớ thầy cô qua câu nào? \textit{Which sentence will remind you of your teacher?}
\end{tuhocnhanh}
