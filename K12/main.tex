	\begin{dn}{Quy tắc cộng}{def1}
		Giả sử một công việc có thể được thực hiện theo phương án $A$ hoặc phương án $B$...
	\end{dn}
	
	\begin{luuy}
		Nếu các phương án $A$ và $B$ không rời nhau, ta phải áp dụng công thức bù trừ.
	\end{luuy}
	
	\subsection{Các dạng toán thường gặp}
	
	\begin{dang}{Bài toán đếm số}{dang1}
		Phương pháp chung:
		\begin{enumerate}
			\item Gọi số cần tìm là $\overline{a_1a_2...a_n}$.
			\item Chọn từng chữ số dựa trên điều kiện bài toán.
		\end{enumerate}
	\end{dang}
	
	\begin{ex}[id=QTC01, stars=2]
		Có bao nhiêu số tự nhiên có 3 chữ số khác nhau lập từ tập $A=\{1,2,3,4,5\}$?
		\choice{12}{\True 60}{125}{10}
		\loigiai{Số cách chọn là $A_5^3 = 60$.}
	\end{ex}
