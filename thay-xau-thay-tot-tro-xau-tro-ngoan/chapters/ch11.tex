\chapter{Nhan cach cua nguoi day, nhan cach cua nguoi hoc}

\begin{leadbox}
Khong co bai hoc nhan cach nao co suc thuyet phuc neu nguoi day khong song noi dung minh day. Va cung khong co su truong thanh nao xay ra neu nguoi hoc chi doi ben ngoai thay doi. Chuong nay noi ve tinh nhat quan giua loi noi, hanh vi va tam the.
\end{leadbox}

\storycard{101. Giao vien di tre}{The Late Teacher}{Uy tin duoc xay tren nhung dieu nho lap di lap lai, khong chi tren bai giang hay.}
\storycard{102. Hoc sinh xin loi that long}{A Real Apology}{Loi xin loi co gia tri khi kem theo hanh dong sua lai can bang.}
\storycard{103. Noi ve trung thuc trong mua diem}{Talking About Honesty During Grade Season}{Day trung thuc la bai kiem tra kho nhat khi ai cung bi ket qua len ap luc.}
\storycard{104. Thay noi it nhung lam ro}{Few Words, Clear Deeds}{Su mau muc am tham thuong giao duc manh hon nhieu bai dien thuyet.}
\storycard{105. Tro tot nhung kieng ki}{The Good Student Who Looks Down on Others}{Thanh tich khong tu dong dong nghia voi nhan cach dep.}
\storycard{106. Mot loi hua voi ca lop}{A Promise to the Class}{Khi loi hua duoc giu, niem tin cua tap the duoc boi dap rat nhanh.}
\storycard{107. Giu bi mat cho hoc sinh}{Keeping a Student's Trust}{Su tin cay duoc xay khi nguoi lon biet rung gioi han va bao ve pham gia cua tre.}
\storycard{108. Noi that ve su thien vi}{Admitting Bias}{Nhan ra do lech cua minh la buoc dau de tro thanh nguoi day cong bang hon.}
\storycard{109. Hanh vi dep khong co giam sat}{Doing Right Unseen}{Nhan cach duoc nhin ro nhat khi khong co ai cham diem.}
\storycard{110. Su ton trong hai chieu}{Respect Both Ways}{Lop hoc manh la noi ton trong khong chi di tu tren xuong, ma con tu duoi len bang su tu giac.}
