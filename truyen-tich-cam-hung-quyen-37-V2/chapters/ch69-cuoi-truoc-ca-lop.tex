\chapter{Cười Trước Cả Lớp}
\chapterintrobi{Có những tiếng cười đến sớm hơn cả nội dung chính}{Laugh before everyone else}{Một số bạn chỉ cần linh cảm có gì vui là đã phản ứng trước toàn lớp.}

\begin{lucbatbox}{Lục bát: Chưa rõ gì đã cười}
Câu kia mới chớm nửa vời\
Em đã cười trước cả nơi học hành\
Bạn còn đang đợi nghĩa thành\
Em như bắt sóng quá nhanh mất rồi\
Có khi chuyện chẳng buồn cười\
Nhưng em nhiệt tình nên vui hộ giùm
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Bắt nhịp quá nhanh}
Người ta còn đang hiểu đầu đuôi\
Em cười như thể rõ từ hồi\
Nếu phần đoán ý dùng vào bài tập\
Điểm số hẳn không đứng lẻ loi
\end{tutuyetbox}

\begin{ghichu}
Cười trước cả lớp thường không nguy hiểm, trừ khi nó biến thành thói quen chen nhịp và kéo sự chú ý lệch hướng.
\end{ghichu}

\begin{englishnote}
\textbf{laugh too early}: cười quá sớm.\par
\textbf{get the joke}: hiểu ý đùa.\par
\textbf{I laughed before I got it.}: Em cười trước khi hiểu hẳn.
\end{englishnote}
