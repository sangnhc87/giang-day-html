\chapter{Thái Độ Học Tập}
\chapterintro{Nhìn cách ngồi là đoán ra mức độ quyết tâm}{Lười học mà giữ mặt nghiêm là một tài lẻ học đường.}

\begin{lucbatbox}{Lục bát: Hứa rồi để đó}
Sáng nay em hứa học chăm
Mở vở được chút lại nằm thở than
Vở ghi mới lật một trang
Tâm tư đã chạy lang thang cuối ngàn
Miệng còn chốt một đôi lời
Mai con bứt phá, nay ngồi dưỡng thân
\end{lucbatbox}

\begin{lucbatbox}{Lục bát: Hứa rồi để đó}
Sáng nay em hứa học chăm
Mở vở chưa ấm đã nằm thở than
Vở ghi mới lật một trang
Tâm tư đã chạy lang thang cuối ngàn
Miệng còn chốt một đôi lời
Mai con bứt phá, nay ngồi dưỡng thân
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Mặt nghiêm, bài rỗng}
Mặt em nghiêm nghị lạ kỳ thay  
Tay cầm cây bút mắt xa bay  
Hỏi sang ý chính vừa xong đó  
Trong đầu gió lộng khắp trời mây
\end{tutuyetbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Mặt nghiêm, bài rỗng}
Mặt em nghiêm nghị tưởng cao siêu  
Tay cầm cây bút mắt liêu xiêu  
Thầy hỏi ý chính vừa xong đó  
Trong đầu gió thổi khắp năm chiều
\end{tutuyetbox}


\begin{ghichu}
Thái độ học tập lệch nhất là lúc quyết tâm phát biểu rất to, còn phần làm thật lại rất mỏng.
\end{ghichu}

\begin{lucbatbox}{Lục bát: Học theo cảm hứng}
Em rằng phải đợi cảm hứng
Không thì chữ nghĩa nửa chừng bỏ ngang
Lạ thay lướt mạng rất hăng
Tinh thần dào dạt, ngập tràn tự nhiên
Đến khi nhìn lại đề cương
Hứng đâu mất sạch, chỉ còn đau đầu
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Trì hoãn thành nghề}
Việc học em hẹn tối nay thôi
Đêm xuống lại dời sáng mai rồi
Ngỡ kế trong đầu luôn kín kẽ
Ngờ đâu điểm rớt tận xa xôi
\end{tutuyetbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Trì hoãn thành nghề}
Việc học em hẹn tối nay thôi
Đêm xuống còn dời sáng nữa rồi
Tưởng kế trong đầu luôn kín kẽ
Thi ra điểm rớt tít xa xôi
\end{tutuyetbox}

\ngancach

\begin{lucbatbox}{Lục bát: Học kiểu đối phó}
Học là học lúc bị kêu
Không ai nhắc nữa là chiều nghỉ ngơi
Sách em sạch đẹp tinh khôi
Nhìn như còn mới hơn hồi khai trường
Bạn bè tưởng rất chăm chăm
Ai ngờ chữ nghĩa xa xăm lâu rồi
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Tự tin hơi sớm}
Em hay gật nhẹ hiểu rồi nhé
Đến khi làm thử mới hay ngay
Phần hiểu tưởng đã dày dặn lắm
Chạm bài là nó gió bay bay
\end{tutuyetbox}
