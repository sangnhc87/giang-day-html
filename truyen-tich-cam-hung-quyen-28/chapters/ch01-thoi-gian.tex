\chapter{Thời gian không đợi ai}
\markboth{Thời gian không đợi ai}{Time Waits for No One}

\section{Người cha hẹn khi nào rảnh sẽ chở con đi chơi.}
\begin{truyen}{Người cha hẹn khi nào rảnh sẽ chở con đi chơi.}{Later can become never.}
\chuhoa{N}{gười cha bận từ sáng tới tối và lần nào con nhắc cũng bảo: chờ ba rảnh. Đến một ngày, con không nhắc nữa vì con đã lớn, và người cha mới hiểu có những lời hứa chỉ có giá trị khi còn đúng độ tuổi của đứa trẻ.}
\textit{(A father stayed busy all day and always said, wait until I am free. One day the child stopped asking because childhood had passed, and the father understood that some promises only matter at the right age.)}
\end{truyen}
\begin{baihoc}
Thời gian không chỉ trôi qua; nó cũng lấy đi đúng khoảnh khắc lẽ ra ta phải có mặt.
\textit{(Time does not only pass; it also removes the exact moment we should have been there.)}
\end{baihoc}
\begin{ghinhoanh}
\textbf{Short English to remember:}

Later can become never.

Be there while it still matters.
\end{ghinhoanh}
\begin{tuhocnhanh}
1. Em đang hẹn điều gì với người thân mà lẽ ra nên làm sớm hơn? \textit{(What are you postponing with your family that should happen sooner?)}

2. Có khoảnh khắc nào nếu qua rồi sẽ không trở lại đúng nghĩa nữa không? \textit{(Is there a moment in your life that will never return in the same way once it passes?)}
\end{tuhocnhanh}

\section{Một sinh viên nghĩ còn nhiều năm để học ngoại ngữ.}
\begin{truyen}{Một sinh viên nghĩ còn nhiều năm để học ngoại ngữ.}{Years disappear quietly.}
\chuhoa{M}{ỗi năm cậu đều nói mình sẽ bắt đầu nghiêm túc vào hè tới. Khi ra trường đi xin việc, cậu mới thấy cái giá của sự chậm trễ không phải là một bài học bỏ lỡ, mà là những cánh cửa đóng lại trước mặt.}
\textit{(Every year he said he would start seriously next summer. At graduation, he realized the price of delay was not a missed lesson but doors closing in front of him.)}
\end{truyen}
\begin{baihoc}
Nhiều người không thất bại vì thiếu năng lực, mà vì tưởng mình còn quá nhiều thời gian để chuẩn bị.
\textit{(Many people do not fail from lack of ability, but from assuming they still have plenty of time to prepare.)}
\end{baihoc}
\begin{ghinhoanh}
\textbf{Short English to remember:}

Years disappear quietly.

Start before the deadline starts you.
\end{ghinhoanh}
\begin{tuhocnhanh}
1. Việc gì em cứ dời mãi vì nghĩ còn sớm? \textit{(What do you keep delaying because it still feels early?)}

2. Nếu phải bắt đầu ngay tuần này, em sẽ bắt đầu bằng bước nào? \textit{(If you had to start this week, what would your first step be?)}
\end{tuhocnhanh}

\section{Một người giữ mãi câu xin lỗi trong lòng.}
\begin{truyen}{Một người giữ mãi câu xin lỗi trong lòng.}{Apologies also expire.}
\chuhoa{A}{nh ta nghĩ vài hôm nữa nói cũng được. Nhưng người cần nghe lời xin lỗi ấy đã mất hẳn niềm tin, rồi đi khỏi mối quan hệ. Câu xin lỗi vẫn còn đó, chỉ là không còn nơi để đến nữa.}
\textit{(He thought he could apologize a few days later. But the person who needed to hear it had already lost trust and left the relationship. The apology remained, but no longer had a place to go.)}
\end{truyen}
\begin{baihoc}
Không phải lời nào cũng có thể nói bất cứ lúc nào; có những lời chậm một nhịp là thành quá muộn.
\textit{(Not every word can be spoken at any time; some words become too late after a single delay.)}
\end{baihoc}
\begin{ghinhoanh}
\textbf{Short English to remember:}

Apologies also expire.

Say the right words in time.
\end{ghinhoanh}
\begin{tuhocnhanh}
1. Em có đang nợ ai một lời xin lỗi không? \textit{(Do you owe someone an apology right now?)}

2. Điều gì làm em chậm nói điều cần nói? \textit{(What makes you slow to say what must be said?)}
\end{tuhocnhanh}

\section{Người trẻ nghĩ thức khuya vài năm không sao.}
\begin{truyen}{Người trẻ nghĩ thức khuya vài năm không sao.}{The body keeps the score.}
\chuhoa{C}{ậu dùng sức trẻ như một khoản vay vô hạn: thức tới hai giờ sáng, ăn uống thất thường, ngủ bù bằng cà phê. Đến khi cơ thể báo động, cậu mới hiểu thời gian không chỉ đi qua lịch, nó còn đi qua từng tế bào.}
\textit{(He treated youth like an endless loan: sleeping at two in the morning, eating irregularly, replacing rest with coffee. When the body finally protested, he understood that time does not move only through calendars but through every cell.)}
\end{truyen}
\begin{baihoc}
Những gì ta tiêu xài bằng sức khỏe hôm nay sẽ được thời gian thu lại với lãi rất cao.
\textit{(What we spend from our health today will be collected by time with very high interest.)}
\end{baihoc}
\begin{ghinhoanh}
\textbf{Short English to remember:}

The body keeps the score.

Youth is not endless credit.
\end{ghinhoanh}
\begin{tuhocnhanh}
1. Em đang đối xử với thời gian nghỉ ngơi của mình ra sao? \textit{(How are you treating your time for rest right now?)}

2. Có thói quen nào nếu giữ tiếp sẽ thành món nợ với cơ thể? \textit{(Which habit will become a debt to your body if you keep it?)}
\end{tuhocnhanh}

\section{Một người đợi đến lúc thành công mới định sống.}
\begin{truyen}{Một người đợi đến lúc thành công mới định sống.}{Life is happening now.}
\chuhoa{A}{nh tự hứa khi có tiền hơn, khi xong việc hơn, khi ổn định hơn sẽ đi thăm nơi mình muốn, đọc sách mình thích, gặp người mình quý. Nhưng đến khi khá hơn thật, anh đã quen với việc hoãn niềm vui và không còn biết sống chậm lại thế nào.}
\textit{(He promised himself that when he had more money, less work, and more stability, he would travel, read, and meet the people he cared about. When life finally improved, he had already become used to postponing joy.)}
\end{truyen}
\begin{baihoc}
Nhiều người hiểu quá muộn rằng sống không phải phần thưởng ở cuối đường, mà là cách ta đi từng ngày trên đường.
\textit{(Many people understand too late that life is not a reward at the end of the road, but the way we walk each day along it.)}
\end{baihoc}
\begin{ghinhoanh}
\textbf{Short English to remember:}

Life is happening now.

Do not postpone being alive.
\end{ghinhoanh}
\begin{tuhocnhanh}
1. Em có đang treo niềm vui của mình vào một ngày rất xa không? \textit{(Are you hanging your joy on a very distant day?)}

2. Điều nhỏ nào em có thể sống trọn hơn ngay hôm nay? \textit{(What small thing can you live more fully today?)}
\end{tuhocnhanh}
