\newcommand{\extractpoemtitle}[1]{%
  \IfBeginWith{#1}{Lục bát: }{\StrBehind{#1}{Lục bát: }[\tempPoemTitle]}{%
    \IfBeginWith{#1}{Thất ngôn tứ tuyệt: }{\StrBehind{#1}{Thất ngôn tứ tuyệt: }[\tempPoemTitle]}{%
      \IfBeginWith{#1}{Tứ tuyệt cà khịa: }{\StrBehind{#1}{Tứ tuyệt cà khịa: }[\tempPoemTitle]}{%
        \def\tempPoemTitle{#1}%
      }%
    }%
  }%
}

\newtcolorbox{tutuyetbox}[1]{
  code={\extractpoemtitle{#1}},
  title={\tempPoemTitle},
  phantom={\addcontentsline{toc}{section}{\tempPoemTitle}},
}
