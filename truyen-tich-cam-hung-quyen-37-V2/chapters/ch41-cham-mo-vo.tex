\chapter{Chậm Mở Vở}
\chapterintrobi{Chưa học gì nhưng nghi thức kéo thời gian đã rất chuyên nghiệp}{Slow to open the notebook}{Một số bạn chưa cần trốn học, chỉ cần mở vở chậm là đã thắng được vài phút đầu giờ.}

\begin{lucbatbox}{Lục bát: Vở còn đang lấy đà}
Chuông vừa dứt một nhịp thôi\
Vở em vẫn ở chân trời cặp kia\
Bạn bên đã chép say sưa\
Em còn gỡ khóa như vừa leo non\
Thầy nhìn, em vẫn rất ngoan\
Chỉ là tốc độ hơi toan tính nhiều
\end{lucbatbox}

\begin{tutuyetbox}{Thất ngôn tứ tuyệt: Khởi động quá lâu}
Mở vở mà như mở chuyện đời
Tay em chuẩn bị rất là lâu
Nếu vào học cũng nhanh như thế
Có khi trang giấy bớt buồn rầu
\end{tutuyetbox}

\begin{ghichu}
Thói quen chậm mở vở nghe nhỏ, nhưng thường là phát súng báo hiệu cho một tiết học trôi khỏi tay rất nhanh.
\end{ghichu}

\end{englishnote}
